### 핵심 로직

```java
@SpringBootTest @Testcontainers
class AuditImmutabilityIT {

    @Autowired DataSource dataSource;
    @Autowired AuditLogRepository auditRepo;
    @Autowired VcScheduleRepository vcRepo;

    @ParameterizedTest
    @ValueSource(strings = {"app_user", "planner_role", "master_admin"})
    void update_audit_table_denied_for_role(String role) throws SQLException {
        // 1) 먼저 audit row 생성 (INSERT via 트리거)
        vcRepo.save(buildDraft());
        var logId = auditRepo.findLatestVcLog().getLogId();

        // 2) 해당 role로 UPDATE 시도
        try (Connection conn = openAsRole(role)) {
            try (PreparedStatement ps = conn.prepareStatement(
                "UPDATE audit.vc_schedule_log SET actor = 'tampered' WHERE log_id = ?")) {
                ps.setObject(1, logId);
                assertThatThrownBy(ps::executeUpdate)
                    .as("%s가 audit UPDATE 차단되어야 함", role)
                    .isInstanceOf(SQLException.class)
                    .hasMessageContaining("permission denied");
            }
        }
    }

    @ParameterizedTest
    @ValueSource(strings = {"app_user", "planner_role", "master_admin"})
    void delete_audit_row_denied_for_role(String role) throws SQLException {
        vcRepo.save(buildDraft());
        var logId = auditRepo.findLatestVcLog().getLogId();

        try (Connection conn = openAsRole(role)) {
            try (PreparedStatement ps = conn.prepareStatement(
                "DELETE FROM audit.vc_schedule_log WHERE log_id = ?")) {
                ps.setObject(1, logId);
                assertThatThrownBy(ps::executeUpdate).isInstanceOf(SQLException.class);
            }
        }
    }

    @ParameterizedTest
    @ValueSource(strings = {"app_user", "planner_role", "master_admin"})
    void truncate_audit_table_denied(String role) throws SQLException {
        try (Connection conn = openAsRole(role)) {
            try (PreparedStatement ps = conn.prepareStatement(
                "TRUNCATE TABLE audit.vc_schedule_log")) {
                assertThatThrownBy(ps::execute).isInstanceOf(SQLException.class);
            }
        }
    }

    @Test
    void audit_reader_can_select() throws SQLException {
        vcRepo.save(buildDraft());

        try (Connection conn = openAsRole("audit_reader")) {
            try (PreparedStatement ps = conn.prepareStatement(
                "SELECT COUNT(*) FROM audit.vc_schedule_log")) {
                try (ResultSet rs = ps.executeQuery()) {
                    rs.next();
                    assertThat(rs.getLong(1)).isGreaterThan(0);
                }
            }
        }
    }

    @Test
    void audit_reader_cannot_insert() throws SQLException {
        try (Connection conn = openAsRole("audit_reader")) {
            try (PreparedStatement ps = conn.prepareStatement(
                "INSERT INTO audit.vc_schedule_log (target_id, operation, actor) VALUES (?, 'UPDATE', 'fake')")) {
                ps.setObject(1, UUID.randomUUID());
                assertThatThrownBy(ps::executeUpdate).isInstanceOf(SQLException.class);
            }
        }
    }

    @Test
    void new_audit_table_inherits_default_privileges() throws SQLException {
        // 신규 audit 테이블 생성 (Flyway 시뮬레이션)
        try (Connection conn = openAsMasterAdmin()) {
            try (Statement s = conn.createStatement()) {
                s.execute("CREATE TABLE audit.test_log_new (log_id UUID PRIMARY KEY, data TEXT)");
            }
        }

        // app_user UPDATE 시도 → 자동 denied (DEFAULT PRIVILEGES)
        try (Connection conn = openAsAppUser()) {
            try (PreparedStatement ps = conn.prepareStatement(
                "UPDATE audit.test_log_new SET data = 'x' WHERE 1=0")) {
                assertThatThrownBy(ps::executeUpdate).isInstanceOf(SQLException.class);
            }
        }

        // cleanup
        try (Connection conn = openAsMasterAdmin()) {
            try (Statement s = conn.createStatement()) {
                s.execute("DROP TABLE audit.test_log_new");
            }
        }
    }
}
```
