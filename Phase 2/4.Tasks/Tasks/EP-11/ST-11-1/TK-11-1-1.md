### Flyway `V4_3__create_audit_schema_and_tables.sql`

```sql
CREATE SCHEMA IF NOT EXISTS audit;

-- VC schedule audit log (partitioned by month)
CREATE TABLE audit.vc_schedule_log (
    log_id          UUID NOT NULL DEFAULT gen_random_uuid(),
    occurred_at     TIMESTAMPTZ NOT NULL DEFAULT now(),
    target_id       UUID NOT NULL,
    operation       VARCHAR(10) NOT NULL CHECK (operation IN ('INSERT','UPDATE','DELETE')),
    old_row         JSONB,
    new_row         JSONB,
    actor           VARCHAR(40) NOT NULL,
    reason          TEXT,
    application     VARCHAR(40) NOT NULL DEFAULT 'scheduling-engine',
    PRIMARY KEY (log_id, occurred_at)
) PARTITION BY RANGE (occurred_at);

-- 2026 ~ 2029 (3년 보존) 월별 파티션 — 본 마이그레이션 시 12개월 사전 생성
DO $$
DECLARE
    v_start DATE := '2026-01-01';
    v_end DATE := '2029-12-31';
    v_part_start DATE;
BEGIN
    v_part_start := v_start;
    WHILE v_part_start <= v_end LOOP
        EXECUTE format(
            'CREATE TABLE IF NOT EXISTS audit.vc_schedule_log_%s PARTITION OF audit.vc_schedule_log ' ||
            'FOR VALUES FROM (%L) TO (%L)',
            to_char(v_part_start, 'YYYYMM'),
            v_part_start,
            v_part_start + INTERVAL '1 month'
        );
        v_part_start := v_part_start + INTERVAL '1 month';
    END LOOP;
END$$;

CREATE INDEX idx_vc_audit_target ON audit.vc_schedule_log(target_id, occurred_at);
CREATE INDEX idx_vc_audit_actor ON audit.vc_schedule_log(actor, occurred_at);

-- ex_schedule_log·order_log 동일 구조
CREATE TABLE audit.ex_schedule_log (LIKE audit.vc_schedule_log INCLUDING ALL)
    PARTITION BY RANGE (occurred_at);
-- ... 파티션 생성

CREATE TABLE audit.order_log (LIKE audit.vc_schedule_log INCLUDING ALL)
    PARTITION BY RANGE (occurred_at);
-- ... 파티션 생성

-- audit role 분리
CREATE ROLE IF NOT EXISTS audit_reader NOLOGIN;
GRANT USAGE ON SCHEMA audit TO audit_reader, app_user;
GRANT SELECT ON ALL TABLES IN SCHEMA audit TO audit_reader;
GRANT INSERT, SELECT ON ALL TABLES IN SCHEMA audit TO app_user;
-- UPDATE/DELETE는 ST-11-2에서 REVOKE
```

### Flyway `V4_3_data__create_audit_triggers.sql`

```sql
-- VC schedule audit trigger
CREATE OR REPLACE FUNCTION audit_vc_schedule()
RETURNS TRIGGER AS $$
DECLARE
    v_actor TEXT := current_setting('audit.actor', TRUE);
    v_reason TEXT := current_setting('audit.reason', TRUE);
BEGIN
    IF v_actor IS NULL OR v_actor = '' THEN v_actor := 'system'; END IF;

    INSERT INTO audit.vc_schedule_log (target_id, operation, old_row, new_row, actor, reason)
    VALUES (
        COALESCE(NEW.id, OLD.id),
        TG_OP,
        CASE WHEN TG_OP = 'INSERT' THEN NULL ELSE to_jsonb(OLD) END,
        CASE WHEN TG_OP = 'DELETE' THEN NULL ELSE to_jsonb(NEW) END,
        v_actor,
        v_reason
    );

    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_audit_vc_schedule
    AFTER INSERT OR UPDATE OR DELETE ON vc_schedule
    FOR EACH ROW EXECUTE FUNCTION audit_vc_schedule();

-- ex_schedule·order 동일 패턴
CREATE OR REPLACE FUNCTION audit_ex_schedule() ...  -- 동일 구조
CREATE OR REPLACE FUNCTION audit_order() ...
CREATE TRIGGER trg_audit_ex_schedule ...
CREATE TRIGGER trg_audit_order ...
```
