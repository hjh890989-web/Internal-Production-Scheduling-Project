### Flyway `V5_3__create_import_queue.sql`

```sql
CREATE TABLE import_queue (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    file_path       TEXT NOT NULL,
    file_size       BIGINT NOT NULL,
    queued_at       TIMESTAMPTZ NOT NULL DEFAULT now(),
    started_at      TIMESTAMPTZ,
    completed_at    TIMESTAMPTZ,
    status          VARCHAR(20) NOT NULL DEFAULT 'QUEUED'
        CHECK (status IN ('QUEUED', 'PROCESSING', 'DONE', 'FAILED')),
    error_message   TEXT,
    rows_imported   INTEGER,
    actor           VARCHAR(40) NOT NULL DEFAULT 'system:watcher'
);

CREATE INDEX idx_import_queue_pending ON import_queue(status, queued_at) WHERE status = 'QUEUED';
```

### `ImportQueueProcessor.java`

```java
@Component @RequiredArgsConstructor @Slf4j
public class ImportQueueProcessor {

    private final ImportQueueRepository queueRepo;
    private final MasterImportService importService;
    private final SlackNotificationService slack;

    /**
     * 5초마다 QUEUED row pick → import 실행.
     */
    @Scheduled(fixedDelay = 5_000)
    @Auditable(reason = "'auto-import queue processing'")
    @Transactional
    public void processQueue() {
        var pending = queueRepo.findByStatusOrderByQueuedAtAsc(QueueStatus.QUEUED).stream()
            .limit(5).toList();

        for (var entry : pending) {
            processEntry(entry);
        }
    }

    private void processEntry(ImportQueueEntry entry) {
        entry.setStatus(QueueStatus.PROCESSING);
        entry.setStartedAt(Instant.now());
        queueRepo.save(entry);

        try {
            byte[] data = Files.readAllBytes(Path.of(entry.getFilePath()));
            var result = importService.importAll(data);

            entry.setStatus(QueueStatus.DONE);
            entry.setCompletedAt(Instant.now());
            entry.setRowsImported(result.totalRowsImported());
            queueRepo.save(entry);

            slack.notifySuccess("자동 ingest 완료: " + entry.getFilePath());
            log.info("Auto-import 완료: file={}, rows={}", entry.getFilePath(), result.totalRowsImported());

        } catch (Exception e) {
            entry.setStatus(QueueStatus.FAILED);
            entry.setCompletedAt(Instant.now());
            entry.setErrorMessage(e.getMessage());
            queueRepo.save(entry);

            slack.notifyError("자동 ingest 실패: " + entry.getFilePath() + " - " + e.getMessage());
            log.error("Auto-import 실패: file={}", entry.getFilePath(), e);
        }
    }
}
```
