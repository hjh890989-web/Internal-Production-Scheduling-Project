### `FolderWatchService.java`

```java
@Service @RequiredArgsConstructor @Slf4j
@ConditionalOnProperty(name = "import.watch.enabled", havingValue = "true")
public class FolderWatchService {

    @Value("${import.watch.folder.path}")
    private String folderPath;

    @Value("${import.watch.interval-ms}")
    private long intervalMs;

    private final ImportQueueListener listener;
    private FileAlterationMonitor monitor;

    @PostConstruct
    public void start() throws Exception {
        var folder = new File(folderPath);
        if (!folder.exists()) folder.mkdirs();

        var observer = new FileAlterationObserver(folder, file -> {
            String name = file.getName().toLowerCase();
            return name.endsWith(".xlsx") || name.endsWith(".xls");
        });
        observer.addListener(listener);

        monitor = new FileAlterationMonitor(intervalMs);
        monitor.addObserver(observer);
        monitor.start();

        log.info("FolderWatchService 시작: path={}, interval={}ms", folderPath, intervalMs);
    }

    @PreDestroy
    public void stop() throws Exception {
        if (monitor != null) monitor.stop();
    }
}
```

### `ImportQueueListener.java`

```java
@Component @RequiredArgsConstructor @Slf4j
public class ImportQueueListener extends FileAlterationListenerAdaptor {

    private final ImportQueueRepository queueRepo;
    private final FileStabilityChecker stabilityChecker;

    @Value("${import.watch.stability-check-ms}")
    private long stabilityMs;

    @Override
    public void onFileCreate(File file) {
        log.info("신규 파일 감지: {}", file.getPath());

        // stability check — partial write 회피
        if (!stabilityChecker.waitUntilStable(file, stabilityMs)) {
            log.warn("파일 stability check 실패 (장시간 쓰기 진행 중): {}", file.getPath());
            return;
        }

        // 큐 등록 (TK-20-1-2)
        var queueEntry = ImportQueueEntry.builder()
            .filePath(file.getAbsolutePath())
            .fileSize(file.length())
            .queuedAt(Instant.now())
            .status(QueueStatus.QUEUED)
            .build();
        queueRepo.save(queueEntry);
    }
}

@Component
public class FileStabilityChecker {

    public boolean waitUntilStable(File file, long checkMs) {
        long initial = file.lastModified();
        try {
            Thread.sleep(checkMs);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            return false;
        }
        return file.lastModified() == initial;     // 변경 없음 = stable
    }
}
```
