### 핵심 로직

```java
@SpringBootTest @Testcontainers
@TestPropertySource(properties = {
    "import.watch.folder.path=${java.io.tmpdir}/test-watch",
    "import.watch.interval-ms=2000",
    "import.watch.stability-check-ms=2000"
})
class FolderWatchSlaIT {

    @Autowired ImportQueueRepository queueRepo;

    @Test
    void file_drop_to_queue_registration_within_60s() throws Exception {
        Path testDir = Path.of(System.getProperty("java.io.tmpdir"), "test-watch");
        Files.createDirectories(testDir);

        var latencies = new ArrayList<Long>();
        for (int i = 0; i < 10; i++) {
            Path testFile = testDir.resolve("test-" + i + ".xlsx");
            long t0 = System.currentTimeMillis();

            Files.write(testFile, loadFixture("sales-order-sample.xlsx"));

            // 큐 등록 대기
            await().atMost(60, SECONDS).until(() ->
                queueRepo.existsByFilePathAndStatusIn(testFile.toString(),
                    List.of(QueueStatus.QUEUED, QueueStatus.PROCESSING, QueueStatus.DONE)));

            long latency = System.currentTimeMillis() - t0;
            latencies.add(latency);

            // cleanup
            queueRepo.deleteByFilePath(testFile.toString());
            Files.deleteIfExists(testFile);
        }

        latencies.sort(Long::compare);
        long p95 = latencies.get((int) (latencies.size() * 0.95));

        System.out.printf("60s SLA latency: p95=%dms%n", p95);
        assertThat(p95).as("TC-XT-003 60초 SLA").isLessThanOrEqualTo(60_000);
    }

    @Test
    void partial_write_skipped_until_stable() throws Exception {
        Path testDir = Path.of(System.getProperty("java.io.tmpdir"), "test-watch");
        Path testFile = testDir.resolve("slow-write.xlsx");

        try (OutputStream out = Files.newOutputStream(testFile)) {
            // 천천히 쓰기 (3초 동안)
            for (int i = 0; i < 30; i++) {
                out.write(new byte[1024]);
                Thread.sleep(100);
            }
        }

        // 5초 후에야 stability 인지 → 큐 등록
        await().atMost(15, SECONDS).until(() ->
            queueRepo.existsByFilePath(testFile.toString()));

        Files.deleteIfExists(testFile);
    }

    @Test
    void non_xlsx_file_ignored() throws Exception {
        Path testDir = Path.of(System.getProperty("java.io.tmpdir"), "test-watch");
        Path txtFile = testDir.resolve("ignored.txt");
        Files.write(txtFile, "test".getBytes());

        Thread.sleep(15_000);     // 15초 대기

        assertThat(queueRepo.existsByFilePath(txtFile.toString()))
            .as(".txt 파일은 ignore")
            .isFalse();

        Files.deleteIfExists(txtFile);
    }
}
```
