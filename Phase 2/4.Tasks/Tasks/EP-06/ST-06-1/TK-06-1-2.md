### 핵심 로직

#### A. `BackwardDeadlineCalculator.java`

```java
package com.scheduling.vc.deadline;

import com.scheduling.calendar.WorkingCalendarService;
import com.scheduling.order.domain.Order;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.*;

@Service @RequiredArgsConstructor
public class BackwardDeadlineCalculator {

    private static final int BACKWARD_DAYS = 2;     // BR-X07: 성형 D-2

    private final WorkingCalendarService calendar;

    /**
     * REQ-FUNC-VC-008: deadline = delivery_date − 2 working days.
     *
     * 같은 hose_id에 여러 수주가 있으면 가장 이른 납기 기준 deadline 사용
     * (가장 빠른 일정이 hard 제약).
     */
    public DeadlineMap compute(List<Order> orders) {
        var earliestDeliveryByHose = new HashMap<String, LocalDate>();
        for (Order o : orders) {
            earliestDeliveryByHose.merge(o.getHoseId(), o.getDeliveryDate(),
                (a, b) -> a.isBefore(b) ? a : b);
        }

        var deadlines = new HashMap<String, LocalDate>();
        for (var entry : earliestDeliveryByHose.entrySet()) {
            LocalDate deadline = calendar.subtractWorkingDays(entry.getValue(), BACKWARD_DAYS);
            deadlines.put(entry.getKey(), deadline);
        }
        return new DeadlineMap(Map.copyOf(deadlines));
    }

    /**
     * 단일 수주의 D-2 deadline.
     */
    public LocalDate deadlineFor(LocalDate deliveryDate) {
        return calendar.subtractWorkingDays(deliveryDate, BACKWARD_DAYS);
    }
}

public record DeadlineMap(Map<String, LocalDate> map) {
    public Optional<LocalDate> get(String hoseId) { return Optional.ofNullable(map.get(hoseId)); }
    public boolean isWithinDeadline(String hoseId, LocalDate productionDate) {
        return get(hoseId).map(d -> !productionDate.isAfter(d)).orElse(true);
    }
}
```
