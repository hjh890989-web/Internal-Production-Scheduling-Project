#### A. `DeadlineRegressionIT.java`

```java
package com.scheduling.vc.deadline;

import com.scheduling.calendar.WorkingCalendarService;
import com.scheduling.vc.allocator.*;
import com.scheduling.vc.schedule.VcScheduleRepository;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.jdbc.Sql;
import org.testcontainers.junit.jupiter.Testcontainers;

import java.time.LocalDate;
import java.util.List;

import static org.assertj.core.api.Assertions.assertThat;

@SpringBootTest @Testcontainers
@Sql("classpath:datasets/DS-VC-DEADLINE-100/master_seed.sql")
class DeadlineRegressionIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired BackwardDeadlineCalculator deadlineCalc;
    @Autowired WorkingCalendarService calendar;
    @Autowired VcScheduleRepository scheduleRepo;
    @Autowired ScenarioLoader loader;

    /**
     * TC-VC-008 — 100 시나리오 회귀.
     * 다양한 납기일 분포 (월/화/.../금, 주말 후 월요일, 연휴 직후 등) 100건.
     * 모든 row.production_date ≤ deadline 강제.
     */
    @Test
    void all_rows_within_d2_deadline_for_100_scenarios() {
        var scenarios = loader.load("DS-VC-DEADLINE-100/scenarios.csv");
        assertThat(scenarios).hasSize(100);

        var ctx = AllocationContext.fromOrders(scenarios, LocalDate.of(2026, 2, 16), 30);
        AllocationResult result = allocator.allocate(ctx);

        // 1. 회귀 — 모든 row production_date ≤ deadline
        var deadlines = deadlineCalc.compute(ctx.allOrders());
        var violations = result.schedule().stream()
            .filter(row -> !deadlines.isWithinDeadline(row.hoseId(), row.productionDate()))
            .toList();

        assertThat(violations)
            .as("BR-X07 위반: %d개 row의 production_date > deadline", violations.size())
            .isEmpty();

        // 2. K-V04 D-2 준수율 — 모든 hose 100% (충돌 제외)
        long totalHoses = ctx.allOrders().stream().map(o -> o.getHoseId()).distinct().count();
        long compliantHoses = totalHoses - result.conflicts().stream()
            .filter(c -> c.reason().contains("BR-X07"))
            .map(AllocationConflict::hoseId).distinct().count();
        double complianceRate = (double) compliantHoses / totalHoses;

        assertThat(complianceRate)
            .as("K-V04 D-2 준수율 (REQ-NF-KPI-008)")
            .isGreaterThanOrEqualTo(0.98);
    }

    @Test
    void single_hose_multi_orders_uses_earliest_delivery() {
        // 같은 hose_id 다중 수주 — 가장 이른 납기 기준 deadline
        var orders = List.of(
            Order.of("28912-2U000", LocalDate.of(2026, 2, 27), 100),
            Order.of("28912-2U000", LocalDate.of(2026, 3, 6),  150),
            Order.of("28912-2U000", LocalDate.of(2026, 3, 13), 100)
        );

        var deadlines = deadlineCalc.compute(orders);
        // earliest 2/27(금) − 2 영업일 = 2/25(수)
        assertThat(deadlines.get("28912-2U000")).contains(LocalDate.of(2026, 2, 25));

        // allocator 실행 후 모든 production_date ≤ 2/25
        var ctx = AllocationContext.fromOrders(orders, LocalDate.of(2026, 2, 16), 30);
        var result = allocator.allocate(ctx);
        assertThat(result.schedule())
            .allMatch(r -> !r.productionDate().isAfter(LocalDate.of(2026, 2, 25)));
    }
}
```

#### B. `DeadlineEdgeCaseIT.java`

```java
@SpringBootTest @Testcontainers
class DeadlineEdgeCaseIT {

    @Autowired BackwardDeadlineCalculator deadlineCalc;
    @Autowired WorkingCalendarService calendar;

    @Test
    void delivery_on_monday_after_weekend() {
        // 납기 월요일 → 2 영업일 차감 = 직전 주 목요일
        LocalDate delivery = LocalDate.of(2026, 3, 9);   // 월
        LocalDate deadline = deadlineCalc.deadlineFor(delivery);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 3, 5));   // 목
    }

    @Test
    void delivery_right_after_lunar_new_year() {
        // 2/16~2/18 설날 연휴 → 2/23(월) 납기 → deadline = 2/19(목)
        LocalDate delivery = LocalDate.of(2026, 2, 23);
        LocalDate deadline = deadlineCalc.deadlineFor(delivery);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 2, 19));
    }

    @Test
    void delivery_friday_normal_case() {
        LocalDate delivery = LocalDate.of(2026, 2, 20);
        LocalDate deadline = deadlineCalc.deadlineFor(delivery);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 2, 18));
    }

    @Test
    void delivery_on_non_working_day_treated_consistently() {
        // 납기가 토요일이라도 거기서부터 영업일 역산 — 결과 deterministic
        LocalDate delivery = LocalDate.of(2026, 2, 21);   // 토
        LocalDate deadline = deadlineCalc.deadlineFor(delivery);
        // 2/21(토)→2/20(금) 1차감 → 2/19(목) 2차감
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 2, 19));
    }

    @Test
    void consecutive_holidays_chuseok() {
        // 추석 9/24~9/26 → 9/29(화) 납기 → 9/24~9/26·9/27토·9/28일 영업일 제외
        // 9/29(화) − 1 = 9/23(수), − 2 = 9/22(화)
        LocalDate delivery = LocalDate.of(2026, 9, 29);
        LocalDate deadline = deadlineCalc.deadlineFor(delivery);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 9, 22));
    }
}
```
