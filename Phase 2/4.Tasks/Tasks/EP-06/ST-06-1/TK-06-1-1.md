#### C. `WorkingCalendarService.java` 정식 구현

```java
package com.scheduling.calendar;

import com.github.benmanes.caffeine.cache.*;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import jakarta.annotation.PostConstruct;
import java.time.*;
import java.util.*;
import java.util.concurrent.atomic.AtomicReference;

@Service @RequiredArgsConstructor
public class WorkingCalendarService {

    private final HolidayRepository repo;
    private final AtomicReference<Set<LocalDate>> holidaysCache = new AtomicReference<>();

    @PostConstruct
    public void initCache() {
        reload();
    }

    public synchronized void reload() {
        var all = repo.findAll();
        holidaysCache.set(all.stream()
            .map(Holiday::getHolidayDate)
            .collect(java.util.stream.Collectors.toUnmodifiableSet()));
    }

    public boolean isWorkingDay(LocalDate date) {
        DayOfWeek d = date.getDayOfWeek();
        if (d == DayOfWeek.SATURDAY || d == DayOfWeek.SUNDAY) return false;
        return !holidaysCache.get().contains(date);
    }

    public boolean isHoliday(LocalDate date) {
        return holidaysCache.get().contains(date);
    }

    /**
     * `date`로부터 `n` 영업일 후 날짜.
     * n=0 → date 자체 (영업일이 아니면 다음 영업일)
     * n=1 → 다음 영업일
     */
    public LocalDate addWorkingDays(LocalDate date, int n) {
        if (n < 0) return subtractWorkingDays(date, -n);
        LocalDate result = date;
        int added = 0;
        while (added < n) {
            result = result.plusDays(1);
            if (isWorkingDay(result)) added++;
        }
        return result;
    }

    /**
     * `date`로부터 `n` 영업일 전 날짜. (D-N 역산)
     * 예: delivery_date 2026-02-20(금) - 2 working days = 2026-02-18(수)
     *     (2/19 목·2/18 수 영업일 카운트)
     */
    public LocalDate subtractWorkingDays(LocalDate date, int n) {
        if (n < 0) return addWorkingDays(date, -n);
        LocalDate result = date;
        int subtracted = 0;
        while (subtracted < n) {
            result = result.minusDays(1);
            if (isWorkingDay(result)) subtracted++;
        }
        return result;
    }

    public long workingDaysBetween(LocalDate from, LocalDate to) {
        if (from.isAfter(to)) return -workingDaysBetween(to, from);
        long count = 0;
        for (LocalDate d = from; !d.isAfter(to); d = d.plusDays(1)) {
            if (isWorkingDay(d)) count++;
        }
        return count;
    }

    /**
     * 호라이즌 [from, to] 안의 영업일 리스트.
     */
    public List<LocalDate> workingDaysInRange(LocalDate from, LocalDate to) {
        var list = new ArrayList<LocalDate>();
        for (LocalDate d = from; !d.isAfter(to); d = d.plusDays(1)) {
            if (isWorkingDay(d)) list.add(d);
        }
        return list;
    }
}
```

#### D. `HolidayController.java` (IT 관리자 endpoint)

```java
@RestController
@RequestMapping("/api/v1/master/holiday")
@RequiredArgsConstructor
public class HolidayController {

    private final HolidayRepository repo;
    private final WorkingCalendarService calendar;

    @GetMapping
    @PreAuthorize("isAuthenticated()")
    public List<Holiday> list(@RequestParam(required = false) Integer year) {
        return year != null
            ? repo.findByYear(year)
            : repo.findAll();
    }

    @PostMapping
    @PreAuthorize("hasRole('IT_OPS')")
    public Holiday add(@Valid @RequestBody Holiday holiday) {
        var saved = repo.save(holiday);
        calendar.reload();          // 캐시 갱신
        return saved;
    }

    @DeleteMapping("/{date}")
    @PreAuthorize("hasRole('IT_OPS')")
    public void remove(@PathVariable LocalDate date) {
        repo.deleteById(date);
        calendar.reload();
    }
}
```
