### `MasterExcelExporter.java`

```java
@Service @RequiredArgsConstructor
public class MasterExcelExporter {

    private final VcConstraintRepository vcRepo;
    private final ExConstraintRepository exRepo;
    private final ProductSettingGroupRepository psgRepo;
    private final HolidayRepository holidayRepo;
    private final ExcelTemplateLoader templateLoader;

    /**
     * 통합 마스터 → MASTER.xlsx (원본 템플릿 기반).
     *
     * 시트:
     *   - 성형공정_제약조건 (VC_CONSTRAINT + LP_LEFT/RIGHT)
     *   - 압출공정_제약조건 (EX_CONSTRAINT + spec)
     *   - 셋팅그룹 (product_setting_group)
     *   - 공휴일 (holiday)
     */
    public byte[] export() {
        try (XSSFWorkbook workbook = templateLoader.load("master-template.xlsx");
             ByteArrayOutputStream baos = new ByteArrayOutputStream()) {

            writeVcConstraintSheet(workbook.getSheet("성형공정_제약조건"));
            writeExConstraintSheet(workbook.getSheet("압출공정_제약조건"));
            writeSettingGroupSheet(workbook.getSheet("셋팅그룹"));
            writeHolidaySheet(workbook.getSheet("공휴일"));

            // 수식 cached value 동기화 (TK-12-1-2)
            XSSFFormulaEvaluator.evaluateAllFormulaCells(workbook);

            workbook.write(baos);
            return baos.toByteArray();
        } catch (IOException e) {
            throw new ExportException("master export 실패", e);
        }
    }

    private void writeVcConstraintSheet(Sheet sheet) {
        var data = vcRepo.findAll();
        int rowIdx = 1;     // header 유지, data row 1번부터
        for (var vc : data) {
            Row row = sheet.getRow(rowIdx);
            if (row == null) row = sheet.createRow(rowIdx);

            setCell(row, 0, vc.getHoseId());
            setCell(row, 1, vc.getMachineId());
            // ... 슬롯 O/X 컬럼
            setCell(row, 10, vc.getLpLeftSetting());    // K열
            setCell(row, 11, vc.getLpRightSetting());   // L열
            rowIdx++;
        }
    }

    private void setCell(Row row, int idx, String value) {
        Cell cell = row.getCell(idx, MissingCellPolicy.CREATE_NULL_AS_BLANK);
        cell.setCellValue(value != null ? value : "");
    }
    // writeExConstraint·writeSettingGroup·writeHoliday 동일 패턴
}
```

### `ExcelTemplateLoader.java`

```java
@Component @RequiredArgsConstructor
public class ExcelTemplateLoader {

    private final ResourceLoader resourceLoader;

    public XSSFWorkbook load(String templateName) throws IOException {
        var resource = resourceLoader.getResource("classpath:templates/" + templateName);
        try (InputStream is = resource.getInputStream()) {
            return new XSSFWorkbook(is);
        }
    }
}
```

### `MasterExportController.java`

```java
@RestController @RequestMapping("/api/v1/export") @RequiredArgsConstructor
public class MasterExportController {

    private final MasterExcelExporter exporter;

    @GetMapping("/master")
    @PreAuthorize("isAuthenticated()")
    @Auditable(reason = "'export master ' + #root.principal")
    public ResponseEntity<byte[]> exportMaster() {
        byte[] data = exporter.export();
        String filename = "master_" + LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd")) + ".xlsx";

        return ResponseEntity.ok()
            .header(HttpHeaders.CONTENT_DISPOSITION, "attachment; filename=" + filename)
            .contentType(MediaType.parseMediaType(
                "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"))
            .body(data);
    }
}
```
