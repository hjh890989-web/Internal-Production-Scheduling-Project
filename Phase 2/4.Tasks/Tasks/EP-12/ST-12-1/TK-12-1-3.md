### `MasterExportDiffIT.java`

```java
@SpringBootTest @Testcontainers
class MasterExportDiffIT {

    @Autowired MasterExcelExporter exporter;
    @Autowired MasterImportService importer;        // EP-01

    @Test
    void original_to_export_cell_diff_under_two_percent() throws IOException {
        // 1) 원본 fixture 로드 → DB import (EP-01 Parser)
        byte[] originalBytes;
        try (var is = getClass().getResourceAsStream("/fixtures/MASTER_original_47.xlsx")) {
            originalBytes = is.readAllBytes();
        }
        importer.importAll(originalBytes);

        // 2) Export
        byte[] exported = exporter.export();

        // 3) Diff 계산
        try (var origWb = new XSSFWorkbook(new ByteArrayInputStream(originalBytes));
             var expWb = new XSSFWorkbook(new ByteArrayInputStream(exported))) {

            double diffRatio = ExcelDiff.diffRatio(origWb, expWb);

            System.out.printf("Cell-level diff ratio: %.4f%n", diffRatio);
            assertThat(diffRatio)
                .as("TC-OC-013: 셀-수준 차이 ≤ 2%%")
                .isLessThan(0.02);
        }
    }

    @Test
    void formula_cells_preserved_exactly() throws IOException {
        byte[] originalBytes = loadFixture("MASTER_original_47.xlsx");
        importer.importAll(originalBytes);

        byte[] exported = exporter.export();

        try (var origWb = new XSSFWorkbook(new ByteArrayInputStream(originalBytes));
             var expWb = new XSSFWorkbook(new ByteArrayInputStream(exported))) {

            // 모든 시트의 FORMULA cell 추출
            var origFormulas = collectFormulas(origWb);
            var expFormulas = collectFormulas(expWb);

            assertThat(expFormulas)
                .as("FORMULA cell 수식 100% 보존")
                .containsExactlyInAnyOrderElementsOf(origFormulas);
        }
    }

    @Test
    void sheet_names_preserved() throws IOException {
        byte[] exported = exporter.export();
        try (var wb = new XSSFWorkbook(new ByteArrayInputStream(exported))) {
            List<String> sheetNames = new ArrayList<>();
            for (int i = 0; i < wb.getNumberOfSheets(); i++) {
                sheetNames.add(wb.getSheetName(i));
            }
            assertThat(sheetNames)
                .containsExactlyInAnyOrder("성형공정_제약조건", "압출공정_제약조건", "셋팅그룹", "공휴일");
        }
    }
}
```

### `RoundTripIT.java`

```java
@SpringBootTest @Testcontainers
class RoundTripIT {

    @Autowired MasterImportService importer;     // EP-01
    @Autowired MasterExcelExporter exporter;
    @Autowired VcConstraintRepository vcRepo;

    @Test
    void parser_to_exporter_to_parser_round_trip_equivalent() throws IOException {
        // 1) 원본 → DB import
        byte[] original = loadFixture("MASTER_original_47.xlsx");
        importer.importAll(original);
        var beforeRows = vcRepo.findAll().stream()
            .collect(Collectors.toMap(VcConstraint::getHoseId, Function.identity()));

        // 2) DB → Excel export
        byte[] exported = exporter.export();

        // 3) Export → DB re-import (delete first)
        vcRepo.deleteAll();
        importer.importAll(exported);
        var afterRows = vcRepo.findAll().stream()
            .collect(Collectors.toMap(VcConstraint::getHoseId, Function.identity()));

        // 4) 비교 — 동일 hose_id에 동일 data
        assertThat(afterRows).hasSameSizeAs(beforeRows);
        for (var entry : beforeRows.entrySet()) {
            var after = afterRows.get(entry.getKey());
            assertThat(after).as("round-trip hose %s", entry.getKey()).isNotNull();
            assertThat(after.getMachineId()).isEqualTo(entry.getValue().getMachineId());
            assertThat(after.getLpLeftSetting()).isEqualTo(entry.getValue().getLpLeftSetting());
            // ... 모든 필드 비교
        }
    }
}
```
