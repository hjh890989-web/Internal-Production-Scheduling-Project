### 핵심 로직

```java
@SpringBootTest @Testcontainers
@Sql("classpath:datasets/DS-EX-MATRIX-7DAY/schedule_seed.sql")
class ExtrusionMatrixRegressionIT {

    @Autowired ExtrusionMatrixExporter exporter;

    @Test
    void seven_day_export_yields_seven_sheets() throws IOException {
        byte[] data = exporter.export(
            LocalDate.of(2026, 3, 2),
            LocalDate.of(2026, 3, 8));

        try (var wb = new XSSFWorkbook(new ByteArrayInputStream(data))) {
            assertThat(wb.getNumberOfSheets()).isEqualTo(7);

            // 시트명 모두 정규식 일치 (BR-E09)
            for (int i = 0; i < wb.getNumberOfSheets(); i++) {
                assertThat(ExtrusionSheetName.matches(wb.getSheetName(i)))
                    .as("sheet %d name regex", i)
                    .isTrue();
            }
        }
    }

    @Test
    void empty_dates_are_skipped() throws IOException {
        // 3월 2일·4일·6일만 스케줄 존재 (mid-empty)
        byte[] data = exporter.export(
            LocalDate.of(2026, 3, 2),
            LocalDate.of(2026, 3, 8));

        try (var wb = new XSSFWorkbook(new ByteArrayInputStream(data))) {
            var sheetNames = new ArrayList<String>();
            for (int i = 0; i < wb.getNumberOfSheets(); i++) sheetNames.add(wb.getSheetName(i));

            // 빈 날짜 (3/3, 3/5, 3/7, 3/8) 시트 없음
            assertThat(sheetNames).doesNotContain("3월3일(압출)", "3월5일(압출)");
        }
    }

    @Test
    void matrix_structure_4_shifts_n_lines() throws IOException {
        byte[] data = exporter.export(
            LocalDate.of(2026, 3, 2), LocalDate.of(2026, 3, 2));

        try (var wb = new XSSFWorkbook(new ByteArrayInputStream(data))) {
            Sheet sheet = wb.getSheetAt(0);

            // Row 0: header
            assertThat(sheet.getRow(0).getCell(0).getStringCellValue()).isEqualTo("Shift");

            // Row 1~4: 4 shift
            assertThat(sheet.getRow(1).getCell(0).getStringCellValue()).isEqualTo("주간전반");
            assertThat(sheet.getRow(2).getCell(0).getStringCellValue()).isEqualTo("주간후반");
            assertThat(sheet.getRow(3).getCell(0).getStringCellValue()).isEqualTo("야간전반");
            assertThat(sheet.getRow(4).getCell(0).getStringCellValue()).isEqualTo("야간후반");
        }
    }

    @Test
    void cell_contains_hose_id_and_yield() throws IOException {
        byte[] data = exporter.export(LocalDate.of(2026, 3, 2), LocalDate.of(2026, 3, 2));

        try (var wb = new XSSFWorkbook(new ByteArrayInputStream(data))) {
            Sheet sheet = wb.getSheetAt(0);

            // 주간전반 × EX-A line cell (row 1, col 1)
            String cellValue = sheet.getRow(1).getCell(1).getStringCellValue();
            assertThat(cellValue).matches("\\S+ \\(\\d+\\)");     // "29673-2R060 (2530)" 같은 패턴
        }
    }

    @Test
    void line_capacity_accounting_matches_schedule() throws IOException {
        // BR-E09 정합: 매트릭스 cell yield 합 = DB 스케줄 yield 합 (line·shift·date 단위)
        byte[] data = exporter.export(LocalDate.of(2026, 3, 2), LocalDate.of(2026, 3, 2));
        var dbTotal = scheduleRepo.findByDate(LocalDate.of(2026, 3, 2)).stream()
            .mapToLong(ExSchedule::getYield).sum();

        long matrixTotal = sumAllCellYields(data);

        assertThat(matrixTotal).as("BR-E09 yield accounting").isEqualTo(dbTotal);
    }

    @Test
    void confirmed_only_draft_and_candidate_excluded() throws IOException {
        // DRAFT·CANDIDATE 스케줄도 DB에 있지만 export에는 미포함
        byte[] data = exporter.export(LocalDate.of(2026, 3, 9), LocalDate.of(2026, 3, 9));

        try (var wb = new XSSFWorkbook(new ByteArrayInputStream(data))) {
            // 3/9는 DRAFT/CANDIDATE만 있음 → 시트 0 (empty 시트 skip)
            assertThat(wb.getNumberOfSheets()).isZero();
        }
    }
}
```
