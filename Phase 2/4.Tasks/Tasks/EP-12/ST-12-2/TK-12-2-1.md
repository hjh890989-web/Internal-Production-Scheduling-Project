### `ExtrusionMatrixExporter.java`

```java
@Service @RequiredArgsConstructor
public class ExtrusionMatrixExporter {

    private static final List<String> SHIFTS = List.of(
        "DAY_FIRST", "DAY_SECOND", "NIGHT_FIRST", "NIGHT_SECOND");
    private static final List<String> SHIFT_LABELS = List.of(
        "주간전반", "주간후반", "야간전반", "야간후반");

    private final ExScheduleRepository scheduleRepo;
    private final MachineRepository machineRepo;
    private final CellTypePreserver cellPreserver;

    public byte[] export(LocalDate from, LocalDate to) {
        try (XSSFWorkbook workbook = new XSSFWorkbook();
             ByteArrayOutputStream baos = new ByteArrayOutputStream()) {

            for (LocalDate date = from; !date.isAfter(to); date = date.plusDays(1)) {
                var rows = scheduleRepo.findByDateAndStatus(date, ExScheduleStatus.CONFIRMED);
                if (rows.isEmpty()) continue;

                String sheetName = ExtrusionSheetName.of(date);    // TK-12-2-2
                Sheet sheet = workbook.createSheet(sheetName);
                writeMatrix(sheet, date, rows);
            }

            workbook.write(baos);
            return baos.toByteArray();
        } catch (IOException e) {
            throw new ExportException("extrusion matrix export 실패", e);
        }
    }

    private void writeMatrix(Sheet sheet, LocalDate date, List<ExSchedule> rows) {
        var machines = machineRepo.findExMachinesOrderById();

        // Header row 0: 빈 셀 + 라인 이름들
        Row headerRow = sheet.createRow(0);
        headerRow.createCell(0).setCellValue("Shift");
        for (int c = 0; c < machines.size(); c++) {
            headerRow.createCell(c + 1).setCellValue(machines.get(c).id());
        }

        // 4 shift × N 라인 매트릭스
        for (int r = 0; r < SHIFTS.size(); r++) {
            String shiftCode = SHIFTS.get(r);
            Row row = sheet.createRow(r + 1);
            row.createCell(0).setCellValue(SHIFT_LABELS.get(r));

            for (int c = 0; c < machines.size(); c++) {
                String machineId = machines.get(c).id();
                var slotRows = rows.stream()
                    .filter(s -> s.getShiftCode().equals(shiftCode) && s.getMachineId().equals(machineId))
                    .toList();

                String cellValue = slotRows.stream()
                    .map(s -> "%s (%d)".formatted(s.getHoseId(), s.getYield()))
                    .collect(Collectors.joining("\n"));

                Cell cell = row.createCell(c + 1);
                cell.setCellValue(cellValue);

                // multi-line 텍스트 wrapping
                CellStyle wrapStyle = sheet.getWorkbook().createCellStyle();
                wrapStyle.setWrapText(true);
                cell.setCellStyle(wrapStyle);
            }
        }

        // column width auto-size
        sheet.autoSizeColumn(0);
        for (int c = 1; c <= machines.size(); c++) sheet.autoSizeColumn(c);
    }
}
```

### `ExtrusionExportController.java`

```java
@RestController @RequestMapping("/api/v1/export") @RequiredArgsConstructor
public class ExtrusionExportController {

    private final ExtrusionMatrixExporter exporter;

    @GetMapping("/extrusion-matrix")
    @PreAuthorize("isAuthenticated()")
    @Auditable(reason = "'export extrusion matrix from ' + #from + ' to ' + #to")
    public ResponseEntity<byte[]> exportMatrix(
        @RequestParam @DateTimeFormat(iso = ISO.DATE) LocalDate from,
        @RequestParam @DateTimeFormat(iso = ISO.DATE) LocalDate to
    ) {
        if (from.isAfter(to)) {
            throw new IllegalArgumentException("from must be ≤ to");
        }
        if (ChronoUnit.DAYS.between(from, to) > 60) {
            throw new IllegalArgumentException("range must be ≤ 60 days");
        }

        byte[] data = exporter.export(from, to);
        String filename = "extrusion_%s_%s.xlsx".formatted(from, to);

        return ResponseEntity.ok()
            .header(HttpHeaders.CONTENT_DISPOSITION, "attachment; filename=" + filename)
            .contentType(MediaType.parseMediaType(
                "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"))
            .body(data);
    }
}
```
