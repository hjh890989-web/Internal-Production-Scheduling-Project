### `ExtrusionSheetName.java`

```java
package com.scheduling.export.excel;

import java.time.LocalDate;
import java.util.regex.Pattern;

/**
 * BR-E09: 압출 시트명 규약 — `{월}월{일}일(압출)`.
 *
 * 예: 2026-03-15 → "3월15일(압출)"
 *
 * 정규식: `^\d{1,2}월\d{1,2}일\(압출\)$`
 *  - 월: 1~12 (1자리 가능)
 *  - 일: 1~31 (1자리 가능)
 *  - 괄호 안 텍스트 "압출" 고정
 */
public record ExtrusionSheetName(String value) {

    private static final Pattern PATTERN = Pattern.compile("^(\\d{1,2})월(\\d{1,2})일\\(압출\\)$");

    public ExtrusionSheetName {
        if (!PATTERN.matcher(value).matches()) {
            throw new IllegalArgumentException(
                "BR-E09 sheet name regex 위반: " + value);
        }
    }

    public static ExtrusionSheetName of(LocalDate date) {
        return new ExtrusionSheetName(
            "%d월%d일(압출)".formatted(date.getMonthValue(), date.getDayOfMonth()));
    }

    /**
     * Parser 측에서 시트명 → LocalDate 추출 (year는 별도 컨텍스트).
     */
    public DateParts parse() {
        var m = PATTERN.matcher(value);
        m.matches();
        return new DateParts(
            Integer.parseInt(m.group(1)),
            Integer.parseInt(m.group(2))
        );
    }

    public record DateParts(int month, int day) {}

    public static boolean matches(String sheetName) {
        return PATTERN.matcher(sheetName).matches();
    }
}
```

### `ExtrusionSheetNameTest.java`

```java
class ExtrusionSheetNameTest {

    @Test
    void march_fifteenth() {
        var name = ExtrusionSheetName.of(LocalDate.of(2026, 3, 15));
        assertThat(name.value()).isEqualTo("3월15일(압출)");
    }

    @Test
    void january_first_single_digits() {
        var name = ExtrusionSheetName.of(LocalDate.of(2026, 1, 1));
        assertThat(name.value()).isEqualTo("1월1일(압출)");
    }

    @Test
    void december_thirty_first_two_digits() {
        var name = ExtrusionSheetName.of(LocalDate.of(2026, 12, 31));
        assertThat(name.value()).isEqualTo("12월31일(압출)");
    }

    @Test
    void regex_validates_correct_format() {
        assertThat(ExtrusionSheetName.matches("3월15일(압출)")).isTrue();
        assertThat(ExtrusionSheetName.matches("12월31일(압출)")).isTrue();
        assertThat(ExtrusionSheetName.matches("1월1일(압출)")).isTrue();
    }

    @Test
    void regex_rejects_invalid_format() {
        assertThat(ExtrusionSheetName.matches("3-15(압출)")).isFalse();          // 구분자 다름
        assertThat(ExtrusionSheetName.matches("3월15일 압출")).isFalse();        // 괄호 없음
        assertThat(ExtrusionSheetName.matches("3월15일(성형)")).isFalse();      // 다른 공정
        assertThat(ExtrusionSheetName.matches("123월15일(압출)")).isFalse();    // 3자리 월
        assertThat(ExtrusionSheetName.matches("")).isFalse();
    }

    @Test
    void invalid_construction_throws() {
        assertThatThrownBy(() -> new ExtrusionSheetName("invalid"))
            .isInstanceOf(IllegalArgumentException.class)
            .hasMessageContaining("BR-E09");
    }

    @Test
    void parse_extracts_month_and_day() {
        var name = ExtrusionSheetName.of(LocalDate.of(2026, 3, 15));
        var parts = name.parse();
        assertThat(parts.month()).isEqualTo(3);
        assertThat(parts.day()).isEqualTo(15);
    }

    @ParameterizedTest
    @CsvSource({
        "1,1,1월1일(압출)",
        "1,15,1월15일(압출)",
        "3,5,3월5일(압출)",
        "12,31,12월31일(압출)",
        "6,15,6월15일(압출)"
    })
    void parameterized_month_day_combinations(int month, int day, String expected) {
        var name = ExtrusionSheetName.of(LocalDate.of(2026, month, day));
        assertThat(name.value()).isEqualTo(expected);
    }
}
```
