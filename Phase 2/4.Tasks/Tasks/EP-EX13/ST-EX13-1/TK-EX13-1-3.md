### `ExScheduleReplanService.java`

```java
@Service @RequiredArgsConstructor @Slf4j
public class ExScheduleReplanService {

    private final ImpactAnalyzer impactAnalyzer;
    private final ExScheduleCandidateRepository candidateRepo;
    private final SettingGroupAllocator allocator;
    private final ExtrusionDemandCalculator demandCalc;
    private final BackwardExtrusionCalculator deadlineCalc;

    @Auditable(reason = "'partial replan from vc.changed event'")
    @Transactional
    public void partialReplan(VcChangedEvent event) {
        var impact = impactAnalyzer.analyzeImpact(event);

        if (impact.isEmpty()) {
            log.info("VC changed мҳҒн–Ҙ EX row м—ҶмқҢ вҖ” replan skip");
            return;
        }

        log.info("Partial replan: {} affected candidates, {} hoses",
            impact.affectedCandidates().size(), impact.affectedHoses().size());

        // 1) мҳҒн–Ҙ candidate мӮӯм ң (DRAFT/CANDIDATE мғҒнғңл§Ң вҖ” CONFIRMEDлҠ” лі„лҸ„ мІҳлҰ¬)
        var deletable = impact.affectedCandidates().stream()
            .filter(c -> c.getStatus() != ExScheduleStatus.CONFIRMED)
            .toList();

        candidateRepo.deleteAll(deletable);

        // 2) мҳҒн–Ҙ hoseмқҳ мғҲ deadline + Q_ext кі„мӮ°
        var newCandidates = new ArrayList<ExScheduleCandidate>();
        for (var row : event.rows()) {
            LocalDate newDeadline = deadlineCalc.deadlineFor(row.newProductionDate());
            long newQExt = demandCalc.computeQExtForHose(row.hoseId(), row.newYield());

            newCandidates.add(ExScheduleCandidate.builder()
                .scheduleId(event.scheduleId())
                .hoseId(row.hoseId())
                .vcRowId(row.rowId())
                .vcProductionDate(row.newProductionDate())
                .extrusionDeadline(newDeadline)
                .vcYield(row.newYield())
                .qExt(newQExt)
                .status(ExScheduleStatus.DRAFT)
                .build());
        }

        candidateRepo.saveAll(newCandidates);

        // 3) Allocator мһ¬мӢӨн–ү вҖ” мҳҒн–Ҙ hoseл§Ң лҢҖмғҒ
        var ctx = ExAllocationContext.builder()
            .candidates(newCandidates)
            .workingDays(computeHorizon(event))
            .build();

        var result = allocator.allocate(ctx);

        log.info("Partial replan мҷ„лЈҢ: schedule={}, conflicts={}",
            result.schedule().size(), result.conflicts().size());

        // 4) Conflicts вҶ’ EP-EX12 м¶©лҸҢ лҢҖм•Ҳ мһҗлҸҷ мІҳлҰ¬ (лі„лҸ„ Service нҳём¶ң)
        // ...
    }

    private List<LocalDate> computeHorizon(VcChangedEvent event) {
        // мҳҒн–Ҙ deadline лІ”мң„ + buffer (м „нӣ„ 3мқј)
        var min = event.rows().stream()
            .map(VcChangedRow::newProductionDate)
            .min(LocalDate::compareTo)
            .orElseThrow();
        var max = event.rows().stream()
            .map(VcChangedRow::newProductionDate)
            .max(LocalDate::compareTo)
            .orElseThrow();

        var list = new ArrayList<LocalDate>();
        for (var d = min.minusDays(3); !d.isAfter(max.plusDays(3)); d = d.plusDays(1)) {
            list.add(d);
        }
        return list;
    }
}
```
