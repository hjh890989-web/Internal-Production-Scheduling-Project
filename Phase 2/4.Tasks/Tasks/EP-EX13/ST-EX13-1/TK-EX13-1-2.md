### `ImpactAnalyzer.java`

```java
@Service @RequiredArgsConstructor
public class ImpactAnalyzer {

    private final ExScheduleCandidateRepository candidateRepo;
    private final BackwardExtrusionCalculator deadlineCalc;

    public ImpactResult analyzeImpact(VcChangedEvent event) {
        var affectedHoses = event.rows().stream()
            .map(VcChangedRow::hoseId)
            .collect(Collectors.toSet());

        // 변경된 hose의 영향 deadline 범위
        Set<LocalDate> impactDates = new HashSet<>();
        for (var row : event.rows()) {
            // old + new 모두 포함 — 둘 다 ex_deadline 영향
            if (row.oldProductionDate() != null) {
                impactDates.add(deadlineCalc.deadlineFor(row.oldProductionDate()));
            }
            impactDates.add(deadlineCalc.deadlineFor(row.newProductionDate()));
        }

        // 영향 deadline 범위의 EX candidates 조회
        var affected = candidateRepo.findByHoseIdsAndDeadlineRange(
            affectedHoses,
            impactDates.stream().min(LocalDate::compareTo).orElseThrow(),
            impactDates.stream().max(LocalDate::compareTo).orElseThrow()
        );

        return new ImpactResult(affected, affectedHoses);
    }
}

public record ImpactResult(
    List<ExScheduleCandidate> affectedCandidates,
    Set<String> affectedHoses
) {
    public boolean isEmpty() { return affectedCandidates.isEmpty(); }
}
```
