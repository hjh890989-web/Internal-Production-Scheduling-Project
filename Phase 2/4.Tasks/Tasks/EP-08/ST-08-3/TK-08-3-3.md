### `ExtrusionDemandIntegrationIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-EX-QEXT-30/inventory_seed.sql"})
class ExtrusionDemandIntegrationIT {

    @Autowired ExtrusionScheduleService extrusionService;
    @Autowired ExtrusionDemandCalculator demandCalc;
    @Autowired ExScheduleCandidateRepository candidateRepo;
    @Autowired EventLoader loader;

    @Test
    void vc_confirmed_triggers_q_ext_computation() {
        // VC 확정: 28912-2U000 yield=100
        var event = loader.eventFor("28912-2U000", 100, LocalDate.of(2026, 3, 9));

        extrusionService.generateCandidates(event);

        // 재고: target=50, current=30 → Q_ext = 100 + 50 - 30 = 120
        var candidate = candidateRepo.findByHoseId("28912-2U000").orElseThrow();
        assertThat(candidate.getQExt()).isEqualTo(120L);
    }

    @Test
    void sufficient_inventory_yields_zero_q_ext() {
        // 잉여 재고: target=50, current=200, yield=100 → Q_ext = 0
        var event = loader.eventFor("29689-2U000", 100, LocalDate.of(2026, 3, 9));
        extrusionService.generateCandidates(event);

        var candidate = candidateRepo.findByHoseId("29689-2U000").orElseThrow();
        assertThat(candidate.getQExt()).isZero();
    }

    @Test
    void unregistered_hose_uses_q_vc_directly() {
        // 재고 미등록 — Q_ext = Q_vc = 80
        var event = loader.eventFor("NEW-PRODUCT", 80, LocalDate.of(2026, 3, 9));
        extrusionService.generateCandidates(event);

        var candidate = candidateRepo.findByHoseId("NEW-PRODUCT").orElseThrow();
        assertThat(candidate.getQExt()).isEqualTo(80L);
    }

    @Test
    void regression_30_scenarios_q_ext_consistency() {
        var events = loader.load("DS-EX-QEXT-30/vc_confirmed_events.csv");

        for (var event : events) {
            extrusionService.generateCandidates(event);
            for (var row : event.rows()) {
                long expected = demandCalc.computeQExtForHose(row.hoseId(), row.yield());
                var candidate = candidateRepo.findByHoseIdAndScheduleId(
                    row.hoseId(), event.scheduleId()).orElseThrow();
                assertThat(candidate.getQExt()).isEqualTo(expected);
            }
        }
    }
}
```

### `QExtCacheInvalidationIT.java`

```java
@SpringBootTest @Testcontainers
class QExtCacheInvalidationIT {

    @Autowired ProductInventoryCache cache;
    @Autowired EntityManager em;

    @Test
    void mes_sync_inventory_update_invalidates_cache_within_500ms()
            throws InterruptedException {
        // 1) 초기 재고
        var before = cache.get("28912-2U000").orElseThrow();
        assertThat(before.getCurrentStock()).isEqualTo(30);

        // 2) MES 동기화 — current_stock 30 → 100
        em.createNativeQuery("UPDATE master.product_inventory " +
            "SET current_stock = 100, updated_by = 'mes-sync' WHERE hose_id = '28912-2U000'")
            .executeUpdate();

        // 3) NOTIFY 도달 대기 (≤ 500ms)
        Thread.sleep(500);

        // 4) 캐시 invalidate → 다시 조회 시 current=100
        var after = cache.get("28912-2U000").orElseThrow();
        assertThat(after.getCurrentStock()).isEqualTo(100);
    }
}
```
