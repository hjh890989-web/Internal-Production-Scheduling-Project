### 핵심 로직

```java
@ExtendWith(MockitoExtension.class)
class ExtrusionDemandCalculatorTest {

    @Mock ProductInventoryCache cache;
    @InjectMocks ExtrusionDemandCalculator calc;

    /**
     * REQ-FUNC-EX-010 4 케이스 매트릭스:
     *   Case 1: 재고 부족 — current < target → Q_ext = Q_vc + (target - current)
     *   Case 2: 재고 = target — Q_ext = Q_vc (안전재고만 유지)
     *   Case 3: 재고 > target, < target + Q_vc — Q_ext = qVc + target - current (양수)
     *   Case 4: 재고 ≥ target + Q_vc — Q_ext = 0 (Math.max floor)
     */
    @ParameterizedTest
    @CsvSource({
        // qVc, target, current, expected
        "100,  50, 30,  120",   // Case 1: 충분히 부족
        "100,  50, 50,  100",   // Case 2: 재고 = target
        "100,  50, 80,   70",   // Case 3: 재고 > target, 부분 부족
        "100,  50, 150,   0",   // Case 4: 잉여 — Q_ext = 0
        "100,  50, 200,   0",   // Case 4 강화: 매우 잉여
        "  0,  50, 30,   20",   // Q_vc=0, 안전재고 부족
        "  0,  50, 50,    0",   // Q_vc=0, 안전재고 충족
        "  0,   0,  0,    0",   // 모두 0
        "1000000, 5000, 0, 1005000"  // 대형 수주
    })
    void four_inventory_cases(long qVc, int target, int current, long expected) {
        assertThat(calc.computeQExt(qVc, target, current)).isEqualTo(expected);
    }

    @Test
    void negative_q_vc_throws() {
        assertThatThrownBy(() -> calc.computeQExt(-1, 50, 30))
            .isInstanceOf(IllegalArgumentException.class)
            .hasMessageContaining("qVc must be ≥ 0");
    }

    @Test
    void negative_target_throws() {
        assertThatThrownBy(() -> calc.computeQExt(100, -1, 30))
            .isInstanceOf(IllegalArgumentException.class);
    }

    @Test
    void negative_current_throws() {
        assertThatThrownBy(() -> calc.computeQExt(100, 50, -1))
            .isInstanceOf(IllegalArgumentException.class);
    }

    @Test
    void unknown_hose_assumes_zero_inventory() {
        when(cache.get("UNKNOWN")).thenReturn(Optional.empty());
        assertThat(calc.computeQExtForHose("UNKNOWN", 100))
            .as("재고 미등록 → target=current=0 → Q_ext = Q_vc")
            .isEqualTo(100L);
    }

    @Test
    void registered_hose_uses_master_inventory() {
        var inv = ProductInventory.builder()
            .hoseId("28912").targetStock(50).currentStock(30).build();
        when(cache.get("28912")).thenReturn(Optional.of(inv));

        assertThat(calc.computeQExtForHose("28912", 100)).isEqualTo(120L);
    }
}
```
