### Flyway `V3_1__create_shift_master.sql`

```sql
CREATE TABLE master.shift (
    shift_code       VARCHAR(20) PRIMARY KEY,
    name             VARCHAR(40) NOT NULL,
    sort_order       INTEGER NOT NULL,
    start_time       TIME NOT NULL,
    end_time         TIME NOT NULL,
    nominal_min      INTEGER NOT NULL CHECK (nominal_min > 0),
    efficiency       NUMERIC(4,3) NOT NULL CHECK (efficiency > 0 AND efficiency <= 1),
    effective_min    INTEGER GENERATED ALWAYS AS (FLOOR(nominal_min * efficiency)) STORED,
    is_night         BOOLEAN NOT NULL DEFAULT FALSE,
    notes            TEXT,
    updated_at       TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_by       VARCHAR(40) NOT NULL DEFAULT 'seed'
);

CREATE INDEX idx_shift_sort ON master.shift(sort_order);

GRANT SELECT ON master.shift TO app_user;
GRANT INSERT, UPDATE, DELETE ON master.shift TO master_admin;

COMMENT ON TABLE master.shift IS 'EP-08: 압출 shift 마스터 (BR-E03·E04 4-shift × 75%)';
COMMENT ON COLUMN master.shift.efficiency IS 'BR-E04: 압출 효율 (기본 0.75)';
COMMENT ON COLUMN master.shift.effective_min IS '계산: floor(nominal_min × efficiency) — yield 수식 입력';

-- LISTEN/NOTIFY 트리거 (TK-21-2-3 패턴)
CREATE OR REPLACE FUNCTION master.notify_shift_change()
RETURNS TRIGGER AS $$
BEGIN
    PERFORM pg_notify('shift_change',
        json_build_object('shift_code', COALESCE(NEW.shift_code, OLD.shift_code), 'op', TG_OP)::text);
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_shift_notify
    AFTER INSERT OR UPDATE OR DELETE ON master.shift
    FOR EACH ROW EXECUTE FUNCTION master.notify_shift_change();
```

### `Shift.java`

```java
@Entity @Table(schema="master", name="shift")
@Data @NoArgsConstructor @AllArgsConstructor @Builder
public class Shift {
    @Id @Column(name="shift_code") private String shiftCode;
    @Column(name="name") private String name;
    @Column(name="sort_order") private Integer sortOrder;
    @Column(name="start_time") private LocalTime startTime;
    @Column(name="end_time") private LocalTime endTime;
    @Column(name="nominal_min") private Integer nominalMin;
    @Column(name="efficiency", precision=4, scale=3) private BigDecimal efficiency;
    @Column(name="effective_min", insertable=false, updatable=false) private Integer effectiveMin;
    @Column(name="is_night") private Boolean isNight;
    @Column(name="notes") private String notes;
}
```

### `ShiftController.java`

```java
@RestController @RequestMapping("/api/v1/master/shifts") @RequiredArgsConstructor
public class ShiftController {

    private final ShiftRepository repo;
    private final ShiftCache cache;

    @GetMapping
    @PreAuthorize("isAuthenticated()")
    public List<Shift> list() { return repo.findAllByOrderBySortOrderAsc(); }

    @GetMapping("/{code}")
    @PreAuthorize("isAuthenticated()")
    public Optional<Shift> get(@PathVariable String code) { return cache.get(code); }

    @PostMapping
    @PreAuthorize("hasRole('IT_OPS')")
    public Shift save(@Valid @RequestBody Shift s) {
        return repo.save(s);   // 트리거가 NOTIFY 자동
    }

    @DeleteMapping("/{code}")
    @PreAuthorize("hasRole('IT_OPS')")
    public void remove(@PathVariable String code) { repo.deleteById(code); }
}
```
