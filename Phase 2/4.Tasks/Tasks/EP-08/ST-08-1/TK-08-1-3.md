```java
@ExtendWith(MockitoExtension.class)
class ShiftEfficiencyServiceUnitTest {

    @Mock ShiftCache cache;
    @InjectMocks ShiftEfficiencyService service;

    @ParameterizedTest
    @CsvSource({
        "DAY_FIRST,    240, 0.750, 180",
        "DAY_SECOND,   240, 0.750, 180",
        "NIGHT_FIRST,  240, 0.750, 180",
        "NIGHT_SECOND, 240, 0.750, 180"
    })
    void four_shifts_effective_min_180(String code, int nominal, String effStr, int expected) {
        var shift = Shift.builder()
            .shiftCode(code).nominalMin(nominal)
            .efficiency(new BigDecimal(effStr))
            .effectiveMin((int) Math.floor(nominal * Double.parseDouble(effStr)))
            .build();
        when(cache.get(code)).thenReturn(Optional.of(shift));

        assertThat(service.effectiveMin(code)).isEqualTo(expected);
    }

    @Test
    void efficiency_change_to_80_percent_yields_192() {
        var shift = Shift.builder()
            .shiftCode("DAY_FIRST").nominalMin(240)
            .efficiency(new BigDecimal("0.800")).effectiveMin(192).build();
        when(cache.get("DAY_FIRST")).thenReturn(Optional.of(shift));

        assertThat(service.effectiveMin("DAY_FIRST")).isEqualTo(192);
    }

    @Test
    void unknown_shift_throws_master_not_found() {
        when(cache.get("UNKNOWN")).thenReturn(Optional.empty());
        assertThatThrownBy(() -> service.effectiveMin("UNKNOWN"))
            .isInstanceOf(MasterNotFoundException.class)
            .hasMessageContaining("shift=UNKNOWN");
    }

    @Test
    void effective_min_all_shifts_returns_four_entries() {
        var shifts = List.of(
            buildShift("DAY_FIRST",    180),
            buildShift("DAY_SECOND",   180),
            buildShift("NIGHT_FIRST",  180),
            buildShift("NIGHT_SECOND", 180)
        );
        when(cache.all()).thenReturn(shifts);

        var map = service.effectiveMinAllShifts();
        assertThat(map).hasSize(4)
            .containsEntry("DAY_FIRST", 180)
            .containsEntry("DAY_SECOND", 180)
            .containsEntry("NIGHT_FIRST", 180)
            .containsEntry("NIGHT_SECOND", 180);
    }

    @Test
    void low_efficiency_floor_behavior() {
        var shift = Shift.builder()
            .shiftCode("TEST").nominalMin(100)
            .efficiency(new BigDecimal("0.333"))
            .effectiveMin(33)                  // floor(100 × 0.333) = 33
            .build();
        when(cache.get("TEST")).thenReturn(Optional.of(shift));

        assertThat(service.effectiveMin("TEST")).isEqualTo(33);
    }

    @Test
    void high_efficiency_one_yields_full_nominal() {
        var shift = Shift.builder()
            .shiftCode("PERFECT").nominalMin(240)
            .efficiency(BigDecimal.ONE).effectiveMin(240).build();
        when(cache.get("PERFECT")).thenReturn(Optional.of(shift));

        assertThat(service.effectiveMin("PERFECT")).isEqualTo(240);
    }

    private Shift buildShift(String code, int effective) {
        return Shift.builder().shiftCode(code).effectiveMin(effective).build();
    }
}

class ShiftCacheUnitTest {

    @Test
    void cache_hit_avoids_repo_call() {
        var repo = mock(ShiftRepository.class);
        var cache = new ShiftCache(repo);
        var shift = Shift.builder().shiftCode("DAY_FIRST").build();
        when(repo.findById("DAY_FIRST")).thenReturn(Optional.of(shift));

        cache.get("DAY_FIRST");
        cache.get("DAY_FIRST");
        cache.get("DAY_FIRST");

        verify(repo, times(1)).findById("DAY_FIRST");
    }

    @Test
    void invalidate_triggers_reload_on_next_get() {
        var repo = mock(ShiftRepository.class);
        var cache = new ShiftCache(repo);
        when(repo.findById("DAY_FIRST")).thenReturn(Optional.of(Shift.builder().build()));

        cache.get("DAY_FIRST");
        cache.invalidate("DAY_FIRST");
        cache.get("DAY_FIRST");

        verify(repo, times(2)).findById("DAY_FIRST");
    }
}
```
