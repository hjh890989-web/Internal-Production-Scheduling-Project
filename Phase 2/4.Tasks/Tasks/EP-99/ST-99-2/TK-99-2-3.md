### 예시 코드 (자동화 부분)

```python
#!/usr/bin/env python3
# tools/master_validation/br_v17_impact_check.py
import sys, io, json
sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8')
from datetime import date
from pathlib import Path

REPORT_DIR = Path('tools/master_validation/reports')

# BR-V14·V15·V16 대상 (WBS §5.5 EP-21 정의)
SPECIAL_HOSES = {'28422-08HA0', '28422-2M800', '28421-2M800'}

def load_under7():
    today = date.today().isoformat()
    json_path = REPORT_DIR / f'spec_under7_{today}.json'
    if not json_path.exists():
        sys.exit(f'❌ TK-99-2-2 산출물 없음: {json_path}')
    return json.loads(json_path.read_text(encoding='utf-8'))

def check_overlap(under_list):
    """규격<7 ↔ BR-V14·V15·V16 대상 중첩."""
    overlap = []
    for p in under_list:
        if p['hose_id'] in SPECIAL_HOSES:
            overlap.append(p['hose_id'])
    return overlap

def write_report(data, overlap):
    today = date.today().isoformat()
    out = REPORT_DIR / f'br_v17_impact_{today}.md'
    out.parent.mkdir(parents=True, exist_ok=True)

    with open(out, 'w', encoding='utf-8') as f:
        f.write(f'# BR-V17 영향 품번 사전 점검 ({today})\n\n')
        f.write(f'## 입력\n- 규격<{data["threshold"]}: {data["under_threshold"]}품번\n')
        f.write(f'- 출처: TK-99-2-2 (`spec_under7_{today}.json`)\n\n')

        f.write(f'## A. 수주 빈도 (P1 인터뷰 필요 — 본 스크립트는 자리표시자)\n')
        f.write(f'| HOSE | 규격 | 월평균 수주 | 빈도 등급 |\n')
        f.write(f'|---|:---:|:---:|:---:|\n')
        for p in data['under_list']:
            f.write(f'| {p["hose_id"]} | {p["spec"]} | <TBD: P1 인터뷰> | <TBD> |\n')

        f.write(f'\n## B. 특수제약(BR-V14·V15·V16) 중첩 검사\n')
        if overlap:
            f.write(f'⚠️ **중첩 발견**: {", ".join(overlap)}\n')
            f.write(f'→ 다중 제약 충돌 위험. EP-21 RuleEngine 설계 시 우선순위 명확화 필요.\n')
        else:
            f.write(f'✅ 중첩 없음 — BR-V17 7품번은 BR-V14·V15·V16 대상과 별도 계열\n')

        f.write(f'\n## C. As-Is 호싱불량 연관 (품질관리팀 데이터 필요)\n')
        f.write(f'| HOSE | 규격 | 호싱불량률 | BR-V17 적용 시 예상 개선 |\n')
        f.write(f'|---|:---:|:---:|---|\n')
        for p in data['under_list']:
            f.write(f'| {p["hose_id"]} | {p["spec"]} | <TBD: 품질팀 데이터> | <TBD> |\n')

        f.write(f'\n## 종합 위험 등급 (P1·STK-08 dual-review 후 확정)\n')
        f.write(f'<TBD: 위 A·B·C 데이터 확보 후 결정>\n')

        f.write(f'\n## 다음 단계\n')
        f.write(f'- [ ] P1 김정훈 주임 인터뷰: 7품번 수주 빈도 확인 (~30분)\n')
        f.write(f'- [ ] 품질관리팀에 As-Is 호싱불량 통계 요청\n')
        f.write(f'- [ ] 데이터 확보 후 위 표 자동 완성 + 등급 산정\n')
        f.write(f'- [ ] dual-review 사인오프 → EP-21 ST-21-5 (S2)에 전달\n')
    return out

if __name__ == '__main__':
    data = load_under7()
    overlap = check_overlap(data['under_list'])
    out = write_report(data, overlap)
    print(f'Report: {out}')
    if overlap:
        print(f'⚠️ 중첩 발견: {overlap}')
    print('TBD 항목: P1 인터뷰 + 품질팀 데이터 필요')
```

> **참고**: 자동화 가능 부분(B 중첩 검사)만 스크립트로 처리. A 수주 빈도·C 호싱불량은 사람 인터뷰·데이터 요청 필요 — 리포트에 TBD 자리표시자로 출력.
