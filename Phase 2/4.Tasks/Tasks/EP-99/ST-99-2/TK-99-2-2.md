### 예시 코드

```python
#!/usr/bin/env python3
# tools/master_validation/identify_spec_under7.py
import sys, io, json
sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8')
from collections import Counter
from datetime import date
from pathlib import Path
from statistics import median
import openpyxl

MASTER_PATH = Path('Phase 1/2.Raw Materials/Extrusion/압출공정_제약조건.xlsx')
REPORT_DIR  = Path('tools/master_validation/reports')
THRESHOLD = 7  # BR-V17 임계값

def load_master():
    wb = openpyxl.load_workbook(MASTER_PATH, data_only=True)
    ws = wb['Sheet1']
    products = []
    for r in range(2, ws.max_row + 1):
        hose = ws.cell(row=r, column=1).value
        if not hose:
            break
        products.append({
            'hose_id': hose,
            'spec':    ws.cell(row=r, column=2).value,
            '내경':   ws.cell(row=r, column=3).value,
            '두께':   ws.cell(row=r, column=4).value,
            '압출셋팅': ws.cell(row=r, column=5).value,
        })
    return products

def analyze(products):
    valid = [p for p in products if isinstance(p['spec'], (int, float))]
    under = [p for p in valid if p['spec'] < THRESHOLD]
    dist  = Counter(p['spec'] for p in valid)
    specs = sorted(dist.keys())
    return {
        'total':            len(valid),
        'under_threshold':  len(under),
        'under_ratio_pct':  round(100 * len(under) / len(valid), 1) if valid else 0,
        'min_spec':         min(specs) if specs else None,
        'max_spec':         max(specs) if specs else None,
        'median_spec':      median([p['spec'] for p in valid]) if valid else None,
        'distribution':     {str(s): dist[s] for s in specs},
        'under_list':       under,
        'threshold':        THRESHOLD,
    }

def write_md_report(result):
    today = date.today().isoformat()
    out = REPORT_DIR / f'spec_under7_{today}.md'
    out.parent.mkdir(parents=True, exist_ok=True)

    with open(out, 'w', encoding='utf-8') as f:
        f.write(f'# 압출 마스터 규격 분포 + 규격<{THRESHOLD} 영향 품번 ({today})\n\n')
        f.write(f'## 요약\n')
        f.write(f'- **총 품번**: {result["total"]}\n')
        f.write(f'- **규격<{THRESHOLD}**: {result["under_threshold"]}품번 ({result["under_ratio_pct"]}%)\n')
        f.write(f'- **규격 범위**: {result["min_spec"]} ~ {result["max_spec"]} (중위값 {result["median_spec"]})\n\n')

        f.write(f'## 규격 분포\n')
        f.write(f'| 규격 | 품번 수 | 비율 | 막대 (1품번=▍) |\n')
        f.write(f'|:---:|:---:|:---:|---|\n')
        for s, n in result['distribution'].items():
            pct = round(100 * n / result['total'], 1)
            bar = '▍' * n
            marker = ' ⭐ BR-V17' if float(s) < THRESHOLD else ''
            f.write(f'| {s}{marker} | {n} | {pct}% | {bar} |\n')

        f.write(f'\n## 규격<{THRESHOLD} 품번 리스트 (BR-V17 직접 영향)\n')
        f.write(f'| HOSE | 규격 | 내경 | 두께 | 압출셋팅 |\n')
        f.write(f'|---|:---:|:---:|:---:|:---:|\n')
        for p in result['under_list']:
            f.write(f"| {p['hose_id']} | {p['spec']} | {p['내경']} | {p['두께']} | {p['압출셋팅']} |\n")
    return out

def write_json_report(result):
    today = date.today().isoformat()
    out = REPORT_DIR / f'spec_under7_{today}.json'
    with open(out, 'w', encoding='utf-8') as f:
        json.dump(result, f, ensure_ascii=False, indent=2)
    return out

if __name__ == '__main__':
    products = load_master()
    result = analyze(products)
    md  = write_md_report(result)
    js  = write_json_report(result)
    print(f'MD : {md}')
    print(f'JSON: {js}')
    print(f'Total: {result["total"]}, <{THRESHOLD}: {result["under_threshold"]} ({result["under_ratio_pct"]}%)')
```
