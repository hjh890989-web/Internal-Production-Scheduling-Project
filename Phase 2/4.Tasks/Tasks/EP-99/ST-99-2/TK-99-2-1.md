### 핵심 로직 / 알고리즘

1. **엑셀 로드 + 헤더 위치 확인**: `openpyxl.load_workbook()` `Sheet1` 시트, **row 1 헤더** (성형과 다름!). B열 헤더가 `규격`인지 assertion. 불일치 시 즉시 fail.

2. **B열 정합성 검증**: row 2부터 마지막 row까지 순회:
   - 값이 `int` 또는 `float` → 정합 (단, `float`은 `13.5` 같은 0.5 배수만 허용 — 사전 조사 결과)
   - 값이 `None`·빈 문자열 → 위반 (NULL)
   - 값이 문자열 (예: `"5"`) → 위반 (타입 미스 — Excel 셀 포맷 오류 의심)
   - 음수 또는 0 → 위반
   - 100 초과 (현실적 호스 규격 범위 외) → 경고

3. **결과 리포트**: 마크다운 표 — HOSE | 규격 | 타입 | 유효 | 비고. 위반 행은 ⚠️ 표시.

4. **종료 코드**: 위반 0건 → exit 0 / ≥1건 → exit 1.

### 예시 코드

```python
#!/usr/bin/env python3
# tools/master_validation/validate_ex_master_spec.py
import sys, io
sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8')
from datetime import date
from pathlib import Path
import openpyxl

MASTER_PATH = Path('Phase 1/2.Raw Materials/Extrusion/압출공정_제약조건.xlsx')
REPORT_DIR  = Path('tools/master_validation/reports')

def is_valid_spec(v):
    """규격 값 정합 검사: 양의 정수 또는 0.5 단위 소수."""
    if v is None or v == '':
        return False, 'NULL'
    if not isinstance(v, (int, float)):
        return False, f'타입 미스 ({type(v).__name__}: {v!r})'
    if isinstance(v, float) and not (v * 2).is_integer():
        return False, f'0.5 단위 아님 ({v})'
    if v <= 0:
        return False, f'양수 아님 ({v})'
    if v > 100:
        return True, f'⚠️ 범위 경고 ({v} > 100)'
    return True, ''

def validate(wb):
    ws = wb['Sheet1']
    assert ws.cell(row=1, column=1).value == 'HOSE', 'A열 헤더 불일치'
    assert ws.cell(row=1, column=2).value == '규격', 'B열 헤더 불일치 — 마스터 구조 변경 감지'

    rows, violations = [], []
    for r in range(2, ws.max_row + 1):
        hose = ws.cell(row=r, column=1).value
        if not hose:
            break
        spec = ws.cell(row=r, column=2).value
        ok, note = is_valid_spec(spec)
        rows.append((hose, spec, type(spec).__name__, ok, note))
        if not ok:
            violations.append((hose, spec, note))
    return rows, violations

def write_report(rows, violations):
    REPORT_DIR.mkdir(parents=True, exist_ok=True)
    today = date.today().isoformat()
    out = REPORT_DIR / f'ex_master_spec_{today}.md'
    with open(out, 'w', encoding='utf-8') as f:
        f.write(f'# 압출 마스터 B열(규격) 검증 리포트 ({today})\n\n')
        f.write(f'- 총 품번: {len(rows)}\n')
        f.write(f'- 위반: {len(violations)}\n\n')
        f.write('| HOSE | 규격 | 타입 | 유효 | 비고 |\n')
        f.write('|---|:---:|:---:|:---:|---|\n')
        for hose, spec, ty, ok, note in rows:
            mark = '⚠️ ' if not ok else ''
            f.write(f'| {mark}{hose} | {spec} | {ty} | {ok} | {note} |\n')
    return out

if __name__ == '__main__':
    wb = openpyxl.load_workbook(MASTER_PATH, data_only=True)
    rows, viol = validate(wb)
    out = write_report(rows, viol)
    print(f'Report: {out}')
    print(f'Total: {len(rows)} 품번, 위반: {len(viol)}')
    sys.exit(0 if not viol else 1)
```
