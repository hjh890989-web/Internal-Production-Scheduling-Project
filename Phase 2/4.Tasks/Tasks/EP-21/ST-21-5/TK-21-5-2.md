### Flyway 트리거 추가 `V2_8_data__create_master_change_triggers.sql`

```sql
-- VC_CONSTRAINT 변경 시 NOTIFY
CREATE OR REPLACE FUNCTION master.notify_vc_constraint_change()
RETURNS TRIGGER AS $$
BEGIN
    PERFORM pg_notify('vc_constraint_change',
        json_build_object('hose_id', COALESCE(NEW.hose_id, OLD.hose_id), 'op', TG_OP)::text);
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_vc_constraint_notify
    AFTER INSERT OR UPDATE OR DELETE ON master.vc_constraint
    FOR EACH ROW EXECUTE FUNCTION master.notify_vc_constraint_change();

-- EX_CONSTRAINT 변경 시 NOTIFY
```

### `ProductSpecCache.java`

```java
@Component @RequiredArgsConstructor
public class ProductSpecCache {

    private final ProductSpecRepository repo;
    private final Cache<String, ProductSpec> cache = Caffeine.newBuilder()
        .maximumSize(500)
        .expireAfterWrite(Duration.ofHours(1))
        .recordStats()
        .build();

    public Optional<ProductSpec> get(String hoseId) {
        var cached = cache.getIfPresent(hoseId);
        if (cached != null) return Optional.of(cached);
        return repo.findById(hoseId).map(p -> { cache.put(hoseId, p); return p; });
    }

    public List<ProductSpec> getAllSpecLt7() {
        // 별도 캐시 키 또는 전체 cache scan (작은 maximumSize)
        return repo.findAllByIsSpecLt7True();
    }

    public void invalidate(String hoseId) { cache.invalidate(hoseId); }
    public void invalidateAll() { cache.invalidateAll(); }
    public CacheStats stats() { return cache.stats(); }
}
```

### `MasterChangeListener.java`

```java
@Component @RequiredArgsConstructor @Slf4j
public class MasterChangeListener {

    private final DataSource dataSource;
    private final ProductSpecCache cache;
    private final ObjectMapper mapper;
    private volatile boolean running = true;

    private static final List<String> CHANNELS = List.of(
        "vc_constraint_change",
        "ex_constraint_change"
    );

    @PostConstruct
    public void start() {
        var executor = Executors.newSingleThreadExecutor(r ->
            new Thread(r, "product-spec-listener"));
        executor.submit(this::listen);
    }

    private void listen() {
        while (running) {
            try (Connection conn = dataSource.getConnection()) {
                try (Statement stmt = conn.createStatement()) {
                    for (String ch : CHANNELS) stmt.execute("LISTEN " + ch);
                }
                PGConnection pgConn = conn.unwrap(PGConnection.class);
                while (running) {
                    var notifications = pgConn.getNotifications(5000);
                    if (notifications == null) continue;
                    for (var n : notifications) handle(n);
                }
            } catch (SQLException e) {
                log.error("LISTEN 끊김 — 5초 후 재시도", e);
                sleepSafely(5000);
            }
        }
    }

    private void handle(PGNotification n) {
        try {
            var event = mapper.readValue(n.getParameter(), Map.class);
            String hoseId = (String) event.get("hose_id");
            cache.invalidate(hoseId);    // VIEW가 invalidate 자동 반영
            log.info("ProductSpec cache invalidated: channel={}, hose_id={}", n.getName(), hoseId);
        } catch (Exception e) {
            log.warn("이벤트 파싱 실패 — invalidateAll", e);
            cache.invalidateAll();
        }
    }
}
```
