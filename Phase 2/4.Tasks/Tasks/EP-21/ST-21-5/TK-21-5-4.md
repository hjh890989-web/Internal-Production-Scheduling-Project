### `SpecLt7RegressionIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-VC-SPEC-LT7-100/master_seed.sql"})
class SpecLt7RegressionIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired ProductSpecRepository specRepo;
    @Autowired ScenarioLoader loader;

    @Test
    void rule_spec_lt7_machine_angle_cap_four() {
        var scenarios = loader.load("DS-VC-SPEC-LT7-100/scenarios.csv");
        var ctx = AllocationContext.fromOrders(scenarios, LocalDate.of(2026, 2, 16), 30);
        var result = allocator.allocate(ctx);

        // spec<7 품번 식별
        var specLt7Hoses = specRepo.findAllByIsSpecLt7True().stream()
            .map(ProductSpec::getHoseId).collect(Collectors.toSet());

        // (machine, date)별 spec<7 angle 누계
        Map<Tuple2<String, LocalDate>, Integer> machineAngles = new HashMap<>();
        for (var row : result.schedule()) {
            if (!specLt7Hoses.contains(row.hoseId())) continue;
            var spec = specRepo.findById(row.hoseId()).orElseThrow();
            machineAngles.merge(Tuple.of(row.machineId(), row.productionDate()),
                spec.getAngleCount(), Integer::sum);
        }

        assertThat(machineAngles.values())
            .as("BR-V17 spec<7 가류기당 ≤ 4")
            .allMatch(angles -> angles <= 4);
    }

    @Test
    void regression_100_scenarios_no_violation() {
        var scenarios = loader.load("DS-VC-SPEC-LT7-100/scenarios.csv");
        var ctx = AllocationContext.fromOrders(scenarios, LocalDate.of(2026, 2, 16), 30);
        var result = allocator.allocate(ctx);

        // 모든 row에 대해 SpecLt7CapRule 검증
        var violations = result.schedule().stream()
            .filter(r -> {
                var spec = specRepo.findById(r.hoseId()).orElse(null);
                return spec != null && spec.isSpecLt7();
            })
            // 누계 계산 (위 방식)
            .filter(this::overFour)
            .toList();

        assertThat(violations).isEmpty();
    }

    @Test
    void conflict_when_spec_lt7_saturated() {
        var ctx = loader.contextSaturatingSpecLt7();
        var result = allocator.allocate(ctx);

        assertThat(result.conflicts())
            .anyMatch(c -> c.reason().contains("BR-V17"));
    }
}
```

@SpringBootTest @Testcontainers
class CrossMasterInvalidationIT {

    @Autowired ProductSpecCache cache;
    @Autowired EntityManager em;

    @Test
    void ex_constraint_update_invalidates_cache_within_500ms() throws InterruptedException {
        // 1) 캐시에 적재
        var before = cache.get("28442-6T010").orElseThrow();
        assertThat(before.getSpec()).isEqualTo(8);

        // 2) EX_CONSTRAINT 직접 UPDATE (spec 8 → 6)
        em.createNativeQuery("UPDATE master.ex_constraint SET spec_value = 6 WHERE hose_id = '28442-6T010'")
            .executeUpdate();

        // 3) NOTIFY 도달 대기 (≤500ms)
