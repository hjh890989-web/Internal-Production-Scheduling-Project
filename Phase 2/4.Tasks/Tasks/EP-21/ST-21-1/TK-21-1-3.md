### 핵심 로직

```java
@SpringBootTest @Testcontainers
@Sql("classpath:datasets/DS-VC-LR-100/master_seed.sql")
class LeftRightRegressionIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired VcScheduleRepository scheduleRepo;
    @Autowired ScenarioLoader loader;

    @Test
    void rule_28421_2M800_only_left() {
        var ctx = loader.contextFor("28421-2M800", 500, 7);
        var result = allocator.allocate(ctx);

        var rows = result.schedule().stream()
            .filter(r -> r.hoseId().equals("28421-2M800")).toList();

        assertThat(rows)
            .as("28421-2M800 모든 배치는 좌측 슬롯이어야 함 (BR-V15)")
            .allMatch(r -> SlotSide.of(r.machineType(), r.slotNumber()) == SlotSide.LEFT);
    }

    @Test
    void rule_28422_2M800_only_right() {
        var ctx = loader.contextFor("28422-2M800", 500, 7);
        var result = allocator.allocate(ctx);

        var rows = result.schedule().stream()
            .filter(r -> r.hoseId().equals("28422-2M800")).toList();

        assertThat(rows)
            .as("28422-2M800 모든 배치는 우측 슬롯이어야 함 (BR-V16)")
            .allMatch(r -> SlotSide.of(r.machineType(), r.slotNumber()) == SlotSide.RIGHT);
    }

    @Test
    void regression_100_scenarios_no_violation() {
        var scenarios = loader.load("DS-VC-LR-100/scenarios.csv");
        var ctx = AllocationContext.fromOrders(scenarios, LocalDate.of(2026, 2, 16), 30);
        var result = allocator.allocate(ctx);

        var violations = result.schedule().stream()
            .filter(r -> !leftRightRule.validate(r.hoseId(), r.machineId(),
                SlotSide.of(r.machineType(), r.slotNumber())))
            .toList();

        assertThat(violations).isEmpty();
    }

    @Test
    void conflict_when_only_wrong_side_available() {
        // 좌측 슬롯이 모두 점유되었고 28421-2M800만 남았을 때 → AllocationConflict
        var ctx = loader.contextSaturatingLeftSlots();
        var result = allocator.allocate(ctx);

        assertThat(result.conflicts())
            .anyMatch(c -> c.reason().contains("BR-V15") || c.reason().contains("좌/우"));
    }
}
```
