### Flyway `V2_6__alter_vc_constraint_left_right.sql`

```sql
ALTER TABLE master.vc_constraint
    ADD COLUMN lp_left_setting  CHAR(1) NOT NULL DEFAULT 'X' CHECK (lp_left_setting  IN ('O','X')),
    ADD COLUMN lp_right_setting CHAR(1) NOT NULL DEFAULT 'X' CHECK (lp_right_setting IN ('O','X'));

CREATE INDEX idx_vc_constraint_left  ON master.vc_constraint(hose_id, machine_id) WHERE lp_left_setting  = 'O';
CREATE INDEX idx_vc_constraint_right ON master.vc_constraint(hose_id, machine_id) WHERE lp_right_setting = 'O';

COMMENT ON COLUMN master.vc_constraint.lp_left_setting  IS 'BR-V15: K열 좌측 셋팅 호환성 (성형공정_제약조건.xlsx)';
COMMENT ON COLUMN master.vc_constraint.lp_right_setting IS 'BR-V16: L열 우측 셋팅 호환성 (성형공정_제약조건.xlsx)';
```

### Entity 수정 `VcConstraint.java`

```java
@Entity @Table(schema="master", name="vc_constraint")
public class VcConstraint {
    @Id @Column(name="hose_id") private String hoseId;
    @Column(name="machine_id") private String machineId;
    @Column(name="lp_left_setting",  length=1, nullable=false) private String lpLeftSetting;
    @Column(name="lp_right_setting", length=1, nullable=false) private String lpRightSetting;
    // ... 기존 슬롯 O/X 필드

    public boolean allowsLeft()  { return "O".equals(lpLeftSetting); }
    public boolean allowsRight() { return "O".equals(lpRightSetting); }
}
```
