### `HoseRuleCache.java`

```java
@Component @RequiredArgsConstructor
public class HoseRuleCache {

    private final VcHoseRuleRepository repo;
    private final Cache<String, VcHoseRule> cache = Caffeine.newBuilder()
        .maximumSize(500)
        .expireAfterWrite(Duration.ofHours(1))        // 안전망 — LISTEN 누락 대비
        .recordStats()
        .build();

    public Optional<VcHoseRule> get(String hoseId) {
        var cached = cache.getIfPresent(hoseId);
        if (cached != null) return Optional.of(cached);
        return repo.findById(hoseId).map(rule -> {
            cache.put(hoseId, rule);
            return rule;
        });
    }

    public void invalidate(String hoseId) { cache.invalidate(hoseId); }
    public void invalidateAll() { cache.invalidateAll(); }

    public CacheStats stats() { return cache.stats(); }
}
```

### `HoseRuleListener.java`

```java
@Component @RequiredArgsConstructor @Slf4j
public class HoseRuleListener {

    private final DataSource dataSource;
    private final HoseRuleCache cache;
    private final ObjectMapper mapper;
    private volatile boolean running = true;

    @PostConstruct
    public void start() {
        var executor = Executors.newSingleThreadExecutor(r ->
            new Thread(r, "vc-hose-rule-listener"));
        executor.submit(this::listen);
    }

    private void listen() {
        while (running) {
            try (Connection conn = dataSource.getConnection()) {
                try (Statement stmt = conn.createStatement()) {
                    stmt.execute("LISTEN vc_hose_rule_change");
                }
                PGConnection pgConn = conn.unwrap(PGConnection.class);
                while (running) {
                    PGNotification[] notifications = pgConn.getNotifications(5000);
                    if (notifications == null) continue;
                    for (PGNotification n : notifications) {
                        handleNotification(n);
                    }
                }
            } catch (SQLException e) {
                log.error("LISTEN 연결 끊김 — 5초 후 재시도", e);
                sleepSafely(5000);
            }
        }
    }

    private void handleNotification(PGNotification n) {
        try {
            var event = mapper.readValue(n.getParameter(), HoseRuleChangeEvent.class);
            cache.invalidate(event.hoseId());
            log.info("Cache invalidated: hose_id={}, op={}", event.hoseId(), event.op());
        } catch (Exception e) {
            log.warn("이벤트 파싱 실패 — 전체 캐시 invalidate", e);
            cache.invalidateAll();
        }
    }

    @PreDestroy
    public void stop() { running = false; }
}

public record HoseRuleChangeEvent(String hoseId, String op) {}
```
