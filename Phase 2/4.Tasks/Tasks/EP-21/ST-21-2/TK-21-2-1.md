### Flyway `V2_7__create_vc_hose_rule.sql`

```sql
CREATE TABLE master.vc_hose_rule (
    hose_id                VARCHAR(40) PRIMARY KEY,
    machine_pin            VARCHAR(10),                       -- 'LP-01' 등 고정 호기 (NULL = 제약 없음)
    max_concurrent_slots   INTEGER NOT NULL DEFAULT 99 CHECK (max_concurrent_slots BETWEEN 1 AND 99),
    side_lock              VARCHAR(5) CHECK (side_lock IN ('LEFT','RIGHT')),  -- NULL = 양쪽
    lp_only                BOOLEAN NOT NULL DEFAULT FALSE,    -- TRUE면 IC 사용 금지
    notes                  TEXT,
    updated_at             TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_by             VARCHAR(40) NOT NULL DEFAULT 'seed'
);

CREATE INDEX idx_vc_hose_rule_machine_pin ON master.vc_hose_rule(machine_pin) WHERE machine_pin IS NOT NULL;

GRANT SELECT ON master.vc_hose_rule TO app_user;
GRANT INSERT, UPDATE, DELETE ON master.vc_hose_rule TO master_admin;

COMMENT ON TABLE master.vc_hose_rule IS 'EP-21: 품번 단위 운영 룰 (BR-V14, REQ-FUNC-VC-024)';
COMMENT ON COLUMN master.vc_hose_rule.machine_pin IS '고정 가류기 (예: 28422-08HA0 → LP-01)';
COMMENT ON COLUMN master.vc_hose_rule.max_concurrent_slots IS '동시 다중 슬롯 상한 (Σ ≤ value)';
COMMENT ON COLUMN master.vc_hose_rule.side_lock IS '좌/우 강제 (28422-2M800 → RIGHT)';
COMMENT ON COLUMN master.vc_hose_rule.lp_only IS 'IC 사용 금지 (BR-V08 LP-first 강화)';

-- PostgreSQL LISTEN/NOTIFY 트리거 (TK-21-2-3에서 캐시 무효화)
CREATE OR REPLACE FUNCTION master.notify_vc_hose_rule_change()
RETURNS TRIGGER AS $$
BEGIN
    PERFORM pg_notify('vc_hose_rule_change',
        json_build_object('hose_id', COALESCE(NEW.hose_id, OLD.hose_id), 'op', TG_OP)::text);
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_vc_hose_rule_notify
    AFTER INSERT OR UPDATE OR DELETE ON master.vc_hose_rule
    FOR EACH ROW EXECUTE FUNCTION master.notify_vc_hose_rule_change();
```

### Entity `VcHoseRule.java`

```java
@Entity @Table(schema="master", name="vc_hose_rule")
@Data @NoArgsConstructor @AllArgsConstructor @Builder
public class VcHoseRule {
    @Id @Column(name="hose_id") private String hoseId;
    @Column(name="machine_pin", length=10) private String machinePin;
    @Column(name="max_concurrent_slots", nullable=false) private Integer maxConcurrentSlots;
    @Enumerated(EnumType.STRING) @Column(name="side_lock") private SlotSide sideLock;
    @Column(name="lp_only", nullable=false) private Boolean lpOnly;
    @Column(name="notes") private String notes;
    @Column(name="updated_at") private Instant updatedAt;
    @Column(name="updated_by") private String updatedBy;

    public boolean hasMachinePin() { return machinePin != null; }
    public boolean hasSideLock()   { return sideLock != null; }
}
```

### Controller `HoseRuleController.java`

```java
@RestController @RequestMapping("/api/v1/master/vc-hose-rule") @RequiredArgsConstructor
public class HoseRuleController {
    private final VcHoseRuleRepository repo;

    @GetMapping
    public List<VcHoseRule> list() { return repo.findAll(); }

    @GetMapping("/{hoseId}")
    public Optional<VcHoseRule> get(@PathVariable String hoseId) { return repo.findById(hoseId); }

    @PostMapping @PreAuthorize("hasRole('IT_OPS')")
    public VcHoseRule save(@Valid @RequestBody VcHoseRule rule) {
        rule.setUpdatedAt(Instant.now());
        return repo.save(rule);    // 트리거가 NOTIFY 자동 발행
    }

    @DeleteMapping("/{hoseId}") @PreAuthorize("hasRole('IT_OPS')")
    public void remove(@PathVariable String hoseId) { repo.deleteById(hoseId); }
}
```
