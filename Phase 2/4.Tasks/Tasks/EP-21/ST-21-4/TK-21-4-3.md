### 핵심 로직

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-VC-SIDECAP-100/master_seed.sql",
      "classpath:db/migration/V2_7_data__seed_vc_hose_rule_47.sql"})
class SideCapRegressionIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired ScenarioLoader loader;

    @Test
    void rule_28422_2M800_right_only_max_two_per_rotation() {
        var ctx = loader.contextFor("28422-2M800", 400, 7);
        var result = allocator.allocate(ctx);

        var rows = result.schedule().stream()
            .filter(r -> r.hoseId().equals("28422-2M800")).toList();

        // 1) side_lock=RIGHT
        assertThat(rows).allMatch(r ->
            SlotSide.of(r.machineType(), r.slotNumber()) == SlotSide.RIGHT);

        // 2) 회전당 ≤ 2
        var grouped = rows.stream().collect(Collectors.groupingBy(
            r -> Tuple.of(r.productionDate(), r.machineId(), r.rotationId()),
            Collectors.counting()));
        assertThat(grouped.values()).allMatch(c -> c <= 2);
    }

    @Test
    void rule_28421_2M800_left_only_max_two_per_rotation() {
        var ctx = loader.contextFor("28421-2M800", 400, 7);
        var result = allocator.allocate(ctx);

        var rows = result.schedule().stream()
            .filter(r -> r.hoseId().equals("28421-2M800")).toList();

        assertThat(rows).allMatch(r ->
            SlotSide.of(r.machineType(), r.slotNumber()) == SlotSide.LEFT);

        var grouped = rows.stream().collect(Collectors.groupingBy(
            r -> Tuple.of(r.productionDate(), r.machineId(), r.rotationId()),
            Collectors.counting()));
        assertThat(grouped.values()).allMatch(c -> c <= 2);
    }

    @Test
    void regression_100_scenarios_no_violation() {
        var scenarios = loader.load("DS-VC-SIDECAP-100/scenarios.csv");
        var ctx = AllocationContext.fromOrders(scenarios, LocalDate.of(2026, 2, 16), 30);
        var result = allocator.allocate(ctx);

        // BR-V15·V16 위반 0건
        long violations = result.schedule().stream()
            .filter(r -> List.of("28421-2M800", "28422-2M800").contains(r.hoseId()))
            .filter(r -> !sideCapValid(r))
            .count();

        assertThat(violations).isZero();
    }

    @Test
    void conflict_when_side_saturated_above_cap() {
        // 우측 슬롯 회전당 2개 모두 점유 후 28422-2M800 추가 시도 → conflict
        var ctx = loader.contextSaturatingRightSlots();
        var result = allocator.allocate(ctx);

        assertThat(result.conflicts())
            .anyMatch(c -> c.hoseId().equals("28422-2M800") &&
                (c.reason().contains("BR-V16") || c.reason().contains("slot")));
    }

    boolean sideCapValid(VcScheduleRow r) {
        // 헬퍼 — side_lock + max_concurrent_slots 검증
        // ...
    }
}
```
