### 핵심 로직

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-VC-PIN-08HA0/master_seed.sql",
      "classpath:db/migration/V2_7_data__seed_vc_hose_rule_47.sql"})
class MachinePinRegressionIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired ScenarioLoader loader;

    @Test
    void rule_28422_08HA0_only_LP01() {
        var ctx = loader.contextFor("28422-08HA0", 300, 7);
        var result = allocator.allocate(ctx);

        var rows = result.schedule().stream()
            .filter(r -> r.hoseId().equals("28422-08HA0")).toList();

        assertThat(rows)
            .as("28422-08HA0 모든 배치는 LP-01이어야 함 (BR-V14)")
            .allMatch(r -> r.machineId().equals("LP-01"));

        assertThat(rows).noneMatch(r -> r.machineType() == MachineType.IC);
    }

    @Test
    void rule_28422_08HA0_max_one_slot_per_rotation() {
        var ctx = loader.contextFor("28422-08HA0", 300, 7);
        var result = allocator.allocate(ctx);

        // (date, machine, rotation) 키별 슬롯 개수 ≤ 1
        var grouped = result.schedule().stream()
            .filter(r -> r.hoseId().equals("28422-08HA0"))
            .collect(Collectors.groupingBy(
                r -> Tuple.of(r.productionDate(), r.machineId(), r.rotationId()),
                Collectors.counting()));

        assertThat(grouped.values())
            .as("28422-08HA0 회전당 동시 슬롯 ≤ 1 (BR-V14)")
            .allMatch(count -> count <= 1);
    }

    @Test
    void regression_50_scenarios_no_violation() {
        var scenarios = loader.load("DS-VC-PIN-08HA0/scenarios.csv");
        var ctx = AllocationContext.fromOrders(scenarios, LocalDate.of(2026, 2, 16), 30);
        var result = allocator.allocate(ctx);

        long violations = result.schedule().stream()
            .filter(r -> r.hoseId().equals("28422-08HA0"))
            .filter(r -> !r.machineId().equals("LP-01"))
            .count();

        assertThat(violations).isZero();
    }

    @Test
    void conflict_when_LP01_saturated() {
        // LP-01 회전 전부 점유 → 28422-08HA0 conflict (다른 머신 우회 불가)
        var ctx = loader.contextSaturatingLP01();
        var result = allocator.allocate(ctx);

        assertThat(result.conflicts())
            .anyMatch(c -> c.hoseId().equals("28422-08HA0") && c.reason().contains("BR-V14"));
    }
}
```
