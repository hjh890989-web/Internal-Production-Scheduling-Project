#### A. `ImportTrackingService` 수정 — ParsedWorkbook 캐싱

```java
package com.scheduling.order.import_;

import com.scheduling.order.parser.ParsedWorkbook;
import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.RequiredArgsConstructor;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;

import java.time.Duration;
import java.util.UUID;

@Service
@RequiredArgsConstructor
public class ImportTrackingService {

    private final RedisTemplate<String, Object> redis;
    private final ObjectMapper json;
    private static final Duration TTL = Duration.ofHours(24);    // REQ-FUNC-OC-004 라운드트립 보존 기간

    public void cacheParsedWorkbooks(UUID trackingId, List<ParsedWorkbook> workbooks) {
        var key = parsedKey(trackingId);
        try {
            // ParsedWorkbook record + 메타데이터를 JSON 직렬화 (Java 21 record 호환)
            String payload = json.writeValueAsString(workbooks);
            redis.opsForValue().set(key, payload, TTL);
        } catch (Exception e) {
            throw new IllegalStateException("Failed to cache parsed workbooks", e);
        }
    }

    public List<ParsedWorkbook> loadParsedWorkbooks(UUID trackingId) {
        var key = parsedKey(trackingId);
        String payload = (String) redis.opsForValue().get(key);
        if (payload == null) {
            throw new NoSuchElementException("No cached workbook for: " + trackingId);
        }
        try {
            return json.readValue(payload, json.getTypeFactory()
                .constructCollectionType(List.class, ParsedWorkbook.class));
        } catch (Exception e) {
            throw new IllegalStateException("Failed to load cached workbooks", e);
        }
    }

    public void cacheMappingResult(UUID trackingId, MappingResult result) {
        // 매핑 결과도 캐시 (UI가 즉시 조회 가능)
        var key = mappingKey(trackingId);
        try {
            redis.opsForValue().set(key, json.writeValueAsString(result), TTL);
        } catch (Exception e) {
            throw new IllegalStateException(e);
        }
    }

    // ... 기존 markStarted·update·get·recordError 유지

    private String parsedKey(UUID id)  { return "import:parsed:" + id; }
    private String mappingKey(UUID id) { return "import:mapping:" + id; }
}
```

#### B. `ImportRetryService.java` — 재매핑 서비스

```java
package com.scheduling.order.import_;

import com.scheduling.order.mapping.*;
import com.scheduling.order.parser.*;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;

import java.util.*;

@Slf4j
@Service
@RequiredArgsConstructor
public class ImportRetryService {

    private final ImportTrackingService tracking;
    private final SchemaMappingService mapper;
    private final SourceClassifierService classifier;

    /**
     * 라운드트립 재시도 — 파싱 생략, 분류 + 매핑만 재실행.
     * REQ-FUNC-OC-004 핵심.
     */
    @Async("excelImportExecutor")
    public void retryAsync(UUID trackingId) {
        try {
            tracking.update(trackingId, ImportStatus.MAPPING, null);

            // 1. 캐시된 ParsedWorkbook 로드 (재업로드 불필요)
            List<ParsedWorkbook> workbooks = tracking.loadParsedWorkbooks(trackingId);
            log.info("Retry mapping for trackingId {}: {} cached workbooks loaded",
                trackingId, workbooks.size());

            // 2. 룰셋이 reload되었음 (TK-01-2-1 SchemaMappingService에서 @RefreshScope)
            //    또는 명시적 reload 호출
            ruleLoader.reload();

            // 3. 분류 + 매핑 재실행
            List<MappingResult> allResults = new ArrayList<>();
            for (ParsedWorkbook wb : workbooks) {
                ClassificationResult c = classifier.classify(wb);
                MappingResult r = mapper.map(wb, c.sourceType());
                allResults.add(r);
            }

            // 4. 통합 결과 저장
            MappingResult merged = MappingResult.merge(allResults);
            tracking.cacheMappingResult(trackingId, merged);
            tracking.update(trackingId,
                merged.failureCount() == 0 ? ImportStatus.MAPPED : ImportStatus.REVIEW_REQUIRED,
                null);

        } catch (NoSuchElementException e) {
            log.warn("Cache miss for trackingId {} (TTL 24h 만료 가능)", trackingId);
            tracking.update(trackingId, ImportStatus.FAILED, "캐시 만료 — 원본 파일 재업로드 필요");
        } catch (Exception e) {
            log.error("Retry failed for trackingId {}", trackingId, e);
            tracking.update(trackingId, ImportStatus.FAILED, e.getMessage());
        }
    }
}
```

#### C. `OrderImportController` 확장 — retry endpoint

```java
// OrderImportController.java 에 메서드 추가

@PostMapping("/import/{trackingId}/retry")
@PreAuthorize("hasAnyRole('PLANNER', 'PLANNER_ASSISTANT')")
@ResponseStatus(HttpStatus.ACCEPTED)
public RetryResponse retry(@PathVariable UUID trackingId) {
    // 캐시 존재 확인 (early fail)
    try {
        trackingService.loadParsedWorkbooks(trackingId);   // throws NoSuchElement if expired
    } catch (NoSuchElementException e) {
        throw new ResponseStatusException(
            HttpStatus.GONE,
            "캐시가 만료되었습니다 (TTL 24h). 원본 파일을 다시 업로드해 주세요.");
    }

    // 비동기 재시도 트리거
    retryService.retryAsync(trackingId);

    return new RetryResponse(
        trackingId,
        "ACCEPTED",
        "/api/v1/orders/import/" + trackingId
    );
}
```

#### E. `MappingRuleService.java` — 룰 CRUD + audit

```java
package com.scheduling.order.mapping;

import com.scheduling.order.audit.AuditLogger;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.concurrent.ConcurrentHashMap;
import java.util.Map;

@Service
@RequiredArgsConstructor
public class MappingRuleService {

    private final MappingRuleLoader loader;
    private final AuditLogger auditLogger;

    private final Map<SourceType, MappingRule> liveRules = new ConcurrentHashMap<>();

    @Transactional
    public MappingRule updateRule(SourceType type, MappingRule updated, String actor) {
        MappingRule before = liveRules.get(type);

        // 룰 검증 (필수 필드·정규식)
        validate(updated);

        // 메모리 갱신 (재시작 없이 즉시 반영)
        liveRules.put(type, updated);

        // Audit 기록 (BR-X02 — 모든 변경)
        auditLogger.logRuleChange(type, before, updated, actor);

        return updated;
    }

    public MappingRule getRule(SourceType type) {
        return liveRules.computeIfAbsent(type, loader::loadFromYaml);
    }

    private void validate(MappingRule rule) {
        if (rule.fieldMappings() == null || rule.fieldMappings().isEmpty()) {
            throw new IllegalArgumentException("fieldMappings 비어있음");
        }
        // 필수 5 필드 (hose_id·delivery_date·qty·order_type·customer) 매핑 존재 확인
        for (String required : List.of("hose_id", "delivery_date", "qty")) {
            if (!rule.fieldMappings().containsKey(required)) {
                throw new IllegalArgumentException("필수 필드 누락: " + required);
            }
        }
    }
}
```

## :memo: Implementation Notes

- **기술 힌트**:
  - Spring Data Redis `RedisTemplate<String, Object>` + JSON Serializer (Jackson)
  - Java 21 record 직렬화는 Jackson 2.16+ 자동 지원
  - **TTL 갱신**: 재시도 시 `redis.expire()` 호출하면 TTL 리셋 — UX 친화적
  - **Idempotency**: `SETNX` (set if not exists)로 같은 trackingId retry 동시 호출 차단
  - **Pub/Sub** 대안: retry 시작·완료를 WebSocket으로 UI에 PUSH (ST-01-2 UI 진행 표시)
- **유사 패턴**:
  - VC/EX 스케줄 후보 캐시도 동일 패턴 — Redis + TTL + audit
- **표준 참조**:
  - Spring Data Redis: https://docs.spring.io/spring-data/redis/reference/redis.html
  - Redis Expiration: https://redis.io/docs/latest/develop/use/keyspace-notifications/
- **참고 외부 자료**:
  - "Caching with Redis" — Baeldung
