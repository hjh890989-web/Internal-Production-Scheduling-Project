```typescript
// frontend/src/features/order-import/types/mapping.ts

export type SourceType = 'MONTHLY_FORECAST' | 'WEEKLY_PLAN' | 'CONFIRMED_ORDER' | 'KD_ORDER' | 'UNRECOGNIZED';
export type ImportStatus = 'QUEUED' | 'PARSING' | 'PARSED' | 'MAPPING' | 'MAPPED' | 'REVIEW_REQUIRED' | 'FAILED';

export interface MappingFailure {
  sheetName: string;
  rowIndex: number;
  originalCells: string[];
  failedField: 'hose_id' | 'delivery_date' | 'qty' | 'customer' | 'HEADER';
  reason: string;             // 한국어
  userMessage: string;        // "{sheet} 시트 {row}행: {field} — {reason}"
}

export interface OrderDraft {
  orderId: string;
  hoseId: string;
  deliveryDate: string;       // ISO-8601
  qty: number;
  orderType: string;
  customer: string;
}

export interface MappingResult {
  trackingId: string;
  sourceType: SourceType;
  confidence: number;
  successes: OrderDraft[];
  failures: MappingFailure[];
  successRate: number;        // 0.0 ~ 1.0
}

export interface MappingRule {
  sourceType: SourceType;
  fieldMappings: Record<string, FieldMapping>;
  defaultCustomer: string;
}

export interface FieldMapping {
  field: 'hose_id' | 'delivery_date' | 'qty' | 'customer';
  aliases: string[];
  required: boolean;
  defaultValue?: string;
}
```

#### B. `MappingReviewModal.tsx` — 메인 모달

```tsx
// frontend/src/features/order-import/components/MappingReviewModal.tsx
import { Modal, Tabs, Alert, Space, Button, Progress } from 'antd';
import { CheckCircleOutlined, WarningOutlined } from '@ant-design/icons';
import { useImportStatus } from '../hooks/useImportStatus';
import { useRetryMapping } from '../hooks/useRetryMapping';
import { FailureSummaryPanel } from './FailureSummaryPanel';
import { MappingRuleEditor } from './MappingRuleEditor';
import { ConfidenceBadge } from './ConfidenceBadge';

interface Props {
  trackingId: string;
  open: boolean;
  onClose: () => void;
}

export function MappingReviewModal({ trackingId, open, onClose }: Props) {
  const { data: status, isLoading } = useImportStatus(trackingId);
  const retry = useRetryMapping(trackingId);

  if (isLoading || !status) return null;

  const successCount = status.result?.successes.length ?? 0;
  const failureCount = status.result?.failures.length ?? 0;
  const total = successCount + failureCount;
  const successPct = total > 0 ? (successCount / total) * 100 : 0;
  const isCriticalFailure = successPct < 95;        // REQ-FUNC-OC-004 threshold

  return (
    <Modal
      title="수주 데이터 매핑 검토"
      open={open}
      onCancel={onClose}
      width={1100}
      destroyOnClose={false}             // 세션 보존을 위해 mount 유지
      footer={
        <Space>
          <Button onClick={onClose}>닫기</Button>
          <Button
            type="primary"
            danger={isCriticalFailure}
            loading={retry.isPending}
            onClick={() => retry.mutate()}
            disabled={failureCount === 0}
          >
            {isCriticalFailure ? '룰 보정 후 재시도' : '재시도'}
          </Button>
          <Button
            type="primary"
            disabled={failureCount > 0}    // 실패 0 일 때만 commit
            // onClick → Diff 화면 (ST-03-1)
          >
            다음: 변경 검토
          </Button>
        </Space>
      }
    >
      <Space direction="vertical" style={{ width: '100%' }}>
        <Alert
          message={
            <Space>
              <span>분류:</span>
              <ConfidenceBadge type={status.result?.sourceType} confidence={status.result?.confidence ?? 0} />
              <span>·</span>
              <span>매핑 성공률: {successPct.toFixed(1)}%</span>
            </Space>
          }
          type={isCriticalFailure ? 'warning' : 'success'}
          showIcon
          icon={isCriticalFailure ? <WarningOutlined /> : <CheckCircleOutlined />}
        />

        <Progress
          percent={successPct}
          status={isCriticalFailure ? 'exception' : 'success'}
          format={() => `${successCount}/${total} (${successPct.toFixed(1)}%)`}
        />

        <Tabs
          items={[
            {
              key: 'failures',
              label: `실패 ${failureCount}건`,
              children: <FailureSummaryPanel failures={status.result?.failures ?? []} />,
              disabled: failureCount === 0,
            },
            {
              key: 'rules',
              label: '매핑 룰 보정',
              children: <MappingRuleEditor sourceType={status.result?.sourceType} />,
            },
            {
              key: 'success',
              label: `성공 ${successCount}건`,
              children: <SuccessPreviewTable successes={status.result?.successes ?? []} />,
            },
          ]}
        />
      </Space>
    </Modal>
  );
}
```

#### C. `FailureSummaryPanel.tsx` — 사유별 그룹

```tsx
import { Collapse, Table, Tag, Typography } from 'antd';
import type { MappingFailure } from '../types/mapping';

const { Text } = Typography;

interface Props { failures: MappingFailure[]; }

export function FailureSummaryPanel({ failures }: Props) {
  // failedField 별로 그룹화
  const grouped = failures.reduce<Record<string, MappingFailure[]>>((acc, f) => {
    (acc[f.failedField] = acc[f.failedField] ?? []).push(f);
    return acc;
  }, {});

  return (
    <Collapse
      items={Object.entries(grouped).map(([field, items]) => ({
        key: field,
        label: (
          <Space>
            <Text strong>{fieldDisplayName(field)}</Text>
            <Tag color="orange">{items.length}건</Tag>
          </Space>
        ),
        children: (
          <Table
            dataSource={items}
            rowKey={(r) => `${r.sheetName}-${r.rowIndex}`}
            size="small"
            pagination={{ pageSize: 10 }}
            columns={[
              { title: '시트', dataIndex: 'sheetName', width: 100 },
              { title: '행', dataIndex: 'rowIndex', width: 80, render: (v) => v + 1 },
              { title: '원본 데이터', dataIndex: 'originalCells',
                render: (cells: string[]) => cells.slice(0, 5).map((c, i) => (
                  <Tag key={i}>{c || '∅'}</Tag>
                )),
              },
              { title: '사유', dataIndex: 'reason' },
            ]}
          />
        ),
      }))}
    />
  );
}

function fieldDisplayName(field: string): string {
  return {
    hose_id: '품번',
    delivery_date: '납기일',
    qty: '수량',
    customer: '거래처',
    HEADER: '헤더 행 식별',
  }[field] ?? field;
}
```

#### D. `MappingRuleEditor.tsx` — 룰 인라인 편집

```tsx
import { Form, Input, Tag, Button, Card, Space, message } from 'antd';
import { PlusOutlined, CloseOutlined } from '@ant-design/icons';
import { useMappingRules } from '../hooks/useMappingRules';

interface Props { sourceType?: string; }

export function MappingRuleEditor({ sourceType }: Props) {
  const { data: rule, mutate: saveRule, isPending } = useMappingRules(sourceType);
  const [form] = Form.useForm();

  if (!rule) return null;

  return (
    <Form form={form} layout="vertical" initialValues={rule}>
      <Space direction="vertical" style={{ width: '100%' }}>
        {Object.entries(rule.fieldMappings).map(([field, mapping]) => (
          <Card key={field} size="small" title={fieldDisplayName(field)} extra={mapping.required && <Tag color="red">필수</Tag>}>
            <Form.List name={['fieldMappings', field, 'aliases']}>
              {(fields, { add, remove }) => (
                <Space wrap>
                  {fields.map((f) => (
                    <Space.Compact key={f.key}>
                      <Form.Item {...f} noStyle>
                        <Input style={{ width: 200 }} />
                      </Form.Item>
                      <Button icon={<CloseOutlined />} onClick={() => remove(f.name)} />
                    </Space.Compact>
                  ))}
                  <Button icon={<PlusOutlined />} onClick={() => add('')}>
                    별칭 추가
                  </Button>
                </Space>
              )}
            </Form.List>
          </Card>
        ))}
        <Button
          type="primary"
          loading={isPending}
          onClick={() => {
            form.validateFields().then((values) => {
              saveRule(values, {
                onSuccess: () => message.success('룰 저장 완료. 재시도하면 적용됩니다.'),
              });
            });
          }}
        >
          룰 저장
        </Button>
      </Space>
    </Form>
  );
}
```

#### E. `useRetryMapping.ts` — 라운드트립 hook (TK-01-2-3 endpoint 호출)

```typescript
// frontend/src/features/order-import/hooks/useRetryMapping.ts
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { retryMapping } from '../api/mappingApi';

export function useRetryMapping(trackingId: string) {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: () => retryMapping(trackingId),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['importStatus', trackingId] });
    },
  });
}
```

## :memo: Implementation Notes

- **기술 힌트**:
  - Ant Design `Form.List` ≫ 수동 array state — 검증·렌더링 자동
  - `Modal` 의 `destroyOnClose={false}` 가 폼 상태 보존의 핵심
  - TanStack Query `invalidateQueries` 로 재시도 후 자동 refetch
  - `Collapse` items API (v5) — 구버전 `panel` 패턴 deprecate
  - **상태 분리**: 서버 상태 (TanStack Query) vs 클라이언트 상태 (Zustand) — 본 컴포넌트는 서버 상태만 사용
- **유사 패턴**:
  - VC/EX 매핑 룰 편집 UI도 동일 패턴 차용 가능 (마스터 데이터 관리)
- **표준 참조**:
  - Ant Design 5: https://ant.design/components/overview
  - TanStack Query v5: https://tanstack.com/query/v5
  - WCAG 2.1 AA: https://www.w3.org/WAI/standards-guidelines/wcag/
- **참고 외부 자료**:
  - "React 18 Concurrent Features" — React 공식
