package com.scheduling.order.mapping;

import com.scheduling.order.parser.*;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.testcontainers.junit.jupiter.Testcontainers;

import java.nio.file.*;
import java.util.stream.Stream;

import static org.assertj.core.api.Assertions.*;

@SpringBootTest
@Testcontainers
class MappingAccuracyIT {

    @Autowired private ExcelParserService parser;
    @Autowired private SourceClassifierService classifier;
    @Autowired private SchemaMappingService mapper;

    /**
     * TC-OC-003: 30 회귀 워크북 (DS-ORDER-3X) 자동 매핑 성공률 ≥95%
     */
    @Test
    void maps_30_workbooks_with_at_least_95_percent_success_rate() throws Exception {
        int totalSuccess = 0;
        int totalAttempts = 0;
        var sourceTypeStats = new java.util.EnumMap<SourceType, int[]>(SourceType.class);

        for (Path workbook : listAllWorkbooks()) {
            try (var is = Files.newInputStream(workbook)) {
                var parsed = parser.parse(workbook.getFileName().toString(), is, Files.size(workbook));
                var classification = classifier.classify(parsed);
                var result = mapper.map(parsed, classification.sourceType());

                int succ = result.successes().size();
                int fail = result.failures().size();
                int tot  = succ + fail;

                totalSuccess += succ;
                totalAttempts += tot;

                sourceTypeStats.merge(classification.sourceType(),
                    new int[]{succ, tot},
                    (a, b) -> new int[]{a[0]+b[0], a[1]+b[1]});
            }
        }

        // 전체 평균 ≥95%
        double avgRate = (double) totalSuccess / totalAttempts;
        assertThat(avgRate)
            .as("Average mapping success rate over %d rows: %.1f%%", totalAttempts, avgRate * 100)
            .isGreaterThanOrEqualTo(0.95);

        // 4 SourceType 각각도 ≥95%
        sourceTypeStats.forEach((type, stats) -> {
            double rate = (double) stats[0] / stats[1];
            assertThat(rate)
                .as("Mapping rate for %s: %.1f%% (success %d / total %d)",
                    type, rate * 100, stats[0], stats[1])
                .isGreaterThanOrEqualTo(0.95);
        });
    }

    private Stream<Path> listAllWorkbooks() throws Exception {
        var root = Paths.get(getClass().getResource("/workbooks/DS-ORDER-3X").toURI());
        return Files.walk(root).filter(p -> p.toString().endsWith(".xlsx"));
    }
}
