```python
"""
DS-ORDER-3X 데이터셋 생성기 — 30개 합성 .xlsx 워크북

각 SourceType별 7~8개 변형 생성:
- 동일 의미·다른 어휘 (예: "월별 예상" vs "월별 발주 예측")
- 동일 의미·다른 시트 구조
- 모서리 케이스: 병합 셀, 다중 헤더 행, 빈 컬럼
"""
import openpyxl
from openpyxl.styles import PatternFill
from pathlib import Path
import random

OUT_DIR = Path(__file__).parent.parent / "backend/order-consolidation/src/test/resources/workbooks/DS-ORDER-3X"

MONTHLY_HEADERS = [
    ["월별 예상 발주량", "품번", "수량", "납기"],
    ["월간 예상", "Item Code", "Qty", "Delivery"],
    ["FORECAST", "PartNo", "Quantity", "Due Date"],
    ...
]

WEEKLY_HEADERS = [
    ["주간 발주 계획", "주차", "품번", "수량"],
    ["WEEKLY PLAN", "Week", "Item", "Qty"],
    ...
]

CONFIRMED_HEADERS = [...]
KD_HEADERS = [...]

def generate_workbook(source_type: str, variant: int) -> Path:
    wb = openpyxl.Workbook()
    ws = wb.active
    ws.title = f"{source_type}_변형{variant}"

    # 헤더 행 (variant별 다른 어휘)
    headers = HEADERS[source_type][variant]
    for col, header in enumerate(headers, start=1):
        ws.cell(row=1, column=col, value=header)

    # 데이터 행 (랜덤 50~200 row)
    n_rows = random.randint(50, 200)
    for r in range(2, n_rows + 2):
        ws.cell(row=r, column=1, value=f"PART-{r:04d}")
        ws.cell(row=r, column=2, value=random.randint(10, 1000))
        ws.cell(row=r, column=3, value=f"2026-02-{r % 28 + 1:02d}")

    out = OUT_DIR / source_type.lower() / f"{source_type}_{variant}.xlsx"
    out.parent.mkdir(parents=True, exist_ok=True)
    wb.save(out)
    return out


if __name__ == "__main__":
    for st in ["monthly", "weekly", "confirmed", "kd"]:
        count = 8 if st in ("confirmed", "kd") else 7
        for i in range(count):
            generate_workbook(st, i)
    print(f"30 workbooks generated under {OUT_DIR}")
```

#### B. `ClassificationAccuracyTest.java` — 30 워크북 회귀

```java
package com.scheduling.order.parser;

import org.junit.jupiter.api.Test;
import org.junit.jupiter.params.ParameterizedTest;
import org.junit.jupiter.params.provider.MethodSource;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.core.io.ClassPathResource;

import java.io.InputStream;
import java.nio.file.*;
import java.util.stream.Stream;

import static org.assertj.core.api.Assertions.*;

@SpringBootTest
class ClassificationAccuracyTest {

    @Autowired private ExcelParserService parser;
    @Autowired private SourceClassifierService classifier;

    /**
     * TC-OC-002: 30개 회귀 워크북에서 분류 정확도 ≥99%
     */
    @Test
    void classifies_30_workbooks_with_at_least_99_percent_accuracy() throws Exception {
        int correct = 0;
        int total = 0;
        StringBuilder errors = new StringBuilder();

        for (Path workbook : listAllTestWorkbooks()) {
            String expectedType = workbook.getParent().getFileName().toString().toUpperCase();
            // monthly → MONTHLY_FORECAST 매핑
            SourceType expected = mapToSourceType(expectedType);

            try (InputStream is = Files.newInputStream(workbook)) {
                var parsed = parser.parse(workbook.getFileName().toString(), is, Files.size(workbook));
                var classification = classifier.classify(parsed);

                if (classification.sourceType() == expected) {
                    correct++;
                } else {
                    errors.append(String.format(
                        "Workbook %s: expected %s, got %s (conf=%.2f)%n",
                        workbook.getFileName(), expected, classification.sourceType(), classification.confidence()));
                }
                total++;
            }
        }

        double accuracy = (double) correct / total;
        assertThat(accuracy)
            .as("Classification accuracy %.1f%% with %d errors:%n%s", accuracy * 100, total - correct, errors)
            .isGreaterThanOrEqualTo(0.99);
    }

    @ParameterizedTest(name = "{0} → {1}")
    @MethodSource("provideEdgeCases")
    void handles_edge_cases(String filename, String expectedBehavior) throws Exception {
        // 손상·병합·수식·빈·초과·위장 케이스 8건 검증
    }

    private Stream<Path> listAllTestWorkbooks() throws Exception {
        var root = Paths.get(getClass().getResource("/workbooks/DS-ORDER-3X").toURI());
        return Files.walk(root).filter(p -> p.toString().endsWith(".xlsx"));
    }

    // 보조 메서드 생략
}
```

```java
        var f2 = new MockMultipartFile("files", "weekly.xlsx",  "application/vnd...", loadBytes("weekly_0.xlsx"));
```

```javascript
// perf/scripts/import_load.js
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
    scenarios: {
        concurrent_users: {
            executor: 'constant-vus',
            vus: 30,                              // NFR-COM-001 30명 동시
            duration: '60s',
        },
    },
    thresholds: {
        http_req_duration: ['p(95)<2000'],        // TC-OC-001 p95 ≤2초
        http_req_failed: ['rate<0.001'],          // 오류율 ≤0.1%
    },
};

export default function () {
    const workbookFile = open('./fixtures/sample.xlsx', 'b');
    const data = {
        files: http.file(workbookFile, 'sample.xlsx', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'),
    };

    const res = http.post('http://localhost:8080/api/v1/orders/import', data, {
        headers: { Authorization: `Bearer ${__ENV.JWT_TOKEN}` },
    });

    check(res, {
        'status 202': (r) => r.status === 202,
        'trackingId present': (r) => r.json('trackingId') !== undefined,
        'response time < 2s': (r) => r.timings.duration < 2000,
    });

    sleep(1);
}
```
