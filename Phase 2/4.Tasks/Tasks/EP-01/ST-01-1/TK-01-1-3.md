#### A. `OrderImportController.java`

```java
package com.scheduling.order.api;

import jakarta.validation.constraints.Size;
import lombok.RequiredArgsConstructor;
import org.springframework.http.HttpStatus;
import org.springframework.security.access.prepost.PreAuthorize;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

import java.util.List;
import java.util.UUID;

@RestController
@RequestMapping("/api/v1/orders")
@RequiredArgsConstructor
public class OrderImportController {

    private final ImportOrchestrator orchestrator;
    private final ImportTrackingService trackingService;

    @PostMapping(value = "/import", consumes = "multipart/form-data")
    @PreAuthorize("hasRole('PLANNER') or hasRole('PLANNER_ASSISTANT')")
    @ResponseStatus(HttpStatus.ACCEPTED)            // 202 — 백그라운드 처리
    public ImportResponse importWorkbooks(
        @RequestParam("files")
        @Size(min = 1, max = 3, message = "1~3 files required")
        List<MultipartFile> files
    ) {
        // 1. 동기 검증 (파일 크기·MIME·확장자) — 본 메서드 응답 ≤2초 보장
        for (MultipartFile file : files) {
            validateFile(file);
        }

        // 2. 추적 ID 발급
        UUID trackingId = UUID.randomUUID();
        trackingService.markStarted(trackingId, files);

        // 3. 비동기 처리 위임 (별도 thread pool)
        orchestrator.processAsync(trackingId, files);

        // 4. 즉시 응답
        return new ImportResponse(
            trackingId,
            "ACCEPTED",
            "/api/v1/orders/import/" + trackingId
        );
    }

    @GetMapping("/import/{trackingId}")
    @PreAuthorize("hasAnyRole('PLANNER', 'PLANNER_ASSISTANT')")
    public ImportStatusResponse status(@PathVariable UUID trackingId) {
        return trackingService.get(trackingId);
    }

    private void validateFile(MultipartFile file) {
        // 1. 크기 — 20MB 초과 시 즉시 거부 (Spring multipart 자체 제한 + 명시 검증)
        if (file.getSize() > 20 * 1024 * 1024) {
            throw new IllegalArgumentException("FILE_TOO_LARGE: " + file.getOriginalFilename());
        }
        // 2. 확장자
        String name = file.getOriginalFilename();
        if (name == null || !name.toLowerCase().endsWith(".xlsx")) {
            throw new IllegalArgumentException("INVALID_EXTENSION: " + name);
        }
        // 3. MIME 타입 (Apache Tika 매직 바이트 검증 — 위장 파일 차단)
        // 실제 구현: org.apache.tika.Tika().detect(file.getInputStream())
        // 결과가 "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet" 인지 확인
    }
}
```

#### B. `ImportOrchestrator.java` — 비동기 처리

```java
package com.scheduling.order.import_;

import com.scheduling.order.parser.*;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;

import java.io.IOException;
import java.util.List;
import java.util.UUID;

@Slf4j
@Service
@RequiredArgsConstructor
public class ImportOrchestrator {

    private final ExcelParserService parser;
    private final SourceClassifierService classifier;
    private final ImportTrackingService tracking;

    @Async("excelImportExecutor")
    public void processAsync(UUID trackingId, List<MultipartFile> files) {
        try {
            tracking.update(trackingId, ImportStatus.PARSING, null);

            for (MultipartFile file : files) {
                try {
                    var workbook = parser.parse(
                        file.getOriginalFilename(),
                        file.getInputStream(),
                        file.getSize()
                    );
                    var classification = classifier.classify(workbook);
                    tracking.recordFile(trackingId, file.getOriginalFilename(), workbook, classification);
                } catch (IOException | ExcelParseException e) {
                    log.warn("Parse failed for {}: {}", file.getOriginalFilename(), e.getMessage());
                    tracking.recordError(trackingId, file.getOriginalFilename(), e.getMessage());
                }
            }

            tracking.update(trackingId, ImportStatus.PARSED, null);
            // 다음 단계 (ST-01-2 스키마 매핑)에 위임 — 이벤트 발행
        } catch (Exception e) {
            log.error("Import failed for tracking {}", trackingId, e);
            tracking.update(trackingId, ImportStatus.FAILED, e.getMessage());
        }
    }
}
```

#### C. `ImportTrackingService.java` — Redis 기반 상태 저장

```java
package com.scheduling.order.import_;

import com.scheduling.order.api.ImportStatusResponse;
import lombok.RequiredArgsConstructor;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;

import java.time.Duration;
import java.time.Instant;
import java.util.*;

@Service
@RequiredArgsConstructor
public class ImportTrackingService {

    private final RedisTemplate<String, Object> redis;
    private static final Duration TTL = Duration.ofHours(24);

    public void markStarted(UUID trackingId, List<MultipartFile> files) {
        var key = key(trackingId);
        redis.opsForHash().put(key, "status", ImportStatus.QUEUED.name());
        redis.opsForHash().put(key, "startedAt", Instant.now().toString());
        redis.opsForHash().put(key, "fileCount", String.valueOf(files.size()));
        redis.opsForHash().put(key, "filenames",
            files.stream().map(MultipartFile::getOriginalFilename).toList());
        redis.expire(key, TTL);
    }

    public void update(UUID id, ImportStatus status, String error) {
        var key = key(id);
        redis.opsForHash().put(key, "status", status.name());
        redis.opsForHash().put(key, "updatedAt", Instant.now().toString());
        if (error != null) {
            redis.opsForHash().put(key, "error", error);
        }
    }

    public ImportStatusResponse get(UUID id) {
        var map = redis.opsForHash().entries(key(id));
        if (map.isEmpty()) {
            throw new NoSuchElementException("Tracking ID not found: " + id);
        }
        return ImportStatusResponse.from(id, map);
    }

    // ... recordFile, recordError 생략

    private String key(UUID id) { return "import:" + id; }
}
```

#### D. `AsyncConfig.java` — Excel 처리 전용 스레드 풀

```java
package com.scheduling.order.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.annotation.EnableAsync;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;

import java.util.concurrent.Executor;

@Configuration
@EnableAsync
public class AsyncConfig {

    @Bean(name = "excelImportExecutor")
    public Executor excelImportExecutor() {
        ThreadPoolTaskExecutor exec = new ThreadPoolTaskExecutor();
        exec.setCorePoolSize(4);                  // 4 동시 import (30명 사용자 가정 충분)
        exec.setMaxPoolSize(8);
        exec.setQueueCapacity(50);
        exec.setThreadNamePrefix("excel-import-");
        exec.setRejectedExecutionHandler(new java.util.concurrent.ThreadPoolExecutor.CallerRunsPolicy());
        exec.initialize();
        return exec;
    }
}
```

## :memo: Implementation Notes

- **기술 힌트**:
  - `@PreAuthorize`는 Spring Security 6 method security 활성 필요 (`@EnableMethodSecurity`)
  - `MultipartFile.getInputStream()`은 디스크 임시 파일 기반 — 메모리 적게 사용
  - Apache Tika 의존성 추가: `org.apache.tika:tika-core:2.9.x` (매직 바이트 검증)
  - `HttpStatus.ACCEPTED (202)`가 비동기 응답 표준
  - **Idempotency** 보장: trackingId 발급 후 같은 요청 재시도해도 새 ID 생성 (Idempotency-Key 헤더는 향후)
- **유사 패턴**:
  - 향후 PDF·CSV 업로드 엔드포인트도 동일 패턴 — multipart + async + tracking
- **표준 참조**:
  - Spring Web 6 Multipart: https://docs.spring.io/spring-framework/reference/web/webmvc/mvc-controller/ann-methods/multipart-forms.html
  - REST 비동기 패턴 (202 Accepted): RFC 7231 §6.3.3
- **참고 외부 자료**:
  - "Building File Upload APIs with Spring Boot" — Baeldung
