### `ConflictCategorizer.java`

```java
@Service
public class ConflictCategorizer {

    private static final Map<String, ConflictCategory> BR_TO_CATEGORY = Map.of(
        "BR-V13", SLOT_OX,
        "BR-V07", ANGLE_CAPA,
        "BR-V03", DAILY_CAPA,
        "BR-X07", DEADLINE_D2,
        "BR-X06", DAY_LOCK,
        "BR-V15", LEFT_RIGHT,
        "BR-V16", LEFT_RIGHT,
        "BR-V14", MACHINE_PIN,
        "BR-V17", SPEC_LT7
    );

    /**
     * AllocationConflict.reason 텍스트에서 BR-Xxx 패턴 추출 → category 매핑.
     * 다중 BR 매칭 시 우선순위 (가장 엄격한 hard 제약 우선).
     */
    public ClassifiedConflict classify(AllocationConflict conflict) {
        var reason = conflict.reason();
        var matched = BR_TO_CATEGORY.entrySet().stream()
            .filter(e -> reason.contains(e.getKey()))
            .map(Map.Entry::getValue)
            .findFirst()
            .orElse(UNKNOWN);

        return new ClassifiedConflict(conflict, matched);
    }

    public List<ClassifiedConflict> classifyAll(List<AllocationConflict> conflicts) {
        return conflicts.stream().map(this::classify).toList();
    }

    /**
     * 카테고리별 conflict 수 집계 — UI 요약 + KPI 데이터 소스.
     */
    public Map<ConflictCategory, Long> countByCategory(List<AllocationConflict> conflicts) {
        return classifyAll(conflicts).stream()
            .collect(Collectors.groupingBy(ClassifiedConflict::category, Collectors.counting()));
    }
}

public record ClassifiedConflict(AllocationConflict conflict, ConflictCategory category) {
    public String hoseId() { return conflict.hoseId(); }
    public String reason() { return conflict.reason(); }
}
```
