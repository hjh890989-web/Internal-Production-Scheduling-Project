### `ConflictReportService.java`

```java
@Service @RequiredArgsConstructor
public class ConflictReportService {

    private final AllocationResultRepository allocRepo;
    private final ConflictCategorizer categorizer;
    private final AlternativeGenerator generator;

    public ConflictReport buildReport(UUID scheduleId) {
        var result = allocRepo.findResultById(scheduleId)
            .orElseThrow(() -> new NotFoundException("schedule=" + scheduleId));

        var classified = categorizer.classifyAll(result.conflicts());
        var items = classified.stream()
            .map(cc -> new ConflictReportItem(
                cc.hoseId(), cc.category(), cc.reason(),
                generator.generate(cc)))
            .toList();

        var summary = categorizer.countByCategory(result.conflicts());

        return new ConflictReport(scheduleId, items, summary, Instant.now());
    }
}

public record ConflictReport(UUID scheduleId, List<ConflictReportItem> items,
                              Map<ConflictCategory, Long> summary, Instant generatedAt) {}

public record ConflictReportItem(String hoseId, ConflictCategory category,
                                  String reason, List<Alternative> alternatives) {}
```

### `ConflictController.java`

```java
@RestController @RequestMapping("/api/v1/schedule/conflicts") @RequiredArgsConstructor
public class ConflictController {

    private final ConflictReportService reportService;

    @GetMapping("/{scheduleId}")
    @PreAuthorize("isAuthenticated()")
    public ConflictReport get(@PathVariable UUID scheduleId) {
        return reportService.buildReport(scheduleId);
    }
}
```

### `ConflictReportIT.java`

```java
@SpringBootTest @Testcontainers
class ConflictReportIT {

    @Autowired ConflictReportService service;
    @Autowired ScenarioLoader loader;

    @Test
    void regression_100_scenarios_all_report_at_least_three_alternatives() {
        var scenarios = loader.loadConflictHeavy("DS-CONFLICT-100/scenarios.csv");
        for (var scenarioId : scenarios) {
            var report = service.buildReport(scenarioId);

            for (var item : report.items()) {
                var distinctTypes = item.alternatives().stream()
                    .map(Alternative::type).distinct().count();
                assertThat(distinctTypes)
                    .as("scenario %s, hose %s: distinct alternatives", scenarioId, item.hoseId())
                    .isGreaterThanOrEqualTo(3);
            }
        }
    }

    @Test
    void p95_response_under_one_second() {
        var ids = loader.preGenerateConflictResults(50);

        var latencies = new ArrayList<Long>();
        for (var id : ids) {
            long start = System.nanoTime();
            service.buildReport(id);
            latencies.add((System.nanoTime() - start) / 1_000_000);
        }

        Collections.sort(latencies);
        long p95 = latencies.get((int) (latencies.size() * 0.95));
        assertThat(p95).isLessThanOrEqualTo(1000);
    }

    @Test
    void api_endpoint_returns_correct_json_structure() throws Exception {
        // MockMvc로 GET /api/v1/schedule/conflicts/{id}
        // 응답 JSON 구조 검증 — items[].alternatives[].type 존재
    }
}
```
