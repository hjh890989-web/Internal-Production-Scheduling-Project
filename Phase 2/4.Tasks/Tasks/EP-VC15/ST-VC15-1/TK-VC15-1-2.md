### `AlternativeGenerator.java`

```java
@Service @RequiredArgsConstructor
public class AlternativeGenerator {

    private static final int MIN_ALTERNATIVES = 3;

    /**
     * 카테고리별 적용 가능한 대안 매트릭스.
     * - SLOT_OX: 어떤 슬롯에도 부적합 → 머신 변경 외 capa 우회 (NIGHT, EXPAND, OUTSOURCE, SWAP)
     * - DEADLINE_D2: 납기 압박 → NIGHT, NEGOTIATE, OUTSOURCE
     * - SPEC_LT7: 가류기당 angle 상한 → NIGHT, EXPAND, SWAP (IC_ROUTING은 적용 불가)
     */
    private static final Map<ConflictCategory, List<AlternativeType>> POLICY = Map.of(
        SLOT_OX,      List.of(NIGHT_ROTATION, EXPAND_CAPA, OUTSOURCE, SWAP_ORDER),
        ANGLE_CAPA,   List.of(NIGHT_ROTATION, IC_ROUTING, EXPAND_CAPA, OUTSOURCE),
        DAILY_CAPA,   List.of(NIGHT_ROTATION, EXPAND_CAPA, OUTSOURCE, SWAP_ORDER),
        DEADLINE_D2,  List.of(NIGHT_ROTATION, DEADLINE_NEGOTIATE, OUTSOURCE, EXPAND_CAPA),
        DAY_LOCK,     List.of(NIGHT_ROTATION, SWAP_ORDER, EXPAND_CAPA, OUTSOURCE),
        LEFT_RIGHT,   List.of(SWAP_ORDER, OUTSOURCE, EXPAND_CAPA, DEADLINE_NEGOTIATE),
        MACHINE_PIN,  List.of(DEADLINE_NEGOTIATE, OUTSOURCE, EXPAND_CAPA, SWAP_ORDER),
        SPEC_LT7,     List.of(NIGHT_ROTATION, EXPAND_CAPA, SWAP_ORDER, OUTSOURCE)
    );

    public List<Alternative> generate(ClassifiedConflict cc) {
        var types = POLICY.getOrDefault(cc.category(), defaultPolicy());
        if (types.size() < MIN_ALTERNATIVES) {
            // 안전망: defaultPolicy 추가
            var merged = new ArrayList<>(types);
            defaultPolicy().forEach(t -> { if (!merged.contains(t)) merged.add(t); });
            types = merged;
        }
        return types.stream()
            .limit(Math.max(MIN_ALTERNATIVES, types.size()))
            .map(t -> Alternative.of(t, cc))
            .toList();
    }

    private List<AlternativeType> defaultPolicy() {
        return List.of(NIGHT_ROTATION, OUTSOURCE, EXPAND_CAPA, DEADLINE_NEGOTIATE, SWAP_ORDER);
    }
}

public record Alternative(AlternativeType type, String label, String description,
                          String suggestedAction) {
    public static Alternative of(AlternativeType t, ClassifiedConflict cc) {
        return new Alternative(t, t.getLabel(), t.getDescription(),
            buildAction(t, cc));
    }
    private static String buildAction(AlternativeType t, ClassifiedConflict cc) {
        return switch (t) {
            case NIGHT_ROTATION   -> "%s 야간 회전 +5 슬롯 시도".formatted(cc.hoseId());
            case DEADLINE_NEGOTIATE -> "%s 납기 D+3 협상".formatted(cc.hoseId());
            case IC_ROUTING       -> "%s IC 가류기 라우팅".formatted(cc.hoseId());
            case OUTSOURCE        -> "%s 외주 발주".formatted(cc.hoseId());
            case EXPAND_CAPA      -> "주말 가동 + %s 우선 배치".formatted(cc.hoseId());
            case SWAP_ORDER       -> "%s 우선순위 ↑ + 후순위 수주 이동".formatted(cc.hoseId());
        };
    }
}
```
