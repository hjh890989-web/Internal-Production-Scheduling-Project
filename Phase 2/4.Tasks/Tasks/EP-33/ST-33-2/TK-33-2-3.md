### `restore-pitr.sh`

```bash
#!/bin/bash
set -e
TARGET_TIMESTAMP=$1                # YYYY-MM-DD HH:MM:SS KST

# 1) STG 환경 stop
docker compose -f docker-compose.stg.yml stop postgres

# 2) 최신 백업 다운로드
LATEST=$(aws s3 ls s3://${BACKUP_BUCKET}/postgres/ | sort | tail -1 | awk '{print $2}')
aws s3 sync s3://${BACKUP_BUCKET}/postgres/${LATEST} /tmp/restore/

# 3) Postgres data 디렉토리 교체
rm -rf /var/lib/postgresql/data/*
tar -xzf /tmp/restore/base.tar.gz -C /var/lib/postgresql/data/

# 4) recovery.conf 작성
cat > /var/lib/postgresql/data/recovery.conf <<EOF
restore_command = 'aws s3 cp s3://${BACKUP_BUCKET}/wal/%f %p'
recovery_target_time = '${TARGET_TIMESTAMP}'
recovery_target_action = 'promote'
EOF

# 5) Postgres start
docker compose -f docker-compose.stg.yml start postgres

# 6) Healthcheck + 데이터 검증
for i in {1..60}; do
    if docker compose exec postgres pg_isready; then
        echo "Postgres ready"
        break
    fi
    sleep 5
done

# 7) RTO 측정 + 보고
END_TIME=$(date +%s)
RTO=$((END_TIME - START_TIME))
echo "RTO: ${RTO}s (target: ≤ 14400s = 4h)"
```
