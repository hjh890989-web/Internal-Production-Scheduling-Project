### `docker-compose.stg.yml`

```yaml
services:
  scheduling-engine:
    image: harbor.internal/scheduling/engine:${TAG}
    environment:
      SPRING_PROFILES_ACTIVE: stg
      DATABASE_URL: jdbc:postgresql://postgres:5432/scheduling_stg
    labels:
      env: stg

  postgres:
    environment:
      POSTGRES_DB: scheduling_stg
    volumes:
      - postgres-stg-data:/var/lib/postgresql/data

  nginx:
    environment:
      - SERVER_NAME=stg.scheduling.internal
    volumes:
      - ./nginx/stg.conf:/etc/nginx/conf.d/default.conf:ro

volumes:
  postgres-stg-data:
```
