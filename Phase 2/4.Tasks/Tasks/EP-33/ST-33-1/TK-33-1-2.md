### `docker-compose.prod.yml`

```yaml
services:
  scheduling-engine-blue:
    image: harbor.internal/scheduling/engine:${TAG}
    environment:
      SPRING_PROFILES_ACTIVE: prod
      DEPLOYMENT_COLOR: blue
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/actuator/health/ready"]
      interval: 10s
      timeout: 3s
      retries: 5

  scheduling-engine-green:
    image: harbor.internal/scheduling/engine:${TAG_PREVIOUS}
    environment:
      SPRING_PROFILES_ACTIVE: prod
      DEPLOYMENT_COLOR: green
    # 동일 healthcheck

  nginx:
    volumes:
      - ./nginx/prod-active.conf:/etc/nginx/conf.d/default.conf:ro
    depends_on:
      scheduling-engine-blue:
        condition: service_healthy
```

### `blue-green-switch.sh`

```bash
#!/bin/bash
set -e
TARGET_COLOR=$1                    # blue or green

# 1) New version 컨테이너 booting
docker compose -f docker-compose.prod.yml up -d scheduling-engine-${TARGET_COLOR}

# 2) Health check 대기 (max 60s)
for i in {1..12}; do
    if docker inspect --format='{{.State.Health.Status}}' scheduling-engine-${TARGET_COLOR} | grep -q healthy; then
        echo "Healthy"
        break
    fi
    sleep 5
done

# 3) NGINX upstream switch
ln -sf nginx/prod-${TARGET_COLOR}.conf nginx/prod-active.conf
docker compose exec nginx nginx -s reload

# 4) 이전 색깔 컨테이너 stop (graceful)
sleep 30                           # 진행 중 요청 마무리
PREVIOUS_COLOR=$([ "$TARGET_COLOR" = "blue" ] && echo "green" || echo "blue")
docker compose stop scheduling-engine-${PREVIOUS_COLOR}

echo "Switch complete: ${PREVIOUS_COLOR} → ${TARGET_COLOR}"
```
