### `ExtrusionPushSoakIT.java`

```java
@SpringBootTest(webEnvironment = WebEnvironment.DEFINED_PORT)
@Testcontainers
class ExtrusionPushSoakIT {

    @LocalServerPort int port;
    @Autowired ExtrusionPushService pushService;

    @Test
    void soak_30_min_p95_under_2s() throws Exception {
        int clientCount = 10;
        int pushPerMin = 5;
        Duration duration = Duration.ofMinutes(30);

        var clients = connectClients(clientCount, port);
        var latencies = Collections.synchronizedList(new ArrayList<Long>());

        // 각 client가 latency 측정
        for (var client : clients) {
            client.subscribe("/topic/extrusion-updates", message -> {
                var receivedAt = Instant.now();
                var payload = (ExtrusionUpdatePayload) message.getPayload();
                long latency = Duration.between(payload.timestamp(), receivedAt).toMillis();
                latencies.add(latency);
            });
        }

        // 30분간 분당 5 PUSH = 1,500 메시지
        var scheduler = Executors.newScheduledThreadPool(1);
        scheduler.scheduleAtFixedRate(() -> {
            pushService.publishReplanned(buildDummyEvent(), buildDummyCandidates());
        }, 0, 12, TimeUnit.SECONDS);    // 12초마다 = 분당 5

        Thread.sleep(duration.toMillis());
        scheduler.shutdown();

        // p95 측정
        Collections.sort(latencies);
        long median = latencies.get(latencies.size() / 2);
        long p95 = latencies.get((int) (latencies.size() * 0.95));
        long p99 = latencies.get((int) (latencies.size() * 0.99));

        System.out.printf("PUSH latency: median=%dms, p95=%dms, p99=%dms (n=%d)%n",
            median, p95, p99, latencies.size());

        // SRS 요구
        assertThat(median).as("중앙값 ≤ 1.5s").isLessThanOrEqualTo(1500);
        assertThat(p95).as("p95 ≤ 2s (REQ-FUNC-EX-014)").isLessThanOrEqualTo(2000);
        assertThat(latencies.size())
            .as("30분간 ≥ 1,400 메시지 (5% drop 허용)")
            .isGreaterThanOrEqualTo(1400);
    }

    @Test
    void burst_load_p95_under_3s() throws Exception {
        // burst 100 PUSH/sec — 일시적 부하 대응
        int burstCount = 100;
        var clients = connectClients(5, port);
        var latencies = Collections.synchronizedList(new ArrayList<Long>());

        // ... 측정 logic

        long p95 = latencies.stream().sorted().toList().get((int) (latencies.size() * 0.95));
        assertThat(p95).as("burst p95 ≤ 3s 마진").isLessThanOrEqualTo(3000);
    }

    @Test
    void disconnect_reconnect_recovery() throws Exception {
        var client = connectClient(port);
        var firstMessages = new ArrayList<ExtrusionUpdatePayload>();
        client.subscribe("/topic/extrusion-updates", m -> firstMessages.add((ExtrusionUpdatePayload) m.getPayload()));

        pushService.publishReplanned(buildDummyEvent(), List.of());
        Thread.sleep(500);
        assertThat(firstMessages).hasSize(1);

        // 연결 끊김
        client.disconnect();
        pushService.publishReplanned(buildDummyEvent(), List.of());  // 이 메시지는 lost (단일 인스턴스 SimpleBroker)

        // 재연결
        client = connectClient(port);
        var afterReconnect = new ArrayList<ExtrusionUpdatePayload>();
        client.subscribe("/topic/extrusion-updates", m -> afterReconnect.add((ExtrusionUpdatePayload) m.getPayload()));

        pushService.publishReplanned(buildDummyEvent(), List.of());
        Thread.sleep(500);
        assertThat(afterReconnect).hasSize(1);
    }
}
```
