### `ExtrusionWebSocketConfig.java`

```java
@Configuration
@EnableWebSocketMessageBroker
public class ExtrusionWebSocketConfig implements WebSocketMessageBrokerConfigurer {

    @Override
    public void registerStompEndpoints(StompEndpointRegistry registry) {
        registry.addEndpoint("/ws/extrusion")
            .setAllowedOriginPatterns("https://*.scheduling.internal")
            .withSockJS();
    }

    @Override
    public void configureMessageBroker(MessageBrokerRegistry registry) {
        registry.enableSimpleBroker("/topic", "/queue");
        registry.setApplicationDestinationPrefixes("/app");
    }
}
```

### `ExtrusionPushService.java`

```java
@Service @RequiredArgsConstructor @Slf4j
public class ExtrusionPushService {

    private final SimpMessagingTemplate messagingTemplate;

    /**
     * EP-EX13 replan 완료 후 호출 — 모든 압출 패드에 PUSH.
     */
    public void publishReplanned(VcChangedEvent event, List<ExScheduleCandidate> updatedCandidates) {
        var payload = ExtrusionUpdatePayload.builder()
            .eventType("REPLAN")
            .scheduleId(event.scheduleId())
            .timestamp(Instant.now())
            .affectedHoses(updatedCandidates.stream()
                .map(ExScheduleCandidate::getHoseId).collect(Collectors.toSet()))
            .updatedCandidates(updatedCandidates)
            .build();

        messagingTemplate.convertAndSend("/topic/extrusion-updates", payload);
        log.info("Extrusion PUSH: scheduleId={}, candidates={}",
            event.scheduleId(), updatedCandidates.size());
    }

    /**
     * EP-EX12 conflict 발생 시도 PUSH.
     */
    public void publishConflict(UUID candidateId, ValidationViolation violation) {
        var payload = ExtrusionUpdatePayload.builder()
            .eventType("CONFLICT")
            .timestamp(Instant.now())
            .conflictCandidateId(candidateId)
            .violationCategory(violation.category().name())
            .violationReason(violation.reason())
            .build();

        messagingTemplate.convertAndSend("/topic/extrusion-updates", payload);
    }
}

public record ExtrusionUpdatePayload(
    String eventType,            // REPLAN · CONFIRMED · CONFLICT
    UUID scheduleId,
    Instant timestamp,
    Set<String> affectedHoses,
    List<ExScheduleCandidate> updatedCandidates,
    UUID conflictCandidateId,
    String violationCategory,
    String violationReason
) {}
```
