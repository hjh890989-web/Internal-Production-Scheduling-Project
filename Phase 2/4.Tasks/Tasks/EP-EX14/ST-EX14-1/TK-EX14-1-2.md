### `ExtrusionWebSocketConfig.java` 수정

```java
@Configuration
@EnableWebSocketMessageBroker
public class ExtrusionWebSocketConfig implements WebSocketMessageBrokerConfigurer {

    @Value("${broker.type:simple}")
    private String brokerType;     // simple | redis

    @Value("${broker.redis.host:redis}")
    private String redisHost;

    @Value("${broker.redis.port:6379}")
    private int redisPort;

    @Override
    public void configureMessageBroker(MessageBrokerRegistry registry) {
        if ("redis".equals(brokerType)) {
            // Phase 2+ 다중 인스턴스 대비 — Redis pub/sub 활용
            registry.enableStompBrokerRelay("/topic", "/queue")
                .setRelayHost(redisHost)
                .setRelayPort(redisPort)
                .setClientLogin(System.getenv("REDIS_USER"))
                .setClientPasscode(System.getenv("REDIS_PASSWORD"));
        } else {
            // 단일 인스턴스 — SimpleBroker
            registry.enableSimpleBroker("/topic", "/queue");
        }
        registry.setApplicationDestinationPrefixes("/app");
    }

    // ... registerStompEndpoints 동일
}
```
