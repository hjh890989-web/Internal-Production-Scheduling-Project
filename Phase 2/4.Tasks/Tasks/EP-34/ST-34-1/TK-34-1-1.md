### Flyway `V5_7__create_master_change_request.sql`

```sql
CREATE TABLE master_change_request (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    table_name      VARCHAR(40) NOT NULL,
    row_id          VARCHAR(40) NOT NULL,
    operation       VARCHAR(10) NOT NULL CHECK (operation IN ('INSERT','UPDATE','DELETE')),
    old_row         JSONB,
    new_row         JSONB,
    requested_by    VARCHAR(40) NOT NULL,
    requested_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
    status          VARCHAR(20) NOT NULL DEFAULT 'PENDING'
        CHECK (status IN ('PENDING', 'APPROVED', 'REJECTED')),
    approved_by     VARCHAR(40),
    approved_at     TIMESTAMPTZ,
    rejection_reason TEXT
);

CREATE INDEX idx_mcr_pending ON master_change_request(status, requested_at) WHERE status='PENDING';
```

### `MasterChangeReviewService.java`

```java
@Service @RequiredArgsConstructor
public class MasterChangeReviewService {

    private final MasterChangeRequestRepository repo;
    private final SecurityContextService securityCtx;

    @Auditable(reason = "'master change request'")
    @Transactional
    public MasterChangeRequest request(MasterChangeRequest req) {
        req.setRequestedBy(securityCtx.currentUserId());
        req.setStatus(RequestStatus.PENDING);
        return repo.save(req);
    }

    @Auditable(reason = "'approve master change'")
    @Transactional
    public void approve(UUID requestId) {
        String approverId = securityCtx.currentUserId();
        var req = repo.findById(requestId).orElseThrow();

        // BR-X05: 동일 actor 거부
        if (req.getRequestedBy().equals(approverId)) {
            throw new DualReviewException("BR-X05: requester와 approver 동일 — 거부");
        }
        if (req.getStatus() != RequestStatus.PENDING) {
            throw new IllegalStateException("이미 처리된 요청");
        }

        req.setStatus(RequestStatus.APPROVED);
        req.setApprovedBy(approverId);
        req.setApprovedAt(Instant.now());
        repo.save(req);

        // 실제 master change 실행
        applyChange(req);
    }
}
```
