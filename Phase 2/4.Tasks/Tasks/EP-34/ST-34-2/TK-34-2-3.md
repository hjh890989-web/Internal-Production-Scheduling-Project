```java
@Service @RequiredArgsConstructor
public class MesReconciliationService {

    @EventListener
    @Auditable(reason = "'MES actual reconciliation'")
    public void onActualReceived(MesActualReceivedEvent event) {
        // 동일 (machine, date, shift, rotation, hose) FALLBACK_ESTIMATED row 갱신
        var existing = repo.findByKey(event.toKey());
        if (existing != null && existing.getStatus() == ESTIMATED) {
            existing.setActualYield(event.yield());
            existing.setStatus(RECEIVED);
            repo.save(existing);

            log.info("MES reconciled: {} estimated={} → actual={}",
                event.toKey(), existing.getYield(), event.yield());
        }
    }
}
```
