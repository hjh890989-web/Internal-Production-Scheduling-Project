### `V5_8__create_mes_actual.sql`

```sql
CREATE TABLE mes_actual (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    machine_id      VARCHAR(10) NOT NULL,
    production_date DATE NOT NULL,
    shift_code      VARCHAR(20) NOT NULL,
    rotation_id     VARCHAR(20) NOT NULL,
    hose_id         VARCHAR(40) NOT NULL,
    actual_yield    INTEGER NOT NULL CHECK (actual_yield >= 0),
    received_at     TIMESTAMPTZ NOT NULL DEFAULT now(),
    status          VARCHAR(20) NOT NULL DEFAULT 'RECEIVED'
        CHECK (status IN ('RECEIVED', 'FALLBACK_ESTIMATED', 'MISSING'))
);

CREATE UNIQUE INDEX uk_mes_actual ON mes_actual(machine_id, production_date, shift_code, rotation_id, hose_id);
```
