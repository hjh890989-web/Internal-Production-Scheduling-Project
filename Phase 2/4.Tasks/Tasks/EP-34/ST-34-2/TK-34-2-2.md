```java
@Component @RequiredArgsConstructor @Slf4j
public class MesShiftBoundaryDetector {

    @Scheduled(cron = "0 5 * * * *")        // 매시 5분
    public void checkAndFallback() {
        // 직전 shift 종료된 모든 (machine, date, shift)
        var endedShifts = findRecentEndedShifts();
        for (var shift : endedShifts) {
            if (!actualReceived(shift)) {
                applyFallback(shift);
            }
        }
    }

    private void applyFallback(ShiftSlot slot) {
        // 직전 shift 평균 또는 schedule 계획값
        double avgYield = lookupRecentAvg(slot.machineId(), slot.hoseId());
        // INSERT mes_actual with status=FALLBACK_ESTIMATED
        log.warn("MES Fallback: {} avg={}", slot, avgYield);
    }
}
```
