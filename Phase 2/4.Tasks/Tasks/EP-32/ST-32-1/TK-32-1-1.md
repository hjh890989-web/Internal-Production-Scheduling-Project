### 핵심 로직 / 알고리즘

#### A. Jenkins 별 컴포즈 (`docker-compose.jenkins.yml`)

```yaml
# infrastructure/jenkins/docker-compose.jenkins.yml
# Jenkins는 별 서버 권장 (SAD §3 line 175). DEV는 같은 호스트 별 컴포즈로 운영
name: scheduling-ci

services:
  jenkins:
    build:
      context: .
      dockerfile: Dockerfile.jenkins
    container_name: scheduling-jenkins
    ports:
      - "127.0.0.1:8090:8080"     # DEV: localhost 8090
      - "127.0.0.1:50000:50000"   # JNLP agent
    environment:
      JAVA_OPTS: "-Djenkins.install.runSetupWizard=false -Dcasc.jenkins.config=/var/jenkins_home/casc/jenkins.yaml"
      CASC_JENKINS_CONFIG: "/var/jenkins_home/casc/jenkins.yaml"
      JENKINS_ADMIN_PASSWORD_FILE: "/run/secrets/jenkins_admin_password"
    volumes:
      - jenkins-data:/var/jenkins_home
      - ./casc:/var/jenkins_home/casc:ro
      - /var/run/docker.sock:/var/run/docker.sock   # Docker-in-Docker
    secrets:
      - jenkins_admin_password
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/login"]
      interval: 30s
      timeout: 5s
      retries: 5
      start_period: 90s   # 첫 부팅 시 플러그인 초기화 시간
    restart: unless-stopped
    networks:
      - scheduling-ci-net

volumes:
  jenkins-data:
    driver: local

networks:
  scheduling-ci-net:
    driver: bridge

secrets:
  jenkins_admin_password:
    file: ./secrets/jenkins_admin_password.txt   # .gitignore 필수
```

#### B. 커스텀 Dockerfile + 플러그인 사전 설치

```dockerfile
# infrastructure/jenkins/Dockerfile.jenkins
FROM jenkins/jenkins:lts-jdk21

USER root

# Docker CLI (Docker-in-Docker 패턴)
RUN apt-get update && apt-get install -y --no-install-recommends \
        docker.io \
        ca-certificates \
        curl \
    && rm -rf /var/lib/apt/lists/*

# 플러그인 사전 설치 (Configuration as Code 패턴)
USER jenkins
COPY plugins.txt /usr/share/jenkins/ref/plugins.txt
RUN jenkins-plugin-cli --plugin-file /usr/share/jenkins/ref/plugins.txt

USER root
```

#### E. 표준 Jenkinsfile 템플릿 (`ci/Jenkinsfile.template`)

```groovy
// infrastructure/ci/Jenkinsfile.template
// Sprint 1+ 모든 Epic이 본 템플릿 복사/재사용
pipeline {
    agent any

    options {
        timeout(time: 30, unit: 'MINUTES')
        timestamps()
        ansiColor('xterm')
        buildDiscarder(logRotator(numToKeepStr: '50'))
        disableConcurrentBuilds()
    }

    environment {
        HARBOR_REGISTRY = 'harbor.internal/scheduling'
        APP_VERSION     = "${env.BUILD_NUMBER}-${env.GIT_COMMIT.take(7)}"
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
                script {
                    env.GIT_AUTHOR = sh(returnStdout: true, script: 'git log -1 --pretty=%an').trim()
                }
            }
        }

        // ↓↓↓ TK-32-1-2에서 상세 정의 ↓↓↓
        stage('Build') {
            steps { /* see TK-32-1-2 */ }
        }
        stage('Test') {
            steps { /* see TK-32-1-2 */ }
        }
        stage('SonarQube') {
            steps { /* see ST-32-2 TK-32-2-2 */ }
        }
        stage('Security Scan') {
            steps { /* see ST-32-2 TK-32-2-1 (Trivy) */ }
        }
        stage('Build Image') {
            steps { /* see TK-32-1-2 */ }
        }
        stage('Push to Harbor') {
            steps { /* see TK-32-1-2 */ }
        }
        stage('Deploy (Blue-Green)') {
            when { branch 'main' }
            steps { /* see TK-32-1-3 */ }
        }
    }

    post {
        success {
            slackSend(channel: '#deploys', color: 'good',
                      message: "✅ ${env.JOB_NAME} #${env.BUILD_NUMBER} 성공 — ${env.GIT_AUTHOR}")
        }
        failure {
            slackSend(channel: '#deploys', color: 'danger',
                      message: "❌ ${env.JOB_NAME} #${env.BUILD_NUMBER} 실패 — ${env.GIT_AUTHOR}")
        }
        always {
            cleanWs()
        }
    }
}
```
