### 핵심 로직 / 알고리즘

#### A. Backend Jenkinsfile (`Jenkinsfile.backend`)

```groovy
// infrastructure/ci/Jenkinsfile.backend
@Library('scheduling-shared-lib') _

pipeline {
    agent any

    tools {
        jdk    'JDK 21'
        gradle 'Gradle 8.10'
    }

    options {
        timeout(time: 30, unit: 'MINUTES')
        timestamps()
        ansiColor('xterm')
        disableConcurrentBuilds()
    }

    environment {
        HARBOR_REGISTRY = 'harbor.internal/scheduling'
        SERVICE_NAME    = 'backend'
        APP_VERSION     = "${env.BUILD_NUMBER}-${env.GIT_COMMIT.take(7)}"
        IMAGE_TAG       = "${HARBOR_REGISTRY}/${SERVICE_NAME}:${APP_VERSION}"
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
                script {
                    env.GIT_AUTHOR = sh(returnStdout: true, script: 'git log -1 --pretty=%an').trim()
                    env.GIT_MESSAGE = sh(returnStdout: true, script: 'git log -1 --pretty=%s').trim()
                }
            }
        }

        stage('Build') {
            steps {
                dir('backend') {
                    sh './gradlew clean build -x test --no-daemon'
                }
            }
        }

        stage('Test (Unit + ArchUnit)') {
            steps {
                dir('backend') {
                    // ArchUnit 테스트 = ST-00-2 TK-00-2-3 모듈 경계 강제
                    // 80% 커버리지 게이트 = ST-32-2 TK-32-2-2 SonarQube와 결합
                    sh './gradlew test jacocoTestReport --no-daemon'
                }
            }
            post {
                always {
                    junit allowEmptyResults: false, testResults: 'backend/**/build/test-results/test/*.xml'
                    publishHTML([
                        reportDir: 'backend/app/build/reports/jacoco/test/html',
                        reportFiles: 'index.html',
                        reportName: 'Jacoco Coverage',
                        keepAll: true,
                    ])
                }
            }
        }

        stage('Integration Test (Testcontainers)') {
            when { not { branch 'feature/*' } }    // feature 브랜치는 skip (속도)
            steps {
                dir('backend') {
                    sh './gradlew integrationTest --no-daemon'
                }
            }
        }

        // SonarQube·Trivy는 ST-32-2에서 채움

        stage('Build Docker Image') {
            steps {
                dir('backend') {
                    sh "docker build -t ${IMAGE_TAG} -t ${HARBOR_REGISTRY}/${SERVICE_NAME}:latest ."
                }
            }
        }

        stage('Push to Harbor') {
            when { anyOf { branch 'main'; branch 'develop' } }
            steps {
                withCredentials([usernamePassword(
                    credentialsId: 'harbor-credentials',
                    usernameVariable: 'HARBOR_USER',
                    passwordVariable: 'HARBOR_PASS'
                )]) {
                    sh '''
                        echo "$HARBOR_PASS" | docker login harbor.internal -u "$HARBOR_USER" --password-stdin
                        docker push ${IMAGE_TAG}
                        docker push ${HARBOR_REGISTRY}/${SERVICE_NAME}:latest
                        docker logout harbor.internal
                    '''
                }
            }
        }

        // Deploy 단계는 TK-32-1-3에서 채움
    }

    post {
        always {
            archiveArtifacts artifacts: 'backend/app/build/libs/*.jar',
                             allowEmptyArchive: true,
                             fingerprint: true
            cleanWs()
        }
    }
}
```

#### B. Frontend Jenkinsfile (`Jenkinsfile.frontend`)

```groovy
// infrastructure/ci/Jenkinsfile.frontend
pipeline {
    agent any

    tools {
        nodejs 'NodeJS 20'
    }

    options {
        timeout(time: 15, unit: 'MINUTES')
        timestamps()
    }

    environment {
        HARBOR_REGISTRY = 'harbor.internal/scheduling'
        SERVICE_NAME    = 'frontend'
        APP_VERSION     = "${env.BUILD_NUMBER}-${env.GIT_COMMIT.take(7)}"
        IMAGE_TAG       = "${HARBOR_REGISTRY}/${SERVICE_NAME}:${APP_VERSION}"
    }

    stages {
        stage('Checkout') { steps { checkout scm } }

        stage('Install') {
            steps {
                dir('frontend') {
                    sh 'npm ci --prefer-offline --no-audit'
                }
            }
        }

        stage('Lint + TypeCheck') {
            steps {
                dir('frontend') {
                    sh 'npm run lint'
                    sh 'npx tsc -b --noEmit'
                }
            }
        }

        stage('Test (Vitest)') {
            steps {
                dir('frontend') {
                    sh 'npm run test:run -- --coverage'
                }
            }
            post {
                always {
                    publishHTML([
                        reportDir: 'frontend/coverage',
                        reportFiles: 'index.html',
                        reportName: 'Vitest Coverage',
                    ])
                }
            }
        }

        stage('Build') {
            steps {
                dir('frontend') {
                    sh 'npm run build'
                }
            }
        }

        // SonarQube·Trivy는 ST-32-2에서 채움

        stage('Build Docker Image') {
            steps {
                dir('frontend') {
                    sh "docker build -t ${IMAGE_TAG} -t ${HARBOR_REGISTRY}/${SERVICE_NAME}:latest ."
                }
            }
        }

        stage('Push to Harbor') {
            when { anyOf { branch 'main'; branch 'develop' } }
            steps {
                withCredentials([usernamePassword(
                    credentialsId: 'harbor-credentials',
                    usernameVariable: 'HARBOR_USER',
                    passwordVariable: 'HARBOR_PASS'
                )]) {
                    sh '''
                        echo "$HARBOR_PASS" | docker login harbor.internal -u "$HARBOR_USER" --password-stdin
                        docker push ${IMAGE_TAG}
                        docker push ${HARBOR_REGISTRY}/${SERVICE_NAME}:latest
                    '''
                }
            }
        }
    }

    post {
        always { cleanWs() }
    }
}
```

#### C. Shared Library 공통 함수 (`vars/buildAndPush.groovy`)

```groovy
// infrastructure/jenkins/shared-library/vars/buildAndPush.groovy
def call(Map config = [:]) {
    def registry = config.registry ?: 'harbor.internal/scheduling'
    def service  = config.service
    def tag      = "${env.BUILD_NUMBER}-${env.GIT_COMMIT.take(7)}"

    sh "docker build -t ${registry}/${service}:${tag} -t ${registry}/${service}:latest ."

    withCredentials([usernamePassword(
        credentialsId: 'harbor-credentials',
        usernameVariable: 'HARBOR_USER',
        passwordVariable: 'HARBOR_PASS'
    )]) {
        sh """
            echo "\$HARBOR_PASS" | docker login harbor.internal -u "\$HARBOR_USER" --password-stdin
            docker push ${registry}/${service}:${tag}
            docker push ${registry}/${service}:latest
        """
    }
    return "${registry}/${service}:${tag}"
}
```

## :memo: Implementation Notes
- **기술 힌트**:
  - `--no-daemon` Gradle 옵션 — Jenkins 환경에서 daemon 메모리 누수 방지
  - `npm ci` (not `npm install`) — lock 파일 엄격 + 재현성 ↑
  - `docker build -t name:tag -t name:latest .` 단일 명령으로 두 태그 — 캐시 효율
  - HTML report publish는 Jenkins 빌드 페이지에서 한 클릭 조회
- **유사 패턴**: Sprint 1+ 새 Epic 추가 시 `Jenkinsfile.backend` 또는 `.frontend` 복사 후 환경변수만 변경
