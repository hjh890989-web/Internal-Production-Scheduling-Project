#### B. 무중단 배포 스크립트 (`blue_green_deploy.sh`)

```bash
#!/usr/bin/env bash
# infrastructure/scripts/blue_green_deploy.sh
# 사용: ./blue_green_deploy.sh <image_tag>
set -euo pipefail

IMAGE_TAG="${1:?image tag 필수}"   # 예: harbor.internal/scheduling/backend:42-abc1234
COMPOSE_FILE="${COMPOSE_FILE:-docker-compose.yml}"
NGINX_CONF="infrastructure/nginx/conf.d/scheduling.conf"

# 1. 현재 활성 색상 검출
CURRENT_COLOR=$(grep -oP 'server backend-\K(blue|green)' "$NGINX_CONF" | head -1)
NEXT_COLOR=$([ "$CURRENT_COLOR" = "blue" ] && echo "green" || echo "blue")
echo "🔄 배포: ${CURRENT_COLOR} → ${NEXT_COLOR}"

# 2. NEXT 색상 컨테이너에 새 이미지 배포
echo "🚀 ${NEXT_COLOR} 컨테이너에 새 이미지 배포..."
BACKEND_IMAGE="$IMAGE_TAG" \
    docker compose -f "$COMPOSE_FILE" \
    up -d --no-deps --build "backend-${NEXT_COLOR}"

# 3. NEXT healthcheck 통과 대기 (최대 60초)
echo "⏳ ${NEXT_COLOR} healthcheck 대기..."
for i in {1..30}; do
    STATUS=$(docker inspect --format='{{.State.Health.Status}}' "scheduling-backend-${NEXT_COLOR}" 2>/dev/null || echo "unknown")
    if [[ "$STATUS" == "healthy" ]]; then
        echo "  ✅ ${NEXT_COLOR} healthy ($i회 시도)"
        break
    fi
    sleep 2
done

if [[ "$STATUS" != "healthy" ]]; then
    echo "❌ ${NEXT_COLOR} healthcheck 실패 — 롤백"
    docker compose -f "$COMPOSE_FILE" stop "backend-${NEXT_COLOR}"
    exit 1
fi

# 4. NEXT smoke test (선택)
echo "🧪 ${NEXT_COLOR} smoke test..."
docker compose exec -T "backend-${NEXT_COLOR}" \
    wget --spider -q http://localhost:8080/actuator/health/readiness \
    || { echo "❌ smoke test 실패"; docker compose -f "$COMPOSE_FILE" stop "backend-${NEXT_COLOR}"; exit 1; }

# 5. NGINX upstream 토글 (≤30초 핵심 단계)
echo "🔀 NGINX upstream 토글 ${CURRENT_COLOR} → ${NEXT_COLOR}..."
sed -i "s/server backend-${CURRENT_COLOR}:8080/server backend-${NEXT_COLOR}:8080/" "$NGINX_CONF"

# 6. NGINX 무중단 reload (config 리로드만, 컨테이너 재기동 안 함)
docker compose exec -T nginx nginx -t   # 설정 검증
docker compose exec -T nginx nginx -s reload

echo "  ✅ NGINX reload 완료 — 새 요청은 ${NEXT_COLOR}로"

# 7. 기존 색상 컨테이너 graceful drain (10초 대기 후 stop)
echo "⏸️ ${CURRENT_COLOR} 컨테이너 drain (기존 요청 완료 대기 10초)..."
sleep 10
docker compose -f "$COMPOSE_FILE" stop "backend-${CURRENT_COLOR}"

echo ""
echo "🎉 배포 완료: ${IMAGE_TAG} → ${NEXT_COLOR} 활성"
echo "   롤백: ./blue_green_rollback.sh"
```

#### C. 롤백 스크립트 (`blue_green_rollback.sh`)

```bash
#!/usr/bin/env bash
# infrastructure/scripts/blue_green_rollback.sh
# 직전 색상으로 즉시 복원 (NGINX 토글만)
set -euo pipefail

NGINX_CONF="infrastructure/nginx/conf.d/scheduling.conf"
COMPOSE_FILE="${COMPOSE_FILE:-docker-compose.yml}"

CURRENT=$(grep -oP 'server backend-\K(blue|green)' "$NGINX_CONF" | head -1)
PREVIOUS=$([ "$CURRENT" = "blue" ] && echo "green" || echo "blue")

# 이전 컨테이너가 살아있는지 확인
STATUS=$(docker inspect --format='{{.State.Status}}' "scheduling-backend-${PREVIOUS}" 2>/dev/null || echo "missing")
if [[ "$STATUS" != "running" ]]; then
    echo "❌ ${PREVIOUS} 컨테이너 미실행 — 자동 롤백 불가"
    echo "   `docker compose up -d backend-${PREVIOUS}` 후 재시도"
    exit 1
fi

echo "↩️ 롤백: ${CURRENT} → ${PREVIOUS}"
sed -i "s/server backend-${CURRENT}:8080/server backend-${PREVIOUS}:8080/" "$NGINX_CONF"
docker compose exec -T nginx nginx -s reload
docker compose -f "$COMPOSE_FILE" stop "backend-${CURRENT}"
echo "✅ 롤백 완료 — ${PREVIOUS} 활성"
```

## :memo: Implementation Notes
- **기술 힌트**:
  - `nginx -s reload`는 worker process를 graceful하게 새로 생성 — 기존 요청 종료까지 대기 + 새 요청은 신규 worker
  - `sed -i`로 NGINX config 동적 변경 — 마커(`# ACTIVE_COLOR_MARKER`) 패턴이 안전, 정규식보다 명확
  - Docker `--no-deps --build` 옵션 — PG·Redis는 건드리지 않고 BE만 재빌드·교체
- **유사 패턴**:
  - Frontend 배포는 단순 정적 파일 교체 — `frontend-static` volume 갱신 + NGINX cache clear로 충분 (Blue-Green 불필요)
- **표준 참조**:
  - NGINX Graceful Reload: https://nginx.org/en/docs/control.html
  - Blue-Green Deployment: https://martinfowler.com/bliki/BlueGreenDeployment.html
