### `pg-backup.sh`

```bash
#!/bin/bash
set -e
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="/var/backups/postgresql/${TIMESTAMP}"
mkdir -p "${BACKUP_DIR}"

pg_basebackup -h ${PG_HOST} -U ${PG_BACKUP_USER} -D "${BACKUP_DIR}" \
  --format=tar --gzip --progress --verbose --checkpoint=fast

# S3 또는 NAS 업로드
aws s3 cp "${BACKUP_DIR}" "s3://${BACKUP_BUCKET}/postgres/${TIMESTAMP}/" --recursive

# 30일 retention
find /var/backups/postgresql -type d -mtime +30 -exec rm -rf {} +
```
