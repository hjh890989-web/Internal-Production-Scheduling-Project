### Flyway `V5_2__create_notification_delivery_log.sql`

```sql
CREATE TABLE notification_delivery_log (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    notification_id UUID NOT NULL,
    channel         VARCHAR(20) NOT NULL CHECK (channel IN ('EMAIL', 'KAKAO')),
    recipient       VARCHAR(200) NOT NULL,
    status          VARCHAR(20) NOT NULL DEFAULT 'PENDING'
        CHECK (status IN ('PENDING', 'SENT', 'DELIVERED', 'READ', 'FAILED')),
    sent_at         TIMESTAMPTZ,
    delivered_at    TIMESTAMPTZ,
    read_at         TIMESTAMPTZ,
    failed_at       TIMESTAMPTZ,
    error_message   TEXT,
    external_id     VARCHAR(100),                -- Kakao tracking ID
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_delivery_log_notif ON notification_delivery_log(notification_id);
CREATE INDEX idx_delivery_log_status ON notification_delivery_log(status, created_at);
CREATE INDEX idx_delivery_log_external ON notification_delivery_log(external_id);
```

### `KakaoBizMessageService.java`

```java
@Service @RequiredArgsConstructor @Slf4j
public class KakaoBizMessageService {

    private final KakaoBizApiClient apiClient;
    private final DeliveryLogRepository logRepo;

    @Transactional
    public DeliveryLog send(UUID notificationId, String recipient, String message) {
        var log = DeliveryLog.builder()
            .notificationId(notificationId)
            .channel(Channel.KAKAO)
            .recipient(recipient)
            .status(DeliveryStatus.PENDING)
            .build();
        log = logRepo.save(log);

        try {
            var response = apiClient.sendBizMessage(recipient, message);
            log.setStatus(DeliveryStatus.SENT);
            log.setSentAt(Instant.now());
            log.setExternalId(response.trackingId());
        } catch (Exception e) {
            log.setStatus(DeliveryStatus.FAILED);
            log.setFailedAt(Instant.now());
            log.setErrorMessage(e.getMessage());
            log.error("Kakao 발송 실패: notification={}", notificationId, e);
        }

        return logRepo.save(log);
    }
}
```

### `KakaoWebhookController.java`

```java
@RestController @RequestMapping("/webhooks/kakao") @RequiredArgsConstructor
public class KakaoWebhookController {

    private final DeliveryLogRepository logRepo;

    @PostMapping("/delivery-callback")
    public ResponseEntity<Void> deliveryCallback(@Valid @RequestBody KakaoDeliveryEvent event) {
        var log = logRepo.findByExternalId(event.trackingId())
            .orElseThrow(() -> new NotFoundException("tracking=" + event.trackingId()));

        switch (event.eventType()) {
            case "DELIVERED" -> {
                log.setStatus(DeliveryStatus.DELIVERED);
                log.setDeliveredAt(event.eventAt());
            }
            case "READ" -> {
                log.setStatus(DeliveryStatus.READ);
                log.setReadAt(event.eventAt());
            }
            case "FAILED" -> {
                log.setStatus(DeliveryStatus.FAILED);
                log.setFailedAt(event.eventAt());
                log.setErrorMessage(event.errorMessage());
            }
        }
        logRepo.save(log);

        return ResponseEntity.ok().build();
    }
}
```
