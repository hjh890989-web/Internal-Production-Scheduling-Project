### `KakaoFallbackIT.java`

```java
@SpringBootTest @Testcontainers
class KakaoFallbackIT {

    @Autowired NotificationDispatcher dispatcher;
    @Autowired DeliveryLogRepository logRepo;
    @Autowired FallbackScheduler scheduler;

    @MockBean KakaoBizApiClient kakaoApi;        // WireMock 대체

    @Test
    void high_priority_email_unread_triggers_kakao_within_2_min() {
        var notificationId = dispatcher.sendHighPriority("planner1@company.com",
            "납기 위반 발생", "010-1234-5678");

        // 이메일 SENT 확인
        await().atMost(5, SECONDS).until(() ->
            logRepo.existsByNotificationIdAndChannelAndStatus(notificationId, EMAIL, SENT));

        // 의도적으로 READ 안 시킴

        // 1분 + scheduler delay 30초 = 1.5분 후 카톡 발송 확인
        await().atMost(2, MINUTES).until(() ->
            logRepo.existsByNotificationIdAndChannelAndStatus(notificationId, KAKAO, SENT));

        verify(kakaoApi).sendBizMessage("010-1234-5678", "납기 위반 발생");
    }

    @Test
    void email_read_within_1_min_no_fallback() {
        var notificationId = dispatcher.sendHighPriority("user@company.com",
            "정보 알림", "010-9999-9999");

        // 30초 후 READ 시뮬레이션 (webhook callback)
        Thread.sleep(30_000);
        simulateEmailRead(notificationId);

        // scheduler 1 cycle 대기 (30초)
        Thread.sleep(30_000);

        // 카톡 미발송 검증
        assertThat(logRepo.existsByNotificationIdAndChannel(notificationId, KAKAO))
            .as("이메일 READ 후 fallback skip")
            .isFalse();
    }

    @Test
    void medium_priority_no_fallback_even_unread() {
        var notificationId = dispatcher.sendMediumPriority("info@company.com",
            "주간 리포트", "010-1111-2222");

        Thread.sleep(90_000);     // 1.5분

        assertThat(logRepo.existsByNotificationIdAndChannel(notificationId, KAKAO))
            .as("MEDIUM priority — fallback 비활성")
            .isFalse();
    }

    @Test
    void kakao_api_failure_logged_no_retry_loop() {
        when(kakaoApi.sendBizMessage(any(), any()))
            .thenThrow(new KakaoApiException("rate limit"));

        var notificationId = dispatcher.sendHighPriority("user@company.com",
            "긴급", "010-3333-4444");

        Thread.sleep(90_000);

        var failedLog = logRepo.findByNotificationIdAndChannel(notificationId, KAKAO).orElseThrow();
        assertThat(failedLog.getStatus()).isEqualTo(FAILED);
        assertThat(failedLog.getErrorMessage()).contains("rate limit");
    }
}
```

### `NotificationKpiIT.java`

```java
@SpringBootTest @Testcontainers
class NotificationKpiIT {

    @Autowired NotificationDispatcher dispatcher;
    @Autowired KpiService kpiService;

    @Test
    void ns_04_delivery_rate_above_95_percent() {
        // 50 알림 시나리오 — 이메일 70% READ·30% 미READ → 카톡 fallback
        for (int i = 0; i < 50; i++) {
            dispatcher.sendHighPriority(
                "user" + i + "@company.com",
                "알림 #" + i,
                "010-0000-" + String.format("%04d", i));
        }

        // scheduler 2 cycle 대기
        Thread.sleep(60_000);

        // 시뮬레이션: 70% READ, 30% 미READ (카톡 fallback)
        // ... read 시뮬레이션 logic

        Thread.sleep(60_000);

        double deliveryRate = kpiService.computeDeliveryRate();
        assertThat(deliveryRate)
            .as("NS-04 알림 도달률 ≥ 95%")
            .isGreaterThanOrEqualTo(0.95);
    }
}
```
