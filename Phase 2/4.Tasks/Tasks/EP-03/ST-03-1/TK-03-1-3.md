CREATE TABLE app.order_change (
    change_id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tracking_id         UUID NOT NULL,
    diff_type           VARCHAR(20) NOT NULL CHECK (diff_type IN ('NEW','MODIFIED','DELETED','UNCHANGED')),
    hose_id             VARCHAR(40) NOT NULL,
    delivery_date       DATE NOT NULL,
    new_order_id        UUID,
    old_order_id        UUID,
    field_diffs         JSONB,                 -- [{field, before, after}, ...]
    previous_version    INT NOT NULL,
    new_version         INT NOT NULL,
    severity            VARCHAR(10),           -- ST-03-2에서 채움 (NULL 허용)
    changed_at          TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_order_change_tracking ON app.order_change(tracking_id);
CREATE INDEX idx_order_change_hose_date ON app.order_change(hose_id, delivery_date);
CREATE INDEX idx_order_change_severity ON app.order_change(severity) WHERE severity IS NOT NULL;

REVOKE UPDATE, DELETE ON app.order_change FROM app_user;
GRANT INSERT, SELECT ON app.order_change TO app_user;
-- severity는 ST-03-2에서 UPDATE 필요 — 예외 정책: UPDATE 허용 OR severity 별 테이블 분리

GRANT UPDATE(severity) ON app.order_change TO app_user;

COMMENT ON TABLE app.order_change IS 'ST-03-1: 마스터 버전 간 row-level diff 영속화';

### Entity + Service

```java
@Entity @Table(schema = "app", name = "order_change")
@Getter @NoArgsConstructor @Builder @AllArgsConstructor
public class OrderChangeEntity {
    @Id @GeneratedValue private UUID changeId;
    private UUID trackingId;
    @Enumerated(EnumType.STRING) private DiffType diffType;
    private String hoseId;
    private LocalDate deliveryDate;
    private UUID newOrderId;
    private UUID oldOrderId;
    @JdbcTypeCode(SqlTypes.JSON) @Column(columnDefinition = "JSONB")
    private JsonNode fieldDiffs;
    private int previousVersion;
    private int newVersion;
    private String severity;       // ST-03-2에서 UPDATE
    private Instant changedAt;
}

@Service @RequiredArgsConstructor
public class DiffPersistenceService {

    private final OrderChangeRepository repo;
    private final ObjectMapper json;

    @Transactional
    public void persist(DiffResult diff) {
        var entities = diff.rows().stream()
            .filter(r -> r.type() != DiffType.UNCHANGED)   // UNCHANGED는 저장 안 함
            .map(r -> toEntity(r, diff))
            .toList();
        repo.saveAll(entities);
    }

    public List<OrderChangeEntity> getByTracking(UUID trackingId) {
        return repo.findByTrackingIdOrderByChangedAtDesc(trackingId);
    }
}
```
