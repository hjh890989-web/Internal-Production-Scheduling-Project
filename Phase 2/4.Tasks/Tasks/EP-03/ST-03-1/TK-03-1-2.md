### `DiffAccuracyIT.java`

```java
@SpringBootTest @Testcontainers
class DiffAccuracyIT {

    @Autowired DiffEngineService diff;
    @Autowired OrderRepository repo;

    @Test
    void diff_100_mutations_with_100_percent_accuracy() throws Exception {
        // 1. 기존 마스터 50 row 로드 (master_version=1)
        var baseMaster = loadOrders("base_master.json", 1);
        repo.saveAll(baseMaster);

        // 2. 변형 batch 로드
        var batch = loadDrafts("mutated_batch.json");

        // 3. 정답 로드
        var expectedMap = loadExpected("expected_diffs.json");  // Map<OrderKey, DiffType>

        // 4. Diff 계산
        var result = diff.compute(UUID.randomUUID(), batch, 1);

        // 5. 100% 정확 분류 검증
        assertThat(result.rows()).hasSize(expectedMap.size() + /*UNCHANGED*/ 20);
        int correct = 0;
        for (RowDiff rd : result.rows()) {
            DiffType expected = expectedMap.get(rd.key());
            if (rd.type() == expected) correct++;
        }
        assertThat(correct).isEqualTo(result.rows().size());

        // 6. 카운트 검증
        assertThat(result.countByType(DiffType.NEW)).isEqualTo(30);
        assertThat(result.countByType(DiffType.DELETED)).isEqualTo(20);
        assertThat(result.countByType(DiffType.MODIFIED)).isEqualTo(30);
        assertThat(result.countByType(DiffType.UNCHANGED)).isEqualTo(20);
    }

    @Test
    void multi_field_modification_captures_all_fields() {
        // 같은 키, 납기+수량+거래처 3 필드 변경
        var oldOrder = order("PART-001", "2026-02-15", 100, "고객사A");
        var newDraft = draft("PART-001", "2026-02-16", 150, "고객사B");
        repo.save(oldOrder);

        var result = diff.compute(UUID.randomUUID(), List.of(newDraft), oldOrder.getMasterVersion());
        assertThat(result.rows()).hasSize(1);
        var rd = result.rows().get(0);
        assertThat(rd.type()).isEqualTo(DiffType.MODIFIED);
        assertThat(rd.fieldDiffs()).extracting(FieldDiff::fieldName)
            .containsExactlyInAnyOrder("delivery_date", "qty", "customer");
    }
}
```
