CREATE TABLE app.notification (
    notification_id   UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_change_id   UUID NOT NULL REFERENCES app.order_change(change_id),
    channel           VARCHAR(20) NOT NULL CHECK (channel IN ('IN_APP','KAKAOTALK','SLACK')),
    severity          VARCHAR(10) NOT NULL,
    target_role       VARCHAR(30) NOT NULL,
    dispatched_at     TIMESTAMPTZ NOT NULL DEFAULT now(),
    sent_at           TIMESTAMPTZ,
    delivered_at      TIMESTAMPTZ,
    acknowledged_at   TIMESTAMPTZ,
    failed_at         TIMESTAMPTZ,
    error_message     TEXT,
    retry_count       INT NOT NULL DEFAULT 0
);

CREATE INDEX idx_notification_change ON app.notification(order_change_id);
CREATE INDEX idx_notification_status ON app.notification(severity, acknowledged_at, failed_at);
CREATE INDEX idx_notification_dispatched ON app.notification(dispatched_at);

-- SLA 위반 모니터링 인덱스
CREATE INDEX idx_notification_undelivered_critical ON app.notification(dispatched_at)
WHERE severity = 'CRITICAL' AND acknowledged_at IS NULL AND failed_at IS NULL;

COMMENT ON TABLE app.notification IS 'EP-03 ST-03-3: 알림 도달 라이프사이클 + SLA 추적';

### Entity + Repository + Controller

```java
@Entity @Table(schema = "app", name = "notification")
@Getter @NoArgsConstructor @AllArgsConstructor @Builder
public class NotificationEntity {
    @Id @GeneratedValue private UUID notificationId;
    private UUID orderChangeId;
    @Enumerated(EnumType.STRING) private NotificationChannel channel;
    @Enumerated(EnumType.STRING) private Severity severity;
    private String targetRole;
    private Instant dispatchedAt;
    private Instant sentAt;
    private Instant deliveredAt;
    private Instant acknowledgedAt;
    private Instant failedAt;
    private String errorMessage;
    @Builder.Default private int retryCount = 0;
}

public enum NotificationStatus { DISPATCHED, SENT, DELIVERED, ACKNOWLEDGED, FAILED }

@RestController
@RequestMapping("/api/v1/notifications")
@RequiredArgsConstructor
public class AckController {

    private final NotificationRepository repo;

    @PostMapping("/{id}/ack")
    @PreAuthorize("isAuthenticated()")
    public NotificationEntity acknowledge(@PathVariable UUID id,
                                          @AuthenticationPrincipal Jwt jwt) {
        var n = repo.findById(id).orElseThrow();
        if (n.getAcknowledgedAt() == null) {
            n.setAcknowledgedAt(Instant.now());
            repo.save(n);
        }
        return n;
    }
}

@Component @RequiredArgsConstructor
public class DeliveryEscalator {

    private final NotificationRepository repo;
    private final SlackClient slack;

    /**
     * 1분마다 SLA 위반 Critical 알림 점검 → Slack 에스컬레이션
     */
    @Scheduled(fixedDelay = 60_000)
    public void escalate() {
        var threshold = Instant.now().minusSeconds(60);  // Critical 1분 SLA
        var overdue = repo.findCriticalOverdue(threshold);
        for (var n : overdue) {
            slack.send(SlackChannel.OPS, buildEscalationMessage(n));
            log.warn("Critical notification {} overdue — escalated to Slack", n.getNotificationId());
        }
    }
}
```

### Repository 쿼리

```java
@Query("""
    SELECT n FROM NotificationEntity n
    WHERE n.severity = 'CRITICAL'
      AND n.acknowledgedAt IS NULL
      AND n.failedAt IS NULL
      AND n.dispatchedAt < :threshold
    """)
List<NotificationEntity> findCriticalOverdue(@Param("threshold") Instant threshold);
```
