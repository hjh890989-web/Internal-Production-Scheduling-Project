### k6 스크립트 (`notification_sla_test.js`)

```javascript
// perf/scripts/notification_sla_test.js
// TC-OC-009: Critical 변경 100건 → SLA <1분 도달률 ≥99%
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Counter, Trend, Rate } from 'k6/metrics';

const dispatchTrend = new Trend('notification_dispatch_to_ack', true);
const slaCompliance = new Rate('sla_compliance');
const totalCritical = new Counter('critical_dispatched');

export const options = {
    scenarios: {
        critical_burst: {
            executor: 'shared-iterations',
            vus: 10,                            // 30명 중 알림 대상 평균 10명
            iterations: 100,                    // 100건 Critical 이벤트
            maxDuration: '5m',
        },
    },
    thresholds: {
        sla_compliance: ['rate>0.99'],          // ≥99% SLA 준수
        notification_dispatch_to_ack: ['p(95)<60000'],  // p95 <60초
        http_req_failed: ['rate<0.01'],
    },
};

const BASE_URL = __ENV.BASE_URL || 'http://stg.intranet/api/v1';
const TOKEN = __ENV.JWT_TOKEN;

export default function () {
    // 1. Critical 변경 이벤트 트리거 (직접 commit 또는 다이렉트 알림 API)
    const dispatchTime = Date.now();
    const event = {
        hose_id: `PART-${__ITER}`,
        delivery_date: '2026-02-15',
        severity: 'CRITICAL',
        change_summary: 'qty 100 → 130 (+30%)'
    };

    const dispatchRes = http.post(
        `${BASE_URL}/orders/_test/dispatch-notification`,    // 테스트 전용 endpoint
        JSON.stringify(event),
        { headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${TOKEN}` } }
    );

    check(dispatchRes, { 'dispatch 202': r => r.status === 202 });
    totalCritical.add(1);

    const notificationId = dispatchRes.json('notificationId');

    // 2. polling — acknowledged_at 채워질 때까지 (or timeout 60s)
    const deadline = dispatchTime + 60_000;
    let acked = false;
    while (Date.now() < deadline) {
        const statusRes = http.get(
            `${BASE_URL}/notifications/${notificationId}`,
            { headers: { Authorization: `Bearer ${TOKEN}` } }
        );
        if (statusRes.json('acknowledgedAt')) {
            const ackDuration = Date.now() - dispatchTime;
            dispatchTrend.add(ackDuration);
            slaCompliance.add(ackDuration < 60_000 ? 1 : 0);
            acked = true;
            break;
        }
        sleep(2);
    }

    if (!acked) {
        slaCompliance.add(0);                   // SLA 위반
    }
}

export function handleSummary(data) {
    return {
        'reports/notification_sla_summary.json': JSON.stringify(data),
        stdout: textSummary(data),
    };
}
```
