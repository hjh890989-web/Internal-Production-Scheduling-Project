### `SeverityClassifierAccuracyIT.java`

```java
@SpringBootTest
class SeverityClassifierAccuracyIT {

    @Autowired SeverityClassifier classifier;

    @Test
    void zero_false_negatives_on_100_critical_candidates() throws Exception {
        var criticals = loadCriticals("critical_candidates.json");
        int falseNegatives = 0;
        for (RowDiff candidate : criticals) {
            if (classifier.classify(candidate) != Severity.CRITICAL) {
                falseNegatives++;
                System.err.println("FN: " + candidate);
            }
        }
        assertThat(falseNegatives)
            .as("Zero False Negatives required (TC-OC-008)")
            .isZero();
    }

    @Test
    void false_positive_rate_under_5_percent() throws Exception {
        var normals = loadNormals("normal_candidates.json");
        long falsePositives = normals.stream()
            .filter(n -> classifier.classify(n) == Severity.CRITICAL)
            .count();
        double fpRate = (double) falsePositives / normals.size();
        assertThat(fpRate).isLessThanOrEqualTo(0.05);
    }
}
```

    var diff = makeQtyDiff(100, 120);  // +20%

    assertThat(classifier.classify(diff)).isEqualTo(Severity.NORMAL);

@Test void qty_negative_20_percent_decrease_is_critical() {
    var diff = makeQtyDiff(100, 80);
    assertThat(classifier.classify(diff)).isEqualTo(Severity.CRITICAL);
}

@Test void qty_zero_to_anything_is_critical() {
    var diff = makeQtyDiff(0, 1);
    assertThat(classifier.classify(diff)).isEqualTo(Severity.CRITICAL);

@Test void delivery_date_change_one_day_is_critical() {
    var diff = makeDateDiff(LocalDate.of(2026, 2, 15), LocalDate.of(2026, 2, 16));

}

@Test void customer_change_only_is_normal() {
