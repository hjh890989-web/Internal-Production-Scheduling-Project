### 핵심 로직

```java
@ConfigurationProperties(prefix = "scheduling.severity")
@Validated @Data
public class SeverityConfig {
    @DecimalMin("0.05") @DecimalMax("1.00")
    private double qtyChangeThresholdPct = 0.20;     // BR-O02
    private boolean deliveryDateAlwaysCritical = true;
    private boolean hoseIdAlwaysCritical = true;
    private boolean newAlwaysCritical = true;
    private boolean deletedAlwaysCritical = true;
}

@RestController
@RequestMapping("/api/v1/master/severity-config")
@RequiredArgsConstructor
public class SeverityConfigController {

    private final SeverityConfigService service;

    @GetMapping
    @PreAuthorize("hasAnyRole('PLANNER', 'IT_OPS')")
    public SeverityConfig get() { return service.current(); }

    @PutMapping
    @PreAuthorize("hasRole('IT_OPS')")                // dual-review 필요 시 별 endpoint
    public SeverityConfig update(@Valid @RequestBody SeverityConfig updated,
                                  @AuthenticationPrincipal Jwt jwt) {
        return service.update(updated, jwt.getSubject());
    }
}

@Service @RequiredArgsConstructor
public class SeverityConfigService {

    private SeverityConfig live;
    private final AuditLogger auditLogger;

    @PostConstruct
    public void init() { this.live = SeverityConfig.builder().build(); }

    public SeverityConfig current() { return live; }

    @Transactional
    public SeverityConfig update(SeverityConfig updated, String actor) {
        var before = clone(live);
        // 검증은 @Valid 어노테이션이 처리
        this.live = updated;
        auditLogger.log("SEVERITY_CONFIG", before, updated, actor);
        return updated;
    }
}
```
