### `ExtrusionConflictReportService.java`

```java
@Service @RequiredArgsConstructor
public class ExtrusionConflictReportService {

    private final ExScheduleCandidateRepository candidateRepo;
    private final ExtrusionValidationGate gate;
    private final ExtrusionAlternativeGenerator generator;
    private final SettingGroupAllocator allocator;

    public ExtrusionConflictReport buildReport(UUID candidateId) {
        var cand = candidateRepo.findById(candidateId)
            .orElseThrow(() -> new NotFoundException("candidate=" + candidateId));

        // 1) 현재 ledger 재구성 (allocator 1회 실행 결과)
        var ctx = buildContextForCandidate(cand);
        var allocResult = allocator.allocate(ctx);
        var ledger = ExtrusionLedger.fromAllocationResult(allocResult);

        // 2) Validation gate 호출
        var validation = gate.validate(cand, ledger);

        // 3) fail 시 대안 enrich
        var alternatives = generator.generateAll(validation, cand);

        return new ExtrusionConflictReport(
            candidateId,
            cand.hoseId(),
            validation.passed(),
            validation.violations(),
            alternatives,
            Instant.now()
        );
    }
}

public record ExtrusionConflictReport(
    UUID candidateId,
    String hoseId,
    boolean passed,
    List<ValidationViolation> violations,
    List<ExtrusionAlternative> alternatives,
    Instant generatedAt
) {}
```

### `ExtrusionConflictController.java`

```java
@RestController @RequestMapping("/api/v1/schedule/extrusion/conflicts") @RequiredArgsConstructor
public class ExtrusionConflictController {

    private final ExtrusionConflictReportService service;

    @GetMapping("/{candidateId}")
    @PreAuthorize("isAuthenticated()")
    public ExtrusionConflictReport get(@PathVariable UUID candidateId) {
        return service.buildReport(candidateId);
    }
}
```

### `ExtrusionConflictReportIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-EX-CONFLICT-100/master_seed.sql",
      "classpath:datasets/DS-EX-CONFLICT-100/candidates_with_violations.sql"})
class ExtrusionConflictReportIT {

    @Autowired ExtrusionConflictReportService service;
    @Autowired ExScheduleCandidateRepository candidateRepo;

    @Test
    void regression_100_failed_candidates_all_have_three_alternatives() {
        // 100 candidate 중 50% fail 의도 (capa 부족·재고 0 등)
        var failedIds = candidateRepo.findAllFailExpected();
        assertThat(failedIds).hasSizeGreaterThan(50);

        for (var id : failedIds) {
            var report = service.buildReport(id);

            assertThat(report.passed()).isFalse();

            long distinctTypes = report.alternatives().stream()
                .map(ExtrusionAlternative::type).distinct().count();
            assertThat(distinctTypes)
                .as("candidate %s: distinct alternatives", id)
                .isGreaterThanOrEqualTo(3);
        }
    }

    @Test
    void passed_candidate_returns_empty_alternatives() {
        var passedIds = candidateRepo.findAllPassExpected();
        for (var id : passedIds) {
            var report = service.buildReport(id);
            assertThat(report.passed()).isTrue();
            assertThat(report.alternatives()).isEmpty();
        }
    }

    @Test
    void p95_response_under_one_second() {
        var ids = candidateRepo.findAll().stream().map(c -> c.getId()).limit(50).toList();

        var latencies = new ArrayList<Long>();
        for (var id : ids) {
            long start = System.nanoTime();
            service.buildReport(id);
            latencies.add((System.nanoTime() - start) / 1_000_000);
        }

        Collections.sort(latencies);
        long p95 = latencies.get((int) (latencies.size() * 0.95));
        assertThat(p95).isLessThanOrEqualTo(1000);
    }

    @Test
    void api_endpoint_returns_correct_json_structure() throws Exception {
        // MockMvc로 GET /api/v1/schedule/extrusion/conflicts/{id}
        // 응답 JSON 구조 검증 — alternatives[].type/label/suggestedAction 존재
    }

    @Test
    void cumulative_yield_short_includes_earlier_start_and_outsource() {
        // 특정 fail 케이스 (CUMULATIVE_YIELD_SHORT) → EARLIER_START·OUTSOURCE 포함 검증
        var cumulativeShortId = candidateRepo.findFirstFailedWithCategory(
            ValidationCategory.CUMULATIVE_YIELD_SHORT).getId();

        var report = service.buildReport(cumulativeShortId);
        var types = report.alternatives().stream()
            .map(ExtrusionAlternative::type).toList();

        assertThat(types)
            .contains(ExtrusionAlternativeType.EARLIER_START)
            .contains(ExtrusionAlternativeType.OUTSOURCE);
    }
}
```
