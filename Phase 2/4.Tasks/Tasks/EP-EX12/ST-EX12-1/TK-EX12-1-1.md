### `ExtrusionAlternativeGenerator.java`

```java
@Service @RequiredArgsConstructor
public class ExtrusionAlternativeGenerator {

    private static final int MIN_ALTERNATIVES = 3;

    /**
     * 카테고리별 적용 가능한 대안 매트릭스.
     * - CUMULATIVE_YIELD_SHORT: 수량 부족 → 시간 확장 (EARLIER, NIGHT, OVERTIME) 또는 협상/외주
     * - SHIFT_CAPACITY_EXCEEDED: shift 과부하 → 후보 이동 (EARLIER, NIGHT, SWAP) 또는 외주
     */
    private static final Map<ValidationCategory, List<ExtrusionAlternativeType>> POLICY = Map.of(
        ValidationCategory.CUMULATIVE_YIELD_SHORT,
            List.of(EARLIER_START, NIGHT_SECOND_BOOST, OUTSOURCE, VC_DATE_NEGOTIATE, OVERTIME),
        ValidationCategory.SHIFT_CAPACITY_EXCEEDED,
            List.of(EARLIER_START, NIGHT_SECOND_BOOST, SWAP_CANDIDATE, OUTSOURCE, OVERTIME)
    );

    public List<ExtrusionAlternative> generate(ValidationViolation violation, ExScheduleCandidate cand) {
        var types = POLICY.getOrDefault(violation.category(), defaultPolicy());

        return types.stream()
            .limit(Math.max(MIN_ALTERNATIVES, types.size()))
            .map(t -> ExtrusionAlternative.of(t, cand, violation))
            .toList();
    }

    public List<ExtrusionAlternative> generateAll(ValidationResult result, ExScheduleCandidate cand) {
        if (result.passed()) return List.of();

        var seen = new HashSet<ExtrusionAlternativeType>();
        var alternatives = new ArrayList<ExtrusionAlternative>();

        for (var v : result.violations()) {
            for (var alt : generate(v, cand)) {
                if (seen.add(alt.type())) alternatives.add(alt);
            }
        }

        // distinct 보장 — 다중 violation에서 중복 제거 후 < 3 이면 default 추가
        if (alternatives.size() < MIN_ALTERNATIVES) {
            for (var t : defaultPolicy()) {
                if (alternatives.size() >= MIN_ALTERNATIVES) break;
                if (seen.add(t)) {
                    alternatives.add(ExtrusionAlternative.of(t, cand,
                        result.violations().get(0)));
                }
            }
        }

        return alternatives;
    }

    private List<ExtrusionAlternativeType> defaultPolicy() {
        return List.of(EARLIER_START, OUTSOURCE, OVERTIME, NIGHT_SECOND_BOOST, SWAP_CANDIDATE);
    }
}

public record ExtrusionAlternative(
    ExtrusionAlternativeType type,
    String label,
    String description,
    String suggestedAction,
    ValidationCategory triggerCategory
) {
    public static ExtrusionAlternative of(ExtrusionAlternativeType t, ExScheduleCandidate cand,
                                            ValidationViolation v) {
        return new ExtrusionAlternative(t, t.getLabel(), t.getDescription(),
            buildAction(t, cand), v.category());
    }

    private static String buildAction(ExtrusionAlternativeType t, ExScheduleCandidate cand) {
        return switch (t) {
            case EARLIER_START       -> "%s deadline %s 이전 가용 shift 검색 → 조기 배치"
                .formatted(cand.hoseId(), cand.getExtrusionDeadline());
            case NIGHT_SECOND_BOOST  -> "%s 야간 후반 shift 추가 배치 시도 (셋팅 그룹 호환 시)"
                .formatted(cand.hoseId());
            case VC_DATE_NEGOTIATE   -> "%s 성형 vc_date %s → +1 영업일 협상 (ex_deadline 연장)"
                .formatted(cand.hoseId(), cand.getVcProductionDate());
            case OUTSOURCE           -> "%s Q_ext=%d 외주 발주"
                .formatted(cand.hoseId(), cand.getQExt());
            case OVERTIME            -> "%s 토 가동 + 압출 임시 라인 활용"
                .formatted(cand.hoseId());
            case SWAP_CANDIDATE      -> "%s 우선순위 ↓ + 후순위 후보로 이동"
                .formatted(cand.hoseId());
        };
    }
}
```
