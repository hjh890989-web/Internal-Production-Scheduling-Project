### `LineRoutingRegressionIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-EX-ROUTING-1WEEK/master_seed.sql"})
class LineRoutingRegressionIT {

    @Autowired SettingGroupAllocator allocator;
    @Autowired LineRoutingPolicy policy;
    @Autowired LineCompatibilityRule compatRule;
    @Autowired ScenarioLoader loader;

    @Test
    void new_line_usage_above_90_percent() {
        var candidates = loader.candidatesFor1Week();
        var ctx = ExAllocationContext.builder()
            .candidates(candidates)
            .workingDays(workingDays(7))
            .build();

        var result = allocator.allocate(ctx);

        long total = result.schedule().size();
        long newLine = result.schedule().stream()
            .filter(row -> policy.isNewLine(row.machineId()))
            .count();

        double usageRate = (double) newLine / total;

        assertThat(usageRate)
            .as("NS-S09 신규 라인 사용률 ≥ 90%%")
            .isGreaterThanOrEqualTo(0.90);
    }

    @Test
    void ford_exclusive_products_zero_misrouting() {
        var fordExclusive = List.of("28912-2U000", "29689-2U000");

        var candidates = loader.candidatesFor1Week();
        var result = allocator.allocate(buildContext(candidates));

        for (var row : result.schedule()) {
            if (fordExclusive.contains(row.hoseId())) {
                assertThat(row.machineId())
                    .as("%s 포드 전용 — NEW 라인 0건", row.hoseId())
                    .isEqualTo("EX-FORD");
            }
        }
    }

    @Test
    void ford_fallback_when_new_lines_saturated() {
        // 신규 capa 폭증 시나리오 — 포드로 fallback
        var saturatedCtx = loader.saturateNewLinesContext();
        var result = allocator.allocate(saturatedCtx);

        long fordCount = result.schedule().stream()
            .filter(row -> row.machineId().equals("EX-FORD"))
            .count();

        assertThat(fordCount).as("NEW 포화 시 FORD fallback").isGreaterThan(0);
    }

    @Test
    void priority_order_new_before_ford() {
        // 동일 capa 시나리오 — priority 정렬에 따라 NEW 우선 배정
        var candidates = loader.smallSetOfCandidates();
        var result = allocator.allocate(buildContext(candidates));

        var firstAllocated = result.schedule().get(0);
        assertThat(policy.isNewLine(firstAllocated.machineId()))
            .as("priority 우선 NEW 배정")
            .isTrue();
    }
}
```
