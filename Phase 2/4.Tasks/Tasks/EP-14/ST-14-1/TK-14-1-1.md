### Flyway `V4_6__create_line_type.sql`

```sql
CREATE TABLE master.line_type (
    line_id          VARCHAR(20) PRIMARY KEY,
    type             VARCHAR(10) NOT NULL CHECK (type IN ('NEW', 'FORD')),
    priority         INTEGER NOT NULL DEFAULT 100,
    is_active        BOOLEAN NOT NULL DEFAULT TRUE,
    notes            TEXT,
    updated_at       TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_line_type_active ON master.line_type(type, priority) WHERE is_active = TRUE;

GRANT SELECT ON master.line_type TO app_user;
GRANT INSERT, UPDATE, DELETE ON master.line_type TO master_admin;
```

### `LineRoutingPolicy.java`

```java
@Service @RequiredArgsConstructor
public class LineRoutingPolicy {

    private final LineMasterRepository lineRepo;

    /**
     * BR-E08: 신규 우선, 포드 fallback.
     *  - 활성 라인 중 priority 오름차순 정렬 (낮을수록 우선)
     *  - NEW type 모두 시도 후 FORD fallback
     */
    public List<LineMaster> candidatesInOrder() {
        return lineRepo.findByIsActiveTrueOrderByPriorityAsc();
    }

    /**
     * 특정 hose의 라인 후보 — 포드 전용 품번 차단은 TK-14-1-2 별도.
     */
    public List<LineMaster> candidatesForHose(String hoseId) {
        return candidatesInOrder();
        // TK-14-1-2가 추가 필터링
    }

    public boolean isNewLine(String lineId) {
        return lineRepo.findById(lineId)
            .map(l -> l.getType() == LineType.NEW)
            .orElse(false);
    }
}
```
