// hooks/useDragValidation.ts
import { useSlotCompatibilityMatrix } from './useSlotCompatibilityMatrix';

export function useDragValidation() {
  const { data: compat } = useSlotCompatibilityMatrix();

  function validate(hoseId: string, slot: SlotPosition): {
    ok: boolean;
    reason?: string;
    alternatives?: SlotPosition[];
  } {
    if (!compat) return { ok: false, reason: '매트릭스 미로딩' };
    const eligible = compat.matrix[hoseId]?.[slot] ?? false;
    if (eligible) return { ok: true };

    // 대안 슬롯 (NFR-USA-002)
    const alts = Object.entries(compat.matrix[hoseId] ?? {})
      .filter(([_, ok]) => ok)
      .map(([s]) => s as SlotPosition);

    return {
      ok: false,
      reason: `품번 ${hoseId}는 ${slot} 슬롯에 적합하지 않습니다`,
      alternatives: alts
    };
  }
  return { validate };
}

// components/VcGanttBoard.tsx — onDragEnd 가드
import { DndContext, DragEndEvent } from '@dnd-kit/core';
import { message, Modal } from 'antd';
import { useDragValidation } from '../hooks/useDragValidation';

export function VcGanttBoard() {
  const { validate } = useDragValidation();
  const [violation, setViolation] = useState<ViolationInfo | null>(null);

  function handleDragEnd(e: DragEndEvent) {
    const hoseId = e.active.id as string;
    const slot = e.over?.id as SlotPosition | undefined;
    if (!slot) return;

    const t0 = performance.now();
    const result = validate(hoseId, slot);
    const elapsed = performance.now() - t0;

    if (!result.ok) {
      // ≤1초 이내 경고 (실제로는 ms 단위)
      message.error(result.reason);
      setViolation({ hoseId, slot, reason: result.reason!, alternatives: result.alternatives! });
      // 자동 원복 (drag 무시)
      return;
    }

    // 통과 → 정상 배치 (TK-05-3 회전 배치로 위임)
    placeOrder(hoseId, slot);
  }

  return (
    <DndContext onDragEnd={handleDragEnd}>
      <ViolationModal violation={violation} onClose={() => setViolation(null)} />
      {/* ... Gantt board UI ... */}
    </DndContext>
  );
}

// components/ViolationModal.tsx — 사유·대안 표시 + override
export function ViolationModal({ violation, onClose }: Props) {
  if (!violation) return null;
  return (
    <Modal title="🔴 슬롯 적합성 위반" open onCancel={onClose}
           footer={[
             <Button key="cancel" onClick={onClose}>취소</Button>,
             <Button key="override" type="primary" danger
                     onClick={() => setShowOverride(true)}>
               override (사유 입력)
             </Button>
           ]}>
      <Alert message={violation.reason} type="error" showIcon />
      <Divider />
      <h4>대안 슬롯 ({violation.alternatives.length}개)</h4>
      <Space wrap>
        {violation.alternatives.map(s => (
          <Tag key={s} color="green">{slotDisplayName(s)}</Tag>
        ))}
      </Space>
      {showOverride && <OverrideJustificationForm violation={violation} />}
    </Modal>
  );
}

// components/OverrideJustificationForm.tsx — REQ-FUNC-CO-010
export function OverrideJustificationForm({ violation }: { violation: ViolationInfo }) {
  const { mutate: overrideWithAudit, isPending } = useOverrideAudit();
  const [form] = Form.useForm();

  return (
    <Form form={form} layout="vertical">
      <Form.Item name="justification" label="override 사유 (필수)"
                 rules={[{ required: true, min: 10, message: '10자 이상 입력' }]}>
        <Input.TextArea rows={4} placeholder="예: 긴급 납기 대응, 현장 협의 완료 (이수진 반장 승인)" />
      </Form.Item>
      <Button type="primary" danger htmlType="submit" loading={isPending}
              onClick={() => form.validateFields().then(v => overrideWithAudit({
                ...violation, justification: v.justification
              }))}>
        강제 배치 + audit 기록
      </Button>
    </Form>
  );
}
