### 핵심 로직

#### A. `slot-drag-guard.spec.ts` — 100건 회귀

```typescript
import { test, expect } from '@playwright/test';
import scenarios from './fixtures/drag-scenarios-100.json';

test.describe('Slot drag guard — TC-VC-004', () => {

  test.beforeEach(async ({ page }) => {
    await page.goto('/vc/board');
    await loginAsPlanner(page, 'p4-test');
    await page.waitForLoadState('networkidle');
  });

  for (const scenario of scenarios) {
    test(`${scenario.case_id}: ${scenario.hose_id} → ${scenario.slot}`, async ({ page }) => {
      const hose = page.locator(`[data-hoseId="${scenario.hose_id}"]`);
      const slot = page.locator(`[data-slot="${scenario.slot}"]`);

      const t0 = Date.now();
      await hose.dragTo(slot);
      const dragElapsed = Date.now() - t0;

      if (scenario.expected_eligible) {
        // 통과 — 슬롯에 배치 완료
        await expect(slot.locator(`[data-hoseId="${scenario.hose_id}"]`)).toBeVisible();
      } else {
        // 차단 — ≤1초 이내 위반 토스트
        await expect(page.getByText(/적합하지 않습니다/)).toBeVisible({ timeout: 1000 });
        expect(dragElapsed).toBeLessThanOrEqualTo(1000);
        // drag 원복 — 슬롯에 미배치
        await expect(slot.locator(`[data-hoseId="${scenario.hose_id}"]`)).not.toBeVisible();
      }
    });
  }
});
```

test('override requires justification ≥10 chars (REQ-FUNC-CO-010)', async ({ page }) => {
  await page.goto('/vc/board');
  await loginAsPlanner(page, 'p1-test');

  // 비적합 드래그
  await page.locator('[data-hoseId="29673-2F900"]').dragTo(

  );
  await page.getByText('🔴 슬롯 적합성 위반').waitFor();

  // override 클릭
  await page.click('button:has-text("override")');

  // 사유 미입력 → 저장 불가
  await page.click('button:has-text("강제 배치 + audit 기록")');
  await expect(page.getByText(/10자 이상 입력/)).toBeVisible();

});

#### C. `p4-persona-uat.spec.ts` — UAT

```typescript
test('P4 페르소나 — 단독 드래그 사용 가능 (US-02 AC-3)', async ({ page }) => {
  await page.goto('/vc/board');
  await loginAsPlanner(page, 'p4-uat-user');

  // 1주 분량 수주 입력 후 드래그앤드롭으로 배치
  const orders = await page.locator('[data-testid="pending-orders"] [data-hoseId]').all();
  expect(orders.length).toBeGreaterThan(0);

  // 처음 5건을 드래그 — 비적합 슬롯 자동 회피
  for (let i = 0; i < 5; i++) {
    const hose = orders[i];
    const hoseId = await hose.getAttribute('data-hoseId');

    // 매트릭스에서 적합 슬롯 자동 추천 (alt 표시)
    await hose.hover();
    const recommended = page.locator(`[data-recommended-slot][data-hoseId="${hoseId}"]`);
    await expect(recommended).toBeVisible();

    await hose.dragTo(recommended);
    await expect(recommended.locator(`[data-hoseId="${hoseId}"]`)).toBeVisible();
  }

  // 5건 모두 배치 성공 → P4 단독 수행 가능 (NS-S03)
  const placed = await page.locator('[data-slot] [data-hoseId]').count();
  expect(placed).toEqual(5);
});
```
