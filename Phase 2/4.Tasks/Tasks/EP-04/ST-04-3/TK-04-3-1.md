```typescript
// types/slotCompat.ts
export type MachineType = 'LP' | 'IC';
export type SlotPosition = 'LP_TOP' | 'LP_UPMID' | 'LP_LOWMID' | 'LP_BOT'
                        | 'IC_TOP' | 'IC_MID' | 'IC_BOT';

export interface CompatibilityResponse {
  version: number;
  builtAt: string;
  matrix: Record<string, Record<SlotPosition, boolean>>;
  bySlot: Record<SlotPosition, string[]>;
  unschedulableHoseIds: string[];
}

// hooks/useSlotCompatibilityMatrix.ts
import { useQuery } from '@tanstack/react-query';
import { fetchCompat } from '../api/compatApi';

export function useSlotCompatibilityMatrix() {
  return useQuery({
    queryKey: ['vc', 'compat'],
    queryFn: fetchCompat,
    staleTime: 5 * 60 * 1000,           // 5분 stale
    gcTime: 30 * 60 * 1000,
    refetchOnWindowFocus: false,
  });
}

// components/DraggableHose.tsx
import { useDraggable } from '@dnd-kit/core';

export function DraggableHose({ hoseId, qty }: { hoseId: string; qty: number }) {
  const { attributes, listeners, setNodeRef, transform } = useDraggable({
    id: hoseId,
    data: { hoseId, qty }
  });
  return (
    <div ref={setNodeRef} style={{ transform: transform ? `translate3d(${transform.x}px, ${transform.y}px, 0)` : undefined }}
         {...listeners} {...attributes}>
      {hoseId} · {qty}EA
    </div>
  );
}

// components/DroppableSlot.tsx
import { useDroppable } from '@dnd-kit/core';
import { useSlotCompatibilityMatrix } from '../hooks/useSlotCompatibilityMatrix';

export function DroppableSlot({ slotPosition, draggedHoseId }: {
  slotPosition: SlotPosition;
  draggedHoseId?: string;
}) {
  const { setNodeRef, isOver } = useDroppable({ id: slotPosition });
  const { data: compat } = useSlotCompatibilityMatrix();

  const isEligible = draggedHoseId && compat
    ? compat.matrix[draggedHoseId]?.[slotPosition] ?? false
    : true;

  return (
    <div ref={setNodeRef}
         className={isOver
           ? (isEligible ? 'slot--hover-ok' : 'slot--hover-blocked')
           : 'slot'}>
      {slotPosition}
      {isOver && !isEligible && <ProhibitedIcon />}
    </div>
  );
}
```
