#### A. Flyway `V2_1__create_vc_constraint.sql`

```sql
-- EP-04 ST-04-1: VC_CONSTRAINT 마스터 (REF-09 47품번 + SAD §6.2.2)

CREATE TABLE master.vc_constraint (
    hose_id              VARCHAR(40) PRIMARY KEY REFERENCES master.product(hose_id),
    mold_qty             INTEGER NOT NULL CHECK (mold_qty >= 0),
    composite_count      SMALLINT NOT NULL CHECK (composite_count IN (1,2,3,6)),

    -- 저압 가류기 (4 슬롯)
    lp_molds_per_angle   SMALLINT CHECK (lp_molds_per_angle IS NULL OR lp_molds_per_angle > 0),
    lp_angle_qty         SMALLINT CHECK (lp_angle_qty IS NULL OR lp_angle_qty >= 0),
    lp_slot_top          BOOLEAN NOT NULL,
    lp_slot_upmid        BOOLEAN NOT NULL,
    lp_slot_lowmid       BOOLEAN NOT NULL,
    lp_slot_bot          BOOLEAN NOT NULL,

    -- IC 가류기 (3 슬롯)
    ic_molds_per_angle   SMALLINT CHECK (ic_molds_per_angle IS NULL OR ic_molds_per_angle > 0),
    ic_angle_qty         SMALLINT CHECK (ic_angle_qty IS NULL OR ic_angle_qty >= 0),
    ic_slot_top          BOOLEAN NOT NULL,
    ic_slot_mid          BOOLEAN NOT NULL,
    ic_slot_bot          BOOLEAN NOT NULL,

    -- 무결성: 저압/IC 둘 중 하나는 가능해야 함 (모두 X면 Unschedulable — ST-04-2)
    -- 위 제약은 application-level (특정 품번은 zero-slot 의도적)

    updated_at           TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_by           VARCHAR(40) NOT NULL
);

CREATE INDEX idx_vc_constraint_lp_slots ON master.vc_constraint(lp_slot_top, lp_slot_upmid, lp_slot_lowmid, lp_slot_bot);
CREATE INDEX idx_vc_constraint_ic_slots ON master.vc_constraint(ic_slot_top, ic_slot_mid, ic_slot_bot);

-- 마스터 변경 LISTEN/NOTIFY (캐시 무효화용 — TK-04-1-2에서 사용)
CREATE OR REPLACE FUNCTION notify_vc_constraint_change() RETURNS TRIGGER AS $$
BEGIN
    PERFORM pg_notify('vc_constraint_changed',
        COALESCE(NEW.hose_id, OLD.hose_id));
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_vc_constraint_notify
AFTER INSERT OR UPDATE OR DELETE ON master.vc_constraint
FOR EACH ROW EXECUTE FUNCTION notify_vc_constraint_change();

-- 권한 (SAD §6.1.1)
GRANT SELECT ON master.vc_constraint TO app_user;
GRANT INSERT, UPDATE ON master.vc_constraint TO master_admin;

COMMENT ON TABLE master.vc_constraint
IS 'EP-04: 47품번 성형 슬롯 적합성 (REF-09 G~J·M~O 컬럼 매핑)';
```

#### B. `MachineType.java` + `SlotPosition.java`

```java
package com.scheduling.master.vc;

public enum MachineType { LP, IC }

public enum SlotPosition {
    LP_TOP, LP_UPMID, LP_LOWMID, LP_BOT,
    IC_TOP, IC_MID, IC_BOT;

    public MachineType machineType() {
        return name().startsWith("LP_") ? MachineType.LP : MachineType.IC;
    }

    /** 같은 머신 유형의 슬롯만 반환 */
    public static SlotPosition[] of(MachineType type) {
        return switch (type) {
            case LP -> new SlotPosition[]{LP_TOP, LP_UPMID, LP_LOWMID, LP_BOT};
            case IC -> new SlotPosition[]{IC_TOP, IC_MID, IC_BOT};
        };
    }
}
```

#### C. `VcConstraint.java` — JPA 엔티티

```java
package com.scheduling.master.vc;

import jakarta.persistence.*;
import lombok.*;

import java.time.Instant;

@Entity
@Table(schema = "master", name = "vc_constraint")
@Getter @NoArgsConstructor @AllArgsConstructor @Builder
public class VcConstraint {

    @Id @Column(name = "hose_id", length = 40)
    private String hoseId;

    @Column(nullable = false) private int moldQty;
    @Column(nullable = false) private short compositeCount;     // 1·2·3·6

    // 저압
    private Short lpMoldsPerAngle;
    private Short lpAngleQty;
    @Column(nullable = false) private boolean lpSlotTop;
    @Column(nullable = false) private boolean lpSlotUpmid;
    @Column(nullable = false) private boolean lpSlotLowmid;
    @Column(nullable = false) private boolean lpSlotBot;

    // IC
    private Short icMoldsPerAngle;
    private Short icAngleQty;
    @Column(nullable = false) private boolean icSlotTop;
    @Column(nullable = false) private boolean icSlotMid;
    @Column(nullable = false) private boolean icSlotBot;

    private Instant updatedAt;
    private String updatedBy;

    /** 슬롯 위치별 O/X 조회 */
    public boolean isEligibleFor(SlotPosition pos) {
        return switch (pos) {
            case LP_TOP -> lpSlotTop;
            case LP_UPMID -> lpSlotUpmid;
            case LP_LOWMID -> lpSlotLowmid;
            case LP_BOT -> lpSlotBot;
            case IC_TOP -> icSlotTop;
            case IC_MID -> icSlotMid;
            case IC_BOT -> icSlotBot;
        };
    }

    /** 모든 슬롯이 X면 Unschedulable (BR-V11) */
    public boolean isUnschedulable() {
        return !lpSlotTop && !lpSlotUpmid && !lpSlotLowmid && !lpSlotBot
            && !icSlotTop && !icSlotMid && !icSlotBot;
    }

    /** 가류기 유형별 yield 단위 (TK-05-2-1에서 사용) */
    public int yieldPerRotation(MachineType type) {
        return switch (type) {
            case LP -> compositeCount * (lpMoldsPerAngle != null ? lpMoldsPerAngle : 0);
            case IC -> compositeCount * (icMoldsPerAngle != null ? icMoldsPerAngle : 0);
        };
    }
}
```

#### D. `VcConstraintRepository.java`

```java
package com.scheduling.master.vc;

import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Query;
import org.springframework.stereotype.Repository;

import java.util.List;

@Repository
public interface VcConstraintRepository extends JpaRepository<VcConstraint, String> {

    @Query("SELECT v FROM VcConstraint v WHERE v.hoseId IN :hoseIds")
    List<VcConstraint> findAllByHoseIds(@Param("hoseIds") List<String> hoseIds);

    /** Unschedulable 품번 — TK-04-2-1에서 사용 */
    @Query("""
        SELECT v FROM VcConstraint v
        WHERE NOT v.lpSlotTop AND NOT v.lpSlotUpmid AND NOT v.lpSlotLowmid AND NOT v.lpSlotBot
          AND NOT v.icSlotTop AND NOT v.icSlotMid AND NOT v.icSlotBot
        """)
    List<VcConstraint> findUnschedulable();
}
```

## :memo: Implementation Notes

- Hibernate 6.x `BOOLEAN` 매핑 자동 — `boolean` primitive 권장 (null 회피)
- `composite_count CHECK IN (1,2,3,6)` — REF-11 도메인 명시
- LISTEN/NOTIFY는 PG-native 패턴 — Redis Pub/Sub 대안은 다중 호스트 시 검토
- `Short` (wrapper) vs `short` (primitive): null 허용 컬럼은 wrapper
