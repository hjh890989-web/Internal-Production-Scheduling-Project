```java
package com.scheduling.master.vc;

import com.fasterxml.jackson.databind.ObjectMapper;
import org.junit.jupiter.api.*;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.core.io.ClassPathResource;
import org.springframework.test.context.jdbc.Sql;
import org.testcontainers.junit.jupiter.*;

import static org.assertj.core.api.Assertions.*;

@SpringBootTest
@Testcontainers
@Sql("classpath:datasets/DS-VC-CONSTRAINT-47/master_seed.sql")
class SlotMatrixAccuracyIT {

    @Autowired SlotCompatibilityMatrixService service;
    @Autowired ObjectMapper json;

    /**
     * TC-VC-002: 100건 슬롯 배정 시도 → 100% 정확 분류
     */
    @Test
    void slot_violations_100_cases_zero_errors() throws Exception {
        var cases = loadCases();
        var matrix = service.current();

        int correct = 0;
        int violations = 0;
        var errors = new StringBuilder();

        for (var c : cases) {
            boolean actual = matrix.isEligible(c.hoseId, SlotPosition.valueOf(c.slotPosition));
            if (actual == c.expectedEligible) {
                correct++;
            } else {
                violations++;
                errors.append("Case %s: hose=%s slot=%s expected=%s actual=%s%n"
                    .formatted(c.caseId, c.hoseId, c.slotPosition, c.expectedEligible, actual));
            }
        }

        assertThat(violations)
            .as("Slot O/X violations should be ZERO (TC-VC-002):%n%s", errors)
            .isZero();
        assertThat(correct).isEqualTo(100);
    }

    @Test
    void matrix_correctly_indexes_47_products() {
        var matrix = service.current();
        // REF-09 정확히 47품번
        assertThat(matrix.byHose()).hasSize(47);
        // 모든 슬롯 위치에 최소 1개 이상의 가능 품번
        for (SlotPosition slot : SlotPosition.values()) {
            assertThat(matrix.eligibleHoseIdsFor(slot))
                .as("Slot %s should have ≥1 eligible product", slot)
                .isNotEmpty();
        }
    }
}
```

#### E. `generate_slot_test_dataset.py` (합성 자동화)

```python
"""
REF-09 47품번 마스터에서 100건 슬롯 배정 시나리오 자동 생성.
각 (품번, 슬롯) 쌍의 실제 O/X를 정답으로 사용.
"""
import openpyxl
import json
from pathlib import Path

REF = Path("Phase 1/2.Raw Materials/Vulcanization/성형공정_제약조건.xlsx")
OUT = Path("backend/scheduling-engine/src/test/resources/datasets/DS-VC-CONSTRAINT-47")

wb = openpyxl.load_workbook(REF, data_only=True)
ws = wb["Sheet1 (2)"]

# REF-09 컬럼 매핑 (V1.4 검증 후)
SLOT_COL = {"LP_TOP": 7, "LP_UPMID": 8, "LP_LOWMID": 9, "LP_BOT": 10,
            "IC_TOP": 13, "IC_MID": 14, "IC_BOT": 15}

cases = []
import random
random.seed(42)

for row in range(3, ws.max_row + 1):
    hose_id = ws.cell(row, 1).value
    if not hose_id: continue

    for slot_name, col in SLOT_COL.items():
        val = ws.cell(row, col).value
        eligible = (val == 'o' or val == 'O' or val == 'O ')
        # 100건 추출 — 적합 50 + 비적합 50
        cases.append({
            "case_id": f"SV-{len(cases)+1:03d}",
            "hose_id": hose_id,
            "slot_position": slot_name,
            "expected_eligible": eligible
        })

# 100건 샘플링 (각 카테고리 균등)
eligible_cases = [c for c in cases if c["expected_eligible"]][:50]
ineligible_cases = [c for c in cases if not c["expected_eligible"]][:50]
final_cases = eligible_cases + ineligible_cases

OUT.mkdir(parents=True, exist_ok=True)
(OUT / "slot_violation_cases.json").write_text(json.dumps(final_cases, indent=2, ensure_ascii=False))

print(f"Generated {len(final_cases)} cases at {OUT}/slot_violation_cases.json")
```
