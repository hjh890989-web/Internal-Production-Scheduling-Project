### 핵심 로직

```java
@Service @RequiredArgsConstructor
public class UnschedulableFilterService {

    private final SlotCompatibilityMatrixService matrixService;

    public FilterResult separate(List<Order> orders) {
        var matrix = matrixService.current();
        var unschedSet = matrix.unschedulableHoseIds();

        var schedulable = new ArrayList<Order>();
        var unschedulable = new ArrayList<Order>();
        for (Order o : orders) {
            if (unschedSet.contains(o.getHoseId())) {
                unschedulable.add(o);
            } else {
                schedulable.add(o);
            }
        }
        return new FilterResult(schedulable, unschedulable, matrix.version());
    }
}

public record FilterResult(
    List<Order> schedulable,
    List<Order> unschedulable,
    int matrixVersion
) {
    public int unschedulableCount() { return unschedulable.size(); }
    public boolean hasUnschedulable() { return !unschedulable.isEmpty(); }
}
```
