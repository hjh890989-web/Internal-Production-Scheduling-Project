### 핵심 로직

```java
@SpringBootTest
@Testcontainers
@Sql("classpath:datasets/DS-VC-CONSTRAINT-47/master_seed.sql")
class UnschedulableFilterIT {

    @Autowired UnschedulableFilterService filter;
    @Autowired UnschedulableReportGenerator reportGen;
    @Autowired ObjectMapper json;

    @Test
    void identifies_ref09_zero_slot_products() throws Exception {
        var expected = loadExpected("unschedulable_expected.json");  // ["7X375-H0020", ...]

        // 47품번 모두를 Order로 입력
        var allOrders = allRef09Orders();

        var result = filter.separate(allOrders);

        assertThat(result.unschedulable())
            .extracting(Order::getHoseId)
            .containsExactlyInAnyOrderElementsOf(expected);
    }

    @Test
    void excel_report_contains_unschedulable_rows(@TempDir Path tempDir) throws Exception {
        var result = filter.separate(allRef09Orders());
        var excelPath = reportGen.generateExcel(result, tempDir);

        assertThat(Files.exists(excelPath)).isTrue();
        // POI로 재읽기 → row 수 검증
        try (var wb = new XSSFWorkbook(Files.newInputStream(excelPath))) {
            int dataRows = wb.getSheetAt(0).getLastRowNum();
            assertThat(dataRows).isEqualTo(result.unschedulableCount());
        }
    }

    @Test
    void empty_unschedulable_does_not_generate_file(@TempDir Path tempDir) {
        var emptyResult = new FilterResult(allRef09Orders(), List.of(), 1);
        // 빈 unschedulable → 파일 미생성
    }
}
```
