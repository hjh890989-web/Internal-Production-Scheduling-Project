### 핵심 로직

```java
@Component @RequiredArgsConstructor
public class UnschedulableReportGenerator {

    public Path generateExcel(FilterResult result, Path outDir) throws IOException {
        var wb = new XSSFWorkbook();
        var sheet = wb.createSheet("Unschedulable");
        // 헤더
        var header = sheet.createRow(0);
        header.createCell(0).setCellValue("품번");
        header.createCell(1).setCellValue("납기일");
        header.createCell(2).setCellValue("수량");
        header.createCell(3).setCellValue("거래처");
        header.createCell(4).setCellValue("사유");
        header.createCell(5).setCellValue("권장 조치");

        int rowIdx = 1;
        for (Order o : result.unschedulable()) {
            var row = sheet.createRow(rowIdx++);
            row.createCell(0).setCellValue(o.getHoseId());
            row.createCell(1).setCellValue(o.getDeliveryDate().toString());
            row.createCell(2).setCellValue(o.getQty());
            row.createCell(3).setCellValue(o.getCustomer());
            row.createCell(4).setCellValue("모든 슬롯 X (BR-V11)");
            row.createCell(5).setCellValue("외주 의뢰 또는 재고 대응");
        }

        Path out = outDir.resolve("unschedulable_" + Instant.now().toEpochMilli() + ".xlsx");
        try (var os = Files.newOutputStream(out)) { wb.write(os); }
        wb.close();
        return out;
    }
}

@Component @RequiredArgsConstructor
public class UnschedulableNotificationPublisher {

    private final NotificationService notificationService;

    public void notifyPlanner(FilterResult result, Path excelPath) {
        if (!result.hasUnschedulable()) return;

        notificationService.sendInApp(
            "PLANNER",
            Severity.NORMAL,
            "Unschedulable 품번 %d건 분리 — 외주·재고 대응 필요"
                .formatted(result.unschedulableCount()),
            Map.of("excelReportUrl", excelPath.toString())
        );
    }
}
```
