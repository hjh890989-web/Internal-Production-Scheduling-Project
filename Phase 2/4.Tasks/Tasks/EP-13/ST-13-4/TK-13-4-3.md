### 핵심 로직

```java
@SpringBootTest @Testcontainers
class LockOverrideAuditIT {

    @Autowired VcLockOverrideService overrideService;
    @Autowired AuditLogRepository auditRepo;
    @Autowired VcScheduleRepository scheduleRepo;

    @Test
    void override_creates_two_audit_rows_delete_and_insert() {
        var schedule = createScheduledOnDate(LocalDate.of(2026, 3, 2));
        long before = auditRepo.countVcLogs();

        var validReason = "긴급 변경 — 고객 28912-2U000 100개 추가 수주 대응으로 LP-01 셋팅 즉시 교체";

        withPlannerRole("planner1", () -> {
            overrideService.override(schedule.getId(),
                new OverrideRequest("29689-2U000", validReason));
        });

        long after = auditRepo.countVcLogs();
        assertThat(after - before).isGreaterThanOrEqualTo(2);     // delete + insert
    }

    @Test
    void override_audit_reason_contains_user_input() {
        var schedule = createScheduledOnDate(LocalDate.of(2026, 3, 2));
        var userReason = "긴급 변경 — 고객 추가 수주 100개 대응 LP-01 즉시 교체";

        withPlannerRole("planner1", () ->
            overrideService.override(schedule.getId(),
                new OverrideRequest("29689-2U000", userReason)));

        var insertLog = auditRepo.findLatestVcLogByOperation("INSERT");
        assertThat(insertLog.getReason())
            .as("override 사용자 사유 audit 기록")
            .contains(userReason.substring(0, 20));    // 첫 20자 match
    }

    @Test
    void override_audit_actor_is_planner() {
        var schedule = createScheduledOnDate(LocalDate.of(2026, 3, 2));

        withPlannerRole("planner-emergency", () ->
            overrideService.override(schedule.getId(),
                new OverrideRequest("29689-2U000",
                    "긴급 — 30자 이상의 의미 있는 사유 텍스트")));

        var log = auditRepo.findLatestVcLog();
        assertThat(log.getActor()).isEqualTo("planner-emergency");
    }

    @Test
    void invalid_short_reason_does_not_create_audit() {
        var schedule = createScheduledOnDate(LocalDate.of(2026, 3, 2));
        long before = auditRepo.countVcLogs();

        assertThatThrownBy(() ->
            withPlannerRole("planner1", () ->
                overrideService.override(schedule.getId(),
                    new OverrideRequest("29689-2U000", "짧음"))))
            .isInstanceOf(ReasonValidationException.class);

        // validation fail 시 audit 미생성 (transaction rollback)
        long after = auditRepo.countVcLogs();
        assertThat(after).isEqualTo(before);
    }
}
```
