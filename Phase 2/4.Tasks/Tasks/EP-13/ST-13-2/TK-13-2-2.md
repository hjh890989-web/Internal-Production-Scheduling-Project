### 핵심 로직

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-VC-LOCK-1WEEK/master_seed.sql"})
class IntraDayLockRegressionIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired ScenarioLoader loader;

    @Test
    void week_horizon_no_intraday_lock_violation() {
        var scenarios = loader.load("DS-VC-LOCK-1WEEK/scenarios.csv");
        var ctx = AllocationContext.fromOrders(scenarios, LocalDate.of(2026, 3, 2), 7);
        var result = allocator.allocate(ctx);

        // (machine, slot, date) 슬롯별 unique hose 수
        Map<Tuple3<String, Integer, LocalDate>, Set<String>> hosesPerSlot = new HashMap<>();
        for (var row : result.schedule()) {
            hosesPerSlot
                .computeIfAbsent(
                    Tuple.of(row.machineId(), row.slotNumber(), row.productionDate()),
                    k -> new HashSet<>())
                .add(row.hoseId());
        }

        var violations = hosesPerSlot.entrySet().stream()
            .filter(e -> e.getValue().size() > 1)
            .toList();

        assertThat(violations)
            .as("BR-V07 일중 교체 0건 (1주 회귀)")
            .isEmpty();
    }

    @Test
    void intraday_swap_attempt_creates_conflict_with_correct_category() {
        // 의도적 충돌: 같은 (LP-01, 슬롯 1, 3/2)에 2 hose 배치 시도
        var ctx = loader.intraDaySwapAttempt();
        var result = allocator.allocate(ctx);

        assertThat(result.conflicts())
            .anyMatch(c -> c.reason().contains("BR-V07") || c.reason().contains("일중"));
    }

    @Test
    void same_hose_multi_rotation_same_slot_allowed() {
        // 같은 hose가 같은 슬롯 다른 회전 → BR-V07 위반 아님
        var ctx = loader.sameHoseMultiRotation();
        var result = allocator.allocate(ctx);

        var slotRows = result.schedule().stream()
            .filter(r -> r.machineId().equals("LP-01") && r.slotNumber() == 1 &&
                          r.productionDate().equals(LocalDate.of(2026, 3, 2)))
            .toList();

        assertThat(slotRows).hasSize(2);   // 주간·야간 회전
        assertThat(slotRows.stream().map(r -> r.hoseId()).distinct()).hasSize(1);
    }

    @Test
    void cross_machine_no_lock() {
        // LP-01, LP-02 각각 동일 슬롯 다른 hose — BR-V07 위반 아님
        var ctx = loader.crossMachineSameSlot();
        var result = allocator.allocate(ctx);

        // 위반 row 0건 검증 (위 week test와 동일 로직)
        var violations = countViolations(result);
        assertThat(violations).isZero();
    }

    @Test
    void next_business_day_swap_allowed() {
        // 같은 (LP-01, 슬롯 1) 다른 날짜 — 일말 경계 후 교체 가능 (BR-V07 일중만 차단)
        var ctx = loader.nextBusinessDaySwap();
        var result = allocator.allocate(ctx);

        var slotRows = result.schedule().stream()
            .filter(r -> r.machineId().equals("LP-01") && r.slotNumber() == 1)
            .toList();

        // 3/2(월) hose A, 3/3(화) hose B — 모두 OK
        assertThat(slotRows.stream().map(r -> r.hoseId()).distinct().count()).isEqualTo(2);
    }
}
```
