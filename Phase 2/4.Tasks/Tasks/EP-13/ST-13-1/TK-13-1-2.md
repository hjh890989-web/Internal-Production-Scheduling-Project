### Flyway `V4_4_5__create_intraday_violations_view.sql`

```sql
-- BR-V07 위반 row 검출 view (V4_5 마이그레이션 사전 검증)
CREATE OR REPLACE VIEW vc_schedule_intraday_violations AS
SELECT
    machine_id,
    slot_position,
    production_date,
    COUNT(*) AS violation_count,
    array_agg(DISTINCT hose_id) AS conflicting_hoses,
    array_agg(id ORDER BY created_at) AS row_ids
FROM vc_schedule
GROUP BY machine_id, slot_position, production_date
HAVING COUNT(*) > 1;

COMMENT ON VIEW vc_schedule_intraday_violations IS
    'EP-13: BR-V07 당일 락 위반 row 검출 — V4_5 마이그레이션 사전 점검';

GRANT SELECT ON vc_schedule_intraday_violations TO app_user, master_admin;
```

### `IntraDayViolationCheckJob.java`

```java
@Component @RequiredArgsConstructor @Slf4j
public class IntraDayViolationCheckJob {

    private final JdbcTemplate jdbc;

    /**
     * Application startup 시 자동 점검 — 마이그레이션 후 데이터 정합 검증.
     */
    @PostConstruct
    public void checkOnStartup() {
        try {
            int count = jdbc.queryForObject(
                "SELECT COUNT(*) FROM vc_schedule_intraday_violations",
                Integer.class);

            if (count > 0) {
                log.error("BR-V07 당일 락 위반 {} 건 발견 — vc_schedule_intraday_violations view 확인 + 정리 필요", count);
                // production 환경에서는 application startup fail 옵션 검토
            } else {
                log.info("BR-V07 당일 락 위반 0건 — 정상");
            }
        } catch (Exception e) {
            // view 미존재 (구버전) — pass
            log.warn("intraday violation 점검 skip — view 미존재 가능", e);
        }
    }

    public List<ViolationReport> findViolations() {
        return jdbc.query(
            "SELECT machine_id, slot_position, production_date, " +
            "       violation_count, conflicting_hoses " +
            "FROM vc_schedule_intraday_violations",
            (rs, rowNum) -> new ViolationReport(
                rs.getString("machine_id"),
                rs.getInt("slot_position"),
                rs.getDate("production_date").toLocalDate(),
                rs.getInt("violation_count"),
                Arrays.asList((String[]) rs.getArray("conflicting_hoses").getArray())
            ));
    }

    public record ViolationReport(
        String machineId,
        int slotPosition,
        LocalDate productionDate,
        int violationCount,
        List<String> conflictingHoses
    ) {}
}
```

3. row N건: 본 가이드 따라 정리

1. 운영팀에 violation row 보고 (machine·slot·date·conflicting_hoses)
2. 현장 확인: 어느 hose가 실제 배치되었는지

   ```sql
   -- 1차 백업
   CREATE TABLE backup_vc_schedule_violations AS
   SELECT * FROM vc_schedule
   WHERE (machine_id, slot_position, production_date) IN
       (SELECT machine_id, slot_position, production_date
        FROM vc_schedule_intraday_violations);

   -- 2차 잘못된 row 삭제 (운영팀 합의 후)
   DELETE FROM vc_schedule WHERE id IN (...);
   ```
4. 재검증: `SELECT COUNT(*) FROM vc_schedule_intraday_violations` → 0
5. V4_5 마이그레이션 실행

## 응급 (운영 환경)
