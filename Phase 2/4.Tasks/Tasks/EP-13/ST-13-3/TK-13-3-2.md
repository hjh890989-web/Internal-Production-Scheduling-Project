### 핵심 로직

```java
@SpringBootTest @Testcontainers
class BoundaryKeyAuditIT {

    @Autowired BusinessDayBoundaryFormatter formatter;
    @Autowired AuditLogRepository auditRepo;
    @Autowired VcLockOverrideService overrideService;

    @Test
    void override_audit_includes_boundary_key() {
        var schedule = createCandidateForDate(LocalDate.of(2026, 3, 2));   // 월

        withPlannerRole("planner1", () ->
            overrideService.override(schedule.getId(),
                "긴급 변경 — 다음 영업일 시작 전 회복"));

        var latest = auditRepo.findLatestVcLogForTarget(schedule.getId());

        // audit.reason 필드 검증
        assertThat(latest.getReason())
            .as("override 사유 audit")
            .contains("긴급 변경");

        // 영업일 경계 키 (DO-04 출력 정합)
        String expectedKey = formatter.formatBoundary(LocalDate.of(2026, 3, 2));
        assertThat(latest.getReason())
            .as("audit reason에 영업일 경계 키 포함")
            .containsAnyOf(expectedKey, "_END", "2026-03-03 시작 전");
    }

    @Test
    void boundary_key_after_holiday() {
        // 2/13(금) override → reason에 "2/19 시작 전" 포함 (설날 후)
        var schedule = createCandidateForDate(LocalDate.of(2026, 2, 13));

        withPlannerRole("planner1", () ->
            overrideService.override(schedule.getId(), "긴급"));

        var latest = auditRepo.findLatestVcLogForTarget(schedule.getId());
        assertThat(latest.getReason()).contains("2026-02-19");
    }
}
```
