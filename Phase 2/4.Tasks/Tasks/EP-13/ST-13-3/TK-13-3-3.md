### 핵심 로직

```java
@ExtendWith(MockitoExtension.class)
class BusinessDayBoundaryFormatterTest {

    @Mock WorkingCalendarService calendar;
    @InjectMocks BusinessDayBoundaryFormatter formatter;

    @Test
    void format_boundary_iso_with_end_suffix() {
        var key = formatter.formatBoundary(LocalDate.of(2026, 3, 2));
        assertThat(key).isEqualTo("2026-03-02_END");
    }

    @ParameterizedTest
    @CsvSource({
        "2026-03-02, 2026-03-03",   // 월 → 화
        "2026-03-06, 2026-03-09",   // 금 → 다음 주 월
        "2026-02-13, 2026-02-19",   // 설날 직전 → 설날 후 (2/16~18 휴일)
        "2026-09-23, 2026-09-29",   // 추석 직전 → 추석 후 (9/24~28 휴일)
        "2026-12-31, 2027-01-02"    // 신정 직전
    })
    void next_business_day_handles_holidays(String input, String expected) {
        var date = LocalDate.parse(input);
        var expDate = LocalDate.parse(expected);
        when(calendar.addWorkingDays(date, 1)).thenReturn(expDate);

        assertThat(formatter.nextBusinessDay(date)).isEqualTo(expDate);
    }

    @ParameterizedTest
    @CsvSource({
        "2026-03-02, 2026-03-03, '2026-03-02 ~ 2026-03-03 시작 전'",
        "2026-02-13, 2026-02-19, '2026-02-13 ~ 2026-02-19 시작 전'"
    })
    void format_lock_period_human_readable(String start, String next, String expected) {
        var startDate = LocalDate.parse(start);
        var nextDate = LocalDate.parse(next);
        when(calendar.addWorkingDays(startDate, 1)).thenReturn(nextDate);

        assertThat(formatter.formatLockPeriod(startDate)).isEqualTo(expected);
    }

    @Test
    void five_weekday_full_matrix() {
        // 월~금 5일 모두 boundary 키 정확
        for (int dayOffset = 0; dayOffset < 5; dayOffset++) {
            var date = LocalDate.of(2026, 3, 2).plusDays(dayOffset);
            assertThat(formatter.formatBoundary(date))
                .as("%s boundary", date)
                .endsWith("_END");
        }
    }
}
```
