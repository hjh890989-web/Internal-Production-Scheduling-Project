### Flyway `V3_3__create_setting_group.sql`

```sql
CREATE TABLE master.setting_group (
    group_number     INTEGER PRIMARY KEY CHECK (group_number BETWEEN 1 AND 8),
    group_name       VARCHAR(40) NOT NULL,
    description      TEXT,
    setup_minutes    INTEGER NOT NULL DEFAULT 30 CHECK (setup_minutes >= 0),
    is_active        BOOLEAN NOT NULL DEFAULT TRUE,
    updated_at       TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_by       VARCHAR(40) NOT NULL DEFAULT 'seed'
);

CREATE TABLE master.product_setting_group (
    hose_id          VARCHAR(40) NOT NULL,
    group_number     INTEGER NOT NULL,
    is_primary       BOOLEAN NOT NULL DEFAULT FALSE,           -- 같은 품번 여러 그룹 호환 시 primary
    notes            TEXT,
    updated_at       TIMESTAMPTZ NOT NULL DEFAULT now(),
    PRIMARY KEY (hose_id, group_number),
    FOREIGN KEY (group_number) REFERENCES master.setting_group(group_number)
);

CREATE INDEX idx_psg_group ON master.product_setting_group(group_number);
CREATE INDEX idx_psg_primary ON master.product_setting_group(hose_id) WHERE is_primary = TRUE;

GRANT SELECT ON master.setting_group, master.product_setting_group TO app_user;
GRANT INSERT, UPDATE, DELETE ON master.setting_group, master.product_setting_group TO master_admin;

COMMENT ON TABLE master.setting_group IS 'EP-09: 압출 셋팅 번호 1~8 (BR-E06·E07)';
COMMENT ON COLUMN master.setting_group.setup_minutes IS '셋업 변경 시 소요 시간 (분) — 회귀 검증용';
COMMENT ON TABLE master.product_setting_group IS 'EP-09: 품번 ↔ 셋팅 그룹 M:N 매핑';

-- LISTEN/NOTIFY (TK-21-2-3 패턴)
CREATE OR REPLACE FUNCTION master.notify_setting_group_change()
RETURNS TRIGGER AS $$
BEGIN
    PERFORM pg_notify('setting_group_change',
        json_build_object('group_number',
            COALESCE(NEW.group_number, OLD.group_number), 'op', TG_OP)::text);
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_setting_group_notify
    AFTER INSERT OR UPDATE OR DELETE ON master.setting_group
    FOR EACH ROW EXECUTE FUNCTION master.notify_setting_group_change();

CREATE TRIGGER trg_product_setting_group_notify
    AFTER INSERT OR UPDATE OR DELETE ON master.product_setting_group
    FOR EACH ROW EXECUTE FUNCTION master.notify_setting_group_change();
```

### `SettingGroup.java` + `ProductSettingGroup.java`

```java
@Entity @Table(schema="master", name="setting_group")
@Data @NoArgsConstructor @AllArgsConstructor @Builder
public class SettingGroup {
    @Id @Column(name="group_number") private Integer groupNumber;
    @Column(name="group_name") private String groupName;
    @Column(name="description") private String description;
    @Column(name="setup_minutes") private Integer setupMinutes;
    @Column(name="is_active") private Boolean isActive;
}

@Entity @Table(schema="master", name="product_setting_group")
@IdClass(ProductSettingGroupId.class)
@Data @NoArgsConstructor @AllArgsConstructor @Builder
public class ProductSettingGroup {
    @Id @Column(name="hose_id") private String hoseId;
    @Id @Column(name="group_number") private Integer groupNumber;
    @Column(name="is_primary") private Boolean isPrimary;
    @Column(name="notes") private String notes;
}

public record ProductSettingGroupId(String hoseId, Integer groupNumber) implements Serializable {}
```

### `SettingGroupCache.java`

```java
@Component @RequiredArgsConstructor
public class SettingGroupCache {

    private final ProductSettingGroupRepository psgRepo;

    private final Cache<String, List<Integer>> hoseToGroups = Caffeine.newBuilder()
        .maximumSize(100)
        .expireAfterWrite(Duration.ofHours(1)).build();

    /**
     * 품번 → 호환 셋팅 그룹 리스트 (primary 우선 정렬).
     */
    public List<Integer> groupsForHose(String hoseId) {
        var cached = hoseToGroups.getIfPresent(hoseId);
        if (cached != null) return cached;

        var groups = psgRepo.findByHoseIdOrderByIsPrimaryDesc(hoseId).stream()
            .map(ProductSettingGroup::getGroupNumber).toList();
        hoseToGroups.put(hoseId, groups);
        return groups;
    }

    public Integer primaryGroup(String hoseId) {
        return groupsForHose(hoseId).stream().findFirst()
            .orElseThrow(() -> new MasterNotFoundException("setting_group=" + hoseId));
    }

    public void invalidate(String hoseId) { hoseToGroups.invalidate(hoseId); }
    public void invalidateAll() { hoseToGroups.invalidateAll(); }
}
```
