### `SettingGroupRegressionIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-EX-4WEEK/master_seed.sql"})
class SettingGroupRegressionIT {

    @Autowired SettingGroupAllocator allocator;
    @Autowired SettingGroupCache settingCache;
    @Autowired EventLoader loader;

    private static final LocalDate START = LocalDate.of(2026, 3, 2);    // 월
    private static final int HORIZON_DAYS = 28;                          // 4주

    @Test
    void four_week_horizon_zero_setup_within_shift() {
        // 1) 4주 VC 확정 이벤트 로드 → 압출 후보 ≈ 1,500
        var candidates = loader.candidatesFor4Week("DS-EX-4WEEK/vc_confirmed_events_28day.csv");

        var ctx = ExAllocationContext.builder()
            .candidates(candidates)
            .workingDays(workingDaysInRange(START, START.plusDays(HORIZON_DAYS)))
            .build();

        // 2) Allocator 실행
        var result = allocator.allocate(ctx);

        // 3) (date, machine, shift) 슬롯별 셋팅 그룹 집계
        Map<ShiftSlot, Set<Integer>> groupsPerSlot = new HashMap<>();
        for (var row : result.schedule()) {
            var slot = row.shiftSlot();
            int group = settingCache.primaryGroup(row.hoseId());
            groupsPerSlot.computeIfAbsent(slot, k -> new HashSet<>()).add(group);
        }

        // 4) BR-E06 핵심: 각 slot당 단일 그룹만 (셋업 0건)
        var setupViolations = groupsPerSlot.entrySet().stream()
            .filter(e -> e.getValue().size() > 1)
            .toList();

        assertThat(setupViolations)
            .as("BR-E06 위반: %d slot에 다중 셋팅 그룹 발견", setupViolations.size())
            .isEmpty();

        // 5) 총 slot 수 검증 — 28일 × 4 머신 × 4 shift = 448 (working days 기준)
        assertThat(groupsPerSlot.size()).isLessThanOrEqualTo(28 * 4 * 4);
    }

    @Test
    void multi_product_same_group_concurrent_in_shift() {
        var candidates = loader.candidatesFor4Week("DS-EX-4WEEK/vc_confirmed_events_28day.csv");
        var ctx = buildCtx(candidates);

        var result = allocator.allocate(ctx);

        // 같은 (slot, group)에 다수 품번 배치 가능 (BR-E07)
        Map<Tuple2<ShiftSlot, Integer>, Set<String>> productsByGroup = new HashMap<>();
        for (var row : result.schedule()) {
            int group = settingCache.primaryGroup(row.hoseId());
            productsByGroup
                .computeIfAbsent(Tuple.of(row.shiftSlot(), group), k -> new HashSet<>())
                .add(row.hoseId());
        }

        // 적어도 일부 slot에 다수 품번 (셋팅 그룹 1에 28912·29689 동시 가능)
        long multiProductSlots = productsByGroup.values().stream()
            .filter(set -> set.size() > 1).count();

        assertThat(multiProductSlots).isGreaterThan(0);
    }

    @Test
    void shift_capacity_not_exceeded() {
        // 각 slot의 누적 actualMin ≤ effective_min (180)
        var candidates = loader.candidatesFor4Week("DS-EX-4WEEK/vc_confirmed_events_28day.csv");
        var result = allocator.allocate(buildCtx(candidates));

        Map<ShiftSlot, Integer> minutesPerSlot = new HashMap<>();
        for (var row : result.schedule()) {
            minutesPerSlot.merge(row.shiftSlot(), row.actualMin(), Integer::sum);
        }

        assertThat(minutesPerSlot.values())
            .as("shift effective_min 180 초과 0건")
            .allMatch(m -> m <= 180);
    }
}
```

### `ZeroSetupKpiIT.java`

```java
@SpringBootTest @Testcontainers
class ZeroSetupKpiIT {

    @Autowired SettingGroupAllocator allocator;
    @Autowired SettingGroupCache settingCache;

    @Test
    void ns_s08_zero_setup_kpi_100_percent() {
        var ctx = loader.full4WeekContext();
        var result = allocator.allocate(ctx);

        // 모든 slot에서 셋업 0건 → KPI 100%
        long totalSlots = countSlots(result);
        long zeroSetupSlots = countZeroSetupSlots(result, settingCache);

        double kpi = (double) zeroSetupSlots / totalSlots;
        assertThat(kpi)
            .as("NS-S08 zero-setup KPI")
            .isEqualTo(1.0);
    }
}
```
