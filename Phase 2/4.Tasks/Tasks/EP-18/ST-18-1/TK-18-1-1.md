### `RankingScore.java`

```java
public record RankingScore(
    double deadlineMargin,        // 평균 (deadline - production_date) days
    int setupCount,               // 셋팅 그룹 변경 수
    double lineBalance,           // 1 / stddev(line usage)
    double totalScore             // 가중 합산
) {
    public static RankingScore of(double margin, int setup, double balance,
                                   double wDeadline, double wSetup, double wBalance) {
        // normalized: margin ↑ good, setup ↓ good, balance ↑ good
        double normalized = margin * wDeadline - setup * wSetup + balance * wBalance;
        return new RankingScore(margin, setup, balance, normalized);
    }
}
```

### `CandidateRankingService.java`

```java
@Service @RequiredArgsConstructor
public class CandidateRankingService {

    private final SettingGroupCache settingCache;

    public RankingScore score(AllocationResult result, AllocationStrategy strategy) {
        double deadlineMargin = computeDeadlineMargin(result);
        int setupCount = computeSetupCount(result);
        double lineBalance = computeLineBalance(result);

        return RankingScore.of(deadlineMargin, setupCount, lineBalance,
            strategy.getWDeadline(), strategy.getWSetup(), strategy.getWBalance());
    }

    private double computeDeadlineMargin(AllocationResult result) {
        if (result.schedule().isEmpty()) return 0;
        return result.schedule().stream()
            .mapToLong(row -> ChronoUnit.DAYS.between(row.productionDate(), row.deadline()))
            .average().orElse(0);
    }

    private int computeSetupCount(AllocationResult result) {
        // (machine, date, shift) 별 unique 셋팅 그룹 수 - 1 합산
        var setupsByShift = result.schedule().stream()
            .collect(Collectors.groupingBy(
                row -> Tuple.of(row.machineId(), row.productionDate(), row.shiftCode()),
                Collectors.mapping(row -> settingCache.primaryGroup(row.hoseId()),
                                    Collectors.toSet())));
        return setupsByShift.values().stream()
            .mapToInt(groups -> Math.max(0, groups.size() - 1))
            .sum();
    }

    private double computeLineBalance(AllocationResult result) {
        var byMachine = result.schedule().stream()
            .collect(Collectors.groupingBy(AllocationRow::machineId, Collectors.counting()));

        if (byMachine.isEmpty()) return 0;

        double mean = byMachine.values().stream().mapToLong(Long::longValue).average().orElse(0);
        double variance = byMachine.values().stream()
            .mapToDouble(c -> Math.pow(c - mean, 2))
            .average().orElse(0);
        double stddev = Math.sqrt(variance);

        return stddev == 0 ? 1.0 : 1.0 / (1.0 + stddev);
    }
}
```

### `MultiCandidateAllocator.java`

```java
@Service @RequiredArgsConstructor
public class MultiCandidateAllocator {

    private final GreedyRotationAllocator allocator;
    private final CandidateRankingService rankingService;

    public List<RankedCandidate> generateCandidates(AllocationContext ctx) {
        // 3 시나리오 실행 (allocator 정책 변경)
        var candidates = new ArrayList<RankedCandidate>();

        for (AllocationStrategy strategy : AllocationStrategy.values()) {
            var strategicCtx = ctx.withStrategy(strategy);   // 정책 파라미터 주입
            var result = allocator.allocate(strategicCtx);
            var score = rankingService.score(result, strategy);
            candidates.add(new RankedCandidate(strategy, result, score));
        }

        candidates.sort(Comparator.comparingDouble(c -> -c.score().totalScore()));
        return candidates;
    }
}

public record RankedCandidate(
    AllocationStrategy strategy,
    AllocationResult result,
    RankingScore score
) {}
```
