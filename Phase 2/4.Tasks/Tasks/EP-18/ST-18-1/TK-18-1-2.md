### `MultiCandidateController.java`

```java
@RestController @RequestMapping("/api/v1/schedule/vc/candidates") @RequiredArgsConstructor
public class MultiCandidateController {

    private final MultiCandidateAllocator allocator;

    @PostMapping
    @PreAuthorize("hasRole('PLANNER')")
    public List<RankedCandidate> generate(@Valid @RequestBody CandidateRequest request) {
        var ctx = AllocationContext.fromRequest(request);
        return allocator.generateCandidates(ctx);
    }
}
```

### `MultiCandidateRegressionIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-VC-RANKING-50/master_seed.sql"})
class MultiCandidateRegressionIT {

    @Autowired MultiCandidateAllocator allocator;
    @Autowired ScenarioLoader loader;

    @Test
    void fifty_scenarios_all_return_three_distinct_candidates() {
        var scenarios = loader.load("DS-VC-RANKING-50/scenarios.csv");
        assertThat(scenarios).hasSize(50);

        for (var ctx : scenarios) {
            var candidates = allocator.generateCandidates(ctx);

            assertThat(candidates)
                .as("scenario: ≥ 3 후보 (TC-XT-001)")
                .hasSize(3);

            // distinct strategy
            assertThat(candidates.stream().map(c -> c.strategy()).distinct().count())
                .as("3 distinct strategy")
                .isEqualTo(3);

            // sorted by score DESC
            for (int i = 1; i < candidates.size(); i++) {
                assertThat(candidates.get(i-1).score().totalScore())
                    .isGreaterThanOrEqualTo(candidates.get(i).score().totalScore());
            }
        }
    }

    @Test
    void candidates_have_distinct_schedule_results() {
        // YIELD_FIRST·SETUP_MIN·BALANCED는 실 schedule 차이 (구성)
        var ctx = loader.singleScenario();
        var candidates = allocator.generateCandidates(ctx);

        var hoseDistributions = candidates.stream()
            .map(c -> c.result().schedule().stream()
                .map(r -> Tuple.of(r.machineId(), r.shiftCode()))
                .collect(Collectors.toSet()))
            .toList();

        // 3 후보의 (machine, shift) distribution이 일정 부분 차이
        assertThat(hoseDistributions.stream().distinct().count())
            .as("후보 간 schedule 차이")
            .isGreaterThanOrEqualTo(2);
    }

    @Test
    void api_endpoint_planner_role_authorized() throws Exception {
        mockMvc.perform(post("/api/v1/schedule/vc/candidates")
            .with(user("planner1").roles("PLANNER"))
            .contentType(MediaType.APPLICATION_JSON)
            .content("{ \"weekStart\": \"2026-03-02\" }"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$").isArray())
            .andExpect(jsonPath("$.length()").value(3));
    }

    @Test
    void user_role_forbidden() throws Exception {
        mockMvc.perform(post("/api/v1/schedule/vc/candidates")
            .with(user("user1").roles("USER"))
            .content("{}"))
            .andExpect(status().isForbidden());
    }
}
```
