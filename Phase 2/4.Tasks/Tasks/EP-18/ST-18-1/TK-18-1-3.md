### `CandidateComparisonPage.tsx`

```tsx
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { Table, Button, Tag, Card } from 'antd';
import { rankingApi } from '@/api/rankingApi';
import { ScoreBreakdown } from './ScoreBreakdown';

export function CandidateComparisonPage() {
  const queryClient = useQueryClient();
  const [request, setRequest] = useState<CandidateRequest>(/* form */);

  const { data: candidates, isLoading } = useQuery({
    queryKey: ['candidates', request],
    queryFn: () => rankingApi.generate(request),
    enabled: !!request,
  });

  const { mutate: selectCandidate } = useMutation({
    mutationFn: rankingApi.selectAndConfirm,
    onSuccess: () => {
      message.success('스케줄이 확정되었습니다');
      queryClient.invalidateQueries({ queryKey: ['vcSchedule'] });
    },
  });

  const columns = [
    { title: 'Strategy', dataIndex: 'strategy', render: (s) => <Tag>{s.label}</Tag> },
    { title: '총 yield', dataIndex: ['result', 'totalYield'] },
    {
      title: 'Conflict',
      dataIndex: ['result', 'conflicts'],
      render: (c) => <Tag color={c.length > 0 ? 'red' : 'green'}>{c.length}</Tag>,
    },
    { title: '점수', dataIndex: ['score', 'totalScore'], render: (s) => s.toFixed(2) },
    {
      title: '세부',
      render: (_, row) => <ScoreBreakdown score={row.score} />,
    },
    {
      title: '선택',
      render: (_, row) => (
        <Button type="primary" onClick={() => selectCandidate(row.result.scheduleId)}>
          이 후보로 확정
        </Button>
      ),
    },
  ];

  return (
    <div>
      <h1>후보 비교</h1>
      <Card title="추천 후보 (점수 순)">
        <Table
          dataSource={candidates}
          columns={columns}
          loading={isLoading}
          pagination={false}
          rowKey={(row) => row.strategy.name}
        />
      </Card>
    </div>
  );
}
```

### `ScoreBreakdown.tsx`

```tsx
import { Tooltip } from 'antd';

interface Props { score: RankingScore; }

export function ScoreBreakdown({ score }: Props) {
  return (
    <div className="score-breakdown">
      <Tooltip title={`납기 여유: ${score.deadlineMargin.toFixed(1)}일`}>
        <div className="bar deadline" style={{ width: `${score.deadlineMargin * 10}%` }} />
      </Tooltip>
      <Tooltip title={`셋업 변경: ${score.setupCount}회`}>
        <div className="bar setup" style={{ width: `${(20 - score.setupCount) * 5}%` }} />
      </Tooltip>
      <Tooltip title={`라인 균형: ${score.lineBalance.toFixed(2)}`}>
        <div className="bar balance" style={{ width: `${score.lineBalance * 100}%` }} />
      </Tooltip>
    </div>
  );
}
```
