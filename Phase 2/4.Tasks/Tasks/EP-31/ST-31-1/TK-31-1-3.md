### `docker-compose.yml` grafana service

```yaml
grafana:
  image: grafana/grafana:10.4.0
  volumes:
    - ./monitoring/grafana/provisioning:/etc/grafana/provisioning:ro
    - ./monitoring/grafana/dashboards:/var/lib/grafana/dashboards:ro
    - grafana-data:/var/lib/grafana
  environment:
    GF_AUTH_GENERIC_OAUTH_ENABLED: "true"
    GF_AUTH_GENERIC_OAUTH_CLIENT_ID: scheduling-app
    GF_AUTH_GENERIC_OAUTH_AUTH_URL: http://keycloak:8080/realms/scheduling-system/protocol/openid-connect/auth
    # ... Keycloak SSO 통합
  ports: ["3000:3000"]
  depends_on: [prometheus, loki, keycloak]
```
