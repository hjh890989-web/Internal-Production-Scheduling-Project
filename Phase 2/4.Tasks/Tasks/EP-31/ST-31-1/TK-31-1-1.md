### `application.yml` 추가

```yaml
management:
  endpoints:
    web:
      exposure:
        include: health, info, prometheus, metrics, loggers
  endpoint:
    health:
      show-details: when-authorized
      probes:
        enabled: true
  metrics:
    distribution:
      percentiles-histogram:
        http.server.requests: true
      percentiles:
        http.server.requests: 0.5, 0.95, 0.99
    tags:
      application: scheduling-engine
      environment: ${SPRING_PROFILES_ACTIVE:dev}
  prometheus:
    metrics:
      export:
        enabled: true
```

### Custom Metric 예시

```java
@Component @RequiredArgsConstructor
public class AllocatorMetrics {

    private final MeterRegistry registry;

    public void recordAllocationDuration(Duration duration, String type) {
        Timer.builder("allocator.duration")
            .tag("type", type)
            .register(registry)
            .record(duration);
    }
}
```
