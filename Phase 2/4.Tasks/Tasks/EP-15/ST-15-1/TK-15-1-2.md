### Backend `VulcanizationPushService.java`

```java
@Service @RequiredArgsConstructor
public class VulcanizationPushService {

    private final SimpMessagingTemplate messagingTemplate;

    @EventListener
    @Async
    public void onVcConfirmed(VcConfirmedEvent event) {
        var payload = VcUpdatePayload.builder()
            .eventType("CONFIRMED")
            .scheduleId(event.scheduleId())
            .timestamp(event.confirmedAt())
            .affectedRows(event.rows().stream()
                .map(r -> new VcRowSummary(r.rowId(), r.hoseId(), r.productionDate(), r.machineId()))
                .toList())
            .build();

        messagingTemplate.convertAndSend("/topic/vulcanization-updates", payload);
    }
}
```

### Frontend `useVulcanizationWebSocket.ts`

```typescript
import { useEffect } from 'react';
import { Client } from '@stomp/stompjs';
import { useQueryClient } from '@tanstack/react-query';
import { useKeycloak } from '@react-keycloak/web';

export function useVulcanizationWebSocket() {
  const queryClient = useQueryClient();
  const { keycloak } = useKeycloak();

  useEffect(() => {
    if (!keycloak.authenticated) return;

    const client = new Client({
      brokerURL: `${import.meta.env.VITE_WS_BASE}/ws/vulcanization`,
      connectHeaders: {
        Authorization: `Bearer ${keycloak.token}`,
      },
      reconnectDelay: 5000,
      onConnect: () => {
        client.subscribe('/topic/vulcanization-updates', (msg) => {
          const payload = JSON.parse(msg.body);
          // VC 스케줄 query invalidate → AG Grid 자동 갱신
          queryClient.invalidateQueries({ queryKey: ['vcSchedule'] });
        });
      },
      debug: (str) => {
        if (import.meta.env.DEV) console.log('STOMP:', str);
      },
    });

    client.activate();
    return () => { client.deactivate(); };
  }, [keycloak.authenticated, queryClient]);
}
```

### Frontend `SimViewPage.tsx`

```tsx
import { useState } from 'react';
import { useKeycloak } from '@react-keycloak/web';
import { Navigate } from 'react-router-dom';
import { RotationMatrix } from './RotationMatrix';
import { useVulcanizationWebSocket } from '@/hooks/useVulcanizationWebSocket';
import { DateRangePicker } from '@/components/DateRangePicker';

export function SimViewPage() {
  const { keycloak } = useKeycloak();
  useVulcanizationWebSocket();    // 자동 구독

  const [range, setRange] = useState({
    from: format(startOfWeek(new Date()), 'yyyy-MM-dd'),
    to: format(endOfWeek(new Date()), 'yyyy-MM-dd'),
  });

  if (!keycloak.hasRealmRole('STK_USER') && !keycloak.hasRealmRole('PLANNER')) {
    return <Navigate to="/unauthorized" replace />;
  }

  return (
    <div className="simview-container">
      <header>
        <h1>성형 시뮬뷰 — STK-03</h1>
        <DateRangePicker value={range} onChange={setRange} />
      </header>
      <RotationMatrix from={range.from} to={range.to} />
    </div>
  );
}
```
