### `simview-publish-sla.spec.ts`

```typescript
import { test, expect } from '@playwright/test';

test('VC confirm → SimView 갱신 p95 ≤ 2초', async ({ page, request }) => {
  const latencies: number[] = [];
  const RUNS = 50;

  // 1) 로그인 + /simview 진입
  await page.goto('/simview');
  await page.waitForSelector('.ag-row');

  // 2) 50회 측정
  for (let i = 0; i < RUNS; i++) {
    // 신규 candidate 생성 (API)
    const candidate = await request.post('/api/v1/schedule/vc/candidate', {
      data: { /* 신규 hose 1건 */ },
    });
    const candidateId = (await candidate.json()).id;

    const t0 = Date.now();

    // confirm
    await request.post(`/api/v1/schedule/vc/${candidateId}/confirm`, {
      headers: { Authorization: `Bearer ${process.env.PLANNER_TOKEN}` },
    });

    // SimView DOM 갱신 대기 — 새 row 등장 확인
    await page.waitForSelector(`[data-row-id="${candidateId}"]`, { timeout: 5000 });

    const latency = Date.now() - t0;
    latencies.push(latency);
  }

  // p95 계산
  latencies.sort((a, b) => a - b);
  const p50 = latencies[Math.floor(RUNS * 0.5)];
  const p95 = latencies[Math.floor(RUNS * 0.95)];
  const p99 = latencies[Math.floor(RUNS * 0.99)];

  console.log(`SimView publish latency: p50=${p50}ms, p95=${p95}ms, p99=${p99}ms`);

  expect(p95).toBeLessThanOrEqualTo(2000);
});
```
