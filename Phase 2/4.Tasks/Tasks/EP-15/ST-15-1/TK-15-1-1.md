### `types.ts`

```typescript
export interface VcRowDto {
  id: string;
  hoseId: string;
  productionDate: string;             // ISO YYYY-MM-DD
  machineId: string;                  // LP-01 ~ LP-04, IC-01
  rotationId: string;                 // DAY_1 ~ DAY_8, NIGHT_1 ~ NIGHT_10
  slotNumber: number;
  yield: number;
  status: 'DRAFT' | 'CANDIDATE' | 'CONFIRMED';
  settingGroup: number;               // 1~8
}

export interface MatrixCell {
  rows: VcRowDto[];     // 다중 hose 가능 (회전·shift 분리)
  totalYield: number;
  colorClass: string;   // 셋팅 그룹별
}
```

### `RotationMatrix.tsx`

```tsx
import { AgGridReact } from 'ag-grid-react';
import { useMemo } from 'react';
import { useVcSchedule } from '@/hooks/useVcSchedule';
import { ScheduleCell } from './ScheduleCell';

interface Props {
  from: string;
  to: string;
  machines?: string[];      // default: 모든 LP
}

export function RotationMatrix({ from, to, machines = ['LP-01','LP-02','LP-03','LP-04'] }: Props) {
  const { data: rows = [], isLoading } = useVcSchedule({ from, to });

  // row key = machineId + rotationId, column = productionDate
  const columnDefs = useMemo(() => {
    const workingDays = computeWorkingDays(from, to);
    return [
      { headerName: '머신·회전', field: 'rowKey', width: 150, pinned: 'left' },
      ...workingDays.map(date => ({
        headerName: formatKoreanDate(date),
        field: date,
        cellRenderer: ScheduleCell,
        width: 200,
        autoHeight: true,
      })),
    ];
  }, [from, to]);

  const rowData = useMemo(() => {
    const rotations = [
      ...Array.from({length: 8}, (_, i) => ({ label: `주간 ${i+1}`, id: `DAY_${i+1}` })),
      ...Array.from({length: 10}, (_, i) => ({ label: `야간 ${i+1}`, id: `NIGHT_${i+1}` })),
    ];
    const machineRotations = machines.flatMap(m =>
      rotations.map(r => ({
        rowKey: `${m} · ${r.label}`,
        machineId: m,
        rotationId: r.id,
      }))
    );

    // 매트릭스 채움
    return machineRotations.map(mr => {
      const cellsByDate: Record<string, MatrixCell> = {};
      for (const row of rows) {
        if (row.machineId !== mr.machineId || row.rotationId !== mr.rotationId) continue;
        if (!cellsByDate[row.productionDate]) {
          cellsByDate[row.productionDate] = { rows: [], totalYield: 0, colorClass: '' };
        }
        cellsByDate[row.productionDate].rows.push(row);
        cellsByDate[row.productionDate].totalYield += row.yield;
        cellsByDate[row.productionDate].colorClass = `setting-group-${row.settingGroup}`;
      }
      return { ...mr, ...cellsByDate };
    });
  }, [rows, machines]);

  if (isLoading) return <div>로딩 중...</div>;

  return (
    <div className="ag-theme-alpine" style={{ height: 'calc(100vh - 100px)' }}>
      <AgGridReact
        columnDefs={columnDefs}
        rowData={rowData}
        rowHeight={60}
        enableRangeSelection={true}
        suppressMenuHide={true}
      />
    </div>
  );
}
```

### `ScheduleCell.tsx`

```tsx
import { ICellRendererParams } from 'ag-grid-community';
import { MatrixCell } from './types';

export function ScheduleCell(params: ICellRendererParams<unknown, MatrixCell>) {
  const cell = params.value;
  if (!cell || cell.rows.length === 0) return <div className="empty-cell">—</div>;

  return (
    <div className={`schedule-cell ${cell.colorClass}`}>
      {cell.rows.map(r => (
        <div key={r.id} className="row-item">
          <span className="hose-id">{r.hoseId}</span>
          <span className="yield">({r.yield})</span>
        </div>
      ))}
      <div className="total">합: {cell.totalYield}</div>
    </div>
  );
}
```

### `useVcSchedule.ts`

```typescript
import { useQuery } from '@tanstack/react-query';
import { vcApi } from '@/api/vcApi';

export function useVcSchedule({ from, to }: { from: string; to: string }) {
  return useQuery({
    queryKey: ['vcSchedule', from, to],
    queryFn: () => vcApi.fetchSchedule(from, to),
    staleTime: 30_000,
    gcTime: 5 * 60_000,
    refetchOnWindowFocus: true,
  });
}
```
