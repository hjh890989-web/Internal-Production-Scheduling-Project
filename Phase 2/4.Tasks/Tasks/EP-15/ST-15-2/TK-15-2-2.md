```sql
CREATE TABLE vc_schedule_swap_proposal (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    source_row_id       UUID NOT NULL REFERENCES vc_schedule(id),
    target_rotation_id  VARCHAR(20) NOT NULL,
    proposed_by         VARCHAR(40) NOT NULL,
    proposed_at         TIMESTAMPTZ NOT NULL DEFAULT now(),
    reason              TEXT NOT NULL,
    status              VARCHAR(20) NOT NULL DEFAULT 'PROPOSED'
        CHECK (status IN ('PROPOSED', 'ACCEPTED', 'REJECTED')),
    resolved_by         VARCHAR(40),
    resolved_at         TIMESTAMPTZ,
    rejection_reason    TEXT
);

CREATE INDEX idx_swap_proposal_status ON vc_schedule_swap_proposal(status, proposed_at);
```

### `SwapProposalService.java`

```java
@Service @RequiredArgsConstructor
public class SwapProposalService {

    private final SwapProposalRepository proposalRepo;
    private final VcScheduleRepository scheduleRepo;
    private final YieldPreservationValidator validator;
    private final VulcanizationPushService pushService;

    @Auditable(reason = "'accept swap proposal ' + #proposalId")
    @Transactional
    public void accept(UUID proposalId) {
        var proposal = proposalRepo.findById(proposalId)
            .orElseThrow(() -> new NotFoundException("proposal=" + proposalId));

        if (proposal.getStatus() != ProposalStatus.PROPOSED) {
            throw new IllegalStateException("이미 처리된 제안: " + proposal.getStatus());
        }

        var sourceRow = scheduleRepo.findById(proposal.getSourceRowId()).orElseThrow();

        // 1) 총량 보존 검증 (변경 전후 hose·일별 yield 동일)
        validator.requirePreserved(sourceRow, proposal.getTargetRotationId());

        // 2) rotation_id 변경
        sourceRow.setRotationId(proposal.getTargetRotationId());
        scheduleRepo.save(sourceRow);

        // 3) Proposal status 갱신
        proposal.setStatus(ProposalStatus.ACCEPTED);
        proposal.setResolvedBy(SecurityContextHolder.getContext().getAuthentication().getName());
        proposal.setResolvedAt(Instant.now());
        proposalRepo.save(proposal);

        // 4) 시뮬뷰 갱신 PUSH
        pushService.publishProposalAccepted(proposal, sourceRow);
    }

    @Auditable(reason = "'reject swap proposal ' + #proposalId + ' — ' + #rejectionReason")
    @Transactional
    public void reject(UUID proposalId, String rejectionReason) {
        if (rejectionReason == null || rejectionReason.length() < 10) {
            throw new IllegalArgumentException("거절 사유 ≥ 10자");
        }

        var proposal = proposalRepo.findById(proposalId).orElseThrow();
        proposal.setStatus(ProposalStatus.REJECTED);
        proposal.setRejectionReason(rejectionReason);
        proposal.setResolvedBy(SecurityContextHolder.getContext().getAuthentication().getName());
        proposal.setResolvedAt(Instant.now());
        proposalRepo.save(proposal);
    }
}
```

### `YieldPreservationValidator.java`

```java
@Component @RequiredArgsConstructor
public class YieldPreservationValidator {

    private final VcScheduleRepository scheduleRepo;

    /**
     * 총량 보존 검증: rotation 변경 후 (hose, date) 별 yield 합 변경 없음.
     * (단순 rotation swap — yield 자체는 안 바뀜)
     */
    public void requirePreserved(VcSchedule sourceRow, String newRotationId) {
        // 단순 검증: 같은 날짜 같은 머신 같은 hose에 새 rotation slot 미점유 확인
        var conflict = scheduleRepo.findByMachineAndDateAndRotation(
            sourceRow.getMachineId(),
            sourceRow.getProductionDate(),
            newRotationId);

        if (conflict.isPresent() && !conflict.get().getId().equals(sourceRow.getId())) {
            throw new ProposalConflictException(
                "대상 회전 %s에 다른 hose 점유 — 단순 swap 불가".formatted(newRotationId));
        }
        // BR-V07 (당일 락) 도 자동 검증 — RuleEngine integration (Phase 2+)
    }
}
```

### `SwapProposalController.java`

```java
@RestController @RequestMapping("/api/v1/schedule/vc/proposals") @RequiredArgsConstructor
public class SwapProposalController {

    private final SwapProposalService service;

    @PostMapping
    @PreAuthorize("hasAnyRole('STK_USER', 'PLANNER')")
    public SwapProposal create(@Valid @RequestBody CreateProposalRequest request) {
        return service.create(request);
    }

    @PostMapping("/{id}/accept")
    @PreAuthorize("hasRole('PLANNER')")
    public void accept(@PathVariable UUID id) {
        service.accept(id);
    }

    @PostMapping("/{id}/reject")
    @PreAuthorize("hasRole('PLANNER')")
    public void reject(@PathVariable UUID id, @Valid @RequestBody RejectRequest request) {
        service.reject(id, request.reason());
    }
}
```
