### `SwapProposalModal.tsx`

```tsx
import { Modal, Form, Select, Input, Button } from 'antd';
import { useMutation } from '@tanstack/react-query';
import { proposalApi } from '@/api/proposalApi';

interface Props {
  open: boolean;
  sourceCell: { rowId: string; rotationId: string };
  onClose: () => void;
}

export function SwapProposalModal({ open, sourceCell, onClose }: Props) {
  const [form] = Form.useForm();

  const { mutate, isPending } = useMutation({
    mutationFn: proposalApi.create,
    onSuccess: () => {
      message.success('제안이 Planner에게 전송되었습니다');
      onClose();
    },
  });

  return (
    <Modal title="회전 변경 제안" open={open} onCancel={onClose} footer={null}>
      <Form
        form={form}
        layout="vertical"
        onFinish={(values) => mutate({
          sourceRowId: sourceCell.rowId,
          targetRotationId: values.targetRotation,
          reason: values.reason,
        })}
      >
        <Form.Item name="targetRotation" label="대상 회전" rules={[{ required: true }]}>
          <Select
            options={[
              { value: 'DAY_1', label: '주간 1' },
              { value: 'DAY_2', label: '주간 2' },
              // ...
              { value: 'NIGHT_10', label: '야간 10' },
            ]}
          />
        </Form.Item>
        <Form.Item name="reason" label="사유" rules={[{ required: true, min: 10 }]}>
          <Input.TextArea rows={3} placeholder="예: 셋팅 변경 발생 가능 — 야간 후반으로 이동 권장" />
        </Form.Item>
        <Button htmlType="submit" type="primary" loading={isPending}>제안 전송</Button>
      </Form>
    </Modal>
  );
}
```

### `ContextMenu.tsx`

```tsx
import { useState } from 'react';
import { Dropdown, MenuProps } from 'antd';

export function CellContextMenu({ children, rowId, rotationId, onProposeSwap }) {
  const items: MenuProps['items'] = [
    { key: 'swap', label: '다른 회전으로 변경 제안', onClick: () => onProposeSwap(rowId, rotationId) },
    { key: 'view-detail', label: '상세 보기' },
  ];

  return (
    <Dropdown menu={{ items }} trigger={['contextMenu']}>
      {children}
    </Dropdown>
  );
}
```

### `ProposalNotificationPanel.tsx` (Planner 측)

```tsx
import { useEffect, useState } from 'react';
import { Drawer, List, Button, Tag } from 'antd';
import { useProposalsWebSocket } from '@/hooks/useProposalsWebSocket';

export function ProposalNotificationPanel({ open, onClose }) {
  const { proposals, accept, reject } = useProposalsWebSocket();

  return (
    <Drawer title="현장 제안 알림" open={open} onClose={onClose} width={400}>
      <List
        dataSource={proposals}
        renderItem={(p) => (
          <List.Item
            actions={[
              <Button type="primary" onClick={() => accept(p.id)}>수용</Button>,
              <Button danger onClick={() => reject(p.id)}>거절</Button>,
            ]}
          >
            <List.Item.Meta
              title={`${p.hoseId} — ${p.sourceRotation} → ${p.targetRotation}`}
              description={
                <>
                  <Tag>{p.proposedBy}</Tag>
                  <span>{p.reason}</span>
                </>
              }
            />
          </List.Item>
        )}
      />
    </Drawer>
  );
}
```
