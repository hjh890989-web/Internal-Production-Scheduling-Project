### 핵심 로직

```typescript
import { test, expect, Browser } from '@playwright/test';

test('STK 제안 → Planner 수용 → 양쪽 시뮬뷰 갱신', async ({ browser }) => {
  // 1) STK 컨텍스트 생성
  const stkContext = await browser.newContext({ storageState: 'stk.json' });
  const stkPage = await stkContext.newPage();

  // 2) Planner 컨텍스트
  const plannerContext = await browser.newContext({ storageState: 'planner.json' });
  const plannerPage = await plannerContext.newPage();

  // 3) 양쪽 SimViewPage 진입
  await stkPage.goto('/simview');
  await plannerPage.goto('/simview');

  await Promise.all([
    stkPage.waitForSelector('.ag-row'),
    plannerPage.waitForSelector('.ag-row'),
  ]);

  // 4) STK: cell 우클릭 → 제안 생성
  const firstCell = stkPage.locator('.schedule-cell').first();
  await firstCell.click({ button: 'right' });
  await stkPage.getByText('다른 회전으로 변경 제안').click();

  await stkPage.getByLabel('대상 회전').click();
  await stkPage.getByText('야간 1').click();
  await stkPage.getByLabel('사유').fill('셋팅 변경 발생 예상 — 야간 후반 권장');
  await stkPage.getByText('제안 전송').click();

  await expect(stkPage.getByText('제안이 Planner에게 전송되었습니다')).toBeVisible();

  // 5) Planner: 알림 자동 등장 (WebSocket)
  await plannerPage.locator('.proposal-notification-btn').click();
  const proposalItem = plannerPage.locator('.proposal-item').first();
  await expect(proposalItem).toContainText('야간 후반 권장');

  // 6) Planner: 수용
  await proposalItem.getByText('수용').click();
  await expect(plannerPage.getByText('수용되었습니다')).toBeVisible();

  // 7) 양쪽 시뮬뷰 갱신 확인 — 해당 cell이 신규 위치로 이동
  await Promise.all([
    expect(stkPage.locator('[data-rotation="NIGHT_1"]')).toContainText(/.+/),
    expect(plannerPage.locator('[data-rotation="NIGHT_1"]')).toContainText(/.+/),
  ]);

  await stkContext.close();
  await plannerContext.close();
});

test('거절 시나리오 — 사유 ≥ 10자 강제', async ({ browser }) => {
  // ... 제안 생성 + Planner 거절 시도
  // 사유 5자 → 400 검증
  // 사유 15자 → 정상 거절 + STK 측 거절 알림
});

test('이미 처리된 제안 재처리 차단', async ({ browser }) => {
  // ... accept 후 다시 accept → 400 Bad Request
});
```
