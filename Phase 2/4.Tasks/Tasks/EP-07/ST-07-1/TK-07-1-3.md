#### A. `ExDeadlineRegressionIT.java`

```java
package com.scheduling.ex.deadline;

import com.scheduling.calendar.WorkingCalendarService;
import com.scheduling.ex.event.ExtrusionScheduleService;
import com.scheduling.ex.schedule.ExScheduleCandidateRepository;
import com.scheduling.vc.event.VcConfirmedEvent;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.jdbc.Sql;
import org.testcontainers.junit.jupiter.Testcontainers;

import java.time.LocalDate;
import java.util.List;

import static org.assertj.core.api.Assertions.assertThat;

@SpringBootTest @Testcontainers
@Sql("classpath:datasets/DS-EX-DEADLINE-100/master_seed.sql")
class ExDeadlineRegressionIT {

    @Autowired ExtrusionScheduleService extrusionService;
    @Autowired BackwardExtrusionCalculator deadlineCalc;
    @Autowired ExScheduleCandidateRepository candidateRepo;
    @Autowired WorkingCalendarService calendar;
    @Autowired EventLoader loader;

    /**
     * TC-EX-001 — 100 시나리오 회귀.
     * 다양한 vc_date 분포 (월/화/.../금, 주말 후 월요일, 연휴 직후 등) 100건.
     * 모든 candidate.extrusion_deadline ≤ vc_production_date − 1 working day 강제.
     */
    @Test
    void all_candidates_within_d1_deadline_for_100_scenarios() {
        var events = loader.load("DS-EX-DEADLINE-100/vc_confirmed_events.csv");
        assertThat(events).hasSize(100);

        // 100 이벤트 발행 → 압출 후보 생성
        for (var event : events) extrusionService.generateCandidates(event);

        var allCandidates = candidateRepo.findAll();

        // 1. 회귀 — 모든 candidate.extrusion_deadline ≤ vc_date - 1 working day
        var violations = allCandidates.stream()
            .filter(c -> {
                LocalDate expected = calendar.subtractWorkingDays(c.getVcProductionDate(), 1);
                return c.getExtrusionDeadline().isAfter(expected);
            })
            .toList();

        assertThat(violations)
            .as("BR-E01 위반: %d개 candidate의 ex_deadline > vc_date − 1", violations.size())
            .isEmpty();

        // 2. NS-S07 D-1 준수율 — 100% (회귀 시점)
        long totalCandidates = allCandidates.size();
        long compliantCandidates = totalCandidates - violations.size();
        double complianceRate = (double) compliantCandidates / totalCandidates;

        assertThat(complianceRate)
            .as("NS-S07 D-1 준수율 (REQ-NF-KPI-009)")
            .isGreaterThanOrEqualTo(0.98);
    }

    @Test
    void single_hose_multi_vc_rows_uses_earliest_vc_date() {
        // 같은 hose_id 다중 vc_date — 가장 이른 production_date 기준 deadline
        var event = new VcConfirmedEvent(
            java.util.UUID.randomUUID(),
            java.time.Instant.now(),
            List.of(
                row("28912-2U000", LocalDate.of(2026, 3, 5),  100),
                row("28912-2U000", LocalDate.of(2026, 3, 11), 150),
                row("28912-2U000", LocalDate.of(2026, 3, 18), 100)
            )
        );

        var deadlines = deadlineCalc.compute(event);
        // earliest 3/5(목) − 1 영업일 = 3/4(수)
        assertThat(deadlines.get("28912-2U000")).contains(LocalDate.of(2026, 3, 4));
    }

    private VcConfirmedEvent.VcConfirmedRow row(String hose, LocalDate date, int yield) {
        return new VcConfirmedEvent.VcConfirmedRow(
            java.util.UUID.randomUUID(), hose, date, "LP-01", "DAY", 1, yield);
    }
}
```

@SpringBootTest @Testcontainers
class ExDeadlineEdgeCaseIT {

    @Autowired BackwardExtrusionCalculator deadlineCalc;
    @Autowired WorkingCalendarService calendar;

    @Test
    void vc_date_monday_deadline_previous_friday() {
        // vc_date 월요일 → 1 영업일 차감 = 직전 금요일
        LocalDate vc = LocalDate.of(2026, 3, 9);   // 월
        LocalDate deadline = deadlineCalc.deadlineFor(vc);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 3, 6));   // 금
    }

    @Test
    void vc_date_right_after_lunar_new_year() {
        // 2/16~2/18 설날 연휴 → vc_date 2/19(목) → deadline = 2/13(금)
        LocalDate vc = LocalDate.of(2026, 2, 19);
        LocalDate deadline = deadlineCalc.deadlineFor(vc);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 2, 13));
    }

    @Test
    void vc_date_tuesday_normal_case() {
        LocalDate vc = LocalDate.of(2026, 3, 10);   // 화
        LocalDate deadline = deadlineCalc.deadlineFor(vc);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 3, 9));   // 월
    }

    @Test
    void consecutive_holidays_chuseok() {
        // 추석 9/24~9/26 → vc_date 9/29(화) → 직전 영업일 = 9/23(수)
        LocalDate vc = LocalDate.of(2026, 9, 29);
        LocalDate deadline = deadlineCalc.deadlineFor(vc);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 9, 23));
    }
}
