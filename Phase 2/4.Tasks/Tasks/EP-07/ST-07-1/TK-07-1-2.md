### 핵심 로직

#### A. `BackwardExtrusionCalculator.java`

```java
package com.scheduling.ex.deadline;

import com.scheduling.calendar.WorkingCalendarService;
import com.scheduling.vc.event.VcConfirmedEvent;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.*;

@Service @RequiredArgsConstructor
public class BackwardExtrusionCalculator {

    private static final int BACKWARD_DAYS = 1;     // BR-E01: 압출 D-1

    private final WorkingCalendarService calendar;

    /**
     * REQ-FUNC-EX-001: ex_deadline = vc_production_date − 1 working day.
     *
     * 같은 hose_id에 여러 vc_date 있을 때 가장 이른 vc_date 기준 (hard 제약).
     */
    public ExDeadlineMap compute(VcConfirmedEvent event) {
        var earliestVcDateByHose = new HashMap<String, LocalDate>();
        for (var row : event.rows()) {
            earliestVcDateByHose.merge(row.hoseId(), row.productionDate(),
                (a, b) -> a.isBefore(b) ? a : b);
        }

        var deadlines = new HashMap<String, LocalDate>();
        for (var entry : earliestVcDateByHose.entrySet()) {
            LocalDate deadline = calendar.subtractWorkingDays(entry.getValue(), BACKWARD_DAYS);
            deadlines.put(entry.getKey(), deadline);
        }
        return new ExDeadlineMap(Map.copyOf(deadlines));
    }

    /**
     * 단일 vc_date의 D-1 deadline.
     */
    public LocalDate deadlineFor(LocalDate vcProductionDate) {
        return calendar.subtractWorkingDays(vcProductionDate, BACKWARD_DAYS);
    }
}

public record ExDeadlineMap(Map<String, LocalDate> map) {
    public Optional<LocalDate> get(String hoseId) { return Optional.ofNullable(map.get(hoseId)); }
    public boolean isWithinDeadline(String hoseId, LocalDate exProductionDate) {
        return get(hoseId).map(d -> !exProductionDate.isAfter(d)).orElse(true);
    }
}
```

#### B. `ExtrusionScheduleService.java` (TK-07-1-1 stub 교체)

```java
@Service @RequiredArgsConstructor
public class ExtrusionScheduleService {

    private final BackwardExtrusionCalculator deadlineCalc;
    private final ExScheduleCandidateRepository candidateRepo;

    @Transactional
    public void generateCandidates(VcConfirmedEvent event) {
        // 1. D-1 deadline 계산 (BR-E01)
        ExDeadlineMap deadlines = deadlineCalc.compute(event);

        // 2. 후보 stub 생성 — EP-08·09에서 정식 yield 계산
        for (var row : event.rows()) {
            LocalDate deadline = deadlines.get(row.hoseId()).orElseThrow();

            var candidate = ExScheduleCandidate.builder()
                .scheduleId(event.scheduleId())
                .hoseId(row.hoseId())
                .vcRowId(row.rowId())
                .vcProductionDate(row.productionDate())
                .extrusionDeadline(deadline)
                .vcYield(row.yield())                  // EP-08 EX-010 입력
                .status(CandidateStatus.PENDING)
                .build();

            candidateRepo.save(candidate);
        }
    }
}
```
