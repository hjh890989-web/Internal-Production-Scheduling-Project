```java
@ExtendWith(MockitoExtension.class)
class BackwardExtrusionCalculatorUnitTest {

    @Mock WorkingCalendarService calendar;
    @InjectMocks BackwardExtrusionCalculator calc;

    @ParameterizedTest
    @CsvSource({
        "2026-03-09, 2026-03-06",     // 월 → 금
        "2026-03-10, 2026-03-09",     // 화 → 월
        "2026-03-11, 2026-03-10",     // 수 → 화
        "2026-03-12, 2026-03-11",     // 목 → 수
        "2026-03-13, 2026-03-12",     // 금 → 목
        "2026-03-07, 2026-03-06",     // 토 → 금
        "2026-03-08, 2026-03-06"      // 일 → 금
    })
    void seven_day_combinations(String vcDateStr, String expectedDeadlineStr) {
        var vcDate = LocalDate.parse(vcDateStr);
        var expected = LocalDate.parse(expectedDeadlineStr);
        when(calendar.subtractWorkingDays(vcDate, 1)).thenReturn(expected);

        assertThat(calc.deadlineFor(vcDate)).isEqualTo(expected);
    }

    @Test
    void empty_event_yields_empty_deadline_map() {
        var event = new VcConfirmedEvent(UUID.randomUUID(), Instant.now(), List.of());
        var deadlines = calc.compute(event);

        assertThat(deadlines.map()).isEmpty();
    }

    @Test
    void multi_hose_multi_row_uses_earliest_per_hose() {
        var event = new VcConfirmedEvent(UUID.randomUUID(), Instant.now(), List.of(
            row("28912", LocalDate.of(2026, 3, 12)),
            row("28912", LocalDate.of(2026, 3, 5)),
            row("28912", LocalDate.of(2026, 3, 19)),
            row("29689", LocalDate.of(2026, 3, 10))
        ));

        when(calendar.subtractWorkingDays(LocalDate.of(2026, 3, 5),  1))
            .thenReturn(LocalDate.of(2026, 3, 4));
        when(calendar.subtractWorkingDays(LocalDate.of(2026, 3, 10), 1))
            .thenReturn(LocalDate.of(2026, 3, 9));

        var deadlines = calc.compute(event);

        assertThat(deadlines.get("28912")).contains(LocalDate.of(2026, 3, 4));    // earliest 3/5
        assertThat(deadlines.get("29689")).contains(LocalDate.of(2026, 3, 9));
    }

    @Test
    void single_row_single_hose() {
        var event = new VcConfirmedEvent(UUID.randomUUID(), Instant.now(), List.of(
            row("A6722", LocalDate.of(2026, 3, 11))
        ));
        when(calendar.subtractWorkingDays(LocalDate.of(2026, 3, 11), 1))
            .thenReturn(LocalDate.of(2026, 3, 10));

        var deadlines = calc.compute(event);
        assertThat(deadlines.get("A6722")).contains(LocalDate.of(2026, 3, 10));
    }

    @Test
    void deadline_map_is_immutable() {
        var event = new VcConfirmedEvent(UUID.randomUUID(), Instant.now(), List.of(
            row("28912", LocalDate.of(2026, 3, 5))
        ));
        when(calendar.subtractWorkingDays(any(), eq(1)))
            .thenReturn(LocalDate.of(2026, 3, 4));

        var deadlines = calc.compute(event);

        assertThatThrownBy(() -> deadlines.map().put("X", LocalDate.now()))
            .isInstanceOf(UnsupportedOperationException.class);
    }

    @Test
    void is_within_deadline_correct_boolean() {
        var map = new ExDeadlineMap(Map.of("28912", LocalDate.of(2026, 3, 5)));
        assertThat(map.isWithinDeadline("28912", LocalDate.of(2026, 3, 5))).isTrue();
        assertThat(map.isWithinDeadline("28912", LocalDate.of(2026, 3, 4))).isTrue();
        assertThat(map.isWithinDeadline("28912", LocalDate.of(2026, 3, 6))).isFalse();
        assertThat(map.isWithinDeadline("unknown", LocalDate.of(2026, 3, 5))).isTrue();   // 룰 없음 = pass
    }

    private VcConfirmedEvent.VcConfirmedRow row(String hose, LocalDate date) {
        return new VcConfirmedEvent.VcConfirmedRow(
            UUID.randomUUID(), hose, date, "LP-01", "DAY", 1, 100);
    }
}
```
