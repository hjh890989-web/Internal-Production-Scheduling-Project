### `CalendarSingleSourceArchTest.java`

```java
package com.scheduling.architecture;

import com.tngtech.archunit.core.domain.JavaClasses;
import com.tngtech.archunit.core.importer.ClassFileImporter;
import org.junit.jupiter.api.Test;

import static com.tngtech.archunit.lang.syntax.ArchRuleDefinition.*;

class CalendarSingleSourceArchTest {

    private static final JavaClasses CLASSES =
        new ClassFileImporter().importPackages("com.scheduling..");

    @Test
    void only_calendar_module_defines_working_day_logic() {
        noClasses().that().resideOutsideOfPackage("com.scheduling.calendar..")
            .should().haveSimpleNameContaining("WorkingCalendar")
            .orShould().haveSimpleNameContaining("HolidayService")
            .check(CLASSES);
    }

    @Test
    void ex_module_uses_calendar_service_only_via_interface() {
        // 압출 모듈은 calendar 패키지 클래스 중 WorkingCalendarService만 의존
        classes().that().resideInAPackage("com.scheduling.ex..")
            .and().areAnnotatedWith(org.springframework.stereotype.Service.class)
            .should().onlyDependOnClassesThat()
                .resideInAnyPackage("com.scheduling.ex..", "com.scheduling.vc.event..",
                                    "com.scheduling.calendar..", "java..", "lombok..",
                                    "org.springframework..", "jakarta..")
            .check(CLASSES);
    }

    @Test
    void ex_module_does_not_define_holiday_entity() {
        noClasses().that().resideInAPackage("com.scheduling.ex..")
            .should().haveSimpleNameContaining("Holiday")
            .check(CLASSES);
    }
}
```
