### 핵심 로직

```java
@SpringBootTest @Testcontainers
class WeekendVcDateRegressionIT {

    @Autowired BackwardExtrusionCalculator deadlineCalc;
    @Autowired WorkingCalendarService calendar;
    @Autowired EventLoader loader;

    @Test
    void saturday_vc_date_yields_previous_friday_deadline() {
        // 2026-03-07(토) → 1 영업일 차감 = 2026-03-06(금)
        LocalDate vc = LocalDate.of(2026, 3, 7);
        LocalDate deadline = deadlineCalc.deadlineFor(vc);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 3, 6));
    }

    @Test
    void sunday_vc_date_yields_previous_friday_deadline() {
        // 2026-03-08(일) → 1 영업일 차감 = 2026-03-06(금)
        // (3/7 토·3/8 일 모두 영업일 아님, 다음 영업일 후보 = 3/6 금)
        LocalDate vc = LocalDate.of(2026, 3, 8);
        LocalDate deadline = deadlineCalc.deadlineFor(vc);
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 3, 6));
    }

    @Test
    void monday_after_long_weekend_vc_date() {
        // 3·1절 일요일 + 월요일 휴일 → vc_date 3/2(월) 휴일 → 3/3(화) 정상 영업일
        // vc_date 3/2 비영업일 입력 시 deadline 계산도 deterministic해야 함
        LocalDate vc = LocalDate.of(2026, 3, 2);   // 월, 3·1절 대체휴일 가정
        // subtractWorkingDays는 vc 자체가 영업일 아니어도 정상 동작 — 그 이전 영업일 카운트
        LocalDate deadline = deadlineCalc.deadlineFor(vc);
        // 3/2(휴), 3/1(일), 2/28(토), 2/27(금) → 1 영업일 차감 = 2/27 금
        assertThat(deadline).isEqualTo(LocalDate.of(2026, 2, 27));
    }

    @Test
    void regression_50_weekend_scenarios() {
        var events = loader.load("DS-EX-WEEKEND-50/vc_confirmed_weekend_events.csv");
        assertThat(events).hasSize(50);

        for (var event : events) {
            var deadlines = deadlineCalc.compute(event);
            for (var entry : deadlines.map().entrySet()) {
                LocalDate deadline = entry.getValue();
                // 모든 deadline은 평일 (BR-E02)
                assertThat(calendar.isWorkingDay(deadline))
                    .as("deadline %s for hose %s must be working day", deadline, entry.getKey())
                    .isTrue();
            }
        }
    }

    @Test
    void deterministic_repeated_invocations() {
        LocalDate vc = LocalDate.of(2026, 3, 7);   // 토
        LocalDate d1 = deadlineCalc.deadlineFor(vc);
        LocalDate d2 = deadlineCalc.deadlineFor(vc);
        LocalDate d3 = deadlineCalc.deadlineFor(vc);
        assertThat(d1).isEqualTo(d2).isEqualTo(d3);
    }
}
```
