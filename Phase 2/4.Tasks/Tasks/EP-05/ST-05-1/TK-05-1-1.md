#### D. `RotationSlot.java` — 도메인 record

```java
package com.scheduling.vc.domain;

import java.time.LocalDate;

/**
 * 회전 격자의 단일 셀. 가류기 회전 시점·위치 표현.
 * (date, machineId, rotationNo) → 가류기 회전 인스턴스
 * + slotPosition → 특정 슬롯
 */
public record RotationSlot(
    LocalDate date,
    String machineId,
    int rotationNo,             // 1~18 (BR-V04)
    int slotPosition            // 1~8 LP / 1~6 IC
) {
    public RotationSlot {
        if (rotationNo < 1 || rotationNo > 18) {
            throw new IllegalArgumentException("rotation_no must be 1..18: " + rotationNo);
        }
        if (slotPosition < 1) {
            throw new IllegalArgumentException("slot_position must be ≥1: " + slotPosition);
        }
    }

    public boolean isDayRotation() { return rotationNo <= 8; }
    public boolean isNightRotation() { return rotationNo >= 9; }
}
```

#### E. `VcMachine.java` 엔티티

```java
package com.scheduling.vc.domain;

import com.scheduling.master.vc.MachineType;
import jakarta.persistence.*;
import lombok.*;

import java.time.Instant;

@Entity @Table(schema = "master", name = "vc_machine")
@Getter @NoArgsConstructor @AllArgsConstructor @Builder
public class VcMachine {

    @Id @Column(length = 10)
    private String machineId;            // LP-01 ~ LP-04, IC-01

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 2)
    private MachineType machineType;

    @Column(nullable = false) private short totalSlots;
    @Column(nullable = false) @Builder.Default private short dayRotations = 8;
    @Column(nullable = false) @Builder.Default private short nightRotations = 10;
    @Builder.Default private boolean active = true;
    private Instant updatedAt;
    private String updatedBy;

    public int totalRotationsPerDay() { return dayRotations + nightRotations; }
    public int dailyCapacity() { return totalRotationsPerDay() * totalSlots; }
}
```

#### F. `VcSchedule.java` 엔티티

```java
package com.scheduling.vc.domain;

import jakarta.persistence.*;
import lombok.*;

import java.time.Instant;
import java.time.LocalDate;
import java.util.UUID;

@Entity @Table(schema = "app", name = "vc_schedule")
@Getter @Setter @NoArgsConstructor @AllArgsConstructor @Builder
public class VcSchedule {

    @Id @GeneratedValue private UUID vcScheduleId;

    @Column(nullable = false, length = 40) private String hoseId;
    @Column(nullable = false, length = 10) private String machineId;
    @Column(nullable = false) private short slotPosition;
    @Column(nullable = false) private LocalDate productionDate;
    @Column(nullable = false) private short rotationNo;      // 1-18

    @Column(nullable = false, length = 40) private String angleId;
    @Column(nullable = false) private int plannedQty;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 20) private VcScheduleStatus status;

    @Column(nullable = false, columnDefinition = "uuid[]")
    private UUID[] linkedOrderIds;

    @Column(nullable = false) private Instant createdAt;
    @Column(nullable = false) private Instant updatedAt;

    public RotationSlot asSlot() {
        return new RotationSlot(productionDate, machineId, rotationNo, slotPosition);
    }
}

public enum VcScheduleStatus { Candidate, Confirmed, Done }
```
