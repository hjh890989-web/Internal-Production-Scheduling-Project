class RotationSlotTest {

    @ParameterizedTest
    @ValueSource(ints = {0, -1, 19, 100})
    void invalid_rotation_no_throws(int n) {
        assertThrows(IllegalArgumentException.class,
            () -> new RotationSlot(LocalDate.now(), "LP-01", n, 1));
    }

    @ParameterizedTest
    @ValueSource(ints = {1, 8, 9, 18})
    void valid_rotation_no_creates(int n) {
        var slot = new RotationSlot(LocalDate.now(), "LP-01", n, 1);
        assertThat(slot.rotationNo()).isEqualTo(n);
    }

    @Test
    void rotation_1_to_8_is_day() {
        for (int r = 1; r <= 8; r++) {
            assertThat(new RotationSlot(LocalDate.now(), "LP-01", r, 1).isDayRotation()).isTrue();
        }
    }

    @Test
    void rotation_9_to_18_is_night() {
        for (int r = 9; r <= 18; r++) {
            assertThat(new RotationSlot(LocalDate.now(), "LP-01", r, 1).isNightRotation()).isTrue();
        }
    }
}

class VcMachineTest {

    @Test
    void lp_capacity_is_144() {
        var lp = VcMachine.builder().machineId("LP-01").machineType(MachineType.LP)
            .totalSlots((short) 8).dayRotations((short) 8).nightRotations((short) 10).build();
        assertThat(lp.totalRotationsPerDay()).isEqualTo(18);
        assertThat(lp.dailyCapacity()).isEqualTo(144);            // 18 × 8
    }

    @Test
    void ic_capacity_is_108() {
        var ic = VcMachine.builder().machineId("IC-01").machineType(MachineType.IC)
            .totalSlots((short) 6).dayRotations((short) 8).nightRotations((short) 10).build();
        assertThat(ic.dailyCapacity()).isEqualTo(108);            // 18 × 6
    }
}

@SpringBootTest @Testcontainers
@Sql({"classpath:db/migration/V2_2__create_vc_machine.sql",
      "classpath:db/migration/V2_3_data__seed_vc_machine.sql"})
class CapacityLedgerIT {

    @Autowired CapacityLedgerBuilder builder;

    @Test
    void daily_capacity_matches_br_v05() {
        // 1일 격자: 저압 4대 × 18회전 × 8슬롯 = 576 + IC 18 × 6 = 108 = 684
        var ledger = builder.build(LocalDate.of(2026, 2, 16), LocalDate.of(2026, 2, 16));   // 월요일
        assertThat(ledger.cells()).hasSize(684);

        // BR-V05: 일일 저압 회전 capa = 72
        long lpRotations = ledger.countLpRotationsAvailable(LocalDate.of(2026, 2, 16));
        assertThat(lpRotations).isEqualTo(72);
    }

    @Test
    void weekend_excluded() {
        // 토요일 (2026-02-14) → 격자 0
        var ledger = builder.build(LocalDate.of(2026, 2, 14), LocalDate.of(2026, 2, 14));
        assertThat(ledger.cells()).isEmpty();
    }

    @Test
    void existing_candidate_marked_reserved() {
        // 미리 Candidate 스케줄 저장 후 build
        // → 해당 RotationSlot이 RESERVED 상태
    }
}
