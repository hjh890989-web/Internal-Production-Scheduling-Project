### 핵심 로직

```java
package com.scheduling.vc.capacity;

import com.scheduling.vc.domain.*;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.*;
import java.util.*;

@Service @RequiredArgsConstructor
public class CapacityLedgerBuilder {

    private final VcMachineRepository machineRepo;
    private final VcScheduleRepository scheduleRepo;
    private final WorkingCalendarService calendar;       // EP-06 (또는 본 Task 임시 stub)

    /**
     * 일자 호라이즌의 가용 회전 격자 빌드.
     * - 저압 4대 × 18 회전 × 8 슬롯 = 576 셀/일
     * - IC 1대 × 18 회전 × 6 슬롯 = 108 셀/일
     * - 총 684 셀/일 (이론 max)
     *
     * BR-V05의 "일일 capa 72 + 18 = 90 회전"은 슬롯이 아닌 회전 단위 — 본 Ledger는 회전×슬롯 격자 모두 추적.
     */
    public CapacityLedger build(LocalDate fromDate, LocalDate toDate) {
        var machines = machineRepo.findByActive(true);
        var existingSchedules = scheduleRepo.findByDateRange(fromDate, toDate);

        var cells = new HashMap<RotationSlot, SlotAvailability>();

        // 영업일만 (월~금, EP-06 calendar 활용)
        for (LocalDate d = fromDate; !d.isAfter(toDate); d = d.plusDays(1)) {
            if (!calendar.isWorkingDay(d)) continue;

            for (VcMachine m : machines) {
                for (int rot = 1; rot <= m.totalRotationsPerDay(); rot++) {
                    for (int slot = 1; slot <= m.totalSlots(); slot++) {
                        var key = new RotationSlot(d, m.machineId(), rot, slot);
                        cells.put(key, SlotAvailability.AVAILABLE);
                    }
                }
            }
        }

        // 기존 스케줄 반영 (CONFIRMED·Candidate·Done)
        for (VcSchedule s : existingSchedules) {
            var key = s.asSlot();
            SlotAvailability avail = switch (s.getStatus()) {
                case Candidate -> SlotAvailability.RESERVED;
                case Confirmed -> SlotAvailability.CONFIRMED;
                case Done -> SlotAvailability.CONFIRMED;
            };
            cells.put(key, avail);
        }

        return new CapacityLedger(fromDate, toDate, Map.copyOf(cells));
    }
}

public record CapacityLedger(
    LocalDate fromDate,
    LocalDate toDate,
    Map<RotationSlot, SlotAvailability> cells
) {
    public SlotAvailability check(RotationSlot slot) {
        return cells.getOrDefault(slot, SlotAvailability.UNAVAILABLE);
    }

    public List<RotationSlot> findAvailableForMachineTypeOnDate(LocalDate date, String machineId) {
        return cells.entrySet().stream()
            .filter(e -> e.getKey().date().equals(date))
            .filter(e -> e.getKey().machineId().equals(machineId))
            .filter(e -> e.getValue() == SlotAvailability.AVAILABLE)
            .map(Map.Entry::getKey)
            .toList();
    }

    /** BR-V05: 저압 일일 capa = 4대 × 18회전 = 72 회전 */
    public long countLpRotationsAvailable(LocalDate date) {
        return cells.entrySet().stream()
            .filter(e -> e.getKey().date().equals(date))
            .filter(e -> e.getKey().machineId().startsWith("LP-"))
            .filter(e -> e.getValue() == SlotAvailability.AVAILABLE)
            .map(e -> Map.entry(e.getKey().machineId(), e.getKey().rotationNo()))
            .distinct()
            .count();
    }
}

public enum SlotAvailability { AVAILABLE, RESERVED, CONFIRMED, UNAVAILABLE }
```

@Service
public class WorkingCalendarService {
    /** 임시: 월~금 영업일 (BR-E02·BR-V09 정합) */
    public boolean isWorkingDay(LocalDate date) {
        DayOfWeek d = date.getDayOfWeek();
        return d != DayOfWeek.SATURDAY && d != DayOfWeek.SUNDAY;
    }
}
