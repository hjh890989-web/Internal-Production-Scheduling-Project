### 핵심 로직

#### A. `AllocationRegressionIT.java`

```java
@SpringBootTest @Testcontainers
@Sql("classpath:datasets/DS-VC-CONSTRAINT-47/master_seed.sql")
class AllocationRegressionIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired CapacityLedgerBuilder ledgerBuilder;
    @Autowired RequiredQuantityCalculator calc;
    @Autowired SlotCompatibilityMatrixService matrixSvc;
    @Autowired AngleCapacityValidator angleVal;

    /**
     * TC-VC-010: 100건 회귀 위반 0건
     */
    @Test
    void allocates_100_scenarios_with_zero_violations() throws Exception {
        var scenarios = loadScenarios("DS-VC-ALLOC-100/scenarios.json");
        int violations = 0;
        var failures = new StringBuilder();

        for (var s : scenarios) {
            // 1. AllocationContext 빌드
            var ledger = ledgerBuilder.build(s.horizonFrom(), s.horizonTo());
            var qRequired = calc.calculate(s.orders(), s.horizonFrom(), s.horizonTo());
            var ctx = new AllocationContext(qRequired, s.ordersByHose(), ledger, s.workingDays());

            // 2. 배치
            var result = allocator.allocate(ctx);

            // 3. 위반 검증 (4축)
            // a) 슬롯 O/X 위반 0
            for (VcSchedule vc : result.schedules()) {
                boolean ok = matrixSvc.current().isEligible(vc.getHoseId(),
                    toSlotPosition(vc.getMachineId(), vc.getSlotPosition()));
                if (!ok) {
                    violations++;
                    failures.append("SLOT_OX violation: ").append(vc).append("\n");
                }
            }

            // b) 앵글 capa 위반 0
            var assignments = result.schedules().stream()
                .collect(Collectors.groupingBy(VcSchedule::getHoseId,
                    Collectors.mapping(VcSchedule::asSlot, Collectors.toList())));
            var angleViols = angleVal.validate(assignments);
            violations += angleViols.size();

            // c) 일일 capa 위반 (회전·슬롯 중복 점유) — UNIQUE 제약이 차단하지만 명시 검증
            var keys = result.schedules().stream()
                .map(vc -> Map.entry(vc.getMachineId(),
                    Map.entry(vc.getProductionDate(),
                        Map.entry(vc.getRotationNo(), vc.getSlotPosition()))))
                .toList();
            assertThat(keys).doesNotHaveDuplicates();

            // d) Conflict 분석 (Unschedulable·용량 부족 등)
            for (var c : result.conflicts()) {
                failures.append("Conflict: ").append(c).append("\n");
            }
        }

        assertThat(violations)
            .as("Allocation violations should be ZERO over 100 scenarios:\n%s", failures)
            .isZero();
    }
}
```

#### B. `AllocationPerformanceTest.java`

```java
@SpringBootTest @Testcontainers
class AllocationPerformanceTest {

    @Autowired GreedyRotationAllocator allocator;
    // ... 의존성

    @Test
    void week_horizon_47_products_p95_under_5_minutes() {
        var durations = new ArrayList<Long>();
        for (int i = 0; i < 20; i++) {              // 20회 측정 (p95 안정성)
            long start = System.nanoTime();
            allocator.allocate(buildFullWeekContext());
            durations.add((System.nanoTime() - start) / 1_000_000);
        }
        Collections.sort(durations);
        long p95 = durations.get(18);                // 95th percentile of 20

        assertThat(p95)
            .as("VC allocator p95 ≤5min for 1-week 47-product horizon")
            .isLessThanOrEqualTo(5 * 60 * 1000L);
    }
}
```

```javascript
// perf/scripts/vc_allocator_load.js — TC-PER-002
import http from 'k6/http';
import { check } from 'k6';
import { Trend } from 'k6/metrics';

const allocationDuration = new Trend('vc_allocation_duration_ms', true);

export const options = {
    scenarios: {
        burst: {
            executor: 'shared-iterations',
            vus: 5,
            iterations: 20,
            maxDuration: '15m',
        },
    },
    thresholds: {
        vc_allocation_duration_ms: ['p(95)<300000'],   // 5분 = 300,000ms
        http_req_failed: ['rate<0.01'],
    },
};

export default function () {
    const t0 = Date.now();
    const res = http.post(`${__ENV.BASE_URL}/api/v1/vc/generate`,
        JSON.stringify({ horizon: '7d' }),
        { headers: { 'Content-Type': 'application/json',
                     Authorization: `Bearer ${__ENV.JWT_TOKEN}` }}
    );
    const elapsed = Date.now() - t0;
    allocationDuration.add(elapsed);

    check(res, {
        'status 202': r => r.status === 202,
        'p95 under 5min': () => elapsed < 300_000,
    });
}
```
