### 핵심 로직

```java
package com.scheduling.vc.allocator;

import com.scheduling.vc.capacity.*;
import com.scheduling.vc.domain.*;
import com.scheduling.vc.yield.*;
import com.scheduling.master.vc.*;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.*;

@Service @RequiredArgsConstructor
public class GreedyRotationAllocator {

    private final SlotCompatibilityMatrixService matrixService;
    private final VcYieldCalculator yieldCalc;
    private final AngleCapacityValidator angleValidator;

    /**
     * REQ-FUNC-VC-010: Q_required → (가류기, 슬롯, 회전) 튜플 할당.
     *
     * 알고리즘 v1 (단순 greedy):
     * 1. Q_required 큰 품번부터 정렬
     * 2. 각 품번에 대해 호라이즌 영업일 순회
     * 3. (machine, rotation, slot) 첫 가능 위치 탐색 (AVAILABLE + 슬롯 O/X + 앵글 capa 통과)
     * 4. 1회전 점유 → 누적 yield += yieldPerRotation(hose, machine_type)
     * 5. 누적 yield ≥ Q_required 도달 시 다음 품번
     * 6. 도달 못 하면 AllocationConflict로 보고
     */
    public AllocationResult allocate(AllocationContext ctx) {
        var matrix = matrixService.current();
        var ledger = new MutableLedger(ctx.ledger());     // 깊은 복사 (가용 격자 변경 추적)
        var schedules = new ArrayList<VcSchedule>();
        var conflicts = new ArrayList<AllocationConflict>();

        // 1. Q_required 큰 품번부터 정렬 (Greedy heuristic)
        var sortedHoses = ctx.qRequired().entrySet().stream()
            .sorted(Map.Entry.<String, Integer>comparingByValue().reversed())
            .toList();

        for (var hoseEntry : sortedHoses) {
            String hose = hoseEntry.getKey();
            int target = hoseEntry.getValue();

            // Unschedulable 사전 제외
            if (matrix.unschedulableHoseIds().contains(hose)) {
                conflicts.add(AllocationConflict.unschedulable(hose, target));
                continue;
            }

            int cumulativeYield = 0;
            var placedForHose = new ArrayList<VcSchedule>();

            // 호라이즌 영업일 순회
            for (LocalDate date : ctx.workingDays()) {
                if (cumulativeYield >= target) break;

                // 머신 유형 순회 (저압 우선 — ST-05-4가 정책 외부화)
                for (MachineType mt : List.of(MachineType.LP, MachineType.IC)) {
                    if (cumulativeYield >= target) break;

                    int yieldPerRot = yieldCalc.yieldPerRotation(hose, mt);
                    if (yieldPerRot == 0) continue;          // 해당 머신 유형 불가

                    // 회전 1~18 + 슬롯 시도
                    for (var slot : ledger.availableSlotsFor(date, mt)) {
                        if (cumulativeYield >= target) break;
                        if (!matrix.isEligible(hose, toSlotPosition(slot))) continue;

                        // 앵글 capa 사전 체크 (동일 회전·기존 점유 슬롯 수 + 1)
                        if (!fitsAngleCapacity(hose, mt, slot, placedForHose)) continue;

                        // 배치
                        var vc = VcSchedule.builder()
                            .hoseId(hose).machineId(slot.machineId())
                            .slotPosition((short) slot.slotPosition())
                            .productionDate(slot.date())
                            .rotationNo((short) slot.rotationNo())
                            .angleId(allocateAngleId(hose, slot))
                            .plannedQty(yieldPerRot)
                            .status(VcScheduleStatus.Candidate)
                            .linkedOrderIds(ctx.ordersByHose().get(hose).stream()
                                .map(Order::getOrderId).toArray(UUID[]::new))
                            .createdAt(java.time.Instant.now())
                            .updatedAt(java.time.Instant.now())
                            .build();

                        placedForHose.add(vc);
                        ledger.reserve(slot);
                        cumulativeYield += yieldPerRot;
                    }
                }
            }

            if (cumulativeYield < target) {
                conflicts.add(AllocationConflict.insufficientCapacity(
                    hose, target, cumulativeYield, placedForHose.size()));
            }
            schedules.addAll(placedForHose);
        }

        // 최종 검증 (방어적 — 앵글 capa 재검사)
        var assignments = schedules.stream()
            .collect(java.util.stream.Collectors.groupingBy(
                VcSchedule::getHoseId,
                java.util.stream.Collectors.mapping(VcSchedule::asSlot, java.util.stream.Collectors.toList())
            ));
        conflicts.addAll(angleValidator.validate(assignments).stream()
            .map(AllocationConflict::fromAngleViolation).toList());

        return new AllocationResult(schedules, conflicts);
    }

    private boolean fitsAngleCapacity(String hose, MachineType mt, RotationSlot newSlot,
                                       List<VcSchedule> alreadyPlaced) {
        long existingAtSameRotation = alreadyPlaced.stream()
            .filter(s -> s.getProductionDate().equals(newSlot.date())
                     && s.getRotationNo() == newSlot.rotationNo()
                     && s.getMachineId().startsWith(mt == MachineType.LP ? "LP-" : "IC-"))
            .count();
        int angleQty = (mt == MachineType.LP)
            ? lookupLpAngleQty(hose) : lookupIcAngleQty(hose);
        return existingAtSameRotation + 1 <= angleQty;
    }

    // 보조: toSlotPosition, allocateAngleId 생략
}

public record AllocationContext(
    Map<String, Integer> qRequired,
    Map<String, List<Order>> ordersByHose,
    CapacityLedger ledger,
    List<LocalDate> workingDays
) {}

public record AllocationResult(
    List<VcSchedule> schedules,
    List<AllocationConflict> conflicts
) {
    public boolean hasConflicts() { return !conflicts.isEmpty(); }
    public int scheduleCount() { return schedules.size(); }
}

public record AllocationConflict(String hoseId, String reason, int targetQty, int placedQty) {
    public static AllocationConflict unschedulable(String hose, int target) {
        return new AllocationConflict(hose, "UNSCHEDULABLE (BR-V11)", target, 0);
    }
    public static AllocationConflict insufficientCapacity(String hose, int target, int placed, int slotsUsed) {
        return new AllocationConflict(hose,
            "용량 부족 — 호라이즌 내 필요 %d 충족 못 함, %d 슬롯 사용 후 %d 배치".formatted(target, slotsUsed, placed),
            target, placed);
    }
    public static AllocationConflict fromAngleViolation(AngleCapacityViolation v) {
        return new AllocationConflict(v.hoseId(), v.userMessage(), 0, 0);
    }
}
```

## :memo: Implementation Notes

- 알고리즘 v1은 **결정적**(deterministic) — 같은 입력 → 같은 출력, 회귀 안정성
- 향후 v2에서 simulated annealing·constraint programming 도입 가능 (Phase 2+)
- `MutableLedger`는 격자 변경 추적용 — 입력 ledger는 불변 보존
- BR-V07 (앵글 교체 페널티)는 ST-21에서 정밀화 — v1은 단순 배치
- 결과 schedules는 Candidate 상태로 저장 (사용자 확정 게이트 BR-X01)
