### 핵심 로직

```java
package com.scheduling.vc.yield;

import com.scheduling.master.vc.*;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.context.event.EventListener;
import com.github.benmanes.caffeine.cache.*;

import java.util.*;
import java.util.concurrent.atomic.AtomicReference;

@Service @RequiredArgsConstructor
public class VcYieldCalculator {

    private final VcConstraintRepository constraintRepo;
    private final AtomicReference<YieldMatrix> current = new AtomicReference<>();

    @jakarta.annotation.PostConstruct
    public void initialBuild() { rebuild(); }

    /**
     * BR-V03: yield per rotation = composite_count × molds_per_angle
     */
    public int yieldPerRotation(String hoseId, MachineType type) {
        var matrix = current.get();
        if (matrix == null) rebuild();
        return matrix.lookup(hoseId, type).orElse(0);
    }

    public synchronized YieldMatrix rebuild() {
        var all = constraintRepo.findAll();
        var lpYields = new HashMap<String, Integer>();
        var icYields = new HashMap<String, Integer>();
        var unschedulableYields = new HashSet<String>();

        for (VcConstraint c : all) {
            int lpY = c.yieldPerRotation(MachineType.LP);    // composite × lp_molds_per_angle
            int icY = c.yieldPerRotation(MachineType.IC);    // composite × ic_molds_per_angle
            if (lpY > 0) lpYields.put(c.getHoseId(), lpY);
            if (icY > 0) icYields.put(c.getHoseId(), icY);
            if (lpY == 0 && icY == 0) unschedulableYields.add(c.getHoseId());
        }

        var matrix = new YieldMatrix(
            Map.copyOf(lpYields),
            Map.copyOf(icYields),
            Set.copyOf(unschedulableYields)
        );
        current.set(matrix);
        return matrix;
    }

    /** TK-04-1-2 LISTEN/NOTIFY 이벤트 수신 시 재계산 */
    @EventListener
    public void onConstraintChanged(VcConstraintChangedEvent event) {
        rebuild();
    }
}

public record YieldMatrix(
    Map<String, Integer> lpYields,           // hose_id → LP yield/rotation
    Map<String, Integer> icYields,           // hose_id → IC yield/rotation
    Set<String> unschedulableYields          // 둘 다 0인 품번
) {
    public Optional<Integer> lookup(String hoseId, MachineType type) {
        Map<String, Integer> map = type == MachineType.LP ? lpYields : icYields;
        return Optional.ofNullable(map.get(hoseId));
    }
}
```
