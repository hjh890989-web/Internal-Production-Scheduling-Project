### 핵심 로직

```java
package com.scheduling.vc.yield;

import com.scheduling.master.vc.*;
import com.scheduling.vc.domain.RotationSlot;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.*;
import java.util.stream.Collectors;

@Service @RequiredArgsConstructor
public class AngleCapacityValidator {

    private final VcConstraintRepository constraintRepo;

    /**
     * 같은 (date, hose_id, machine_type)의 동시 점유 슬롯 수가
     * 앵글 보유량을 초과하지 않는지 검증.
     *
     * BR-V06: hose_id별로 동시에 사용할 수 있는 슬롯 = lp_angle_qty (저압) 또는 ic_angle_qty (IC)
     */
    public List<AngleCapacityViolation> validate(Map<String, List<RotationSlot>> assignments) {
        var violations = new ArrayList<AngleCapacityViolation>();

        // 1. (date, hose_id, machine_type) → 동시 점유 슬롯 수 집계
        var concurrentUsage = new HashMap<UsageKey, Set<Integer>>();
        for (var entry : assignments.entrySet()) {
            String hoseId = entry.getKey();
            for (RotationSlot slot : entry.getValue()) {
                MachineType mt = slot.machineId().startsWith("LP-") ? MachineType.LP : MachineType.IC;
                var key = new UsageKey(slot.date(), hoseId, mt, slot.rotationNo());
                concurrentUsage.computeIfAbsent(key, k -> new HashSet<>())
                    .add(slot.slotPosition());
            }
        }

        // 2. 각 키별 앵글 가용량 비교
        for (var entry : concurrentUsage.entrySet()) {
            UsageKey k = entry.getKey();
            int concurrentSlots = entry.getValue().size();
            int allowed = getAngleQty(k.hoseId(), k.machineType());

            if (concurrentSlots > allowed) {
                violations.add(new AngleCapacityViolation(
                    k.date(), k.hoseId(), k.machineType(), k.rotationNo(),
                    concurrentSlots, allowed
                ));
            }
        }
        return violations;
    }

    public boolean isWithinCapacity(String hoseId, MachineType type, int requestedSlots) {
        return requestedSlots <= getAngleQty(hoseId, type);
    }

    private int getAngleQty(String hoseId, MachineType type) {
        return constraintRepo.findById(hoseId).map(c -> switch (type) {
            case LP -> c.getLpAngleQty() != null ? c.getLpAngleQty() : 0;
            case IC -> c.getIcAngleQty() != null ? c.getIcAngleQty() : 0;
        }).orElse(0);
    }
}

public record UsageKey(LocalDate date, String hoseId, MachineType machineType, int rotationNo) {}

public record AngleCapacityViolation(
    LocalDate date,
    String hoseId,
    MachineType machineType,
    int rotationNo,
    int actualSlotsUsed,
    int allowedAngles
) {
    public String userMessage() {
        return "앵글 과초과: %s %s 회전 %d에서 %s 가류기 슬롯 %d개 점유, 보유 %d개 한도 초과"
            .formatted(date, hoseId, rotationNo, machineType, actualSlotsUsed, allowedAngles);
    }
}
```
