### 핵심 로직

```java
package com.scheduling.vc.yield;

import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;

import java.time.LocalDate;
import java.util.*;

import static org.assertj.core.api.Assertions.*;

@SpringBootTest
class AngleCapacityStressTest {

    @Autowired AngleCapacityValidator validator;

    /**
     * TC-VC-007: 1000회 무작위 배치 시도에서 앵글 과초과 case 100% 식별
     */
    @Test
    void detects_all_angle_over_subscriptions_in_1000_random_trials() {
        var random = new Random(42);
        int knownViolations = 0;
        int detected = 0;

        for (int trial = 0; trial < 1000; trial++) {
            var assignments = generateRandomAssignment(random);
            boolean expectViolation = shouldViolateAngleCapa(assignments);

            var violations = validator.validate(assignments);

            if (expectViolation) {
                knownViolations++;
                if (!violations.isEmpty()) detected++;
            } else {
                // 무위반 시나리오에서 false-positive 0건
                assertThat(violations)
                    .as("Trial %d: expected no violations but got %d", trial, violations.size())
                    .isEmpty();
            }
        }

        // 100% detection
        assertThat(detected)
            .as("Detection rate: %d/%d", detected, knownViolations)
            .isEqualTo(knownViolations);
    }

    @Test
    void no_violation_within_capacity_simple_case() {
        // 29673-2F900 lp_angle_qty=1, 슬롯 1개 점유 → 위반 X
        var assignments = Map.of("29673-2F900",
            List.of(slot(LocalDate.of(2026, 2, 16), "LP-01", 1, 2)));
        assertThat(validator.validate(assignments)).isEmpty();
    }

    @Test
    void detects_violation_simple_case() {
        // 29673-2F900 lp_angle_qty=1, 같은 회전·다른 슬롯 2개 → 위반
        var assignments = Map.of("29673-2F900", List.of(
            slot(LocalDate.of(2026, 2, 16), "LP-01", 1, 2),
            slot(LocalDate.of(2026, 2, 16), "LP-02", 1, 3)        // 같은 회전, 다른 머신
        ));
        // 하지만 같은 회전이라도 다른 머신이면 별 앵글 → 위반 아닐 수도
        // 실제 정책 검증 (BR-V06 정확 해석에 따라)
    }

    // 보조 메서드 생략 (generateRandomAssignment, shouldViolateAngleCapa 등)
}
```

### 데이터셋 generator (`generate_angle_stress_dataset.py`)

```python
"""
DS-ANGLE-STRESS-1000: REF-09 47품번 + 임의 슬롯 배정 1000건.
shouldViolateAngleCapa 정답을 함께 생성.
"""
import json
import random
from pathlib import Path

random.seed(42)
REF_PRODUCTS = load_ref09_47_products()        # (hose_id, lp_angle_qty, ic_angle_qty)

scenarios = []
for trial in range(1000):
    hose = random.choice(REF_PRODUCTS)
    # 의도적으로 50% 케이스는 위반 만들기
    n_slots = random.choice([
        random.randint(1, hose.lp_angle_qty),               # 위반 X
        hose.lp_angle_qty + random.randint(1, 3)            # 위반 O
    ])
    slots = [{"machine_id": "LP-01", "rotation_no": 1, "slot_position": s}
             for s in range(1, n_slots + 1)]
    expected_violation = n_slots > hose.lp_angle_qty

    scenarios.append({
        "trial": trial,
        "hose_id": hose.hose_id,
        "slots": slots,
        "expected_violation": expected_violation
    })

# 합성 → JSON
(Path("backend/scheduling-engine/src/test/resources/datasets/DS-ANGLE-STRESS-1000")
 / "stress_scenarios.json").write_text(json.dumps(scenarios, indent=2))
```
