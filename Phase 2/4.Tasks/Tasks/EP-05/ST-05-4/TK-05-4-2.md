### Flyway `V2_4`

```sql
CREATE TABLE audit.machine_decision (
    decision_id        UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    decided_at         TIMESTAMPTZ NOT NULL DEFAULT now(),
    hose_id            VARCHAR(40) NOT NULL,
    production_date    DATE NOT NULL,
    machine_id         VARCHAR(10) NOT NULL,
    machine_type       VARCHAR(2) NOT NULL,
    decision_type      VARCHAR(20) NOT NULL CHECK (decision_type IN
        ('LP_PRIMARY','LP_FALLBACK','IC_PRIMARY','IC_FALLBACK')),
    policy_id          VARCHAR(20) NOT NULL,
    reason             TEXT,
    actor              VARCHAR(40) NOT NULL DEFAULT 'system:allocator'
);

CREATE INDEX idx_machine_decision_date ON audit.machine_decision(production_date);
CREATE INDEX idx_machine_decision_type ON audit.machine_decision(decision_type, decided_at);
REVOKE UPDATE, DELETE ON audit.machine_decision FROM app_user;
GRANT INSERT, SELECT ON audit.machine_decision TO app_user;
```

### `RoutingAuditLogger.java`

```java
@Service @RequiredArgsConstructor
public class RoutingAuditLogger {

    private final MachineDecisionRepository repo;

    public void logDecision(String hoseId, LocalDate date, String machineId, MachineType type,
                            DecisionType decisionType, String policyId, String reason) {
        var audit = MachineDecisionAudit.builder()
            .decidedAt(Instant.now())
            .hoseId(hoseId)
            .productionDate(date)
            .machineId(machineId)
            .machineType(type.name())
            .decisionType(decisionType)
            .policyId(policyId)
            .reason(reason)
            .actor(currentActor())
            .build();
        repo.save(audit);
    }
}

public enum DecisionType {
    LP_PRIMARY,    // BR-V08 정책에 따라 첫 시도 → LP
    LP_FALLBACK,   // IC_FIRST 정책인데 IC 포화로 LP 폴백
    IC_PRIMARY,    // IC_FIRST 또는 LP 자격 없음
    IC_FALLBACK    // LP 포화 (BR-V08 fallback)
}
```
