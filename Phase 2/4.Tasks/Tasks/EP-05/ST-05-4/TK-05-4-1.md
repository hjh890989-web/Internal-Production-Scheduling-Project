### 핵심 로직

```java
package com.scheduling.vc.routing;

import com.scheduling.master.vc.MachineType;
import java.util.List;

public interface MachineTypeRoutingPolicy {
    /**
     * 주어진 hose_id에 대해 머신 유형 우선순위 반환.
     * 첫 번째 원소부터 시도 → 포화 시 다음.
     */
    List<MachineType> prioritize(String hoseId, RoutingContext ctx);

    String policyId();
}

@Component
public class LpFirstThenIcRoutingPolicy implements MachineTypeRoutingPolicy {

    @Override
    public List<MachineType> prioritize(String hoseId, RoutingContext ctx) {
        // BR-V08: 둘 다 가능하면 LP 우선
        return List.of(MachineType.LP, MachineType.IC);
    }

    @Override
    public String policyId() { return "LP_FIRST"; }
}

// 대안 정책 (정책 변경 시 사용)
@Component
public class IcFirstRoutingPolicy implements MachineTypeRoutingPolicy {
    public List<MachineType> prioritize(String h, RoutingContext c) {
        return List.of(MachineType.IC, MachineType.LP);
    }
    public String policyId() { return "IC_FIRST"; }
}

@ConfigurationProperties(prefix = "scheduling.routing")
@Data
public class RoutingPolicyConfig {
    private String activePolicy = "LP_FIRST";    // BR-V08 기본
}

public record RoutingContext(
    java.time.LocalDate date,
    int currentLpUtilizationPct,
    int currentIcUtilizationPct
) {}
```

### `application.yaml`

```yaml
scheduling:
  routing:
    active-policy: LP_FIRST              # BR-V08 (ADR-006 성형판)
    # 대안: IC_FIRST, BALANCED
```
