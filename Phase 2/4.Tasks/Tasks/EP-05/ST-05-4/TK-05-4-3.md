### 핵심 로직

```java
@SpringBootTest @Testcontainers
@Sql("classpath:datasets/DS-VC-CONSTRAINT-47/master_seed.sql")
class LpFirstFallbackIT {

    @Autowired GreedyRotationAllocator allocator;
    @Autowired MachineDecisionRepository auditRepo;

    @Test
    void lp_saturates_first_then_ic_fallback() {
        // 1주 호라이즌, LP 자격 + IC 자격 둘 다 있는 품번만, Q_required 폭증
        var ctx = buildDualCapableContext(LocalDate.of(2026, 2, 16), 7,
            /*qPerHose*/ 500);     // 의도적 LP 포화

        var result = allocator.allocate(ctx);

        // 1. LP 사용량 확인 — 호라이즌 가용 capa(저압 4 × 18 회전 × 8슬롯 × 5일 = 2880 슬롯·회전)에 근접
        long lpDecisions = auditRepo.countByDecisionType(DecisionType.LP_PRIMARY);
        long icFallbacks = auditRepo.countByDecisionType(DecisionType.IC_FALLBACK);

        // 2. LP_PRIMARY가 IC_FALLBACK보다 훨씬 많음 (BR-V08)
        assertThat(lpDecisions).isGreaterThan(icFallbacks);

        // 3. IC 사용은 LP 포화 이후만 — 시간 순서 검증
        var first = auditRepo.findFirstByOrderByDecidedAtAsc();
        var firstIc = auditRepo.findFirstByDecisionTypeOrderByDecidedAtAsc(DecisionType.IC_FALLBACK);
        if (firstIc.isPresent()) {
            assertThat(firstIc.get().getDecidedAt()).isAfterOrEqualTo(first.get().getDecidedAt());
        }
    }

    @Test
    void ic_only_eligible_product_routes_to_ic_primary() {
        // IC 자격만 있는 품번 (예: 29689-2U000) → IC_PRIMARY
        var ctx = buildIcOnlyContext();
        allocator.allocate(ctx);

        var decisions = auditRepo.findByHoseId("29689-2U000");
        assertThat(decisions).extracting(MachineDecisionAudit::getDecisionType)
            .containsOnly(DecisionType.IC_PRIMARY);
    }

    @Test
    void lp_only_eligible_product_routes_to_lp_primary() {
        // LP 자격만 (예: A6722030900) → LP_PRIMARY
        var ctx = buildLpOnlyContext();
        allocator.allocate(ctx);

        var decisions = auditRepo.findByHoseId("A6722030900");
        assertThat(decisions).extracting(MachineDecisionAudit::getDecisionType)
            .containsOnly(DecisionType.LP_PRIMARY);
    }
}
```
