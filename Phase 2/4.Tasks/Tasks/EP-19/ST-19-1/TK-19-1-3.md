### 핵심 로직

```java
@SpringBootTest @Testcontainers
class MasterTimeTravelLoadIT {

    @Autowired MasterTimeTravelService service;
    @Autowired JdbcTemplate jdbc;

    private static final int RUNS = 30;

    @BeforeAll
    void setupLoadData() {
        // 5년치 audit row 생성 (10^6 / 5 = 200K / year ≈ 550/day)
        for (int year = 2026; year <= 2030; year++) {
            jdbc.update("""
                INSERT INTO audit.vc_schedule_log (target_id, operation, old_row, new_row, actor, occurred_at)
                SELECT
                    gen_random_uuid(),
                    (ARRAY['INSERT', 'UPDATE', 'DELETE'])[1 + floor(random() * 3)::int],
                    NULL,
                    jsonb_build_object('id', gen_random_uuid(), 'hose_id', 'TEST-' || s, 'production_date', current_date - (s % 365)),
                    'load-test',
                    timestamp ? || ' ' || (s::text || ':00')::interval
                FROM generate_series(1, 200000) s
            """, year + "-01-01");
        }
    }

    @Test
    void snapshot_5_years_back_p95_under_5s() {
        var oldTimestamp = Instant.parse("2026-06-01T00:00:00Z");
        var latencies = new ArrayList<Long>();

        // Warm-up
        for (int i = 0; i < 5; i++) service.snapshotAt("vc_schedule_log", oldTimestamp);

        // Measure
        for (int i = 0; i < RUNS; i++) {
            long start = System.nanoTime();
            service.snapshotAt("vc_schedule_log", oldTimestamp);
            latencies.add((System.nanoTime() - start) / 1_000_000);
        }

        Collections.sort(latencies);
        long p50 = latencies.get(RUNS / 2);
        long p95 = latencies.get((int) (RUNS * 0.95));

        System.out.printf("5-year snapshot: p50=%dms, p95=%dms%n", p50, p95);

        assertThat(p95).as("TC-XT-002 p95 ≤ 5초").isLessThanOrEqualTo(5000);
    }

    @Test
    void recent_timestamp_p95_under_2s() {
        var recent = Instant.now().minusSeconds(86400);     // 1일 전
        var latencies = new ArrayList<Long>();

        for (int i = 0; i < RUNS; i++) {
            long start = System.nanoTime();
            service.snapshotAt("vc_schedule_log", recent);
            latencies.add((System.nanoTime() - start) / 1_000_000);
        }

        long p95 = latencies.stream().sorted().toList().get((int) (RUNS * 0.95));
        assertThat(p95).as("recent snapshot 빠름").isLessThanOrEqualTo(2000);
    }

    @Test
    void partition_pruning_active() {
        // EXPLAIN ANALYZE 결과에 partition pruning 확인
        var plan = jdbc.queryForList(
            "EXPLAIN ANALYZE SELECT * FROM audit.vc_schedule_log WHERE occurred_at > ?",
            Instant.parse("2029-01-01T00:00:00Z"));

        String planText = plan.stream().map(r -> r.values().toString()).collect(Collectors.joining());
        assertThat(planText)
            .as("partition pruning 확인 — 2029년 partition만 scan")
            .contains("vc_schedule_log_2029");
    }
}
```
