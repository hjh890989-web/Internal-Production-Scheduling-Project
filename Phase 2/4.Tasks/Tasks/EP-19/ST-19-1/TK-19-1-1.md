### `MasterTimeTravelService.java`

```java
@Service @RequiredArgsConstructor @Slf4j
public class MasterTimeTravelService {

    private final JdbcTemplate jdbc;
    private final ObjectMapper mapper;

    /**
     * 특정 timestamp의 master 상태 snapshot.
     *  - 현재 master 상태에서 시작
     *  - audit.occurred_at > timestamp 변경 역재생
     */
    public SnapshotResult snapshotAt(String auditTable, Instant timestamp) {
        // 1) 현재 master 상태 수집 (target_id별)
        String masterTable = inferMasterTable(auditTable);     // vc_schedule_log → vc_schedule
        var currentRows = jdbc.queryForList("SELECT * FROM " + masterTable);
        var byId = currentRows.stream()
            .collect(Collectors.toMap(r -> r.get("id").toString(), r -> r));

        // 2) audit row > timestamp DESC 순회 (역재생)
        var auditRows = jdbc.queryForList(
            "SELECT operation, target_id, old_row, new_row FROM audit." + auditTable +
            " WHERE occurred_at > ? ORDER BY occurred_at DESC",
            timestamp);

        for (var audit : auditRows) {
            String op = (String) audit.get("operation");
            String targetId = audit.get("target_id").toString();

            switch (op) {
                case "INSERT" -> byId.remove(targetId);    // INSERT 역 = DELETE
                case "DELETE" -> {
                    var oldRow = parseJsonb(audit.get("old_row"));
                    if (oldRow != null) byId.put(targetId, oldRow);
                }
                case "UPDATE" -> {
                    var oldRow = parseJsonb(audit.get("old_row"));
                    if (oldRow != null) byId.put(targetId, oldRow);
                }
            }
        }

        return new SnapshotResult(timestamp, masterTable, byId.values().stream().toList());
    }

    /**
     * 미리보기 diff: snapshot vs 현재 master.
     */
    public RestoreDiff previewRestore(String auditTable, Instant timestamp) {
        var snapshot = snapshotAt(auditTable, timestamp);
        var currentRows = jdbc.queryForList("SELECT * FROM " + snapshot.masterTable());

        var snapshotIds = snapshot.rows().stream()
            .map(r -> r.get("id").toString())
            .collect(Collectors.toSet());
        var currentIds = currentRows.stream()
            .map(r -> r.get("id").toString())
            .collect(Collectors.toSet());

        var toDelete = currentIds.stream().filter(id -> !snapshotIds.contains(id)).toList();
        var toAdd = snapshotIds.stream().filter(id -> !currentIds.contains(id)).toList();
        // UPDATE 감지는 row diff 필요 (간소화)

        return new RestoreDiff(timestamp, toAdd.size(), toDelete.size(),
            currentIds.size() - toDelete.size());     // 변경 안 됨
    }

    /**
     * 실 복원 — Master Admin only, transaction 내.
     * 중요: 본 작업은 audit 추가 row 생성 (역시 추적성)
     */
    @Auditable(reason = "'restore master to ' + #timestamp")
    @Transactional
    public void restore(String auditTable, Instant timestamp) {
        var snapshot = snapshotAt(auditTable, timestamp);
        String masterTable = snapshot.masterTable();

        // 단순 구현: DELETE ALL + INSERT FROM snapshot
        // 실 운영: 부분 UPDATE 권장 (현재는 PoC 수준)
        jdbc.update("DELETE FROM " + masterTable);
        for (var row : snapshot.rows()) {
            // INSERT row — JSONB → SQL 변환
            // ... 동적 INSERT 생성
        }

        log.warn("Master 복원 실행: table={}, restored to {}", masterTable, timestamp);
    }

    private Map<String, Object> parseJsonb(Object jsonb) {
        if (jsonb == null) return null;
        try {
            return mapper.readValue(jsonb.toString(), new TypeReference<>() {});
        } catch (Exception e) {
            log.error("JSONB parse 실패", e);
            return null;
        }
    }

    private String inferMasterTable(String auditTable) {
        // "vc_schedule_log" → "vc_schedule"
        return auditTable.replace("_log", "").replace("audit.", "");
    }
}
```
