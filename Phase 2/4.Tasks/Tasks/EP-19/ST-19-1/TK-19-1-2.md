### `TimeTravelPage.tsx`

```tsx
import { useState } from 'react';
import { useKeycloak } from '@react-keycloak/web';
import { Navigate } from 'react-router-dom';
import { Card, Select, Alert } from 'antd';
import { TimestampSlider } from './TimestampSlider';
import { DiffPreview } from './DiffPreview';
import { RestoreConfirmModal } from './RestoreConfirmModal';
import { useQuery } from '@tanstack/react-query';
import { timeTravelApi } from '@/api/timeTravelApi';

const AUDIT_TABLES = [
  { value: 'vc_schedule_log',  label: '성형 스케줄' },
  { value: 'ex_schedule_log',  label: '압출 스케줄' },
  { value: 'order_log',         label: '수주' },
];

export function TimeTravelPage() {
  const { keycloak } = useKeycloak();
  const [auditTable, setAuditTable] = useState(AUDIT_TABLES[0].value);
  const [timestamp, setTimestamp] = useState<Date>(/* 1 week ago default */);
  const [restoreOpen, setRestoreOpen] = useState(false);

  const { data: diff } = useQuery({
    queryKey: ['masterDiff', auditTable, timestamp],
    queryFn: () => timeTravelApi.preview(auditTable, timestamp.toISOString()),
    enabled: !!timestamp,
  });

  if (!keycloak.hasRealmRole('PLANNER') && !keycloak.hasRealmRole('MASTER_ADMIN')) {
    return <Navigate to="/unauthorized" replace />;
  }

  const isAdmin = keycloak.hasRealmRole('MASTER_ADMIN');

  return (
    <div>
      <h1>마스터 시점 복원</h1>
      <Alert
        message="시점 슬라이더로 과거 시점 마스터 상태를 미리볼 수 있습니다. 실제 복원은 Master Admin만 가능합니다."
        type="info"
      />
      <Card>
        <Select
          value={auditTable}
          onChange={setAuditTable}
          options={AUDIT_TABLES}
          style={{ width: 200 }}
        />
        <TimestampSlider value={timestamp} onChange={setTimestamp} />
      </Card>

      {diff && <DiffPreview diff={diff} />}

      {isAdmin && diff && (
        <Button danger onClick={() => setRestoreOpen(true)}>
          {timestamp.toLocaleString()} 시점으로 복원 실행
        </Button>
      )}

      <RestoreConfirmModal
        open={restoreOpen}
        timestamp={timestamp}
        auditTable={auditTable}
        diff={diff}
        onClose={() => setRestoreOpen(false)}
      />
    </div>
  );
}
```

### `TimestampSlider.tsx`

```tsx
import { Slider, Radio } from 'antd';
import { useMemo } from 'react';

interface Props { value: Date; onChange: (d: Date) => void; }

export function TimestampSlider({ value, onChange }: Props) {
  const [zoom, setZoom] = useState<'week' | 'month' | 'year'>('week');

  const { min, max } = useMemo(() => {
    const now = Date.now();
    switch (zoom) {
      case 'week':  return { min: now - 7 * 86400000, max: now };
      case 'month': return { min: now - 30 * 86400000, max: now };
      case 'year':  return { min: now - 365 * 86400000, max: now };
    }
  }, [zoom]);

  return (
    <div>
      <Radio.Group value={zoom} onChange={(e) => setZoom(e.target.value)}>
        <Radio.Button value="week">1주</Radio.Button>
        <Radio.Button value="month">1개월</Radio.Button>
        <Radio.Button value="year">1년</Radio.Button>
      </Radio.Group>
      <Slider
        min={min}
        max={max}
        value={value.getTime()}
        onChange={(v) => onChange(new Date(v))}
        tooltip={{ formatter: (v) => new Date(v!).toLocaleString('ko-KR') }}
      />
    </div>
  );
}
```

### `RestoreConfirmModal.tsx`

```tsx
import { Modal, Form, Input, Alert, Button } from 'antd';
import { useMutation } from '@tanstack/react-query';

interface Props { open: boolean; timestamp: Date; auditTable: string; diff: RestoreDiff; onClose: () => void; }

export function RestoreConfirmModal({ open, timestamp, auditTable, diff, onClose }: Props) {
  const [form] = Form.useForm();

  const { mutate, isPending } = useMutation({
    mutationFn: timeTravelApi.restore,
    onSuccess: () => {
      message.success('복원 완료. 감사 로그에 기록됩니다.');
      onClose();
    },
  });

  return (
    <Modal title="⚠️ 마스터 복원 — 위험 작업" open={open} onCancel={onClose} footer={null}>
      <Alert
        message="이 작업은 되돌릴 수 없습니다. 30자 이상 사유 입력 후 진행하세요."
        type="error"
        showIcon
      />
      <p>대상 테이블: <b>{auditTable}</b></p>
      <p>복원 시점: <b>{timestamp.toLocaleString('ko-KR')}</b></p>
      <p>변경: +{diff.toAdd} 행 추가, -{diff.toDelete} 행 삭제</p>

      <Form form={form} onFinish={(v) => mutate({ auditTable, timestamp: timestamp.toISOString(), reason: v.reason })}>
        <Form.Item name="reason" label="복원 사유 (≥ 30자)"
                    rules={[{ required: true, min: 30 }]}>
          <Input.TextArea rows={3} placeholder="예: 2026-03-10 09:00에 잘못된 데이터 일괄 입력 — 그 이전으로 복원" />
        </Form.Item>
        <Form.Item name="confirm" label="확인 문구 — '복원' 입력"
                    rules={[{ required: true, pattern: /^복원$/, message: '"복원"을 정확히 입력하세요' }]}>
          <Input />
        </Form.Item>
        <Button danger htmlType="submit" loading={isPending}>최종 복원 실행</Button>
      </Form>
    </Modal>
  );
}
```
