### `OrderLoadFixtureGenerator.java`

```java
@Component
public class OrderLoadFixtureGenerator {

    public byte[] generate10K(long seed) throws IOException {
        var random = new Random(seed);
        try (XSSFWorkbook wb = new XSSFWorkbook();
             ByteArrayOutputStream baos = new ByteArrayOutputStream()) {

            Sheet sheet = wb.createSheet("Orders");
            createHeader(sheet);

            for (int i = 0; i < 10_000; i++) {
                Row row = sheet.createRow(i + 1);
                row.createCell(0).setCellValue("ORD-" + i);
                row.createCell(1).setCellValue(pickHose(random));
                row.createCell(2).setCellValue(1 + random.nextInt(1000));        // quantity
                row.createCell(3).setCellValue(deliveryDate(random));
                row.createCell(4).setCellValue("CUSTOMER-" + (i % 50));

                // 의도적 invalid row 20% — 검증 실패 케이스 측정
                if (i % 5 == 0) {
                    row.createCell(2).setCellValue(-1);    // 음수 quantity
                }
            }

            wb.write(baos);
            return baos.toByteArray();
        }
    }
}
```
