### `import-load.js`

```javascript
import http from 'k6/http';
import { check } from 'k6';
import { Trend } from 'k6/metrics';

export const options = {
  vus: 5,
  duration: '10m',
  thresholds: {
    'http_req_duration{stage:import}': ['p(95)<60000'],
  },
};

const importLatency = new Trend('import_latency_ms');

export function setup() {
  const token = authenticate();
  const fixture = open('../../backend/scheduling-engine/src/test/resources/load-data/orders_10k.xlsx', 'b');
  return { token, fixture };
}

export default function (data) {
  const res = http.post(
    `${__ENV.API_BASE}/api/v1/import`,
    { file: http.file(data.fixture, 'orders_10k.xlsx') },
    { headers: { Authorization: `Bearer ${data.token}` }, tags: { stage: 'import' } }
  );

  check(res, {
    'status 200': (r) => r.status === 200,
    'p95 ≤ 60s': (r) => r.timings.duration < 60000,
  });

  importLatency.add(res.timings.duration);
}
```
