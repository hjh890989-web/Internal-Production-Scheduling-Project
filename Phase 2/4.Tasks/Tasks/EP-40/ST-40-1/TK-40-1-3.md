### `Jenkinsfile.performance-nightly`

```groovy
pipeline {
    agent { label 'k6' }
    triggers { cron('H 2 * * *') }      // nightly 02:00 KST

    stages {
        stage('Setup') {
            steps {
                sh 'docker compose -f infra/docker-compose.test.yml up -d'
                sh 'k6 version'
            }
        }
        stage('Load Test') {
            steps {
                sh 'k6 run --out=experimental-prometheus-rw load-tests/k6/import-load.js'
            }
        }
        stage('Verify SLO') {
            steps {
                script {
                    def report = readJSON file: 'k6-results.json'
                    def p95 = report.metrics.http_req_duration.values['p(95)']
                    if (p95 > 60000) {
                        error "p95 ${p95}ms 초과 (REQ-NF-PER-001 ≤ 60s)"
                    }
                }
            }
        }
    }
    post {
        failure {
            slackSend channel: '#scheduling-alerts',
                color: 'danger',
                message: "Performance Regression: PER-001 p95 violation (build ${BUILD_NUMBER})"
        }
    }
}
```
