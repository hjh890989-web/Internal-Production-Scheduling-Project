### `vc-allocator-load.js`

```javascript
import http from 'k6/http';
import { Trend } from 'k6/metrics';

export const options = { iterations: 50, thresholds: { vc_latency: ['p(95)<300000'] } };
const vcLatency = new Trend('vc_latency');

export default function () {
  const token = __ENV.AUTH_TOKEN;
  const res = http.post(`${__ENV.API_BASE}/api/v1/schedule/vc/candidates`,
    JSON.stringify({ weekStart: '2026-03-02' }),
    { headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` } });
  vcLatency.add(res.timings.duration);
}
```
