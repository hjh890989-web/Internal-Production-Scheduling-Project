### `AllocatorSloVerifier.java`

```java
@Component @RequiredArgsConstructor
public class AllocatorSloVerifier {

    @Test
    void vc_p95_under_5min() {
        var latencies = runVcAllocator(50);
        long p95 = percentile(latencies, 0.95);
        System.out.printf("VC allocator p95: %dms (SLO: 300000)%n", p95);
        assertThat(p95).isLessThanOrEqualTo(300_000L);
    }

    @Test
    void ex_p95_under_2min() {
        var latencies = runExAllocator(50);
        long p95 = percentile(latencies, 0.95);
        assertThat(p95).isLessThanOrEqualTo(120_000L);
    }
}
```
