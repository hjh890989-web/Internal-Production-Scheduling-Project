### `ExtrusionGatePerformanceIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-EX-GATE-600/master_seed.sql",
      "classpath:datasets/DS-EX-GATE-600/candidates_seed.sql"})
class ExtrusionGatePerformanceIT {

    @Autowired ExtrusionValidationGate gate;
    @Autowired ExScheduleCandidateRepository candidateRepo;
    @Autowired SettingGroupAllocator allocator;

    private static final int WARMUP_RUNS = 5;
    private static final int MEASURE_RUNS = 50;

    @Test
    void p95_per_candidate_under_two_seconds() {
        var candidates = candidateRepo.findAll();
        assertThat(candidates).hasSize(600);

        var ctx = ExAllocationContext.builder().candidates(candidates)
            .workingDays(workingDays(7)).build();
        var allocResult = allocator.allocate(ctx);
        var ledger = ExtrusionLedger.fromAllocationResult(allocResult);

        // Warm-up — JIT + 캐시 적재
        for (int i = 0; i < WARMUP_RUNS; i++) gate.validateAll(candidates, ledger);

        var latenciesPerCandidate = new ArrayList<Long>();
        for (int i = 0; i < MEASURE_RUNS; i++) {
            long batchStart = System.nanoTime();
            gate.validateAll(candidates, ledger);
            long batchMs = (System.nanoTime() - batchStart) / 1_000_000;

            // 후보당 평균 — 600 candidate batch / 600 = per candidate avg
            latenciesPerCandidate.add(batchMs / candidates.size());
        }

        Collections.sort(latenciesPerCandidate);
        long p50 = latenciesPerCandidate.get((int) (MEASURE_RUNS * 0.50));
        long p95 = latenciesPerCandidate.get((int) (MEASURE_RUNS * 0.95));
        long p99 = latenciesPerCandidate.get((int) (MEASURE_RUNS * 0.99));

        System.out.printf("validation per candidate: p50=%dms, p95=%dms, p99=%dms%n", p50, p95, p99);

        // SRS 요구: 후보당 p95 ≤ 2,000ms
        assertThat(p95)
            .as("REQ-FUNC-EX-011 후보당 p95 ≤ 2초")
            .isLessThanOrEqualTo(2000);
    }

    @Test
    void p95_batch_600_candidates_under_60_seconds() {
        // 600 candidate batch 전체 — 후보당 2초 × 600 / 4-core = ≈ 300s (worst-case)
        // 현실 목표: 30 ~ 60초 (parallelStream + cache hit)
        var candidates = candidateRepo.findAll();
        var ledger = buildLedger(candidates);

        long start = System.nanoTime();
        gate.validateAll(candidates, ledger);
        long ms = (System.nanoTime() - start) / 1_000_000;

        assertThat(ms).isLessThanOrEqualTo(60_000);
    }

    @Test
    void cold_cache_first_validation_under_5_seconds() {
        // 캐시 비어있는 worst-case first call
        var sample = candidateRepo.findAll().get(0);
        var ledger = new ExtrusionLedger();

        long start = System.nanoTime();
        gate.validate(sample, ledger);
        long ms = (System.nanoTime() - start) / 1_000_000;

        assertThat(ms).isLessThanOrEqualTo(5_000);
    }
}
```
