### `ExtrusionMatrixPage.tsx`

```tsx
import { useState } from 'react';
import { useKeycloak } from '@react-keycloak/web';
import { Navigate } from 'react-router-dom';
import { Tabs } from 'antd';
import { ShiftMatrix } from './ShiftMatrix';
import { GanttView } from './GanttView';
import { DownloadExcelButton } from './DownloadExcelButton';
import { DateRangePicker } from '@/components/DateRangePicker';
import { useExtrusionSchedule } from '@/hooks/useExtrusionSchedule';
import { useExtrusionWebSocket } from '@/hooks/useExtrusionWebSocket';

export function ExtrusionMatrixPage() {
  const { keycloak } = useKeycloak();
  useExtrusionWebSocket();

  const [range, setRange] = useState(defaultWeekRange());
  const { data: rows = [] } = useExtrusionSchedule(range);

  if (!keycloak.hasRealmRole('STK_USER') && !keycloak.hasRealmRole('PLANNER')) {
    return <Navigate to="/unauthorized" replace />;
  }

  return (
    <div className="matrix-page">
      <header>
        <h1>압출 매트릭스 — S-05</h1>
        <DateRangePicker value={range} onChange={setRange} />
        <DownloadExcelButton from={range.from} to={range.to} />
      </header>

      <Tabs
        items={[
          { key: 'matrix', label: '매트릭스', children: <ShiftMatrix rows={rows} from={range.from} to={range.to} /> },
          { key: 'gantt', label: 'Gantt', children: <GanttView rows={rows} viewMode="Week" /> },
        ]}
      />
    </div>
  );
}
```

### `ShiftMatrix.tsx`

```tsx
import { useMemo } from 'react';
import { AgGridReact } from 'ag-grid-react';
import { ExScheduleRow } from '@/types/extrusion';
import { ScheduleCell } from '@/pages/simview/ScheduleCell';

interface Props {
  rows: ExScheduleRow[];
  from: string;
  to: string;
}

const SHIFT_LABELS = [
  { id: 'DAY_FIRST',    label: '주간전반' },
  { id: 'DAY_SECOND',   label: '주간후반' },
  { id: 'NIGHT_FIRST',  label: '야간전반' },
  { id: 'NIGHT_SECOND', label: '야간후반' },
];

export function ShiftMatrix({ rows, from, to }: Props) {
  const columnDefs = useMemo(() => {
    const dates = computeWorkingDays(from, to);
    return [
      { headerName: 'Shift', field: 'shiftLabel', width: 120, pinned: 'left' },
      ...dates.map(d => ({
        headerName: formatKoreanDate(d),
        field: d,
        cellRenderer: ScheduleCell,
        width: 220,
        autoHeight: true,
      })),
    ];
  }, [from, to]);

  const rowData = useMemo(() => {
    return SHIFT_LABELS.map(shift => {
      const cellsByDate: Record<string, MatrixCell> = {};
      for (const row of rows) {
        if (row.shiftCode !== shift.id) continue;
        if (!cellsByDate[row.productionDate]) {
          cellsByDate[row.productionDate] = { rows: [], totalYield: 0, colorClass: '' };
        }
        cellsByDate[row.productionDate].rows.push({
          ...row,
          hoseId: `${row.hoseId} [${row.machineId}]`,    // 라인 표시
        });
        cellsByDate[row.productionDate].totalYield += row.yield;
        cellsByDate[row.productionDate].colorClass = `setting-group-${row.settingGroup}`;
      }
      return { shiftLabel: shift.label, shiftId: shift.id, ...cellsByDate };
    });
  }, [rows]);

  return (
    <div className="ag-theme-alpine" style={{ height: 'calc(100vh - 200px)' }}>
      <AgGridReact columnDefs={columnDefs} rowData={rowData} rowHeight={120} />
    </div>
  );
}
```

### `DownloadExcelButton.tsx`

```tsx
import { Button } from 'antd';
import { DownloadOutlined } from '@ant-design/icons';
import { extrusionExportApi } from '@/api/extrusionExportApi';

interface Props { from: string; to: string; }

export function DownloadExcelButton({ from, to }: Props) {
  const handleDownload = async () => {
    const blob = await extrusionExportApi.downloadMatrix(from, to);
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `extrusion_${from}_${to}.xlsx`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  };

  return (
    <Button icon={<DownloadOutlined />} onClick={handleDownload}>
      엑셀 다운로드
    </Button>
  );
}
```
