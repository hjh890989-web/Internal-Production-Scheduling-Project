### `GanttView.tsx`

```tsx
import { useEffect, useRef } from 'react';
import Gantt from 'frappe-gantt';
import { ExScheduleRow } from '@/types/extrusion';
import { GanttTaskAdapter } from './GanttTaskAdapter';

interface Props {
  rows: ExScheduleRow[];
  viewMode?: 'Day' | 'Week' | 'Month';
  onClick?: (row: ExScheduleRow) => void;
}

export function GanttView({ rows, viewMode = 'Week', onClick }: Props) {
  const containerRef = useRef<HTMLDivElement>(null);
  const ganttRef = useRef<Gantt | null>(null);

  useEffect(() => {
    if (!containerRef.current) return;

    const tasks = GanttTaskAdapter.toFrappeTasks(rows);

    ganttRef.current = new Gantt(containerRef.current, tasks, {
      view_mode: viewMode,
      header_height: 50,
      column_width: 30,
      step: 24,
      bar_height: 20,
      bar_corner_radius: 3,
      padding: 18,
      custom_popup_html: (task) => {
        const row = rows.find(r => r.id === task.id);
        if (!row) return '';
        return `
          <div class="gantt-popup">
            <h5>${row.hoseId}</h5>
            <p>라인: ${row.machineId} · shift: ${row.shiftCode}</p>
            <p>yield: ${row.yield}</p>
            <p>셋팅 그룹: ${row.settingGroup}</p>
          </div>
        `;
      },
    });

    if (onClick) {
      ganttRef.current.on('click', (task: Gantt.Task) => {
        const row = rows.find(r => r.id === task.id);
        if (row) onClick(row);
      });
    }

    return () => {
      // Frappe Gantt 명시적 cleanup 없음 — innerHTML 비움
      if (containerRef.current) containerRef.current.innerHTML = '';
    };
  }, [rows, viewMode, onClick]);

  return <div ref={containerRef} className="gantt-container" />;
}
```

### `GanttTaskAdapter.ts`

```typescript
import { ExScheduleRow } from '@/types/extrusion';

export class GanttTaskAdapter {

  private static SHIFT_START_HOURS: Record<string, number> = {
    'DAY_FIRST':    8,
    'DAY_SECOND':   13,
    'NIGHT_FIRST':  20,
    'NIGHT_SECOND': 0,    // 다음날 0~4
  };

  private static SHIFT_DURATION_MIN = 240;     // 4시간 (nominal — 효율 적용 안 함)

  /**
   * ExSchedule rows → Frappe Gantt tasks
   *  - start: production_date + shift_start_time
   *  - end: start + actualMin
   *  - progress: 0% (placeholder — Phase 2+ 실 진행률)
   *  - color: 셋팅 그룹별 hex
   */
  static toFrappeTasks(rows: ExScheduleRow[]): Gantt.Task[] {
    return rows.map(row => {
      const startHour = this.SHIFT_START_HOURS[row.shiftCode];
      const baseDate = new Date(row.productionDate);
      const startDate = new Date(baseDate);
      startDate.setHours(startHour, 0, 0);

      // NIGHT_SECOND은 다음날
      if (row.shiftCode === 'NIGHT_SECOND') {
        startDate.setDate(startDate.getDate() + 1);
      }

      const endDate = new Date(startDate.getTime() + (row.actualMin || this.SHIFT_DURATION_MIN) * 60_000);

      return {
        id: row.id,
        name: `${row.hoseId} (${row.yield})`,
        start: formatYMD(startDate),
        end: formatYMD(endDate),
        progress: 0,
        custom_class: `setting-group-${row.settingGroup}`,
      };
    });
  }
}

function formatYMD(d: Date): string {
  return d.toISOString().split('T')[0];     // YYYY-MM-DD
}
```
