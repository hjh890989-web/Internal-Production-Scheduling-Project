### 핵심 로직

```typescript
import { test, expect } from '@playwright/test';
import * as XLSX from 'xlsx';
import { readFileSync } from 'fs';

const SHEET_NAME_REGEX = /^\d{1,2}월\d{1,2}일\(압출\)$/;

test('Excel 다운로드 시트명 정규식 일치 (BR-E09)', async ({ page }) => {
  // 1) 페이지 진입
  await page.goto('/extrusion-matrix');
  await page.waitForSelector('.ag-row');

  // 2) 날짜 범위 설정 — 1주
  await page.locator('[data-testid="date-range-from"]').fill('2026-03-02');
  await page.locator('[data-testid="date-range-to"]').fill('2026-03-08');

  // 3) Download 버튼 클릭 → 다운로드 대기
  const downloadPromise = page.waitForEvent('download');
  await page.getByText('엑셀 다운로드').click();
  const download = await downloadPromise;

  // 4) 파일 저장
  const path = await download.path();

  // 5) 파일 분석 — 시트명 검증
  const buffer = readFileSync(path!);
  const workbook = XLSX.read(buffer);

  expect(workbook.SheetNames.length).toBeGreaterThan(0);

  for (const sheetName of workbook.SheetNames) {
    expect(sheetName).toMatch(SHEET_NAME_REGEX);
  }
});

test('빈 일자 sheet 미생성', async ({ page }) => {
  // 스케줄 없는 일자가 포함된 범위 다운로드
  // 다운로드 후 sheet 수 < 일자 수 확인
  // ...
});

test('60일 초과 range — 400 응답', async ({ page, request }) => {
  const response = await request.get('/api/v1/export/extrusion-matrix?from=2026-01-01&to=2026-12-31');
  expect(response.status()).toBe(400);
});

test('시트 내용 — yield accounting (UI 매트릭스와 일치)', async ({ page }) => {
  await page.goto('/extrusion-matrix');
  await page.waitForSelector('.ag-row');

  // UI yield 합산
  const uiTotal = await page.$$eval('.schedule-cell .yield', els =>
    els.reduce((sum, el) => sum + parseInt(el.textContent!.replace(/[^\d]/g, '')), 0)
  );

  const downloadPromise = page.waitForEvent('download');
  await page.getByText('엑셀 다운로드').click();
  const download = await downloadPromise;

  const buffer = readFileSync((await download.path())!);
  const workbook = XLSX.read(buffer);
  let xlsxTotal = 0;
  for (const sheetName of workbook.SheetNames) {
    const sheet = workbook.Sheets[sheetName];
    const data = XLSX.utils.sheet_to_json<any>(sheet, { header: 1 });
    // cell yield 합산 — "hose (yield)" 패턴 parse
    for (const row of data) {
      for (const cell of row.slice(1)) {     // 첫 column은 Shift label
        if (typeof cell === 'string') {
          const matches = cell.matchAll(/\((\d+)\)/g);
          for (const m of matches) xlsxTotal += parseInt(m[1]);
        }
      }
    }
  }

  expect(xlsxTotal).toBe(uiTotal);
});
```
