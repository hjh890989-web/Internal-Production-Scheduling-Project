### `ValidateAllPerformanceIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-VC-VALIDATE-1500/master_seed.sql",
      "classpath:datasets/DS-VC-VALIDATE-1500/schedule_rows.sql"})
class ValidateAllPerformanceIT {

    @Autowired ScheduleValidatorService validator;

    private static final UUID SCHEDULE_ID = UUID.fromString("11111111-1111-1111-1111-111111111111");
    private static final int WARMUP_RUNS = 5;
    private static final int MEASURE_RUNS = 50;

    @Test
    void p95_under_three_seconds_for_1500_rows() {
        // Warm-up — JIT + 캐시 적재
        for (int i = 0; i < WARMUP_RUNS; i++) validator.validateAll(SCHEDULE_ID);

        var latenciesMs = new ArrayList<Long>();
        for (int i = 0; i < MEASURE_RUNS; i++) {
            long start = System.nanoTime();
            validator.validateAll(SCHEDULE_ID);
            latenciesMs.add((System.nanoTime() - start) / 1_000_000);
        }

        Collections.sort(latenciesMs);
        long p50 = latenciesMs.get((int) (MEASURE_RUNS * 0.50));
        long p95 = latenciesMs.get((int) (MEASURE_RUNS * 0.95));
        long p99 = latenciesMs.get((int) (MEASURE_RUNS * 0.99));

        System.out.printf("validateAll latency: p50=%dms, p95=%dms, p99=%dms%n", p50, p95, p99);

        // SRS 요구: p95 ≤ 3,000ms
        assertThat(p95).as("REQ-FUNC-VC-016 p95 ≤ 3초").isLessThanOrEqualTo(3000);
    }

    @Test
    void cold_cache_first_call_under_5_seconds() {
        // 캐시 비어있는 상태에서 첫 호출 (worst-case)
        // 안전 마진 — 5초까지 허용 (warm 이후 정상)
        long start = System.nanoTime();
        validator.validateAll(SCHEDULE_ID);
        long ms = (System.nanoTime() - start) / 1_000_000;

        assertThat(ms).isLessThanOrEqualTo(5000);
    }
}
```
