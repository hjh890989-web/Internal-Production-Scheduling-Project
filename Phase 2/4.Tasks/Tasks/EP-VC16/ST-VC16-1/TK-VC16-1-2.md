### `ValidateAllRegressionIT.java`

```java
@SpringBootTest @Testcontainers
@Sql({"classpath:datasets/DS-VC-VALIDATE-1500/master_seed.sql",
      "classpath:datasets/DS-VC-VALIDATE-1500/schedule_rows.sql"})
class ValidateAllRegressionIT {

    @Autowired ScheduleValidatorService validator;

    @Test
    void detects_all_expected_violations() {
        UUID scheduleId = UUID.fromString("11111111-1111-1111-1111-111111111111");
        var result = validator.validateAll(scheduleId);

        // 기대 위반 25건 (모든 카테고리 mix)
        assertThat(result.violations()).hasSize(25);

        // 카테고리 mix 검증 — 8 카테고리 중 ≥ 5
        assertThat(result.summary().keySet()).hasSizeGreaterThanOrEqualTo(5);

        // 특정 위반 식별 — DEADLINE_D2 5건 등 (시나리오별)
        assertThat(result.summary()).containsEntry(ConflictCategory.DEADLINE_D2, 5L);
        assertThat(result.summary()).containsEntry(ConflictCategory.SLOT_OX, 4L);
    }

    @Test
    void idempotent_across_multiple_calls() {
        UUID scheduleId = UUID.fromString("11111111-1111-1111-1111-111111111111");
        var first = validator.validateAll(scheduleId);
        var second = validator.validateAll(scheduleId);

        assertThat(first.violations()).hasSameSizeAs(second.violations());
        assertThat(first.summary()).isEqualTo(second.summary());
    }

    @Test
    void clean_schedule_returns_no_violations() {
        UUID scheduleId = UUID.fromString("22222222-2222-2222-2222-222222222222");
        var result = validator.validateAll(scheduleId);

        assertThat(result.violations()).isEmpty();
        assertThat(result.summary()).isEmpty();
    }

    @Test
    void master_change_introduces_new_violations() {
        UUID scheduleId = UUID.fromString("33333333-3333-3333-3333-333333333333");
        var before = validator.validateAll(scheduleId);
        assertThat(before.violations()).isEmpty();

        // 마스터 변경: 신규 휴일 추가 (deadline 압축)
        em.createNativeQuery("INSERT INTO master.holiday (holiday_date, holiday_name, holiday_type) " +
            "VALUES ('2026-02-19', '임시 공휴일', 'COMPANY')").executeUpdate();
        calendar.reload();

        var after = validator.validateAll(scheduleId);
        assertThat(after.violations())
            .as("신규 휴일로 인한 D-2 위반 발견")
            .isNotEmpty();
        assertThat(after.summary()).containsKey(ConflictCategory.DEADLINE_D2);
    }
}
```
