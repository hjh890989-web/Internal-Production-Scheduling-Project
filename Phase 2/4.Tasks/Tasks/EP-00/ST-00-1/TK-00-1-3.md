# infrastructure/docker-compose.yml (nginx 부분 추가)
services:
  # ... postgres, redis 위에 정의됨

    container_name: scheduling-nginx

      - "127.0.0.1:80:80"     # DEV: localhost only (HTTP → HTTPS 리다이렉트)

      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/conf.d:/etc/nginx/conf.d:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro

      interval: 10s
      timeout: 3s
      retries: 5
      start_period: 5s
    depends_on:
      backend:
        condition: service_healthy   # BE 준비된 후 시작 (BE 서비스는 ST-00-2)
    restart: unless-stopped

      - scheduling-net

  postgres-data:
  redis-data:

#### B. `nginx.conf` 메인 설정

```nginx
# infrastructure/nginx/nginx.conf
user nginx;
worker_processes auto;
worker_rlimit_nofile 65535;

events {
    worker_connections 8192;
    use epoll;
    multi_accept on;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # Logging (NFR-OPS-001)
    log_format main '$remote_addr - $remote_user [$time_iso8601] '
                    '"$request" $status $body_bytes_sent '
                    '"$http_referer" "$http_user_agent" '
                    'rt=$request_time uct=$upstream_connect_time '
                    'uht=$upstream_header_time urt=$upstream_response_time';
    access_log /var/log/nginx/access.log main;
    error_log  /var/log/nginx/error.log warn;

    # Performance
    sendfile        on;
    tcp_nopush      on;
    tcp_nodelay     on;
    keepalive_timeout 65;
    types_hash_max_size 2048;
    client_max_body_size 20m;          # Excel 업로드 한도 (SAD-RSK-005)

    # gzip (NFR-PER-005)
    gzip on;
    gzip_comp_level 6;
    gzip_min_length 1000;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml;

    # TLS 글로벌 (NFR-SEC-006)
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;

    # HSTS (NFR-SEC-006 1년 max-age)
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Frame-Options "DENY" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    include /etc/nginx/conf.d/*.conf;
}
```

#### C. 사이트 설정 (`scheduling.conf`)

```nginx
# infrastructure/nginx/conf.d/scheduling.conf

# Backend upstream
upstream backend {
    server backend:8080 max_fails=3 fail_timeout=30s;
    keepalive 32;
}

# HTTP → HTTPS 리다이렉트
server {
    listen 80 default_server;
    server_name _;
    return 301 https://$host$request_uri;
}

# HTTPS 메인
server {
    listen 443 ssl http2 default_server;
    server_name _;

    # TLS 인증서 (DEV: 자체 서명, STG/PROD: 사내 CA)
    ssl_certificate     /etc/nginx/ssl/dev-self-signed.crt;
    ssl_certificate_key /etc/nginx/ssl/dev-self-signed.key;

    # Healthcheck
    location = /health {
        access_log off;
        return 200 "OK\n";
        add_header Content-Type text/plain;
    }

    # WebSocket (NFR-PER-004 p95 ≤2초)
    location /ws/ {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 3600s;          # WebSocket long-lived
        proxy_send_timeout 3600s;
    }

    # API
    location /api/ {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_connect_timeout 5s;
        proxy_read_timeout 60s;            # NFR-PER-001 1만 row 60초
        proxy_send_timeout 60s;
    }

    # Frontend SPA (정적 파일)
    location / {
        root /usr/share/nginx/html;
        index index.html;
        try_files $uri $uri/ /index.html;  # React Router fallback
        expires 1h;
        add_header Cache-Control "public, no-transform";
    }
}
```

#### D. DEV 자체 서명 인증서 생성 (`generate-dev-cert.sh`)

```bash
#!/usr/bin/env bash
# infrastructure/nginx/scripts/generate-dev-cert.sh
# DEV 환경 전용 — STG/PROD는 사내 CA 발급 인증서 사용
set -euo pipefail

SSL_DIR="$(dirname "$0")/../ssl"
mkdir -p "$SSL_DIR"

openssl req -x509 -nodes -days 365 -newkey rsa:4096 \
    -keyout "$SSL_DIR/dev-self-signed.key" \
    -out    "$SSL_DIR/dev-self-signed.crt" \
    -subj "/C=KR/ST=Local/L=Dev/O=Internal/CN=localhost" \
    -addext "subjectAltName=DNS:localhost,IP:127.0.0.1"

chmod 600 "$SSL_DIR/dev-self-signed.key"
echo "✅ DEV 인증서 생성 완료 (1년 유효)"
echo "   key: $SSL_DIR/dev-self-signed.key (600)"
echo "   crt: $SSL_DIR/dev-self-signed.crt"
echo ""
echo "⚠️ STG/PROD는 사내 CA 발급 인증서로 교체 (docs/infrastructure/nginx_setup.md 참조)"
```

## :memo: Implementation Notes
- **기술 힌트**:
  - `nginx:1.25-alpine`는 약 40 MB — 디스크·이미지 풀 시간 절약 (NFR-COS-001)
  - `proxy_read_timeout 3600s` — WebSocket long-lived 위해 필수 (기본 60s면 1분 후 끊김)
  - `try_files $uri $uri/ /index.html` — React Router(SPA) 새로고침 404 방지 핵심
- **유사 패턴**:
  - STG/PROD compose 파일은 동일 NGINX 서비스 + `.env.staging`·`.env.prod`로 ports·인증서·도메인만 변경

  - NGINX Docs: https://nginx.org/en/docs/
  - Mozilla TLS Configuration: https://ssl-config.mozilla.org/
  - HSTS Preload: https://hstspreload.org/
