import i18n from 'i18next'
import { initReactI18next } from 'react-i18next'
import ko from './locales/ko.json'

void i18n
  .use(initReactI18next)
  .init({
    resources: {
      ko: { translation: ko },
    },
    lng: 'ko',
    fallbackLng: 'ko',
    interpolation: {
      escapeValue: false,    // React가 자체 escape
    },

      useSuspense: false,

export default i18n

#### D. Ant Design ConfigProvider + theme (`src/main.tsx`)

```tsx
import React from 'react'
import ReactDOM from 'react-dom/client'
import { ConfigProvider } from 'antd'
import koKR from 'antd/locale/ko_KR'
import dayjs from 'dayjs'
import 'dayjs/locale/ko'
import App from './App'
import './i18n'
import { theme } from './theme'
import './styles/global.css'

dayjs.locale('ko')

ReactDOM.createRoot(document.getElementById('root')!).render(
  <React.StrictMode>
    <ConfigProvider locale={koKR} theme={theme}>
      <App />
    </ConfigProvider>
  </React.StrictMode>,
)
```

#### G. App.tsx — 한국어 + Ant Design 사용 예시

```tsx
import { Button, Space, Typography, message } from 'antd'
import { useTranslation } from 'react-i18next'

const { Title, Paragraph } = Typography

function App() {
  const { t } = useTranslation()

  const handleClick = () => {
    void message.success(t('messages.save_success'))
  }

  return (
    <div style={{ padding: 24, maxWidth: 1280, margin: '0 auto' }}>
      <Title level={1}>{t('app.title')}</Title>
      <Paragraph>{t('app.subtitle')}</Paragraph>
      <Space>
        <Button type="primary" onClick={handleClick}>
          {t('common.save')}
        </Button>
        <Button>{t('common.cancel')}</Button>
      </Space>
    </div>
  )
}

export default App
```
