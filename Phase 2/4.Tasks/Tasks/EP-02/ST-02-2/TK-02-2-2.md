#### A. Flyway `V1_5`

```sql
CREATE TABLE audit.precedence_resolution (
    audit_id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_key_hose   VARCHAR(40) NOT NULL,
    order_key_date   DATE NOT NULL,
    winner_order_id  UUID NOT NULL,
    winner_type      VARCHAR(20) NOT NULL,
    losers_count     INT NOT NULL,
    losers_details   JSONB NOT NULL,            -- [{orderId, type, source_workbook}, ...]
    existing_master_id UUID,
    decision         VARCHAR(30) NOT NULL CHECK (decision IN ('KEPT_EXISTING','REPLACED_EXISTING','NEW_WINS')),
    rule_id          VARCHAR(20) NOT NULL DEFAULT 'BR-O01',
    resolved_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    actor            VARCHAR(40) NOT NULL
);

CREATE INDEX idx_precedence_resolved_at ON audit.precedence_resolution(resolved_at);
CREATE INDEX idx_precedence_key ON audit.precedence_resolution(order_key_hose, order_key_date);

REVOKE UPDATE, DELETE ON audit.precedence_resolution FROM app_user;
GRANT INSERT, SELECT ON audit.precedence_resolution TO app_user;
GRANT SELECT ON audit.precedence_resolution TO auditor;

COMMENT ON TABLE audit.precedence_resolution
IS 'BR-X02·BR-O01: (품번,납기) 중복 우선순위 해소 audit';
```

@Entity @Table(schema = "audit", name = "precedence_resolution")

public class PrecedenceResolutionAudit {
    @Id @GeneratedValue private UUID auditId;
    private String orderKeyHose;
    private LocalDate orderKeyDate;
    private UUID winnerOrderId;
    @Enumerated(EnumType.STRING) private OrderType winnerType;
    private int losersCount;
    @JdbcTypeCode(SqlTypes.JSON) @Column(columnDefinition = "JSONB")
    private JsonNode losersDetails;
    private UUID existingMasterId;
    @Enumerated(EnumType.STRING) private Resolution.Decision decision;
    @Builder.Default private String ruleId = "BR-O01";
    private Instant resolvedAt;
    private String actor;
}

#### C. `PrecedenceAuditLogger.java`

```java
@Service @RequiredArgsConstructor
public class PrecedenceAuditLogger {

    private final PrecedenceResolutionRepository repo;
    private final ObjectMapper json;
    private final ApplicationContext ctx;

    /**
     * Resolution → audit row.
     * @Transactional propagation REQUIRED — 부모 commit과 결합 (BR-X02)
     */
    @Transactional(propagation = Propagation.REQUIRED)
    public void log(Resolution resolution) {
        try {
            var losersJson = json.valueToTree(resolution.losers().stream()
                .map(l -> Map.of(
                    "orderId", l.orderId(),
                    "type", l.orderType().name(),
                    "source", l.sourceWorkbook()
                )).toList());

            var audit = PrecedenceResolutionAudit.builder()
                .orderKeyHose(resolution.key().hoseId())
                .orderKeyDate(resolution.key().deliveryDate())
                .winnerOrderId(resolution.winner().orderId())
                .winnerType(resolution.winner().orderType())
                .losersCount(resolution.losers().size())
                .losersDetails(losersJson)
                .existingMasterId(resolution.existingMaster() != null
                    ? resolution.existingMaster().getOrderId() : null)
                .decision(resolution.decision())
                .resolvedAt(Instant.now())
                .actor(currentActor())
                .build();

            repo.save(audit);
        } catch (Exception e) {
            throw new IllegalStateException("Failed to log precedence resolution — commit blocked", e);
        }
    }

    private String currentActor() {
        // SecurityContext에서 사용자 추출, 시스템 호출은 "system:resolver"
        return SecurityContextHolder.getContext().getAuthentication() != null
            ? SecurityContextHolder.getContext().getAuthentication().getName()
            : "system:resolver";
    }
}
```

## :memo: Implementation Notes

- Hibernate 6.x의 `@JdbcTypeCode(SqlTypes.JSON)` + `JsonNode` 컬럼 — Jackson 통합 자동
- `Propagation.REQUIRED`로 부모 트랜잭션에 참여 → audit 실패 시 메인 rollback
- audit 테이블은 partition by month 권장 (3년 보존, 데이터 폭증 대비)
