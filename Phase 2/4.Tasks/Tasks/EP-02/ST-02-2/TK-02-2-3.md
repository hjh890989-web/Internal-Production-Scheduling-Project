#### A. `PrecedenceResolverTest.java` — 진리표 + 4 Decision 분기

```java
@SpringBootTest
class PrecedenceResolverTest {

    @Autowired PrecedenceResolver resolver;

    /** 6 페어 (4C2) × 양방향 = 12 + 동일 4 = 16 케이스 */
    @ParameterizedTest
    @MethodSource("allPairs")
    void priority_truth_table(OrderType a, OrderType b, OrderType expectedWinner) {
        var group = new DuplicateGroup(
            new OrderKey("PART-001", LocalDate.of(2026, 2, 15)),
            List.of(draftOfType(a), draftOfType(b)),
            null
        );
        var resolution = resolver.resolve(group);
        assertThat(resolution.winner().orderType()).isEqualTo(expectedWinner);
    }

    static Stream<Arguments> allPairs() {
        return Stream.of(
            // FORECAST < KD < WEEKLY < CONFIRMED
            Arguments.of(FORECAST, KD, KD),
            Arguments.of(FORECAST, WEEKLY, WEEKLY),
            Arguments.of(FORECAST, CONFIRMED, CONFIRMED),
            Arguments.of(KD, WEEKLY, WEEKLY),
            Arguments.of(KD, CONFIRMED, CONFIRMED),
            Arguments.of(WEEKLY, CONFIRMED, CONFIRMED),
            // 역방향
            Arguments.of(KD, FORECAST, KD),
            Arguments.of(WEEKLY, FORECAST, WEEKLY),
            Arguments.of(CONFIRMED, FORECAST, CONFIRMED),
            Arguments.of(WEEKLY, KD, WEEKLY),
            Arguments.of(CONFIRMED, KD, CONFIRMED),
            Arguments.of(CONFIRMED, WEEKLY, CONFIRMED),
            // 동률 (4건 — 어느 쪽이든 자기 자신 winner)
            Arguments.of(FORECAST, FORECAST, FORECAST),
            Arguments.of(KD, KD, KD),
            Arguments.of(WEEKLY, WEEKLY, WEEKLY),
            Arguments.of(CONFIRMED, CONFIRMED, CONFIRMED)
        );
    }

    @Test
    void existing_confirmed_kept_against_new_forecast() {
        var existing = orderEntity(CONFIRMED);
        var group = new DuplicateGroup(
            new OrderKey("PART-001", LocalDate.of(2026, 2, 15)),
            List.of(draftOfType(FORECAST)),
            existing
        );
        var resolution = resolver.resolve(group);
        assertThat(resolution.decision()).isEqualTo(Resolution.Decision.KEPT_EXISTING);
    }

    @Test
    void existing_forecast_replaced_by_new_confirmed() {
        var existing = orderEntity(FORECAST);
        var group = new DuplicateGroup(
            new OrderKey("PART-001", LocalDate.of(2026, 2, 15)),
            List.of(draftOfType(CONFIRMED)),
            existing
        );
        var resolution = resolver.resolve(group);
        assertThat(resolution.decision()).isEqualTo(Resolution.Decision.REPLACED_EXISTING);
    }
}
```

#### B. `ResolutionServiceIT.java` — 100쌍 회귀

```java
@SpringBootTest
@Testcontainers
class ResolutionServiceIT {

    @Autowired ResolutionService service;
    @Autowired PrecedenceResolutionRepository auditRepo;

    @Test
    void resolves_100_pairs_with_correct_winners_and_audit_rows() throws Exception {
        var pairs = loadPairs("DS-PRECEDENCE-100/precedence_pairs.json");

        for (var pair : pairs) {
            var resolutions = service.resolveAll(List.of(pair.a(), pair.b()));
            assertThat(resolutions).hasSize(1);
            var r = resolutions.get(0);
            assertThat(r.winner().orderType().ordinal())
                .isGreaterThanOrEqualTo(r.losers().get(0).orderType().ordinal());
        }

        // audit 100건 (각 pair당 1건)
        assertThat(auditRepo.count()).isEqualTo(100);

        // 모든 audit row의 winner.ordinal >= losers max ordinal
        auditRepo.findAll().forEach(audit -> {
            assertThat(audit.getDecision()).isIn(
                Resolution.Decision.NEW_WINS,
                Resolution.Decision.KEPT_EXISTING,
                Resolution.Decision.REPLACED_EXISTING
            );
        });
    }
}
```
