#### A. `DuplicateDetectionIT.java` — 100사이클 회귀

```java
package com.scheduling.order.domain;

import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.testcontainers.junit.jupiter.Testcontainers;

import java.time.LocalDate;
import java.util.*;
import java.util.stream.IntStream;

import static org.assertj.core.api.Assertions.*;

@SpringBootTest
@Testcontainers
class DuplicateDetectionIT {

    @Autowired private OrderCommitService commitService;
    @Autowired private OrderRepository repo;

    /**
     * TC-OC-005: 100 사이클 commit → 중복 0
     */
    @Test
    void no_duplicates_after_100_cycles() {
        var pairs = loadDuplicateDataset();   // 50쌍, 각 쌍 = (orig, dup)
        int expectedUnique = pairs.size();    // 50

        long totalCommitted = 0;
        long totalRejected = 0;

        for (int cycle = 0; cycle < 100; cycle++) {
            // 매 사이클 임의 순서로 중복 시도
            var shuffled = new ArrayList<>(pairs);
            Collections.shuffle(shuffled);

            for (DuplicatePair pair : shuffled) {
                // 첫 row commit
                try {
                    commitService.commit(pair.orig());
                    totalCommitted++;
                } catch (DuplicateOrderException e) {
                    totalRejected++;
                }
                // 두 번째 row commit — DuplicateOrderException 예상
                try {
                    commitService.commit(pair.dup());
                    totalCommitted++;
                } catch (DuplicateOrderException e) {
                    totalRejected++;
                }
            }
        }

        // DB 검증
        long uniqueRowCount = repo.count();
        assertThat(uniqueRowCount)
            .as("After 100 cycles, master must contain exactly %d unique rows", expectedUnique)
            .isEqualTo(expectedUnique);

        assertThat(totalRejected)
            .as("Should have rejected at least 50 duplicates per cycle × 100 cycles = 5000")
            .isGreaterThanOrEqualTo(5000);
    }

    @Test
    void concurrent_inserts_only_one_wins() throws Exception {
        // 10개 thread가 동일 (hose_id, delivery_date) row INSERT 시도
        var threadPool = Executors.newFixedThreadPool(10);
        var successes = new AtomicInteger();
        var failures = new AtomicInteger();

        var sameOrder = OrderDraft.builder()
            .hoseId("29673-2F900").deliveryDate(LocalDate.of(2026, 2, 15))
            .qty(100).build();

        var futures = IntStream.range(0, 10).mapToObj(i -> threadPool.submit(() -> {
            try {
                commitService.commit(sameOrder);
                successes.incrementAndGet();
            } catch (DuplicateOrderException e) {
                failures.incrementAndGet();
            }
        })).toList();
        for (var f : futures) f.get();

        assertThat(successes.get()).isEqualTo(1);
        assertThat(failures.get()).isEqualTo(9);
    }
}
```

#### B. `DuplicateMetricsTest.java` — K-O03 부속

```java
package com.scheduling.order.domain;

import io.micrometer.core.instrument.MeterRegistry;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;

import static org.assertj.core.api.Assertions.*;

@SpringBootTest
class DuplicateMetricsTest {

    @Autowired MeterRegistry registry;
    @Autowired OrderCommitService service;

    @Test
    void duplicate_detection_emits_counter() {
        service.commit(makeOrder(1));
        try { service.commit(makeOrder(1)); } catch (Exception e) {}

        var counter = registry.find("order.duplicate.detected.total").counter();
        assertThat(counter).isNotNull();
        assertThat(counter.count()).isGreaterThan(0);
    }
}
```
