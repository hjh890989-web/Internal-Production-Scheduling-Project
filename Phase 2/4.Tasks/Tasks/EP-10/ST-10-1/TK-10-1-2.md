### Flyway `V4_1_data__create_planner_role_function.sql`

```sql
-- Planner role 생성 (CON-07)
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_roles WHERE rolname = 'planner_role') THEN
        CREATE ROLE planner_role NOLOGIN;
    END IF;
END$$;

-- app_user에는 status UPDATE 권한 없음 (DB 직접 차단)
REVOKE UPDATE (status, confirmed_at, confirmed_by) ON vc_schedule FROM app_user;

-- planner_role만 confirm function 호출 가능
CREATE OR REPLACE FUNCTION vc_schedule_confirm(
    p_schedule_id UUID,
    p_planner_id VARCHAR(40)
) RETURNS VOID AS $$
DECLARE
    v_old_status VARCHAR(20);
BEGIN
    -- 1) 현재 상태 조회
    SELECT status INTO v_old_status FROM vc_schedule WHERE id = p_schedule_id;

    IF v_old_status IS NULL THEN
        RAISE EXCEPTION 'BR-X01: vc_schedule % not found', p_schedule_id;
    END IF;

    -- 2) Candidate만 Confirmed 가능 (trigger도 검증하지만 명시적)
    IF v_old_status != 'CANDIDATE' THEN
        RAISE EXCEPTION 'BR-X01: cannot confirm from %, must be CANDIDATE', v_old_status;
    END IF;

    -- 3) 전이
    UPDATE vc_schedule SET
        status = 'CONFIRMED',
        confirmed_at = now(),
        confirmed_by = p_planner_id
    WHERE id = p_schedule_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

REVOKE EXECUTE ON FUNCTION vc_schedule_confirm(UUID, VARCHAR) FROM PUBLIC;
GRANT EXECUTE ON FUNCTION vc_schedule_confirm(UUID, VARCHAR) TO planner_role;
GRANT planner_role TO app_user;             -- app_user가 planner_role을 SET ROLE로 활성화
```

### `VcScheduleConfirmService.java`

```java
@Service @RequiredArgsConstructor
public class VcScheduleConfirmService {

    private final JdbcTemplate jdbc;
    private final VcConfirmedPublisher publisher;
    private final SecurityContextService securityCtx;

    /**
     * BR-X01: VC schedule Candidate → Confirmed 전이.
     *  1. planner_role 세션 활성화
     *  2. DB function 호출 (function 내부 트리거가 검증)
     *  3. VcConfirmedEvent 발행 — EP-07 listener가 압출 후보 생성
     *
     * @PreAuthorize("hasRole('PLANNER')")는 controller 측에서
     */
    @Transactional
    public void confirm(UUID scheduleId) {
        String plannerId = securityCtx.currentUserId();

        // 1) planner_role 활성화 (SET ROLE)
        jdbc.execute("SET LOCAL ROLE planner_role");

        // 2) DB function 호출
        jdbc.update("SELECT vc_schedule_confirm(?, ?)", scheduleId, plannerId);

        // 3) 이벤트 발행 (Spring Modulith)
        publisher.publishConfirmed(scheduleId);
    }
}
```

### `VcScheduleStateController.java`

```java
@RestController @RequestMapping("/api/v1/schedule/vc") @RequiredArgsConstructor
public class VcScheduleStateController {

    private final VcScheduleConfirmService confirmService;

    @PostMapping("/{id}/confirm")
    @PreAuthorize("hasRole('PLANNER')")
    public ResponseEntity<Void> confirm(@PathVariable UUID id) {
        confirmService.confirm(id);
        return ResponseEntity.ok().build();
    }
}
```

## :shield: Risk Mitigation

- DB function SECURITY DEFINER — app_user가 planner_role 컬럼 직접 UPDATE 불가, function 통한 간접 접근만
- `SET LOCAL ROLE` — transaction scope, leak 없음
- 이중 검증 (function 내부 status check + trigger) — defense in depth
