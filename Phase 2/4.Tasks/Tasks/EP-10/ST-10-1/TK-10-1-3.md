### `VcConfirmGateNegativeIT.java`

```java
@SpringBootTest @Testcontainers
class VcConfirmGateNegativeIT {

    @Autowired DataSource dataSource;
    @Autowired VcScheduleConfirmService service;
    @Autowired VcScheduleRepository repo;

    @Test
    void app_user_direct_update_status_denied() throws SQLException {
        var schedule = createCandidateSchedule();

        // app_user로 직접 UPDATE 시도
        try (Connection conn = openAsAppUser()) {
            try (PreparedStatement ps = conn.prepareStatement(
                "UPDATE vc_schedule SET status = 'CONFIRMED' WHERE id = ?")) {
                ps.setObject(1, schedule.getId());
                assertThatThrownBy(ps::executeUpdate)
                    .as("app_user가 status 컬럼 UPDATE 차단되어야 함")
                    .isInstanceOf(SQLException.class)
                    .hasMessageContaining("permission denied");
            }
        }
    }

    @Test
    void draft_to_confirmed_skip_blocked_by_trigger() {
        var draftSchedule = createDraftSchedule();

        assertThatThrownBy(() -> service.confirm(draftSchedule.getId()))
            .as("DRAFT → CONFIRMED skip 차단")
            .isInstanceOf(Exception.class)
            .hasMessageContaining("must be CANDIDATE");
    }

    @Test
    void already_confirmed_cannot_reconfirm() {
        var confirmed = createConfirmedSchedule();

        assertThatThrownBy(() -> service.confirm(confirmed.getId()))
            .isInstanceOf(Exception.class)
            .hasMessageContaining("cannot confirm");
    }

    @Test
    void insert_with_confirmed_status_blocked() throws SQLException {
        try (Connection conn = openAsAppUser()) {
            try (PreparedStatement ps = conn.prepareStatement(
                "INSERT INTO vc_schedule (id, status, ...) VALUES (?, 'CONFIRMED', ...)")) {
                ps.setObject(1, UUID.randomUUID());
                assertThatThrownBy(ps::executeUpdate)
                    .as("신규 INSERT는 DRAFT만 허용")
                    .isInstanceOf(SQLException.class);
            }
        }
    }

    @Test
    void confirm_with_planner_role_succeeds() {
        var schedule = createCandidateSchedule();

        // SecurityContext에 PLANNER role 주입 (테스트 helper)
        withPlannerRole("planner1", () -> {
            service.confirm(schedule.getId());
        });

        var confirmed = repo.findById(schedule.getId()).orElseThrow();
        assertThat(confirmed.getStatus()).isEqualTo(VcScheduleStatus.CONFIRMED);
        assertThat(confirmed.getConfirmedBy()).isEqualTo("planner1");
        assertThat(confirmed.getConfirmedAt()).isNotNull();
    }

    @Test
    void user_role_cannot_invoke_confirm_api() throws Exception {
        // MockMvc + USER role
        var schedule = createCandidateSchedule();

        mockMvc.perform(post("/api/v1/schedule/vc/{id}/confirm", schedule.getId())
                .with(user("user1").roles("USER")))
            .andExpect(status().isForbidden());
    }

    @Test
    void delete_blocked_via_trigger() throws SQLException {
        var schedule = createConfirmedSchedule();

        try (Connection conn = openAsAppUser()) {
            try (PreparedStatement ps = conn.prepareStatement(
                "DELETE FROM vc_schedule WHERE id = ?")) {
                ps.setObject(1, schedule.getId());
                // DELETE 권한 자체가 제거되어 있다면 SQLException, 트리거 차단이라면 다른 메시지
                // 본 시스템 정책: app_user는 DELETE 권한 없음
                assertThatThrownBy(ps::executeUpdate)
                    .isInstanceOf(SQLException.class);
            }
        }
    }
}
```

### `VcStateTransitionUnitTest.java`

```java
class VcStateTransitionUnitTest {

    @Test
    void draft_can_transition_to_candidate_only() {
        assertThat(VcScheduleStatus.DRAFT.canTransitionTo(VcScheduleStatus.CANDIDATE)).isTrue();
        assertThat(VcScheduleStatus.DRAFT.canTransitionTo(VcScheduleStatus.CONFIRMED)).isFalse();
        assertThat(VcScheduleStatus.DRAFT.canTransitionTo(VcScheduleStatus.DRAFT)).isFalse();
    }

    @Test
    void candidate_can_transition_to_confirmed_only() {
        assertThat(VcScheduleStatus.CANDIDATE.canTransitionTo(VcScheduleStatus.CONFIRMED)).isTrue();
        assertThat(VcScheduleStatus.CANDIDATE.canTransitionTo(VcScheduleStatus.DRAFT)).isFalse();
    }

    @Test
    void confirmed_is_terminal() {
        assertThat(VcScheduleStatus.CONFIRMED.isTerminal()).isTrue();
        assertThat(VcScheduleStatus.CONFIRMED.canTransitionTo(VcScheduleStatus.DRAFT)).isFalse();
        assertThat(VcScheduleStatus.CONFIRMED.canTransitionTo(VcScheduleStatus.CANDIDATE)).isFalse();
    }
}
```
