### Flyway `V4_1__alter_vc_schedule_status.sql`

```sql
-- Status enum-like CHECK м ңм•Ҫ
ALTER TABLE vc_schedule
    ADD COLUMN status VARCHAR(20) NOT NULL DEFAULT 'DRAFT'
        CHECK (status IN ('DRAFT', 'CANDIDATE', 'CONFIRMED')),
    ADD COLUMN confirmed_at TIMESTAMPTZ,
    ADD COLUMN confirmed_by VARCHAR(40);

CREATE INDEX idx_vc_schedule_status ON vc_schedule(status);
CREATE INDEX idx_vc_schedule_confirmed ON vc_schedule(confirmed_at) WHERE status = 'CONFIRMED';

COMMENT ON COLUMN vc_schedule.status IS 'EP-10 мғҒнғң лЁёмӢ : DRAFT вҶ’ CANDIDATE вҶ’ CONFIRMED (BR-X01)';
COMMENT ON COLUMN vc_schedule.confirmed_at IS 'EP-10: Planner нҷ•м • мӢңк°Ғ (audit мһ…л Ҙ)';
COMMENT ON COLUMN vc_schedule.confirmed_by IS 'EP-10: Planner role мӮ¬мҡ©мһҗ ID';

-- DB level м „мқҙ кІҖмҰқ нҠёлҰ¬кұ° (мһҳлӘ»лҗң м „мқҙ м°ЁлӢЁ)
CREATE OR REPLACE FUNCTION enforce_vc_status_transition()
RETURNS TRIGGER AS $$
BEGIN
    -- мӢ к·ң INSERTлҠ” DRAFTл§Ң н—Ҳмҡ©
    IF TG_OP = 'INSERT' AND NEW.status != 'DRAFT' THEN
        RAISE EXCEPTION 'BR-X01: new VC schedule must start as DRAFT, got: %', NEW.status;
    END IF;

    -- UPDATEлҠ” valid м „мқҙл§Ң
    IF TG_OP = 'UPDATE' AND OLD.status != NEW.status THEN
        IF NOT (
            (OLD.status = 'DRAFT' AND NEW.status = 'CANDIDATE') OR
            (OLD.status = 'CANDIDATE' AND NEW.status = 'CONFIRMED')
        ) THEN
            RAISE EXCEPTION 'BR-X01: invalid state transition % вҶ’ %', OLD.status, NEW.status;
        END IF;
    END IF;

    -- CONFIRMED м „мқҙ мӢң confirmedAtВ·confirmedBy н•„мҲҳ
    IF NEW.status = 'CONFIRMED' AND (NEW.confirmed_at IS NULL OR NEW.confirmed_by IS NULL) THEN
        RAISE EXCEPTION 'BR-X01: CONFIRMED requires confirmed_at and confirmed_by';
    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_vc_status_transition
    BEFORE INSERT OR UPDATE OF status ON vc_schedule
    FOR EACH ROW EXECUTE FUNCTION enforce_vc_status_transition();
```

### `VcScheduleStatus.java`

```java
public enum VcScheduleStatus {
    DRAFT,
    CANDIDATE,
    CONFIRMED;

    public boolean canTransitionTo(VcScheduleStatus target) {
        return switch (this) {
            case DRAFT -> target == CANDIDATE;
            case CANDIDATE -> target == CONFIRMED;
            case CONFIRMED -> false;        // final state
        };
    }

    public boolean isTerminal() { return this == CONFIRMED; }
}
```

### `StateTransitionRule.java`

```java
@Component
public class StateTransitionRule {

    public void requireTransition(VcScheduleStatus from, VcScheduleStatus to) {
        if (!from.canTransitionTo(to)) {
            throw new InvalidStateTransitionException(
                "BR-X01 invalid VC state transition: %s вҶ’ %s".formatted(from, to));
        }
    }
}

public class InvalidStateTransitionException extends RuntimeException {
    public InvalidStateTransitionException(String msg) { super(msg); }
}
```
