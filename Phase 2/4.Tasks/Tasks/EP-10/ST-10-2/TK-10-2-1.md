### Flyway `V4_2__alter_ex_schedule_status.sql`

```sql
ALTER TABLE ex_schedule
    ADD COLUMN status VARCHAR(20) NOT NULL DEFAULT 'DRAFT'
        CHECK (status IN ('DRAFT', 'CANDIDATE', 'CONFIRMED')),
    ADD COLUMN confirmed_at TIMESTAMPTZ,
    ADD COLUMN confirmed_by VARCHAR(40);

CREATE INDEX idx_ex_schedule_status ON ex_schedule(status);
CREATE INDEX idx_ex_schedule_confirmed ON ex_schedule(confirmed_at) WHERE status = 'CONFIRMED';

-- 전이 트리거 (VC 패턴 동일)
CREATE OR REPLACE FUNCTION enforce_ex_status_transition()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' AND NEW.status != 'DRAFT' THEN
        RAISE EXCEPTION 'BR-X01: new EX schedule must start as DRAFT';
    END IF;
    IF TG_OP = 'UPDATE' AND OLD.status != NEW.status THEN
        IF NOT (
            (OLD.status = 'DRAFT' AND NEW.status = 'CANDIDATE') OR
            (OLD.status = 'CANDIDATE' AND NEW.status = 'CONFIRMED')
        ) THEN
            RAISE EXCEPTION 'BR-X01: invalid EX state transition % → %', OLD.status, NEW.status;
        END IF;
    END IF;
    IF NEW.status = 'CONFIRMED' AND (NEW.confirmed_at IS NULL OR NEW.confirmed_by IS NULL) THEN
        RAISE EXCEPTION 'BR-X01: CONFIRMED requires confirmed_at and confirmed_by';
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_ex_status_transition
    BEFORE INSERT OR UPDATE OF status ON ex_schedule
    FOR EACH ROW EXECUTE FUNCTION enforce_ex_status_transition();

REVOKE UPDATE (status, confirmed_at, confirmed_by) ON ex_schedule FROM app_user;
```

### Flyway `V4_2_data__create_ex_confirm_function.sql`

```sql
CREATE OR REPLACE FUNCTION ex_schedule_confirm(
    p_schedule_id UUID,
    p_planner_id VARCHAR(40)
) RETURNS VOID AS $$
DECLARE v_old_status VARCHAR(20);
BEGIN
    SELECT status INTO v_old_status FROM ex_schedule WHERE id = p_schedule_id;
    IF v_old_status IS NULL THEN
        RAISE EXCEPTION 'BR-X01: ex_schedule % not found', p_schedule_id;
    END IF;
    IF v_old_status != 'CANDIDATE' THEN
        RAISE EXCEPTION 'BR-X01: cannot confirm from %, must be CANDIDATE', v_old_status;
    END IF;
    UPDATE ex_schedule SET status='CONFIRMED', confirmed_at=now(), confirmed_by=p_planner_id
     WHERE id = p_schedule_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

REVOKE EXECUTE ON FUNCTION ex_schedule_confirm(UUID, VARCHAR) FROM PUBLIC;
GRANT EXECUTE ON FUNCTION ex_schedule_confirm(UUID, VARCHAR) TO planner_role;
```

### `ExScheduleStatus.java`

```java
public enum ExScheduleStatus {
    DRAFT, CANDIDATE, CONFIRMED;
    public boolean canTransitionTo(ExScheduleStatus target) {
        return switch (this) {
            case DRAFT -> target == CANDIDATE;
            case CANDIDATE -> target == CONFIRMED;
            case CONFIRMED -> false;
        };
    }
    public boolean isTerminal() { return this == CONFIRMED; }
}
```

### `ExScheduleConfirmService.java`

```java
@Service @RequiredArgsConstructor
public class ExScheduleConfirmService {

    private final JdbcTemplate jdbc;
    private final ExConfirmedPublisher publisher;
    private final SecurityContextService securityCtx;

    @Transactional
    public void confirm(UUID scheduleId) {
        String plannerId = securityCtx.currentUserId();
        jdbc.execute("SET LOCAL ROLE planner_role");
        jdbc.update("SELECT ex_schedule_confirm(?, ?)", scheduleId, plannerId);
        publisher.publishConfirmed(scheduleId);
    }
}
```

### `ExScheduleStateController.java`

```java
@RestController @RequestMapping("/api/v1/schedule/ex") @RequiredArgsConstructor
public class ExScheduleStateController {

    private final ExScheduleConfirmService confirmService;

    @PostMapping("/{id}/confirm")
    @PreAuthorize("hasRole('PLANNER')")
    public ResponseEntity<Void> confirm(@PathVariable UUID id) {
        confirmService.confirm(id);
        return ResponseEntity.ok().build();
    }
}
```
