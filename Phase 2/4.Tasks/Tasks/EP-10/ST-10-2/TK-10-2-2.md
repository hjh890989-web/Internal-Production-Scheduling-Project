### 핵심 로직

```java
@SpringBootTest @Testcontainers
class ExConfirmGateNegativeIT {

    @Autowired DataSource dataSource;
    @Autowired ExScheduleConfirmService service;
    @Autowired ExScheduleRepository repo;

    @Test
    void app_user_direct_update_status_denied() throws SQLException {
        var schedule = createCandidateExSchedule();
        try (Connection conn = openAsAppUser()) {
            try (PreparedStatement ps = conn.prepareStatement(
                "UPDATE ex_schedule SET status='CONFIRMED' WHERE id=?")) {
                ps.setObject(1, schedule.getId());
                assertThatThrownBy(ps::executeUpdate)
                    .isInstanceOf(SQLException.class)
                    .hasMessageContaining("permission denied");
            }
        }
    }

    @Test
    void draft_to_confirmed_skip_blocked() {
        var draft = createDraftExSchedule();
        assertThatThrownBy(() -> service.confirm(draft.getId()))
            .hasMessageContaining("must be CANDIDATE");
    }

    @Test
    void planner_role_confirms_and_publishes_event() {
        var schedule = createCandidateExSchedule();
        var eventCapture = new ExConfirmedEventCapture();

        withPlannerRole("planner1", () -> service.confirm(schedule.getId()));

        var confirmed = repo.findById(schedule.getId()).orElseThrow();
        assertThat(confirmed.getStatus()).isEqualTo(ExScheduleStatus.CONFIRMED);
        assertThat(eventCapture.events).hasSize(1);
        assertThat(eventCapture.events.get(0).scheduleId()).isEqualTo(schedule.getId());
    }

    @Test
    void user_role_api_call_forbidden() throws Exception {
        var schedule = createCandidateExSchedule();
        mockMvc.perform(post("/api/v1/schedule/ex/{id}/confirm", schedule.getId())
                .with(user("user1").roles("USER")))
            .andExpect(status().isForbidden());
    }

    @Test
    void insert_with_confirmed_status_blocked() throws SQLException {
        try (Connection conn = openAsAppUser()) {
            try (PreparedStatement ps = conn.prepareStatement(
                "INSERT INTO ex_schedule (id, status, ...) VALUES (?, 'CONFIRMED', ...)")) {
                ps.setObject(1, UUID.randomUUID());
                assertThatThrownBy(ps::executeUpdate)
                    .isInstanceOf(SQLException.class);
            }
        }
    }

    @Test
    void already_confirmed_cannot_reconfirm() {
        var confirmed = createConfirmedExSchedule();
        assertThatThrownBy(() -> service.confirm(confirmed.getId()))
            .hasMessageContaining("cannot confirm");
    }
}

class ExStateTransitionUnitTest {

    @Test
    void draft_can_transition_to_candidate_only() {
        assertThat(ExScheduleStatus.DRAFT.canTransitionTo(ExScheduleStatus.CANDIDATE)).isTrue();
        assertThat(ExScheduleStatus.DRAFT.canTransitionTo(ExScheduleStatus.CONFIRMED)).isFalse();
    }

    @Test
    void candidate_can_transition_to_confirmed_only() {
        assertThat(ExScheduleStatus.CANDIDATE.canTransitionTo(ExScheduleStatus.CONFIRMED)).isTrue();
        assertThat(ExScheduleStatus.CANDIDATE.canTransitionTo(ExScheduleStatus.DRAFT)).isFalse();
    }

    @Test
    void confirmed_is_terminal() {
        assertThat(ExScheduleStatus.CONFIRMED.isTerminal()).isTrue();
        assertThat(ExScheduleStatus.CONFIRMED.canTransitionTo(ExScheduleStatus.DRAFT)).isFalse();
    }
}
```
